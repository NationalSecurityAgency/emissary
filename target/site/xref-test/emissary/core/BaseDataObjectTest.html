116     void testLargestFile() throws IOException {
117         final BaseDataObject bdo = new BaseDataObject();
118         final long fileSize = Long.MAX_VALUE;
119         final SeekableByteChannelFactory sbcf = FillChannelFactory.create(fileSize, (byte) 0);
120         bdo.setChannelFactory(sbcf);
121         assertEquals(fileSize, bdo.getChannelSize());
122         assertEquals(BaseDataObject.MAX_BYTE_ARRAY_SIZE, bdo.dataLength());
123 
124         final SeekableByteChannel sbc = sbcf.create();
125         final long newPosition = ThreadLocalRandom.current().nextLong(Integer.MAX_VALUE, Long.MAX_VALUE);
126         sbc.position(newPosition);
127         assertEquals(newPosition, sbc.position());
128         final ByteBuffer buff = ByteBuffer.allocate(16);
129         final int bytesRead = sbc.read(buff);
130         assertEquals(16, bytesRead);
131         final byte[] zeroByteArray = new byte[16];
132         Arrays.fill(zeroByteArray, (byte) 0);
133         assertArrayEquals(zeroByteArray, buff.array());
134     }

66      @Test
67      void testWithNoValidOutputTypes() throws Exception {
68          final IBaseDataObject payload = DataObjectFactory.getInstance();
69          payload.setData("This is the data".getBytes());
70          payload.setFileType("FTYPE");
71          payload.setFilename("/this/is/a/testfile");
72          final List<IBaseDataObject> payloadList = new ArrayList<>();
73          payloadList.add(payload);
74          final List<IBaseDataObject> val = this.place.agentProcessHeavyDuty(payloadList);
75          assertEquals(1, payloadList.size(), "All payloads still on list");
76          assertEquals(0, val.size(), "Nothing returned from drop off");
77          assertEquals(0, payloadList.get(0).currentFormSize(), "All current forms removed");
78      }

81      @Test
82      void testOutputMessageContainsPayloadSize() throws Exception {
83          final LoggerContext lc = (LoggerContext) LoggerFactory.getILoggerFactory();
84          final Logger rootLogger = lc.getLogger(Logger.ROOT_LOGGER_NAME);
85  
86          ListAppender<ILoggingEvent> appender = new ListAppender<>();
87          byte[] content = "This is the data".getBytes();
88          String expected = "payload size: " + content.length + " bytes";
89  
90          try {
91  
92              appender.start();
93              rootLogger.addAppender(appender);
94  
95              final IBaseDataObject payload = DataObjectFactory.getInstance();
96              payload.setData(content);
97  
98              payload.setCurrentForm("FOO");
99              payload.setFileType("FTYPE");
100             payload.setFilename("/this/is/a/testfile");
101             final List<IBaseDataObject> payloadList = new ArrayList<>();
102             payloadList.add(payload);
103 
104             final List<IBaseDataObject> val = this.place.agentProcessHeavyDuty(payloadList);
105             assertTrue(appender.list.stream().anyMatch(i -> i.getFormattedMessage().endsWith(expected)));
106         } finally {
107             rootLogger.detachAppender(appender);
108             appender.stop();
109         }
110     }

145     public static void cleanupDirectoryRecursively(Path path) throws IOException {
146         Files.walkFileTree(path, new SimpleFileVisitor<>() {
147             @Override
148             public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
149                 Files.delete(file);
150                 return FileVisitResult.CONTINUE;
151             }
152 
153             @Override
154             public FileVisitResult postVisitDirectory(Path dir, IOException exc) throws IOException {
155                 Files.delete(dir);
156                 return FileVisitResult.CONTINUE;
157             }
158         });

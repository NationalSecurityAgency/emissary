1   package emissary.core;

3   import com.codahale.metrics.Snapshot;
4   import com.codahale.metrics.Timer;
5   
6   import java.util.concurrent.TimeUnit;
7   
8   /**
9    * Formatter for metrics data, following the existing format emissary format, established previously. Employs the
10   * builder pattern of the Reporters from coda hale's metrics package. Currently supports Timers, but could be expanded
11   * to support any number of metrics classes.
12   */
13  public class MetricsFormatter {
14      /** Create a builder */
15      public static Builder builder() {
16          return new Builder();
17      }
18  
19      public static class Builder {
20          private TimeUnit durationUnit;
21          private TimeUnit rateUnit;
22  
23          /** Specify the duration unit for the formatter to be built */
24          public Builder withDurationUnit(final TimeUnit t) {
25              this.durationUnit = t;
26              return this;
27          }
28  
29          /** Specify the rate unit for the formatter to be build */
30          public Builder withRateUnit(final TimeUnit t) {
31              this.rateUnit = t;
32              return this;
33          }
34  
35          /** Build the formatter */
36          public MetricsFormatter build() {
37              return new MetricsFormatter(this.rateUnit, this.durationUnit);
38          }
39      }
40  
41      private final double durationFactor;
42  
43      /** Use a builder to create MetricsFormatter instances */
44      private MetricsFormatter(final TimeUnit rateUnit, final TimeUnit durationUnit) {
45          this.durationFactor = 1.0 / durationUnit.toNanos(1);
46      }
47  
48      public String formatTimer(final String name, final Timer timer) {
49          final Snapshot snapshot = timer.getSnapshot();
50          return String.format("STAT: %s => min=%2.2f,  max=%2.2f, avg=%2.2f, events=%d", name, convertDuration(snapshot.getMin()),
51                  convertDuration(snapshot.getMax()), convertDuration(snapshot.getMean()), timer.getCount());
52      }
53  
54      protected double convertDuration(final double duration) {
55          return duration * this.durationFactor;
56      }
57  }

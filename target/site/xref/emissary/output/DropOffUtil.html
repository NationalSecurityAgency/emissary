100      * Set up config for this class
101      * <ul>
102      * <li>ID_PARAMETER : multiple parameter values, ordered list of how to build a EMISSARY_ID</li>
103      * <li>ID : backwards compatibility for ID_PARAMETER only used if ID_PARAMETER does not exist</li>
104      * <li>DATE_PARAMETER : multiple parameter values, ordered list of how to build a date path</li>
105      * <li>DATE_FORMAT: string to use for SimpleDateFormat, ordered list</li>
106      * <li>OUTPUT_FILE_PREFIX: string to use when generating random filenames, dflt: TXT</li>
107      * <li>UUID_IN_OUTPUT_FILENAMES: boolean [true]</li>
108      * <li>AUTO_GENERATED_ID_PREFIX: prefix to use for an auto-generated id</li>
109      * </ul>

111     protected void configure(final Configurator configG) {
112         Configurator actualConfigG = configG;
113         if (actualConfigG == null) {
114             try {
115                 actualConfigG = ConfigUtil.getConfigInfo(DropOffUtil.class);
116             } catch (IOException e) {
117                 logger.error("Cannot open default config file", e);
118             }
119         }
120 
121         if (actualConfigG != null) {
122             this.placeOutputData = actualConfigG.findStringEntry("OUTPUT_DATA", "outputData");
123             this.unixRoot = actualConfigG.findStringEntry("UNIX_ROOT", null);
124             this.executrix = new Executrix(actualConfigG);
125             this.idTokens = actualConfigG.findEntries("ID_PARAMETER");
126             this.autoGeneratedIdPrefix = actualConfigG.findStringEntry("AUTO_GENERATED_ID_PREFIX");
127 
128             // truncate the prefix if necessary
129             if (!StringUtils.isBlank(this.autoGeneratedIdPrefix) && (this.autoGeneratedIdPrefix.length() > 4)) {
130                 this.autoGeneratedIdPrefix = this.autoGeneratedIdPrefix.substring(0, 4);
131             }
132             if (this.idTokens.size() == 0) {
133                 this.idTokens = actualConfigG.findEntries("ID");
134             }
135             this.dateTokens = actualConfigG.findEntries("DATE_PARAMETER");
136             this.parentParams = actualConfigG.findEntries("PARENT_PARAM");
137 
138 
139             this.defaultEventDateToNow = actualConfigG.findBooleanEntry(DEFAULT_EVENT_DATE_TO_NOW, this.defaultEventDateToNow);
140 
141             prefix = actualConfigG.findStringEntry("OUTPUT_FILE_PREFIX", prefix);
142             this.uuidInOutputFilenames = actualConfigG.findBooleanEntry("UUID_IN_OUTPUT_FILENAMES", this.uuidInOutputFilenames);
143             // limit the length of the FILEXT param
144             this.maxFilextLen = actualConfigG.findIntEntry("MAX_FILEXT_LEN", this.maxFilextLen);
145             if (this.maxFilextLen < 0) {
146                 this.maxFilextLen = Integer.MAX_VALUE;
147             }
148         } else {
149             logger.debug("Configuration is null for DropOffUtil, using defaults");
150             this.executrix = new Executrix();
151         }

181     public String makeWayForIncomingFile(final IBaseDataObject DO, final String suffix, final String spec) {
182         // Construct output file from DropOff output path and IBaseDataObject
183         // filename
184         final String outputFile = getShortOutputFileName(DO, spec);
185 
186         // Return value when finally set
187         final String fileName = outputFile + suffix;
188 
189         // Set up path and make it writable
190         if (!setupPath(fileName)) {
191             return null;
192         }
193 
194         if (removeExistingFile(fileName)) {
195             return fileName;
196         }
197 
198         return null;
199     }

205         // If a file already exists under this name, we're going
206         // to have to move it aside. This is going to break the link
207         // if it is an attachment to another parent document.
208         final Path theFile = Paths.get(fileName);
209         try {
210             Files.deleteIfExists(theFile);
211         } catch (IOException e) {
212             logger.error("Trouble removing file: {}", theFile, e);
213         }
214         return !Files.exists(theFile);

217     /**
218      * mkdir -p to a path and make sure it is writable
219      *
220      * @param fileName the file name, including directory and filename parts
221      * @return true iff it works
222      */
223     public boolean setupPath(final String fileName) {
224         final String pathName = fileName.substring(0, fileName.lastIndexOf(SEPARATOR));
225         final Path thePath = Paths.get(pathName);
226 
227         // If the specified output directory doesn't exist try creating it
228         if (!Files.exists(thePath)) {
229             int tryCount = 1;
230             do {
231                 try {
232                     Files.createDirectories(thePath);
233                 } catch (IOException e) {
234                     logger.warn("Trouble setting up directories:{}", thePath, e);
235                 }
236                 if (!Files.exists(thePath)) {
237                     try {
238                         Thread.sleep(50L * tryCount);
239                     } catch (InterruptedException e) {
240                         Thread.currentThread().interrupt();
241                     }
242                 }
243                 tryCount++;
244             } while (!Files.exists(thePath) && tryCount <= 10);
245 
246             if (!Files.exists(thePath)) {
247                 logger.warn("Cannot create directory for output: {} in {} attempts", thePath, tryCount);
248                 return false;
249             }
250 
251             if (tryCount > 2 && logger.isDebugEnabled()) {
252                 logger.debug("Output path created for {} but it took {} attempts", thePath, tryCount);
253             }
254         }
255 
256         // If the specified output directory doesn't have write
257         // permission try to fix it
258         if (!Files.isWritable(thePath)) {
259             logger.warn("Dont have write permission for {}, setting it now", pathName);
260             if (!thePath.toFile().setWritable(true) || !Files.isWritable(thePath)) {
261                 logger.warn("Cannot write to directory for output: {}", thePath);
262                 return false;
263             }
264         }
265 
266         return true;
267     }

285     /**
286      * Create a path name from the spec for the specified spec Specs understand the following 'language' of replacement
287      * stuff. Anything not understood is a literal
288      *
289      * <pre>
290      * <code>@TLD{'KEY'}</code> is to pull the named KEY from the top level document Metadata
291      * <code>@META{'KEY'}</code> is to pull the named KEY from the MetaData
292      * %U% = USER
293      * %I% = INPUT_FILE_NAME (whole thing)
294      * %S% = INPUT_FILE SHORT NAME
295      * %P% = INPUT_FILE PATH (all but short name)
296      * %i% = INPUT_FILE_NAME with slashes converted to underscores
297      * %p% = INPUT_FILE PATH with slashes converted to underscores
298      * %F% = FILETYPE
299      * %L% = LANGUAGE
300      * %G% = DTG multi directory layout yyyy-mm-dd/hh/mi(div)10
301      * %R% = ROOT (Unix or Win depending on OS)
302      * %B% = ID for the payload depending on type (no -att-)
303      * %b% = ID for the payload depending on type (with -att-)
304      * %Y% = Four digit year
305      * %M% = Two digit month
306      * %D% = Two digit day of month
307      * %J% = Three digit ordinal day of the year
308      * </pre>
309      *
310      * @param specArg the incoming specification
311      * @param d the payload we are making a path for
312      * @param tld the top level document in the d family, possibly null
313      * @return string path name with correct separators for this OS
314      */
315     public String getPathFromSpec(final String specArg, @Nullable final IBaseDataObjectt">IBaseDataObject d, @Nullable final IBaseDataObject tld) {
316         final StringBuffer sb = new StringBuffer(128);
317 
318         // Provide a default spec, just like the old days...
319         String spec = specArg;
320         if (spec == null) {
321             spec = "%R%/@TLD{'TARGETBIN'}/%S%";
322         }
323 
324         for (int i = 0; i < spec.length(); i++) {
325             final char c = spec.charAt(i);
326 
327             if (c == '%' && i < spec.length() - 2) {
328                 final char t = spec.charAt(i + 1);
329                 final char x = spec.charAt(i + 2);
330 
331                 if (x == c) {
332                     switch (t) {
333                         case 'U':
334                             if (tld != null) {
335                                 sb.append(nvl(tld.getParameter("UserName"), "no-userid"));
336                             } else if (d != null) {
337                                 sb.append(nvl(d.getParameter("UserName"), "no-userid"));
338                             }
339                             break;
340                         case 'S':
341                             if (d != null) {
342                                 sb.append(d.shortName());
343                             }
344                             break;
345                         case 'I':
346                             if (d != null) {
347                                 sb.append(d.getFilename());
348                             }
349                             break;
350                         case 'i':
351                             if (d != null) {
352                                 sb.append(d.getFilename().replaceAll("[/\\\\]", "_"));
353                             }
354                             break;
355                         case 'P':
356                             if (d != null) {
357                                 sb.append(d.getFilename().substring(0, d.getFilename().length() - d.shortName().length()));
358                             }
359                             break;
360                         case 'p':
361                             if (d != null) {
362                                 sb.append(d.getFilename().substring(0, d.getFilename().length() - d.shortName().length()).replaceAll("[/\\\\]", "_"));
363                             }
364                             break;
365                         case 'F':
366                             if (d != null) {
367                                 sb.append(nvl(cleanSpecPath(d.getFileType()), "NONE"));
368                             }
369                             break;
370                         case 'L':
371                             if (d != null) {
372                                 sb.append(nvl(d.getParameter("LANGUAGE"), "NONE"));
373                             }
374                             break;
375                         case 'G':
376                             if (tld != null) {
377                                 sb.append(datePath(cleanSpecPath(tld.getStringParameter("DTG"))));
378                             } else if (d != null) {
379                                 sb.append(datePath(cleanSpecPath(d.getStringParameter("DTG"))));
380                             }
381                             break;
382                         case 'R':
383                             sb.append(getRootPath());
384                             break;
385                         case 'B':
386                             if (tld != null) {
387                                 sb.append(cleanSpecPath(getBestIdFrom(tld)));
388                             } else if (d != null) {
389                                 sb.append(cleanSpecPath(getBestIdFrom(d)));
390                             }
391                             break;
392                         case 'b':
393                             sb.append(cleanSpecPath((tld != null) ? getBestIdFrom(tld) : getBestIdFrom(d)));
394                             final String sn = d.shortName();
395                             final int pos = sn.indexOf(emissary.core.Family.SEP);
396                             if (pos > 0) {
397                                 sb.append(sn.substring(pos));
398                             }
399                             break;
400                         case 'Y':
401                             sb.append(TimeUtil.getDate("yyyy", "GMT"));
402                             break;
403                         case 'M':
404                             sb.append(TimeUtil.getDate("MM", "GMT"));
405                             break;
406                         case 'D':
407                             sb.append(TimeUtil.getDate("dd", "GMT"));
408                             break;
409                         case 'J':
410                             sb.append(TimeUtil.getDate("DDD", "GMT"));
411                             break;
412                         default:
413                             sb.append(c).append(t).append(x);
414                     }
415                     i += 2; // SUPPRESS CHECKSTYLE ModifiedControlVariable
416                 } else {
417                     // No trailing % after character
418                     sb.append(c);
419                 }
420 
421             } else if (c == '@' && i < spec.length() - 8 && spec.charAt(i + 1) == 'M' && spec.charAt(i + 2) == 'E' && spec.charAt(i + 3) == 'T'
422                     && spec.charAt(i + 4) == 'A' && spec.charAt(i + 5) == '{' && spec.charAt(i + 6) == '\'') {
423 
424                 final int endpos = spec.indexOf("'", i + 7);
425                 if (endpos > i + 7) {
426                     final String token = spec.substring(i + 7, endpos);
427                     final String value = cleanSpecPath(d.getStringParameter(token));
428                     sb.append(nvl(value, "NO-" + token));
429                     i += 8 + token.length(); // META{'token'} SUPPRESS CHECKSTYLE ModifiedControlVariable
430                 } else {
431                     sb.append(c);
432                 }
433             } else if (c == '@' && i < spec.length() - 7 && spec.charAt(i + 1) == 'T' && spec.charAt(i + 2) == 'L' && spec.charAt(i + 3) == 'D'
434                     && spec.charAt(i + 4) == '{' && spec.charAt(i + 5) == '\'' && tld != null) {
435                 final int endpos = spec.indexOf("'", i + 6);
436                 if (endpos > i + 6) {
437                     final String token = spec.substring(i + 6, endpos);
438                     final String value = cleanSpecPath(tld.getStringParameter(token));
439                     sb.append(nvl(value, "NO-" + token));
440                     i += 7 + token.length(); // TLD{'token'} SUPPRESS CHECKSTYLE ModifiedControlVariable
441                 } else {
442                     sb.append(c);
443                 }
444             } else {
445                 sb.append(c);
446             }
447 
448         }
449 
450         String answer = sb.toString();
451 
452         // Set the proper path separator
453         answer = answer.replace('\\', '/');
454         answer = answer.replaceAll("\\.([/\\\\])", "_$1");
455 
456         return answer;
457     }

463     /**
464      * Extract the ID from the payload. The ID from the payload is specified in the cfg file. An ID = SHORTNAME will use the
465      * shortname. An ID = AUTO_GENERATED_ID will use an auto gen uuid. If no id value is found, defaults to using an auto
466      * generated id.
467      *
468      * @return id
469      */
470     public String getBestIdFrom(final IBaseDataObject d) {
471         for (final String s : this.idTokens) {
472             if (!StringUtils.isBlank(d.getStringParameter(s))) {
473                 return d.getStringParameter(s);
474             }
475             if (SHORTNAME.equals(s)) {
476                 final String shortName = d.shortName();
477                 // if shortname is not blank use it, if blank move on...
478                 if (!StringUtils.isBlank(shortName)) {
479                     return shortName;
480                 }
481             }
482             if (AUTO_GENERATED_ID.equals(s)) {
483                 return getRandomUUID(d);
484             }
485         }
486         // if nothing else worked, use an auto gen id
487         return getRandomUUID(d);
488     }

582     /**
583      * Replace any file separators that are not for this platform with the correct one
584      */
585     public String fixFileNameSeparators(final String s) {
586         // other platform;
587         String badfs = null;
588 
589         if ("/".equals(SEPARATOR)) {
590             badfs = "\\";
591         } else {
592             badfs = "/";
593         }
594 
595         final StringBuilder ret = new StringBuilder(s.length());
596 
597         for (int i = 0; i < s.length(); i++) {
598             if (s.charAt(i) == badfs.charAt(0)) {
599                 ret.append(SEPARATOR);
600             } else {
601                 ret.append(s.charAt(i));
602             }
603         }
604 
605         return ret.toString();
606     }

632      * Make a dot file name from the supplied path Ex: /path/to/file.txt --&gt; /path/to/.file.txt
633      *
634      * @param fullName the full path and filename of the file
635      * @return the path as is but with a leading dot in the filename
636      */
637     public String makeDotFile(final String fullName) {
638         final int fpos = fullName.lastIndexOf("/");
639         final int rpos = fullName.lastIndexOf("\\");
640         if (fpos == -1 && rpos == -1) {
641             return "." + fullName;
642         }
643         final int pos = Math.max(fpos, rpos);
644         return fullName.substring(0, pos + 1) + "." + fullName.substring(pos + 1);
645     }

657     /**
658      * Get the file type from the IBaseDataObject or the form string passed in
659      *
660      * @param bdo IBaseDataObject
661      * @param metaData Optional map of metadata that might be modified.
662      * @param formsArg Optional space separated string of current forms
663      * @return the file type
664      */
665     public static String getAndPutFileType(final IBaseDataObject bdo, @Nullable final Map<String, String> metaData, final String formsArg) {
666         String forms = formsArg;
667         if (forms == null) {
668             forms = bdo.getStringParameter(FileTypeCheckParameter.POPPED_FORMS.getFieldName());
669             if (forms == null) {
670                 forms = "";
671             }
672         }
673 
674         String fileType;
675         if (bdo.hasParameter(FileTypeCheckParameter.FILETYPE.getFieldName())) {
676             fileType = bdo.getStringParameter(FileTypeCheckParameter.FILETYPE.getFieldName());
677         } else if (bdo.hasParameter(FileTypeCheckParameter.FINAL_ID.getFieldName())) {
678             fileType = bdo.getStringParameter(FileTypeCheckParameter.FINAL_ID.getFieldName());
679             logger.debug("FINAL_ID FileType is ({})", fileType);
680             if (metaData != null) {
681                 metaData.put(FileTypeCheckParameter.FILETYPE.getFieldName(), fileType);
682             }
683         } else {
684             if (forms.contains(" ")) {
685                 fileType = forms.substring(0, forms.indexOf(" ")).trim();
686                 if (metaData != null) {
687                     metaData.put(FileTypeCheckParameter.COMPLETE_FILETYPE.getFieldName(), forms);
688                 }
689             } else {
690                 fileType = forms;
691             }
692             if (StringUtils.isEmpty(fileType)) {
693                 if (bdo.hasParameter(FileTypeCheckParameter.FONT_ENCODING.getFieldName())) {
694                     fileType = TEXT;
695                 } else {
696                     fileType = UNKNOWN;
697                 }
698             }
699 
700             if (metaData != null) {
701                 metaData.put(FileTypeCheckParameter.FILETYPE.getFieldName(), fileType);
702             }
703         }
704 
705         if (UNKNOWN.equals(fileType) && forms.contains("MSWORD")) {
706             fileType = "MSWORD_FRAGMENT";
707         }
708 
709         if ("QUOTED-PRINTABLE".equals(fileType) || fileType.startsWith(PREFIXES_LANG) || fileType.startsWith("ENCODING(")) {
710             fileType = TEXT;
711         }
712 
713         return fileType;
714     }

734     /**
735      * Extract the ID from the payload. The ID from the payload is specified in the cfg file. An ID = SHORTNAME will use the
736      * shortname. An ID = AUTO_GENERATED_ID will use an auto gen uuid. If no id value is found, defaults to using an auto
737      * generated id.
738      *
739      * @param d the payload
740      * @param tld if a param is specified and cannot be found in the d method parameter, tries to get param from tld.
741      * @return the id based the best ID available, shortname or auto gen id. If no id is found, defaults to auto gen.
742      */
743     public String getBestId(final IBaseDataObjectt">IBaseDataObject d, @Nullable final IBaseDataObject tld) {
744 
745         for (final String s : this.idTokens) {
746             if (AUTO_GENERATED_ID.equals(s)) {
747                 String parentAutoGeneratedId = null;
748                 if (tld != null) {
749                     parentAutoGeneratedId = tld.getStringParameter(PARENT_AUTO_GENERATED_ID);
750                 }
751                 if (StringUtils.isBlank(parentAutoGeneratedId)) {
752                     String uuid = getRandomUUID(d);
753                     if (tld != null) {
754                         tld.setParameter(PARENT_AUTO_GENERATED_ID, uuid);
755                     }
756                     if (!StringUtils.isBlank(uuid)) {
757                         final String component = d.shortName();
758                         final int pos = component.indexOf(Family.SEP);
759                         if (pos > -1) {
760                             uuid += component.substring(pos);
761                         }
762                         return uuid;
763                     }
764                 }
765                 String uuid = null;
766                 // try getting param from the tld
767                 if (!StringUtils.isBlank(parentAutoGeneratedId) && (tld != null)) {
768                     uuid = tld.getStringParameter(PARENT_AUTO_GENERATED_ID);
769                 }
770                 if (!StringUtils.isBlank(uuid)) {
771                     final String component = d.shortName();
772                     final int pos = component.indexOf(Family.SEP);
773                     if (pos > -1) {
774                         uuid += component.substring(pos);
775                     }
776                     d.setParameter(AUTO_GENERATED_ID, "yes");
777                     return uuid;
778                 }
779 
780             }
781 
782             if (SHORTNAME.equals(s)) {
783                 final String shortName = d.shortName();
784 
785                 // if shortname is not blank use it, if blank move on...
786                 if (!StringUtils.isBlank(shortName)) {
787                     return shortName;
788                 }
789 
790                 String path = d.getStringParameter(s);
791                 // try getting shortname from the tld
792                 if (StringUtils.isBlank(path) && (tld != null)) {
793                     path = tld.getStringParameter(s);
794                 }
795                 if (!StringUtils.isBlank(path)) {
796                     final String component = d.shortName();
797                     final int pos = component.indexOf(Family.SEP);
798                     if (pos > -1) {
799                         path += component.substring(pos);
800                     }
801                     return path;
802                 }
803 
804             }
805             // the param from the tld has priority over any child
806             if ((tld != null) && !StringUtils.isBlank(tld.getStringParameter(s))) {
807                 String path = tld.getStringParameter(s);
808                 if (!StringUtils.isBlank(path)) {
809                     final String component = d.shortName();
810                     final int pos = component.indexOf(Family.SEP);
811                     if (pos > -1) {
812                         path += component.substring(pos);
813                     }
814                     return path;
815                 }
816             }
817             // if the param is not in the tld
818             if (!StringUtils.isBlank(d.getStringParameter(s))) {
819                 return d.getStringParameter(s);
820             }
821 
822         }
823 
824         // if nothing works, auto gen an id
825         final String uuid = getRandomUUID(d);
826         if (tld != null) {
827             tld.setParameter(PARENT_AUTO_GENERATED_ID, uuid);
828         }
829         return uuid;
830     }

833      * Creates a UUID. Includes a prefix if specified.
834      *
835      * @param d Adds a parameter indicating an auto gen id.
836      * @return uuid
837      */
838     private String getRandomUUID(final IBaseDataObject d) {
839         String uuid = UUID.randomUUID().toString();
840         // use the prefix
841         if (!StringUtils.isBlank(this.autoGeneratedIdPrefix)) {
842             uuid = uuid.substring(this.autoGeneratedIdPrefix.length());
843             uuid = this.autoGeneratedIdPrefix + uuid;
844         }
845         d.setParameter(PARENT_AUTO_GENERATED_ID, uuid);
846         d.setParameter(AUTO_GENERATED_ID, "yes");
847         return uuid;
848     }

897     /**
898      * Extracts from the provided {@link IBaseDataObject} the last file extension from each value in the "Original-Filename"
899      * parameter, if that value contains "." before its last character. If one or more file extensions are extracted, the
900      * IBaseDataObject's "FILEXT" parameter is set as the unique set of extracted file extensions, converted to lowercase.
901      *
902      * @param p IBaseDataObject to process
903      *
904      */
905     void extractUniqueFileExtensions(IBaseDataObject p) {
906         if (p.hasParameter(ORIGINAL_FILENAME)) {
907             final Set<String> extensions = new HashSet<>();
908             for (Object filename : p.getParameter(ORIGINAL_FILENAME)) {
909                 final String fn = (String) filename;
910                 if (StringUtils.isNotEmpty(fn) && fn.lastIndexOf('.') > -1) {
911                     final int pos = fn.lastIndexOf('.') + 1;
912                     if (pos < fn.length()) {
913                         final String fext = fn.substring(pos).toLowerCase();
914                         if (fext.length() > 0 && fext.length() <= this.maxFilextLen) {
915                             extensions.add(fext);
916                         }
917                     }
918                 }
919             }
920             if (!extensions.isEmpty()) {
921                 p.setParameter("FILEXT", extensions);
922             }
923         }
924     }

926     /**
927      * Process metadata before doing any output
928      *
929      * @param payloadList list of items being dropped off that may need initial metadata computations
930      */
931     public void processMetadata(final List<IBaseDataObject> payloadList) {
932 
933         // Keep track of parent's filetype to output
934         // relies on the attachments being sorted
935         final Map<String, String> parentTypes = new HashMap<>();
936         final IBaseDataObject tld = payloadList.get(0);
937         final List<String> extended_filetypes = new ArrayList<>();
938         parentTypes.put("1", tld.getFileType());
939         for (int i = 0; i < parentParams.size(); i++) {
940             final String param = parentParams.get(i);
941             if (tld.hasParameter(param)) {
942                 parentTypes.put("1" + param, tld.getStringParameter(param));
943             }
944         }
945 
946         for (final IBaseDataObject p : payloadList) {
947             final int level = StringUtils.countMatches(p.shortName(), emissary.core.Family.SEP) + 1;
948             // save specified metadata items for children to grab
949             parentTypes.put("" + level, p.getFileType());
950 
951             extractUniqueFileExtensions(p);
952 
953             if (p.getStringParameter("EXTENDED_FILETYPE") == null) {
954                 extended_filetypes.clear();
955                 for (final Map.Entry<String, Collection<Object>> entry : p.getParameters().entrySet()) {
956                     final String key = entry.getKey();
957                     if (key != null && key.endsWith("_FILETYPE")) {
958                         for (final Object value : entry.getValue()) {
959                             final String vs = value.toString();
960                             if (!extended_filetypes.contains(vs)) {
961                                 extended_filetypes.add(vs);
962                             }
963                         }
964                     }
965                 }
966                 if (!extended_filetypes.isEmpty()) {
967                     final StringBuilder extft = new StringBuilder(getFileType(p));
968                     for (int j = 0; j < extended_filetypes.size(); j++) {
969                         final String s = extended_filetypes.get(j);
970                         extft.append("//").append(s);
971                     }
972                     p.setParameter("EXTENDED_FILETYPE", extft.toString());
973                 }
974             }
975 
976             for (int j = 0; j < parentParams.size(); j++) {
977                 final String param = parentParams.get(j);
978                 if (p.hasParameter(param)) {
979                     parentTypes.put("" + level + param, p.getStringParameter(param));
980                 } else {
981                     // Must clear to keep my children from getting their uncle's value
982                     parentTypes.remove("" + level + param);
983                 }
984 
985             }
986 
987             if (level > 1) {
988                 // then it might want some PARENT info
989 
990                 final int parentLevel = level - 1;
991                 final String pType = parentTypes.get("" + parentLevel);
992                 if (pType != null && pType.length() > 0) {
993                     p.setParameter("PARENT_FILETYPE", pType);
994                 } else {
995                     p.setParameter("PARENT_FILETYPE", parentTypes.get("1"));
996                 }
997                 for (int j = 0; j < parentParams.size(); j++) {
998                     final String param = parentParams.get(j);
999                     int plvl = parentLevel;
1000                     while (plvl > 1 && !parentTypes.containsKey("" + plvl + param)) {
1001                         plvl--;
1002                     }
1003                     if (StringUtils.isNotBlank(parentTypes.get(plvl + param))) {
1004                         p.setParameter("PARENT_" + param, parentTypes.get(plvl + param));
1005                     }
1006                 }
1007             }
1008 
1009             if (p.hasExtractedRecords()) {
1010                 final List<IBaseDataObject> childObjList = p.getExtractedRecords();
1011                 Collections.sort(childObjList, new ShortNameComparator());
1012                 for (final IBaseDataObject child : childObjList) {
1013                     final int parentLevel = StringUtils.countMatches(child.shortName(), emissary.core.Family.SEP);
1014                     final String parentFileType = parentTypes.get("" + parentLevel);
1015                     if (parentFileType != null) {
1016                         child.setParameter("PARENT_FILETYPE", parentFileType);
1017                     }
1018                     for (int k = 0; k < parentParams.size(); k++) {
1019                         final String param = parentParams.get(k);
1020                         int plvl = parentLevel;
1021                         while (plvl > 1 && !parentTypes.containsKey("" + plvl + param)) {
1022                             plvl--;
1023                         }
1024                         if (StringUtils.isNotBlank(parentTypes.get(plvl + param))) {
1025                             child.setParameter("PARENT_" + param, parentTypes.get(plvl + param));
1026                         }
1027                     }
1028                 }
1029             }
1030         }
1031     }

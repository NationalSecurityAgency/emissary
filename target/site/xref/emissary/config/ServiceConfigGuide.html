202     protected void readConfigData(final InputStream is, final String filename) throws IOException, ConfigSyntaxException {
203         final Reader r = new BufferedReader(new InputStreamReader(is));
204         final StreamTokenizer in = new StreamTokenizer(r);
205         int nextToken = StreamTokenizer.TT_WORD;
206         String parmName;
207         String sval;
208 
209         in.commentChar('#');
210         in.wordChars(33, 33);
211         in.wordChars(36, 47);
212         in.wordChars(58, 64);
213         in.wordChars(91, 96);
214         in.wordChars(123, 65536);
215 
216         while (nextToken != StreamTokenizer.TT_EOF) {
217             // Read three tokens at a time (X = Y)
218             nextToken = in.nextToken();
219 
220             // Make sure the first token in the tuple is a word
221             if (nextToken == StreamTokenizer.TT_EOF) {
222                 break;
223             }
224             if (nextToken == StreamTokenizer.TT_NUMBER) {
225                 throw new ConfigSyntaxException("Illegal token " + in.sval + ", missing quote on line " + in.lineno() + "?");
226             }
227 
228             parmName = in.sval;
229 
230             nextToken = in.nextToken();
231             this.operator = in.sval;
232 
233             nextToken = in.nextToken();
234             if (nextToken == StreamTokenizer.TT_NUMBER) {
235                 sval = Long.toString((long) in.nval);
236             } else {
237                 sval = in.sval;
238             }
239 
240             if (sval == null) {
241                 // Problem is likely on previous line
242                 throw new ConfigSyntaxException("Illegal token " + parmName + ", missing space or value on line " + (in.lineno() - 1) + "?");
243             }
244 
245             handleNewEntry(parmName, sval, this.operator, filename, in.lineno() - 1, false);
246         }
247         r.close();
248         is.close();
249     }

251     /**
252      * Handle a newly parsed or passed in entry. Substitutions are handled on both the LHS and RHS, then the values are
253      * stored as a ConfigEntry in our local list and map. Only the last value in the map is available for substitutions. LHS
254      * is analyzed before RHS and in L to R order.
255      *
256      * @param parmNameArg the LHS
257      * @param svalArg the raw RHS
258      * @param operatorArg the equation
259      * @param filename the filename we are parsing for error reporting
260      * @param lineno the line number we are currently reporting the error on
261      * @param merge true when adding in from a merge
262      * @return a new config entry with the expanded key and value
263      * @throws IOException when the key or value is malformed
264      */
265     protected ConfigEntry handleNewEntry(final String parmNameArg, final String svalArg, final String operatorArg, final String filename,
266             final int lineno, final boolean merge) throws IOException {
267         final String parmName = handleReplacements(parmNameArg, filename, lineno);
268         final String sval = handleReplacements(svalArg, filename, lineno);
269 
270         // Create a config entry from this
271         final ConfigEntryntry">ConfigEntry anEntry = new ConfigEntry(parmName, sval);
272 
273         if ("!=".equals(operatorArg)) {
274             if ("*".equals(sval)) {
275                 removeAllEntries(parmName);
276                 this.VALUES.remove(parmName);
277             } else {
278                 removeEntry(anEntry);
279                 if (sval.equals(this.VALUES.get(parmName))) {
280                     this.VALUES.remove(parmName);
281                 }
282             }
283             this.p_remove_parameters.add(anEntry);
284         } else {
285             // Save the entry in the list
286             if (merge) {
287                 this.p_service_parameters.add(0, anEntry);
288             } else {
289                 this.p_service_parameters.add(anEntry);
290             }
291 
292             // Save this pair in the map
293             this.VALUES.put(parmName, sval);
294         }
295 
296         if ("IMPORT_FILE".equals(parmName) || "OPT_IMPORT_FILE".equals(parmName)) {
297             final List<String> fileFlavorList = new ArrayList<>();
298             // Add the base file and then add all the flavor versions
299             fileFlavorList.add(sval);
300             final String[] fileFlavors = ConfigUtil.addFlavors(sval);
301             if (ArrayUtils.isNotEmpty(fileFlavors)) {
302                 fileFlavorList.addAll(Arrays.asList(fileFlavors));
303             }
304             logger.debug("ServiceConfigGuide::handleNewEntry -- FileFlavorList = {}", fileFlavorList);
305 
306             // loop through the files and attempt to read/merger the configurations.
307             for (int i = 0; i < fileFlavorList.size(); i++) {
308                 final String fileFlavor = fileFlavorList.get(i);
309                 // recursion alert: This could lead to getFile being called
310                 try {
311                     readConfigData(ConfigUtil.getConfigStream(fileFlavor), fileFlavor);
312                 } catch (ConfigSyntaxException e) {
313                     // whether opt or not, syntax errors are a problem
314                     throw new IOException(parmName + " = " + sval + " from " + filename + " failed " + e.getMessage(), e);
315                 } catch (IOException e) {
316                     // Throw exception if it is an IMPORT_FILE and the base file is not found
317                     if ("IMPORT_FILE".equals(parmName) && i == 0) {
318                         String importFileName = Paths.get(svalArg).getFileName().toString();
319                         throw new IOException("In " + filename + ", cannot find IMPORT_FILE: " + sval
320                                 + " on the specified path. Make sure IMPORT_FILE (" + importFileName + ") exists, and the file path is correct.",
321                                 e);
322                     }
323                 }
324             }
325             return anEntry;
326         } else if ("CREATE_DIRECTORY".equals(parmName) && !createDirectory(sval)) {
327             logger.warn("{}: Cannot create directory {}", filename, sval);
328         } else if ("CREATE_FILE".equals(parmName) && !createFile(sval)) {
329             logger.warn("{}: Cannot create file {}", filename, sval);
330         }
331 
332         return anEntry;
333     }

335     /**
336      * Handle all the possible replacements in a string value
337      *
338      * @param svalArg the raw value
339      * @param filename the filename we are parsing for error reporting
340      * @param lineno the line number we are currently reporting the error on
341      * @return the expanded value with all legal @{..} values replaced
342      * @throws IOException when the value is malformed
343      */
344     protected String handleReplacements(final String svalArg, final String filename, final int lineno) throws IOException {
345         String sval = svalArg;
346         int startpos = 0;
347         while (sval != null && sval.indexOf(VSTART, startpos) > -1) {
348             final int ndx = sval.indexOf(VSTART, startpos);
349             final int edx = sval.indexOf(VEND, ndx + VSTART.length());
350             if (ndx == -1 && ndx >= edx) {
351                 throw new IOException("Problem parsing line " + lineno + " " + sval);
352             }
353             final String tok = sval.substring(ndx + VSTART.length(), edx);
354             logger.debug("Replacement token is {}", tok);
355             final String mapval = this.VALUES.get(tok);
356             if (mapval != null) {
357                 sval = sval.substring(0, ndx) + mapval + sval.substring(edx + VEND.length());
358             } else {
359                 logger.warn("Did not find replacement for '{}' in file {} at line {}", tok, filename, lineno);
360                 startpos = edx + VEND.length();
361             }
362         }
363 
364         // This is obsolete
365         if (sval != null && sval.contains(ENVSTART)) {
366             sval = substEnvProps(sval, filename, lineno);
367         }
368 
369         // Do unicode stuff
370         if (sval != null && (sval.contains("\\u") || sval.contains("\\U"))) {
371             sval = substUTFChars(sval, filename, lineno);
372         }
373 
374         // This is obsolete
375         if (sval != null && sval.equals(this.NULL_VALUE)) {
376             sval = null;
377             logger.debug("Using {} is deprecated, please just use {}NULL{}", this.NULL_VALUE, VSTART, VEND);
378         }
379         return sval;
380     }

382     /**
383      * Substitute any java unicode character values: \\uxxxx
384      *
385      * @param s the string to process
386      * @param filename the name of the file we are in for error reporting
387      * @param lnum the current line number for error reporting
388      * @return string with character values replaced
389      */
390     protected String substUTFChars(final String s, final String filename, final int lnum) throws IOException {
391         final int slen = s.length();
392         final StringBuilder sb = new StringBuilder(slen);
393         for (int i = 0; i < slen; i++) {
394             if (s.charAt(i) != '\\') {
395                 sb.append(s.charAt(i));
396             } else if ((i + 4) < slen && (s.charAt(i + 1) == 'u' || s.charAt(i + 1) == 'U')) {
397                 int epos = i + 2;
398                 final int max = (s.charAt(epos) == '1' || s.charAt(epos) == '0') ? (i + 7) : (i + 6);
399                 while (epos < slen
400                         && epos < max
401                         && ((s.charAt(epos) >= '0' && s.charAt(epos) <= '9') || (s.charAt(epos) >= 'A' && s.charAt(epos) <= 'F')
402                                 || (s.charAt(epos) >= 'a' && s
403                                         .charAt(epos) <= 'f'))) {
404                     epos++;
405                 }
406                 if (epos <= slen) {
407                     try {
408                         final int digit = Integer.parseInt(s.substring(i + 2, epos), 16);
409                         sb.appendCodePoint(digit);
410                         i = epos - 1;
411                     } catch (Exception ex) {
412                         throw new IOException("Unable to convert characters in " + s + ", from filename=" + filename + " line " + lnum, ex);
413                     }
414                 }
415             } else {
416                 sb.append(s.charAt(i));
417             }
418         }
419 
420         return sb.toString();
421     }

423     /**
424      * Substitute any referenced env properties with their values. Look for @ENV{'foo'} and replace foo with
425      * System.getProperty("foo") or System.getenv("foo") in that order.
426      *
427      * @param str the string to process
428      * @param filename the name of the file we are in for error reporting
429      * @param lnum the current line number for error reporting
430      * @return string with env values replaced
431      */
432     protected String substEnvProps(final String str, final String filename, final int lnum) throws IOException {
433         int lastPos = -1;
434         int thisPos = 0;
435         int count = 0;
436 
437         logger.debug("{}{} style substitution is deprecated. Please just use {}yourvalue{}", ENVSTART, ENVSTOP, VSTART, VEND);
438 
439         String currentStr = str;
440         while ((thisPos = currentStr.indexOf(ENVSTART, thisPos)) > lastPos) {
441             final int start = thisPos + ENVSTART.length();
442             final int stop = currentStr.indexOf(ENVSTOP, thisPos);
443             count++;
444             // Pull out the env name they specified
445             if (stop > start) {
446                 final String envName = currentStr.substring(start, stop);
447                 String envVal = System.getProperty(envName);
448                 if (envVal == null) {
449                     envVal = System.getenv(envName);
450                 }
451                 // We got a replacement, do the subst
452                 if (envVal != null) {
453                     currentStr = currentStr.substring(0, thisPos) + // before
454                             envVal + // replacement value
455                             currentStr.substring(stop + ENVSTOP.length()); // tail
456                     logger.debug("Replaced {} with {} at {}: {}", envName, envVal, filename, lnum);
457                 } else {
458                     logger.debug("No env value for {} at {}: {}", envName, filename, lnum);
459                 }
460             } else {
461                 throw new IOException("Runaway string on line ->" + currentStr + "<- at " + filename + ": " + lnum);
462             }
463 
464             lastPos = thisPos;
465         }
466         logger.debug("Found {} env vars to subst --> {}", count, currentStr);
467         return currentStr;
468     }

484     /**
485      * Create a file as specified by the config driver
486      */
487     protected boolean createFile(final String sval) {
488 
489         final String fixedSval = sval.replace('\\', '/');
490         logger.debug("Trying to create file {}", fixedSval);
491         final File d = new File(fixedSval);
492         FileWriter newFile = null;
493         if (!d.exists()) {
494             try {
495                 // Ensure the directory exists to hold the file
496                 final File parent = new File(new File(d.getCanonicalPath()).getParent());
497                 if (!parent.exists() && !createDirectory(parent.toString())) {
498                     logger.debug("Failed to create parent directory for {}", fixedSval);
499                     return false;
500                 }
501                 // Create the file in the directory
502                 newFile = new FileWriter(d);
503             } catch (IOException e) {
504                 logger.debug("Failed to create file {}", fixedSval, e);
505                 return false;
506             } finally {
507                 if (newFile != null) {
508                     try {
509                         newFile.close();
510                     } catch (IOException ioe) {
511                         logger.debug("Error closing file", ioe);
512                     }
513                 }
514             }
515         }
516         return true;
517     }

605      *

607      */
608     public void removeEntry(final ConfigEntry anEntry) {
609         // NB: enhanced for loop does not support remove
610         for (final Iterator<ConfigEntry> i = this.p_service_parameters.iterator(); i.hasNext();) {
611             final ConfigEntry curEntry = i.next();
612             if ((anEntry.getKey().equals(curEntry.getKey()))
613                     && ((anEntry.getValue() == null && curEntry.getValue() == null) || (anEntry.getValue() != null && anEntry.getValue().equals(
614                             curEntry.getValue())))) {
615                 logger.debug("Removing {} = {}", curEntry.getKey(), curEntry.getValue());
616                 i.remove();
617             }
618         }

655     /**
656      * Remove all entries from the list of parameters matching the String argument passed in.
657      *
658      * @param theParameter key name to match, all matching will be removed
659      */
660     public void removeAllEntries(final String theParameter) {
661         // NB: enhanced for loop does not support remove
662         for (final Iterator<ConfigEntry> i = this.p_service_parameters.iterator(); i.hasNext();) {

780      * @return map where key is remainder after match and value is the config value, or an empty map if none found

784         if (theParameter == null) {

787

789         final List<ConfigEntry> parameters = this.findStringMatchEntries(theParameter);

802     /**
803      * Find entries beginning with the specified string and return a hash keyed on the remainder of the string with the
804      * value of the config line as the value of the hash Multiple values for the same hash are allowed and returned as a
805      * Set.
806      *
807      * <pre>
808      * {@code
809      * Example config entries
810      *    FOO_ONE: AAA
811      *    FOO_TWO: BBB
812      *    FOO_TWO: CCC
813      * Calling findStringMatchMap("FOO_",true)
814      * will yield a map with Sets
815      *     ONE -> {AAA}
816      *     TWO -> {BBB,CCC}
817      * }
818      * </pre>
819      *
820      * @param param the key to look for in the config file
821      * @return map where key is remainder after match and value is a Set of all found config values, or an empty map if none
822      *         found
823      */
824     @Override
825     public Map<String, Set<String>> findStringMatchMultiMap(@Nullable final String param) {
826 
827         if (param == null) {
828             return Collections.emptyMap();
829         }
830 
831         final Map<String, Set<String>> theHash = new HashMap<>();
832         final List<ConfigEntry> parameters = this.findStringMatchEntries(param);
833 
834         for (final ConfigEntry el : parameters) {
835             final String key = el.getKey().substring(param.length()).toUpperCase();
836 
837             if (theHash.containsKey(key)) {
838                 theHash.get(key).add(el.getValue());
839             } else {
840                 final Set<String> values = new HashSet<>();
841                 values.add(el.getValue());
842                 theHash.put(key, values);
843             }
844         }
845         return theHash;
846     }

895      * Return a long from a string entry representing either an int, a long, a double, with or without a final letter
896      * designation such as "m" or "M" for megabytes, "g" or "G" for gigabytes, etc. Legal designations are bBkKmMgGTt or
897      * just a number.
898      *
899      * @param theParameter the config entry name
900      * @param dflt the default value when nothing found in config
901      * @return the long value of the size parameter
902      */
903     @Override
904     public long findSizeEntry(final String theParameter, final long dflt) {
905         final List<String> matchingEntries = findEntries(theParameter);
906         if (!matchingEntries.isEmpty()) {
907             long val = dflt;
908             final String s = matchingEntries.get(0);
909             final char c = Character.toUpperCase(s.charAt(s.length() - 1));
910             final String ss = s.substring(0, s.length() - 1);
911             boolean broken = false;
912             switch (c) {
913                 case 'T':
914                     val = Long.parseLong(ss) * 1024 * 1024 * 1024 * 1024;
915                     break;
916                 case 'G':
917                     val = Long.parseLong(ss) * 1024 * 1024 * 1024;
918                     break;
919                 case 'M':
920                     val = Long.parseLong(ss) * 1024 * 1024;
921                     break;
922                 case 'K':
923                     val = Long.parseLong(ss) * 1024;
924                     break;
925                 case 'B':
926                     val = Long.parseLong(ss);
927                     break;
928                 case '0':
929                 case '1':
930                 case '2':
931                 case '3':
932                 case '4':
933                 case '5':
934                 case '6':
935                 case '7':
936                 case '8':
937                 case '9':
938                     val = Long.parseLong(s);
939                     break;
940                 default:
941                     broken = true;
942             }
943 
944             if (!broken) {
945                 return val;
946             }
947             return dflt;
948         }
949         return dflt;

1092     /**
1093      * Merge in a new configuration set with this one. New things are supposed to override older things in the sense of
1094      * findStringEntry which only picks the top of the list, the new things should get added to the top. If the merged in
1095      * Configurator contains remove entries (operator of '!=') then it only applies to entries in this instance, not in
1096      * "other". This is slightly different than when a config is read in directly, but without that there would be no way to
1097      * remove entries from a super-config and continue to supply entries here and still be able to use the wildcard remove
1098      * (value of '*') The order in the merged config file is important. Any '!= "*"' operations must precede the new value
1099      * being supplied since normal remove operations take place in each config before the merge.
1100      *
1101      * @param other the new entries to merge in
1102      */
1103     @Override
1104     public void merge(final Configurator other) throws IOException {
1105         int i = 1;
1106 
1107         // First handle the remove entries from "other"
1108         if (other instanceof ServiceConfigGuide) {
1109             for (final ConfigEntry entry : ((ServiceConfigGuide) other).getRemoveEntries()) {
1110                 handleNewEntry(entry.getKey(), entry.getValue(), "!=", "<merge>", i++, true);
1111             }
1112         }
1113 
1114         // Add in new entries from "other" at the top of the list
1115         for (final ConfigEntry entry : other.getEntries()) {
1116             handleNewEntry(entry.getKey(), entry.getValue(), "=", "<merge>", i++, true);
1117         }
1118     }

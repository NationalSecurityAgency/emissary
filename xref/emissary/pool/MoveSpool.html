21  /**
22   * Provide a storage area for incoming "moveTo(here)" payloads so that the http transfer can become more asnychronous.
23   * This class provides a FIFO for payloads that are arriving and a thread that will put them into agents from the pool
24   * as agents become available
25   */
26  
27  public class MoveSpool implements Runnable {

30      private static final Logger logger = LoggerFactory.getLogger(MoveSpool.class);

33      protected final LinkedList<SpoolItem> spool = new LinkedList<SpoolItem>();

74      /**
75       * Configure stuff
76       */
77      private void configure() {
78          // Get the agent pool
79          resetPool();
80  
81          // start the watcher thread
82          watcher = new Thread(this, "MoveSpool");
83          watcher.setPriority(Thread.MAX_PRIORITY - 2);
84          watcher.setDaemon(true);
85          watcher.start();
86      }

88      public void resetPool() {
89          // grab the default pool
90          try {
91              pool = AgentPool.lookup();
92              logger.debug("Found the AgentPool on MoveSpool#resetPool");
93          } catch (NamespaceException nex) {
94              logger.error("Unable to find agent pool, " + "please create the agent pool before creating the MoveSpool");
95          }

98      /**
99       * Shut down the spooling thread and clear out any remaining payloads.
100      */
101     public void quit() {
102         logger.warn("Purging the spool...");
103         synchronized (spool) {
104             if (spool.size() > 0) {
105                 spool.clear();
106             }
107             spool.notifyAll();
108         }
109         timeToQuit = true;
110         Namespace.unbind(NAMESPACE_NAME);
111         logger.info("Done stopping the move spool");
112     }

114     /**
115      * Get a reference to the local directory on this machine
116      *
117      * @return a reference to the local directory from the namespace
118      */
119     private IDirectoryPlace getLocalDirectory() {
120         if (localDirectory == null) {
121             for (Iterator<String> i = Namespace.keySet().iterator(); i.hasNext();) {
122                 String key = i.next();
123                 try {
124                     Object value = Namespace.lookup(key);
125                     if (value instanceof IDirectoryPlace) {
126                         localDirectory = (IDirectoryPlace) value;
127                         break;
128                     }
129                 } catch (NamespaceException ex) {
130                     logger.info("Problem in namespace", ex);
131                 }
132             }
133         }
134         return localDirectory;
135     }

137     /**
138      * Run the thread to watch the spool
139      */
140     @Override
141     public void run() {
142         int consecutiveSendCounter = 0;
143 
144         // Run until we are told to quit
145         while (!timeToQuit) {
146             // Check the spool for work to be done
147             int sz = 0;
148             synchronized (spool) {
149                 sz = spool.size();
150             }
151 
152             if (sz == 0) {
153                 // No payloads to look at. Sleep a while
154                 consecutiveSendCounter = 0;
155                 try {
156                     logger.debug("Nothing in spool, time to wait...");
157                     Thread.yield();
158                     synchronized (spool) {
159                         if (spool.size() == 0) {
160                             spool.wait(60000);
161                         }
162                     }
163                 } catch (InterruptedException ignore) {
164                     Thread.currentThread().interrupt();
165                 }
166                 continue;
167             }
168 
169             // Get an agent and a sool item
170             IMobileAgent agent = null;
171             SpoolItem item = null;
172             String itemName = null;
173 
174             try {
175                 // This may block for the max time the
176                 // pool is configured to use if no
177                 // agents available
178                 agent = pool.borrowAgent();
179                 if (agent == null) {
180                     logger.debug("Got a null agent from pool!");
181                     continue;
182                 }
183 
184                 // Get the oldest payload from the spool
185                 item = removeFirstPayload();
186                 if (item == null) {
187                     logger.debug("Got a null item from move spool!");
188                     pool.returnAgent(agent);
189                     continue;
190                 }
191 
192                 // We have both an agent and a spool item
193                 // so hook em up and send it on the way
194                 itemName = PayloadUtil.getName(item.getPayload());
195 
196                 logger.debug("Handing over " + itemName + " to an agent, method=" + item.getMethod());
197 
198                 if (item.getMethod() == Method.GO) {
199                     IServiceProviderPlace place = item.getPlace();
200                     if (place == null) {
201                         place = getLocalDirectory();
202                     }
203 
204                     Object payload = item.getPayload();
205                     agent.go(payload, place);
206                     consecutiveSendCounter++;
207                 } else if (item.getMethod() == Method.ARRIVE) {
208                     agent.arrive(item.getPayload(), item.getPlace(), item.getErrorCount(), item.getItineraryItems());
209                     consecutiveSendCounter++;
210                 } else {
211                     logger.error("Illegal spooler method specified " + item.getMethod() + ", payload=" + item.getPayload()
212                             + " will be irretreivably lost");
213                 }
214 
215                 if (consecutiveSendCounter % 10 == 0) {
216                     logger.debug("Sent 10 consecutive entries, " + "time to yield the MoveSpool");
217                     Thread.yield();
218                 }
219             } catch (Throwable t) {
220                 if (agent != null) {
221                     logger.error("Unable to start agent, payload " + itemName + " is irretrievably lost", t);
222                     try {
223                         pool.returnAgent(agent);
224                     } catch (Exception ex) {
225                         logger.error("Unable to return agent to the pool", ex);
226                     }
227                 } else {
228                     logger.debug("Cannot get agent from pool, trying again", t);
229                 }
230             } finally {
231                 // hold no references to this stuff
232                 agent = null;
233                 item = null;
234             }
235         }
236     }

238     /**
239      * Remove the oldest payload item on the spool
240      *
241      * @return SpoolItem from the spool
242      */
243     protected SpoolItem removeFirstPayload() {
244         SpoolItem s = null;
245         synchronized (spool) {
246             // Do some stats
247             if (spool.size() > highWaterMark) {
248                 highWaterMark = spool.size();
249             }
250 
251             s = spool.removeFirst();
252             dequeCount++;
253         }
254         return s;
255     }

293     /**
294      * Add an item to the spool. When an agent becomes available it is assigned to this payload in turn. We call
295      * MobileAgent.go or MobileAgent.arrive depending on the Method requested
296      *
297      * @param method ARRIVE or GO
298      * @param payload the data object or Collection to save
299      * @param place IServiceProviderPlace ref for the agent to visit, possibly null
300      * @param errorCount state from the transferred MobileAgent or null for GO
301      * @param itineraryItems state from the transferred MobileAgent or empty for GO
302      * @return number of items on the queue
303      */
304     protected int enqueue(Method method, Object payload, IServiceProviderPlace place, int errorCount, List<DirectoryEntry> itineraryItems) {
305 
306         String itemName = PayloadUtil.getName(payload);
307         logger.debug("Enqueue item " + itemName + " for place " + place + ", method=" + method);
308         SpoolItem s = new SpoolItem(method, payload, place, errorCount, itineraryItems);
309         int size = 0;
310 
311         synchronized (spool) {
312             spool.addLast(s);
313             enqueCount++;
314             size = spool.size();
315             spool.notifyAll();
316         }
317 
318         // Collect the stats
319         synchronized (moveCountMap) {
320             String serviceName = s.getServiceName();
321             if (moveCountMap.containsKey(serviceName)) {
322                 Integer count = moveCountMap.get(serviceName);
323                 moveCountMap.put(serviceName, Integer.valueOf(count.intValue() + 1));
324             } else {
325                 moveCountMap.put(serviceName, Integer.valueOf(1));
326             }
327         }
328 
329         logger.debug("Done enqueue of " + itemName + ", size=" + size);
330         return size;
331     }

352     /**
353      * Provide statistics in string form
354      *
355      * @return list of types and counts that have spooled here
356      */
357     public String getStatPairs() {
358         StringBuilder sb = new StringBuilder();
359         sb.append("{");
360         synchronized (moveCountMap) {
361             for (String key : moveCountMap.keySet()) {
362                 if (sb.length() > 1) {
363                     sb.append(",");
364                 }
365                 sb.append(key).append("=").append(moveCountMap.get(key));
366             }
367         }
368         sb.append("}");
369         return sb.toString();
370     }

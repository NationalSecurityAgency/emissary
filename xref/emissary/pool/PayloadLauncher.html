12  import org.slf4j.LoggerFactory;
13  import org.slf4j.MDC;
14  
15  import java.util.List;
16  
17  /**
18   * Launch an incoming payload the best way possible.
19   */
20  public class PayloadLauncher {
21      private static final Logger logger = LoggerFactory.getLogger(PayloadLauncher.class);
22  
23  
24      @SuppressWarnings("CatchingUnchecked")
25      public static boolean launch(Object payload, IServiceProviderPlace place, int errorCount, List<DirectoryEntry> itineraryItems)
26              throws EmissaryException {
27          String payloadName = PayloadUtil.getName(payload);
28          // Try to grab the arrival spool and save the payload and place
29          // because this will allow us to give an asynchronous response
30  
31          MDC.put(MDCConstants.SHORT_NAME, payloadName);
32          try {
33              MoveSpool spool = null;
34              try {
35                  spool = MoveSpool.lookup();
36              } catch (NamespaceException ignored) {
37                  // empty catch block
38              }
39  
40              if (spool != null) {
41                  int sz = spool.arrive(payload, place, errorCount, itineraryItems);
42                  logger.debug("Payload " + payloadName + " spooled out, " + sz + " on the spool");
43  
44                  return true;
45              }
46              // Use the agent pool directly, this may block the caller
47              AgentPool pool = null;
48              try {
49                  pool = AgentPool.lookup();
50              } catch (NamespaceException nse) {
51                  throw new EmissaryException("No agent pool available for " + payloadName, nse);
52              }
53  
54              IMobileAgent agent = null;
55              try {
56                  agent = pool.borrowAgent();
57              } catch (Exception e) {
58                  logger.error("Cannot get agent from pool for " + payloadName, e);
59                  throw new EmissaryException("Cannot get agent from pool for " + payloadName, e);
60              }
61  
62              try {
63                  agent.arrive(payload, place, errorCount, itineraryItems);
64                  logger.debug("Finished setting up agent with place=" + place + ", payload=" + payloadName + ", isInUse=" + agent.isInUse());
65                  return true;
66              } catch (Exception e) {
67                  logger.error("Cannot get agent started on payload, " + "is the same implementation of IMobileAgent in use "
68                          + "on both sides of the moveTo? payload=" + payloadName, e);
69                  try {
70                      pool.returnAgent(agent);
71                  } catch (Exception poolex) {
72                      logger.error("Could not return agent to pool when it " + "failed to start, killing agent " + agent.getName(), poolex);
73                      agent.killAgent();
74                  }
75  
76                  throw new EmissaryException("Cannot get agent started on payload for " + payloadName, e);
77              }
78          } finally {
79              MDC.remove(MDCConstants.SHORT_NAME);
80          }
81      }
82  
83      /** This class is not meant to be instantiated. */
84      private PayloadLauncher() {}
85  }

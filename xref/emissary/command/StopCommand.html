15  @Deprecated
16  @Command(description = "Stop an Emissary jetty server", subcommands = {HelpCommand.class})
17  public class StopCommand extends HttpCommand {
18  
19      public static String COMMAND_NAME = "stop";
20  
21      @Override
22      public String getCommandName() {
23          return COMMAND_NAME;
24      }
25  
26      public static int DEFAULT_PORT = 8001;
27  
28      @Override
29      public int getDefaultPort() {
30          return DEFAULT_PORT;
31      }
32  
33      public String getShutdownEndpoint() {
34          return SERVICE_SHUTDOWN_ENDPOINT;
35      }
36  
37      private static final Logger LOG = LoggerFactory.getLogger(StopCommand.class);
38  
39  
40      @Override
41      public void run(CommandLine c) {
42          setup();
43          LOG.info("Stopping Emissary Server at {}://{}:{}", getScheme(), getHost(), getPort());
44          EmissaryResponse response = performPost(getShutdownEndpoint());
45          if (response.getStatus() != 200) {
46              LOG.error("Problem shutting down");
47              LOG.error(response.getContentString());
48          } else {
49              LOG.info("EmissaryServer stopped");
50          }
51      }
52  
53  
54      @Override
55      public void setupCommand() {
56          setupHttp();
57          reinitLogback();
58          setupStop();
59      }

62

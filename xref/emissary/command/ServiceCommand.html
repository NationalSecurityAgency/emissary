21  public abstract class ServiceCommand extends HttpCommand {
22  
23      static final Logger LOG = LoggerFactory.getLogger(ServiceCommand.class);

25      public static final String COMMAND_NAME = "ServiceCommand";
26      public static final String SERVICE_HEALTH_ENDPOINT = "/api/" + HEALTH;
27      public static final String SERVICE_SHUTDOWN_ENDPOINT = "/api/" + SHUTDOWN;
28      public static final String SERVICE_KILL_ENDPOINT = SERVICE_SHUTDOWN_ENDPOINT + "/force";
29      public static final String SERVICE_INVALIDATE_ENDPOINT = "/api/" + INVALIDATE;
30      public static final String SERVICE_REFRESH_ENDPOINT = "/api/" + REFRESH;

32      @Option(names = {"--csrf"}, description = "disable csrf protection\nDefault: ${DEFAULT-VALUE}", arity = "1")
33      private boolean csrf = true;
34  
35      @Option(names = {"--stop"}, description = "Shutdown the service\nDefault: ${DEFAULT-VALUE}")
36      private boolean stop = false;
37  
38      @Option(names = {"--invalidate"}, description = "Invalidate services that are refreshable\nDefault: ${DEFAULT-VALUE}")
39      private boolean invalidate = false;
40  
41      @Option(names = {"--refresh"}, description = "Force refresh of services\nDefault: ${DEFAULT-VALUE}")
42      private boolean refresh = false;
43  
44      @Option(names = {"--kill"}, description = "Force the shutdown of the service\nDefault: ${DEFAULT-VALUE}")
45      private boolean kill = false;
46  
47      @Option(names = {"--pause"}, description = "Stop the service from taking work\nDefault: ${DEFAULT-VALUE}")
48      private boolean pause = false;
49  
50      @Option(names = {"--unpause"}, description = "Allow a paused service to take work\nDefault: ${DEFAULT-VALUE}")
51      private boolean unpause = false;

97      /**
98       * Startup the service
99       */
100     protected abstract void startService();

102     @Override
103     public void setupHttp() {
104         super.setupHttp();
105 
106         // check to see if this is set -- feed will call server command and reset this value
107         if (StringUtils.isBlank(System.getProperty(EmissaryNode.NODE_SERVICE_TYPE_PROPERTY))) {
108             logInfo("Setting {} to {} ", EmissaryNode.NODE_SERVICE_TYPE_PROPERTY, getCommandName());
109             System.setProperty(EmissaryNode.NODE_SERVICE_TYPE_PROPERTY, getCommandName());
110         }
111     }
112 
113     @Override
114     public void run(CommandLine c) {
115         setup();
116 
117         // let's check to see if the server is already running
118         LOG.debug("Checking to see if Emissary {} is running at {}", getServiceName(), getServiceHealthEndpoint());
119         EmissaryResponse response = performGet(getServiceHealthEndpoint());
120         boolean isRunning = response.getStatus() == 200;
121         if (isStop() || isKill()) {
122             if (isRunning) {
123                 stopService();
124             } else {
125                 LOG.warn("Error stopping service: no service {} running", getServiceName());
126             }
127         } else {
128             if (isRunning) {
129                 // the server is already running so pause/unpause or fail
130                 if (isPause()) {
131                     pauseService();
132                 } else if (isUnpause()) {
133                     unpauseService();
134                 } else if (isRefresh()) {
135                     refreshService();
136                 } else {
137                     throw new EmissaryRuntimeException("Emissary " + getServiceName() + " is already running");
138                 }
139             } else {
140                 if (isPause()) {
141                     // we hadn't intended to start the service, so don't try to do so now
142                     throw new EmissaryRuntimeException("Error pausing service: request returned status " + response.getStatus());
143                 } else {
144                     // no running server so fire it up
145                     startService();
146                 }
147             }
148         }
149     }
150 
151     /**
152      * Shutdown method that uses an endpoint to stop a running service
153      */
154     protected void stopService() {
155         LOG.info("Stopping Emissary {} at {}", getServiceName(), getServiceShutdownEndpoint());
156         EmissaryResponse response = performPost(getServiceShutdownEndpoint());
157         if (response.getStatus() != 200) {
158             LOG.error("Problem shutting down {} -- {}", getServiceName(), response.getContentString());
159         } else {
160             LOG.info("Emissary {} stopped", getServiceName());
161         }
162     }
163 
164     /**
165      * A method that stops a running service from taking work
166      */
167     protected void pauseService() {
168         throw new UnsupportedOperationException("Pause not implemented for " + getServiceName());
169     }

171     /**
172      * A method that allows a paused service to take work

174     protected void unpauseService() {
175         throw new UnsupportedOperationException("Unpause not implemented for " + getServiceName());
176     }
177 
178     /**
179      * A method that refreshes services
180      */
181     protected void refreshService() {
182         throw new UnsupportedOperationException("Refresh not implemented for " + getServiceName());
183     }

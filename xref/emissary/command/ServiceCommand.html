3   import emissary.client.EmissaryResponse;
4   import emissary.core.EmissaryRuntimeException;
5   import emissary.directory.EmissaryNode;
6   
7   import org.apache.commons.lang3.StringUtils;
8   import org.slf4j.Logger;
9   import org.slf4j.LoggerFactory;
10  import picocli.CommandLine;
11  import picocli.CommandLine.Option;
12  
13  import static emissary.server.api.HealthCheckAction.HEALTH;
14  import static emissary.server.api.Shutdown.SHUTDOWN;
15  
16  /**
17   * Abstract command to control service components (stop/pause/unpause)
18   */
19  public abstract class ServiceCommand extends HttpCommand {
20  
21      static final Logger LOG = LoggerFactory.getLogger(ServiceCommand.class);
22  
23      public static String COMMAND_NAME = "ServiceCommand";
24      public static String SERVICE_HEALTH_ENDPOINT = "/api/" + HEALTH;
25      public static String SERVICE_SHUTDOWN_ENDPOINT = "/api/" + SHUTDOWN;
26      public static String SERVICE_KILL_ENDPOINT = SERVICE_SHUTDOWN_ENDPOINT + "/force";
27  
28      @Option(names = {"--csrf"}, description = "disable csrf protection\nDefault: ${DEFAULT-VALUE}", arity = "1")
29      private boolean csrf = true;
30  
31      @Option(names = {"--stop"}, description = "Shutdown the service\nDefault: ${DEFAULT-VALUE}")
32      private boolean stop = false;
33  
34      @Option(names = {"--kill"}, description = "Force the shutdown of the service\nDefault: ${DEFAULT-VALUE}")
35      private boolean kill = false;
36  
37      @Option(names = {"--pause"}, description = "Stop the service from taking work\nDefault: ${DEFAULT-VALUE}")
38      private boolean pause = false;
39  
40      @Option(names = {"--unpause"}, description = "Allow a paused service to take work\nDefault: ${DEFAULT-VALUE}")
41      private boolean unpause = false;
42  
43      public boolean isCsrf() {
44          return csrf;
45      }
46  
47      public boolean isStop() {
48          return stop;
49      }
50  
51      public boolean isKill() {
52          return kill;
53      }
54  
55      public boolean isPause() {
56          return pause;
57      }
58  
59      public boolean isUnpause() {
60          return unpause;
61      }
62  
63      public String getServiceHealthEndpoint() {
64          return SERVICE_HEALTH_ENDPOINT;
65      }
66  
67      public String getServiceShutdownEndpoint() {
68          return isKill() ? SERVICE_KILL_ENDPOINT : SERVICE_SHUTDOWN_ENDPOINT;
69      }
70  
71      public String getServiceName() {
72          return getCommandName();
73      }
74  
75      /**
76       * Startup the service
77       */
78      protected abstract void startService();
79  
80      @Override
81      public void setupHttp() {
82          super.setupHttp();
83  
84          // check to see if this is set -- feed will call server command and reset this value
85          if (StringUtils.isBlank(System.getProperty(EmissaryNode.NODE_SERVICE_TYPE_PROPERTY))) {
86              logInfo("Setting {} to {} ", EmissaryNode.NODE_SERVICE_TYPE_PROPERTY, getCommandName());
87              System.setProperty(EmissaryNode.NODE_SERVICE_TYPE_PROPERTY, getCommandName());
88          }
89      }
90  
91      @Override
92      public void run(CommandLine c) {
93          setup();
94  
95          // let's check to see if the server is already running
96          LOG.debug("Checking to see if Emissary {} is running at {}", getServiceName(), getServiceHealthEndpoint());
97          EmissaryResponse response = performGet(getServiceHealthEndpoint());
98          boolean isRunning = response.getStatus() == 200;
99          if (isStop() || isKill()) {
100             if (isRunning) {
101                 stopService();
102             } else {
103                 LOG.warn("Error stopping service: no service {} running", getServiceName());
104             }
105         } else {
106             if (isRunning) {
107                 // the server is already running so pause/unpause or fail
108                 if (isPause()) {
109                     pauseService();
110                 } else if (isUnpause()) {
111                     unpauseService();
112                 } else {
113                     throw new EmissaryRuntimeException("Emissary " + getServiceName() + " is already running");
114                 }
115             } else {
116                 // no running server so fire it up
117                 startService();
118             }
119         }
120     }
121 
122     /**
123      * Shutdown method that uses an endpoint to stop a running service
124      */
125     protected void stopService() {
126         LOG.info("Stopping Emissary {} at {}", getServiceName(), getServiceShutdownEndpoint());
127         EmissaryResponse response = performPost(getServiceShutdownEndpoint());
128         if (response.getStatus() != 200) {
129             LOG.error("Problem shutting down {} -- {}", getServiceName(), response.getContentString());
130         } else {
131             LOG.info("Emissary {} stopped", getServiceName());
132         }
133     }
134 
135     /**
136      * A method that stops a running service from taking work
137      */
138     protected void pauseService() {
139         throw new UnsupportedOperationException("Pause not implemented for " + getServiceName());
140     }
141 
142     /**
143      * A method that allows a paused service to take work
144      */
145     protected void unpauseService() {
146         throw new UnsupportedOperationException("Unpause not implemented for " + getServiceName());
147     }
148 
149 }

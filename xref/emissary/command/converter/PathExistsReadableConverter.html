1   package emissary.command.converter;

3   import org.slf4j.Logger;
4   import org.slf4j.LoggerFactory;
5   
6   import java.nio.file.Files;
7   import java.nio.file.Path;
8   
9   public class PathExistsReadableConverter extends PathExistsConverter {
10      public PathExistsReadableConverter() {
11          this(null);
12      }
13  
14      public PathExistsReadableConverter(String optionName) {
15          super(optionName);
16      }
17  
18      private static final Logger LOG = LoggerFactory.getLogger(PathExistsReadableConverter.class);
19  
20      @Override
21      public Path convert(String value) {
22          Path p = super.convert(value);
23          if (!Files.isReadable(p)) {
24              String msg = String.format("The option '%s' was configured with path '%s' which is not readable", getOptionName(), p);
25              LOG.error(msg);
26              throw new IllegalArgumentException(msg);
27          }
28          return p;
29      }
30  }

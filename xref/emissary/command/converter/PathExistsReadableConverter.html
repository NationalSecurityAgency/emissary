19      private static final Logger LOG = LoggerFactory.getLogger(PathExistsReadableConverter.class);
20  
21      @Override
22      public Path convert(String value) {
23          Path p = super.convert(value);
24          if (!Files.isReadable(p)) {
25              String msg = String.format("The option '%s' was configured with path '%s' which is not readable", getOptionName(), p);
26              LOG.error(msg);
27              throw new IllegalArgumentException(msg);
28          }
29          return p;
30      }

11  public class ProjectBaseConverter extends PathExistsConverter implements ITypeConverter<Path> {
12      public ProjectBaseConverter() {
13          this(null);
14      }
15  
16      public ProjectBaseConverter(@Nullable String optionName) {
17          super(optionName);
18      }
19  
20      private static final Logger LOG = LoggerFactory.getLogger(ProjectBaseConverter.class);
21  
22      @Override
23      public Path convert(String value) {
24          String origValue = value;
25          String projectBaseEnv = System.getenv("PROJECT_BASE");
26          Path projectBaseEnvPath;
27          String projectBaseEnvString = null;
28  
29          // if PROJECT_BASE not null, set some variables
30          if (projectBaseEnv != null) {
31              projectBaseEnvPath = Paths.get(projectBaseEnv);
32              projectBaseEnvString = projectBaseEnvPath.toAbsolutePath().toString();
33          }
34  
35  
36          // try to use the value from PROJECT_BASE
37          if (origValue == null) {
38              if (projectBaseEnvString == null) {
39                  throw new IllegalArgumentException("You set neither PROJECT_BASE nor passed in a directory with -b, --projectBase.  One is required");
40              } else {
41                  value = projectBaseEnvString;
42              }
43          }
44  
45          Path p = super.convert("-b", value);
46          String pString = p.toAbsolutePath().toString();
47  
48          // both -b and PROJECT_BASE were set, make sure they are pointing to the same place
49          if (origValue != null && projectBaseEnv != null && !projectBaseEnvString.equals(pString)) {
50              String msg = "You passed in " + pString + " but PROJECT_BASE was set to " + projectBaseEnvString;
51              LOG.error(msg);
52              throw new IllegalArgumentException(msg);
53          }
54          return p;
55      }

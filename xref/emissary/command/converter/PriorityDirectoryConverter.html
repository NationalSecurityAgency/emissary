1   package emissary.command.converter;
2   
3   import emissary.pickup.Priority;
4   import emissary.pickup.PriorityDirectory;
5   
6   import org.apache.commons.lang3.StringUtils;
7   import picocli.CommandLine.ITypeConverter;
8   
9   public class PriorityDirectoryConverter implements ITypeConverter<PriorityDirectory> {
10  
11      public PriorityDirectoryConverter() {
12          super();
13      }
14  
15      public static final String PRIORITY_DIR_REGEX = ".*:\\d+$";
16  
17      @Override
18      public PriorityDirectory convert(String value) {
19          final String dirName;
20          final int priority;
21          if (value.matches(PRIORITY_DIR_REGEX)) {
22              dirName = StringUtils.substringBeforeLast(value, ":");
23              priority = Integer.parseInt(StringUtils.substringAfterLast(value, ":"));
24          } else {
25              dirName = value;
26              priority = Priority.DEFAULT;
27          }
28          return new PriorityDirectory(StringUtils.appendIfMissing(dirName, "/"), priority);
29      }
30  }

1   package emissary.command;

3   import org.slf4j.Logger;
4   import org.slf4j.LoggerFactory;
5   import picocli.CommandLine;
6   import picocli.CommandLine.Command;
7   import picocli.CommandLine.ParameterException;
8   import picocli.CommandLine.Parameters;
9   
10  import java.util.ArrayList;
11  import java.util.List;
12  import java.util.Map.Entry;
13  
14  @Command(name = "help", description = "Print commands or usage for subcommand")
15  public class HelpCommand implements EmissaryCommand {
16  
17      static final Logger LOG = LoggerFactory.getLogger(HelpCommand.class);
18  
19      public static final String COMMAND_NAME = "help";
20  
21      // @Option(names = "placeholder", arity = "1", description = "display usage for this subcommand ")
22      @Parameters(
23              paramLabel = "COMMAND",
24              arity = "0..1",
25              description = {"display usage for this subcommand "})
26      public List<String> subcommands = new ArrayList<>();
27  
28  
29      public List<String> getSubcommands() {
30          return subcommands;
31      }
32  
33      public String getSubcommand() {
34          return subcommands.get(0);
35      }
36  
37      @Override
38      public String getCommandName() {
39          return COMMAND_NAME;
40      }
41  
42      @Override
43      public void run(CommandLine c) {
44          setup();
45          if (subcommands.isEmpty()) {
46              dumpCommands(c);
47          } else if (subcommands.size() > 1) {
48              LOG.error("You can only see help for 1 command at a time");
49              dumpCommands(c);
50          } else {
51              String subcommand = getSubcommand();
52              dumpHelp(c, subcommand);
53          }
54      }
55  
56      @Override
57      public void setupCommand() {
58          // nothing for this command
59      }
60  
61      public static void dumpCommands(CommandLine cl) {
62          LOG.info("Available commands:");
63          for (Entry<String, CommandLine> cmd : cl.getSubcommands().entrySet()) {
64              final String name = cmd.getKey();
65              String[] descList = cl.getSubcommands().get(name).getCommandSpec().usageMessage().description();
66              final String description = descList.length == 0 ? "" : descList[0];
67              if (LOG.isInfoEnabled()) {
68                  LOG.info("\t {} {}", String.format("%1$-15s", name), description);
69              }
70          }
71          LOG.info("Use 'help <command-name>' to see more detailed info about that command");
72      }
73  
74      public static void dumpHelp(CommandLine c, String subcommand) {
75          LOG.info("Detailed help for: {}", subcommand);
76          try {
77              CommandLine help = c.getSubcommands().get(subcommand);
78              help.usage(System.out);
79          } catch (ParameterException | NullPointerException e) {
80              LOG.error("ERROR: invalid command name: {}", subcommand);
81              dumpCommands(c);
82          }
83      }
84  
85      @Override
86      public void outputBanner() {
87          new Banner().dump();
88      }
89  }

8   import java.io.BufferedOutputStream;

10  import java.io.IOException;
11  import java.io.InputStream;
12  import java.nio.file.Files;
13  import java.nio.file.Paths;
14  
15  /**
16   * This class assists users and integrators by extracting the named resource and putting it into the config directory
17   */
18  public class ExtractResource {
19      /** Our logger */
20      protected static final Logger logger = LoggerFactory.getLogger(ExtractResource.class);
21  
22      /** The output path for extraction, the first config dir */
23      protected String outputDirectory = ConfigUtil.getFirstConfigDir();
24  
25      /**
26       * Set the output directory for extraction to a new value Default is obtained from ConfigUtil#getConfigDir
27       */
28      public void setOutputDirectory(final String dir) {
29          this.outputDirectory = dir;
30      }
31  
32      /**
33       * Return the extraction output directory
34       */
35      public String getOutputDirectory() {
36          return this.outputDirectory;
37      }
38  
39      public String getResource(final String theResource) throws IOException {
40          String resource = theResource;
41          if (!hasFileEnding(resource)) {
42              resource += ConfigUtil.CONFIG_FILE_ENDING;
43          }
44          logger.debug("Reading " + resource);
45          final String result;
46          try (final InputStream is = ConfigUtil.getConfigStream(resource);
47                  final ByteArrayOutputStream baos = new ByteArrayOutputStream()) {
48              final byte[] buf = new byte[4096];
49              int thisReadOp = 0;
50              while ((thisReadOp = is.read(buf)) > -1) {
51                  baos.write(buf, 0, thisReadOp);
52              }
53              result = baos.toString();
54          }
55          return result;
56      }
57  
58      public void writeResource(final String theResource) throws IOException {
59          String resource = theResource;
60          if (!hasFileEnding(resource)) {
61              resource += ConfigUtil.CONFIG_FILE_ENDING;
62          }
63          final String rezdata = getResource(resource);
64          final String outputPath = this.outputDirectory + "/" + resource.replaceAll("/", ".");
65          logger.debug("Writing " + outputPath);
66          try (final BufferedOutputStream os = new BufferedOutputStream(Files.newOutputStream(Paths.get(outputPath)))) {
67              os.write(rezdata.getBytes());
68          }
69      }
70  
71      private static boolean hasFileEnding(final String resource) {
72          return resource.endsWith(ResourceReader.CONFIG_SUFFIX) || resource.endsWith(ResourceReader.PROP_SUFFIX)
73                  || resource.endsWith(ResourceReader.XML_SUFFIX) || resource.endsWith(ResourceReader.JS_SUFFIX);
74      }
75  
76      /**
77       * Run from the command line specifying a set of resources to be extracted
78       */
79      @SuppressWarnings("SystemOut")
80      public static void main(final String[] args) {
81          final ExtractResource ex = new ExtractResource();
82  
83          if (args.length == 0) {
84              printUsage();
85              return;
86          }
87  
88          int i = 0;
89  
90          // Process arguments
91          while (i < args.length && args[i].startsWith("-")) {
92              if ("-o".equals(args[i])) {
93                  if (args.length >= i + 1) {
94                      ex.setOutputDirectory(args[++i]);
95                  } else {
96                      printUsage();
97                      return;
98                  }
99              } else if ("--".equals(args[i])) {
100                 i++;
101                 break;
102             } else {
103                 printUsage();
104                 return;
105             }
106             i++;
107         }
108 
109         // Process remaining args as resource names
110         for (; i < args.length; i++) {
111             try {
112                 ex.writeResource(args[i]);
113             } catch (IOException iox) {
114                 logger.error("{}: {}", args[i], iox.toString());
115             }
116         }
117     }
118 
119     private static void printUsage() {
120         System.out.println("Usage: scripts/run.sh " + ExtractResource.class.getName() + " [ -o output_directory ] package/to/Resource[.cfg] ...");
121     }
122 }

1   package emissary.config;
2   
3   import java.io.IOException;
4   import java.util.List;
5   import java.util.Map;
6   import java.util.Set;
7   
8   /**
9    * Interface for dealing with configured properties
10   */
11  public interface Configurator {
12  
13      // Values for findStringMatchMap(String,boolean)
14      boolean PRESERVE_CASE = true;
15      boolean IGNORE_CASE = false;
16  
17      /**
18       * Find all entries matching name
19       *
20       * @param param the name of the config entry
21       * @param dflt default value when none found
22       * @return list of string values of matching entries
23       */
24      List<String> findEntries(String param, String dflt);
25  
26      /**
27       * Find all entries matching name
28       *
29       * @param param the name of the config entry
30       * @return list of string values of matching entries
31       */
32      List<String> findEntries(String param);
33  
34      /**
35       * Return a set with all parameter values as members
36       */
37      Set<String> findEntriesAsSet(String param);
38  
39      /**
40       * Find all entry keys that begin with specified string
41       *
42       * @param param the leading key startsWith
43       * @return the values of matching items
44       */
45      List<ConfigEntry> findStringMatchEntries(String param);
46  
47      /**
48       * Find entries beginning with the specified string and put them into a list with the specified part of the name
49       * stripped off like #findStringMatchMap
50       *
51       * @param theParameter key to match with a startsWith
52       * @return list of ConfigEntry
53       */
54      List<ConfigEntry> findStringMatchList(final String theParameter);
55  
56      /**
57       * Find all entry keys that begin with the specified string Remaining key portion is upper cased
58       *
59       * @param param the leading key startsWith
60       * @return map with remaining key portion as key, value as value
61       */
62      Map<String, String> findStringMatchMap(String param);
63  
64      /**
65       * Find all entry keys that begin with the specified string
66       *
67       * @param param the leading key startsWith
68       * @param preserveCase Case of key is preserved when true
69       * @return map with remaining key portion as key, value as value
70       */
71      Map<String, String> findStringMatchMap(String param, boolean preserveCase);
72  
73      /**
74       * Find all entry keys that begin with the specified string
75       *
76       * @param param the leading key startsWith
77       * @param preserveCase Case of key is preserved when true
78       * @param preserveOrder ordering of keys is preserved
79       * @return map with remaining key portion as key, value as value
80       */
81      Map<String, String> findStringMatchMap(String param, boolean preserveCase, boolean preserveOrder);
82  
83      /**
84       * Find all entry keys that begin with the specified string Remaining key portion is upper cased
85       *
86       * @param theParameter the parameter the leading key startsWith
87       * @return map with remaining key portion as key, value as a set of multiple values
88       */
89      Map<String, Set<String>> findStringMatchMultiMap(final String theParameter);
90  
91  
92      String findRequiredStringEntry(String theParameter);
93  
94      /**
95       * Find the first config entry matching with default value when none
96       *
97       * @param param the config entry name
98       * @param dflt the default value when no entry found
99       * @return string value of entry or default when no entry
100      */
101     String findStringEntry(String param, String dflt);
102 
103     /**
104      * Find the first config entry matching the name
105      *
106      * @param param the config entry name
107      * @return string value of entry or null when no entry
108      */
109     String findStringEntry(String param);
110 
111     /**
112      * Find the last config entry (newest) matching the name
113      *
114      * @param param the config entry name
115      * @return string value of entry or empty string when none
116      */
117     String findLastStringEntry(String param);
118 
119     /**
120      * Return a long from a string entry representing either an int, a long, a double, with or without a final letter
121      * designation such as "m" or "M" for megabytes, "g" or "G" for gigabytes, etc. Legal designations are bBkKmMgGTt or
122      * just a number.
123      *
124      * @param theParameter the config entry name
125      * @param dflt the default value when nothing found in config
126      * @return the long value of the size parameter
127      */
128     long findSizeEntry(String theParameter, long dflt);
129 
130     /**
131      * Return the string canonical file name of a matching entry
132      *
133      * @param theParameter the key to match
134      * @param dflt the string to use as a default when no matches are found
135      * @return the first matching value run through File.getCanonicalPath
136      */
137     String findCanonicalFileNameEntry(String theParameter, String dflt);
138 
139     /**
140      * Find config entry as an int
141      *
142      * @param param name of config entry
143      * @param dflt default value when none found
144      * @return int value or dflt when none found or not an integer
145      */
146     int findIntEntry(String param, int dflt);
147 
148     /**
149      * Find config entry as a long
150      *
151      * @param param name of config entry
152      * @param dflt default value when none found
153      * @return long value or dflt when none found or not a long
154      */
155     long findLongEntry(String param, long dflt);
156 
157     /**
158      * Find config entry as an double
159      *
160      * @param param name of config entry
161      * @param dflt default value when none found
162      * @return double value or dflt when none found or not a double
163      */
164     double findDoubleEntry(String param, double dflt);
165 
166     /**
167      * Find config entry as an boolean
168      *
169      * @param param name of config entry
170      * @param dflt default value when none found
171      * @return boolean value or dflt when none found or not a boolean
172      */
173     boolean findBooleanEntry(String param, boolean dflt);
174 
175 
176     /**
177      * Find config entry as an boolean
178      *
179      * @param param name of config entry
180      * @param dflt default value when none found
181      * @return boolean value or dflt when none found or not a boolean
182      */
183     boolean findBooleanEntry(String param, String dflt);
184 
185     /**
186      * Get the names of all entries for this config. This set is not backed by the configuration and any changes to it are
187      * not relflected in the configuration.
188      */
189     Set<String> entryKeys();
190 
191     /**
192      * Get all of the entries for this config This is a copy and changes to it are not reflected in the configuration
193      */
194     List<ConfigEntry> getEntries();
195 
196     /**
197      * Add an entry to this config
198      *
199      * @param key the name of the entry
200      * @param value the value
201      * @return the new entry or null if it fails
202      */
203     ConfigEntry addEntry(String key, String value);
204 
205     /**
206      * Add a list of entries for the same key
207      *
208      * @param key the name of the entry
209      * @param values the values
210      * @return the new entries or null if it fails
211      */
212     List<ConfigEntry> addEntries(String key, List<String> values);
213 
214     /**
215      * Remove all entries by the given name
216      *
217      * @param key the name of the entry or entries
218      * @param value the value
219      */
220     void removeEntry(String key, String value);
221 
222     /**
223      * Merge in a new configuration set with this one. New things are supposed to override older things in the sense of
224      * findStringEntry which only picks the top of the list, the new things should get added to the top.
225      *
226      * @param other the new entries to merge in
227      */
228     void merge(Configurator other) throws IOException;
229 }

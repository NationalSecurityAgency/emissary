3   import emissary.core.NamespaceException;
4   import emissary.server.EmissaryServer;
5   
6   import jakarta.servlet.http.HttpServletRequest;
7   import jakarta.ws.rs.POST;
8   import jakarta.ws.rs.Path;
9   import jakarta.ws.rs.Produces;
10  import jakarta.ws.rs.core.Context;
11  import jakarta.ws.rs.core.MediaType;
12  import jakarta.ws.rs.core.Response;
13  import org.slf4j.Logger;
14  import org.slf4j.LoggerFactory;
15  
16  @Path("")
17  // context is api
18  public class Pause {
19  
20      private static final Logger LOG = LoggerFactory.getLogger(Pause.class);
21      public static final String PAUSE = "pause";
22      public static final String UNPAUSE = "unpause";
23  
24      @POST
25      @Path("/" + PAUSE)
26      @Produces(MediaType.TEXT_HTML)
27      public Response pause(@Context HttpServletRequest request) {
28          return doAction(true);
29      }
30  
31      protected String pause() throws NamespaceException {
32          EmissaryServer.pause();
33          return "server paused";
34      }
35  
36      @POST
37      @Path("/" + UNPAUSE)
38      @Produces(MediaType.TEXT_HTML)
39      public Response unpause(@Context HttpServletRequest request) {
40          return doAction(false);
41      }
42  
43      protected String unpause() throws NamespaceException {
44          EmissaryServer.unpause();
45          return "server unpaused";
46      }
47  
48      private Response doAction(boolean pause) {
49          try {
50              return Response.ok(pause ? pause() : unpause()).build();
51          } catch (Exception e) {
52              LOG.warn("Exception trying to initiate {}", pause ? PAUSE : UNPAUSE, e);
53              return Response.serverError().entity("error trying to " + (pause ? PAUSE : UNPAUSE)).build();
54          }
55      }
56  }

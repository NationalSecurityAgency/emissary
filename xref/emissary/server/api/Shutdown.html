12  import org.slf4j.Logger;

15  @Path("")
16  // context is api
17  public class Shutdown {
18  
19      private static final Logger LOG = LoggerFactory.getLogger(Shutdown.class);
20  
21      public static final String SHUTDOWN = "shutdown";
22  
23      @POST
24      @Path("/" + SHUTDOWN)
25      @Produces(MediaType.TEXT_HTML)
26      public Response shutdownNow(@Context HttpServletRequest request) {
27          return shutdown(request, false);
28      }
29  
30      @POST
31      @Path("/" + SHUTDOWN + "/force")
32      @Produces(MediaType.TEXT_HTML)
33      public Response forceShutdown(@Context HttpServletRequest request) {
34          return shutdown(request, true);
35      }
36  
37      @SuppressWarnings("SystemExitOutsideMain")
38      protected Response shutdown(HttpServletRequest request, boolean force) {
39          try {
40              LOG.debug("Calling the stop method");
41              // need a new thread so the response will return
42              new Thread(() -> {
43                  try {
44                      if (force) {
45                          EmissaryServer.stopServerForce();
46                      } else {
47                          EmissaryServer.stopServer();
48                      }
49                  } catch (RuntimeException ignored) {
50                      // swallow
51                  }
52                  System.exit(0);
53              }).start();
54              return Response.ok("Shutdown initiated. Come again soon!").build();
55          } catch (RuntimeException e) {
56              LOG.warn("Exception trying to initiate shutdown: {}", e.getMessage());
57              return Response.serverError().entity("Error trying to initiate shutdown").build();
58          }
59      }

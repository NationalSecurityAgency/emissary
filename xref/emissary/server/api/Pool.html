25  /**
26   * The agents Emissary API endpoint. Currently contains the local (/api/pool) call and cluster (/api/clusterPool) calls.
27   */
28  @Path("")
29  // context is /api and is set in EmissaryServer.java
30  public class Pool {
31      private final Logger logger = LoggerFactory.getLogger(this.getClass());
32  
33      public static final String POOL_ENDPOINT = "api/pool";
34      public static final String POOL_CLUSTER_ENDPOINT = "api/cluster/pool";
35  
36      @GET
37      @Path("/pool")
38      @Produces(MediaType.APPLICATION_JSON)
39      public Response pool() {
40          return Response.ok().entity(this.lookupPool()).build();
41      }
42  
43      @GET
44      @Path("/cluster/pool")
45      @Produces(MediaType.APPLICATION_JSON)
46      public Response clusterPool() {
47          MapResponseEntity entity = new MapResponseEntity();
48          try {
49              // Get our local mobile agents
50              entity.append(this.lookupPool());
51              // Get all of our peers agents
52              EmissaryClient client = new EmissaryClient();
53              for (String peer : lookupPeers()) {
54                  String remoteEndPoint = stripPeerString(peer) + "api/pool";
55                  MapResponseEntity remoteEntity = client.send(new HttpGet(remoteEndPoint)).getContent(MapResponseEntity.class);
56                  entity.append(remoteEntity);
57              }
58              return Response.ok().entity(entity).build();
59          } catch (EmissaryException e) {
60              // This should never happen since we already saw if it exists
61              return Response.serverError().entity(e.getMessage()).build();
62          }
63      }
64  
65      private MapResponseEntity lookupPool() {
66          MapResponseEntity entity = new MapResponseEntity();
67          // Get the server so we can be ready to talk to ourself then the peers
68          try {
69              EmissaryServer emissaryServer = (EmissaryServer) Namespace.lookup("EmissaryServer");
70              EmissaryNode localNode = emissaryServer.getNode();
71              String nodeName = localNode.getNodeName() + ":" + localNode.getNodePort();
72              int active = 0;
73              int idle = 0;
74              try {
75                  for (int i = 0; i < AgentPool.lookup().getMaxTotal(); i++) {
76                      String agentKey = MobileAgentFactory.AGENT_NAME + "-" + String.format("%02d", i);
77                      if (Namespace.exists(agentKey)) {
78                          if (Namespace.lookup(agentKey).toString().startsWith("Idle")) {
79                              idle++;
80                          } else {
81                              active++;
82                          }
83                      } else {
84                          // TODO Handle this any better?
85                          logger.error("Missing an agent in the Namespace: {}", agentKey);
86                          entity.addError("Missing an agent in the Namespace: " + agentKey);
87                          idle++;
88                      }
89                  }
90                  entity.addKeyValue(nodeName, "Poolsize active/idle: " + active + "/" + idle);
91              } catch (EmissaryException e) {
92                  // TODO Figure out what we really want to do here in the event a node crashes
93                  logger.error("Problem when looking up the pool", e);
94                  entity.addError("Problem when looking up the pool on " + localNode + ": " + e.getMessage());
95              }
96          } catch (NamespaceException e) {
97              // should never happen
98              logger.error("Problem finding the emissary server in the namespace, something is majorly wrong", e);
99              entity.addError("Problem finding the emissary server in the namespace: " + e.getMessage());
100         }
101         return entity;
102     }

24      @GET
25      @Path("/Threaddump.action")
26      @Produces(MediaType.TEXT_HTML)
27      @Template(name = "/threaddumps")
28      public Map<String, Object> getThreaddumps() {
29          ThreadMXBean tmbean = ManagementFactory.getThreadMXBean();
30  
31          Map<String, Object> model = new HashMap<>();
32          model.put("emissary.version", new Version());
33          model.put("java.version", System.getProperty("java.vm.version"));
34          model.put("java.name", System.getProperty("java.vm.name"));
35          Map<String, Object> threadcount = new HashMap<>();
36          threadcount.put("current", tmbean.getThreadCount());
37          threadcount.put("max", tmbean.getPeakThreadCount());
38          threadcount.put("daemon", tmbean.getDaemonThreadCount());
39          model.put("threadcount", threadcount);
40  
41          Set<ThreadDumpInfo> deadlocks = new HashSet<>();
42          long[] tids = tmbean.findMonitorDeadlockedThreads();
43          if (tids != null) {
44              for (ThreadInfo ti : tmbean.getThreadInfo(tids, Integer.MAX_VALUE)) {
45                  deadlocks.add(new ThreadDumpInfo(ti));
46              }
47          }
48          model.put("deadlocks", deadlocks);
49  
50          Set<ThreadDumpInfo> threads = new HashSet<>();
51          for (ThreadInfo ti : tmbean.getThreadInfo(tmbean.getAllThreadIds(), Integer.MAX_VALUE)) {
52              threads.add(new ThreadDumpInfo(ti));
53          }
54          model.put("threads", threads);
55  
56          return model;
57      }
58  
59      public static class ThreadDumpInfo {
60          public String stack;
61  
62          public ThreadDumpInfo(@Nullable ThreadInfo ti) {
63              StringBuilder sb = new StringBuilder();
64              if (ti == null) {
65                  sb.append("A null thread?");
66              } else {
67                  sb.append("\"").append(ti.getThreadName()).append("\" tid=").append(ti.getThreadId()).append("\n");
68                  sb.append("   thread state ").append(ti.getThreadState()); // no new line
69                  if (ti.getLockName() != null) {
70                      sb.append(" (on ").append(ti.getLockName()).append(" owned by ").append(ti.getLockOwnerId()).append(")\n");
71                  }
72                  if (ti.isSuspended()) {
73                      sb.append("   SUSPENDED\n");
74                  }
75                  if (ti.isInNative()) {
76                      sb.append("   IN NATIVE CODE\n");
77                  }
78                  for (StackTraceElement ste : ti.getStackTrace()) {
79                      sb.append("      ").append(ste.toString()).append("\n");
80                  }
81              }
82              stack = sb.toString();
83          }
84      }

14  /**
15   * Stuff for adapting the Directory heartbeat calls to HTTP
16   */
17  public class HeartbeatAdapter extends EmissaryClient {
18  
19      private static final Logger logger = LoggerFactory.getLogger(HeartbeatAdapter.class);
20  
21      public static final String FROM_PLACE_NAME = "hbf";
22      public static final String TO_PLACE_NAME = "hbt";
23  
24      /**
25       * Process the heartbeat call coming remotely over HTTP request params onto the specified (local) directory place
26       *
27       * @return reference to the place when found, null otherwise
28       */
29      @Nullable
30      public IServiceProviderPlace inboundHeartbeat(final String fromName, final String toName) throws NamespaceException {
31          if (toName == null) {
32              throw new IllegalArgumentException("No place specified in msg from " + fromName);
33          }
34  
35          logger.debug("Servicing inbound heartbeat for {} to {}", fromName, toName);
36  
37          final DirectoryPlace d = (DirectoryPlace) Namespace.lookup(toName);
38          if (!d.isStaticPeer(KeyManipulator.getDefaultDirectoryKey(fromName))) {
39              logger.warn("Contact attempted from {} but it is not a configured peer", fromName);
40              return null;
41          }
42  
43          IServiceProviderPlace place = null;
44  
45          try {
46              place = (IServiceProviderPlace) Namespace.lookup(toName);
47          } catch (NamespaceException ne) {
48              throw ne;
49          }
50  
51          if (place == null) {
52              throw new IllegalArgumentException("No place found using name " + toName + " from " + fromName);
53          }
54  
55          logger.debug("Heartbeat returning {}", place);
56  
57          return place;
58      }
59  }

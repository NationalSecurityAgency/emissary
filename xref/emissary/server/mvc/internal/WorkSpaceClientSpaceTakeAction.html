40      @POST
41      @Path("/WorkSpaceClientSpaceTake.action")
42      @Consumes(MediaType.APPLICATION_FORM_URLENCODED)
43      @Produces(MediaType.APPLICATION_XML)
44      public Response clientSpaceTake(@FormParam(CLIENT_NAME) String placeName, @FormParam(SPACE_NAME) String spaceName) {
45          String cleanPlaceName = RequestUtil.sanitizeParameter(placeName);
46          String cleanSpaceName = RequestUtil.sanitizeParameter(spaceName);
47          if (StringUtils.isBlank(cleanPlaceName) || StringUtils.isBlank(cleanSpaceName)) {
48              return Response.serverError().entity(HtmlEscaper.escapeHtml(
49                      "Bad params: " + CLIENT_NAME + " - " + cleanPlaceName + ", or " + SPACE_NAME + " - " + cleanSpaceName)).build();
50          }
51  
52          try {
53              return doClientSpaceTake(cleanPlaceName, cleanSpaceName);
54          } catch (EmissaryException | IllegalArgumentException e) {
55              logger.warn("There was an exception in the WorkSpaceClientSpaceTake", e);
56              return Response.serverError().entity("There was an exception in the WorkSpaceClientSpaceTake").build();
57          }
58      }
59  
60      private static Response doClientSpaceTake(String placeName, String spaceName) throws EmissaryException {
61          // TODO Figure out why we have to remove the key prefix now
62          String workspaceKey = spaceName.substring("WORKSPACE.WORK_SPACE.INPUT.".length());
63          final WorkSpace space = (WorkSpace) Namespace.lookup(workspaceKey);
64          if (space == null) {
65              throw new IllegalArgumentException("No WorkSpace found using name " + spaceName);
66          }
67  
68          final WorkBundle path = space.take(placeName);
69          if (path == null) {
70              throw new EmissaryException("WorkSpaceClientSpaceTakeWorker failed, no bundle to retrieve");
71          }
72  
73          // old return from WorkSpaceClientSpaceTakeWorker
74          // return WORKER_SUCCESS;
75          return Response.ok().entity(path.toXml()).build();
76      }

142     /**
143      * Creates and starts a server that is bound into the local Namespace using DEFAULT_NAMESPACE_NAME and returned
144      *
145      *
146      */
147     public Server startServer() {
148         // do what StartJetty and then JettyServer did to start
149         try {
150             // Resource.setDefaultUseCaches(false);
151 
152             // needs to be loaded first into the server as it setups up Emissary stuff
153             ContextHandler emissaryHandler = buildEmissaryHandler();
154             // TODO: rework this, no need for it be set with a context path but if this
155             // is left out, it matches / and nothing works correctly
156             emissaryHandler.setContextPath("/idontreallyservecontentnowdoi");
157             ContextHandler lbConfigHandler = buildLogbackConfigHandler();
158             lbConfigHandler.setContextPath("/lbConfig");
159             ContextHandler apiHandler = buildApiHandler();
160             apiHandler.setContextPath("/api");
161             ContextHandler mvcHandler = buildMvcHandler();
162             mvcHandler.setContextPath("/emissary");
163             // needs to be loaded last into the server so other contexts can match or fall through
164             ContextHandler staticHandler = buildStaticHandler();
165             staticHandler.setContextPath("/");
166 
167             LoginService loginService = buildLoginService();
168             ConstraintSecurityHandler security = buildSecurityHandler();
169             security.setLoginService(loginService);
170 
171             // secure some of the contexts
172             final HandlerList securedHandlers = new HandlerList();
173             securedHandlers.addHandler(lbConfigHandler);
174             securedHandlers.addHandler(apiHandler);
175             securedHandlers.addHandler(mvcHandler);
176             securedHandlers.addHandler(staticHandler);
177             security.setHandler(securedHandlers);
178 
179             final HandlerList handlers = new HandlerList();
180             handlers.addHandler(emissaryHandler); // not secured, no endpoints and must be loaded first
181             handlers.addHandler(security);
182 
183             Server configuredServer = configureServer();
184             configuredServer.setHandler(handlers);
185             configuredServer.addBean(loginService);
186             configuredServer.setStopAtShutdown(true);
187             configuredServer.setStopTimeout(10000L);
188             if (this.cmd.shouldDumpJettyBeans() && LOG.isInfoEnabled()) {
189                 LOG.info(configuredServer.dump());
190             }
191             this.server = configuredServer;
192             bindServer(); // emissary specific
193 
194             configuredServer.start();
195             // server.join(); // don't join so we can shutdown
196 
197             String serverLocation = cmd.getScheme() + "://" + cmd.getHost() + ":" + cmd.getPort();
198 
199             // write out env.sh file here
200             Path envsh = Paths.get(ConfigUtil.getProjectBase() + File.separator + "env.sh");
201             if (Files.exists(envsh)) {
202                 LOG.debug("Removing old {}", envsh.toAbsolutePath());
203                 Files.delete(envsh);
204             }
205             String envUri = serverLocation + "/api/env.sh";
206             EmissaryResponse er = new EmissaryClient().send(new HttpGet(envUri));
207             String envString = er.getContentString();
208             Files.createFile(envsh);
209             Files.write(envsh, envString.getBytes());
210             LOG.info("Wrote {}", envsh.toAbsolutePath());
211             LOG.debug(" with \n{}", envString);
212 
213             if (cmd.isPause()) {
214                 pause(true);
215             } else {
216                 unpause(true);
217             }
218 
219             LOG.info("Started EmissaryServer at {}", serverLocation);
220 
221             // check if invisible place start-ups occurred on strict server start-up, and shut down server if so.
222             if (Startup.isInvisPlacesStartedInStrictMode() && this.server.isStarted()) {
223                 EmissaryServer.stopServer(true);
224                 LOG.info("Server shut down due to invisible place startups on strict-mode: {}", Startup.getInvisPlaces());
225             }
226 
227             return configuredServer;
228         } catch (Throwable t) {
229             String errorMsg = "Emissary server didn't start";
230             throw new EmissaryRuntimeException(errorMsg, t);
231         }
232     }

294     /**
295      * Refresh reloadable places
296      */
297     public static void refresh() {
298         try {
299             pause();
300             waitForAgentsToDrain();
301             Namespace.lookup(ServiceProviderRefreshablePlace.class)
302                     .forEach(p -> {
303                         p.invalidate();
304                         p.refresh(true);
305                     });
306             unpause();
307         } catch (Throwable e) {
308             LOG.error("There was an error trying to refresh services, shutting down!!", e);
309             stopServer();
310         }
311     }

313     private static void waitForAgentsToDrain() throws NamespaceException, TimeoutException {
314         long agentCount;
315         try {
316             long maxWait = getInstance().getNode().getNodeRefreshTimeout();
317             long timeout = System.currentTimeMillis() + maxWait;
318             final Set<MobileAgent> agents = Namespace.lookup(MobileAgent.class, true);
319             LOG.info("Waiting for server to drain, waiting max of {} minutes...", TimeUnit.MILLISECONDS.toMinutes(maxWait));
320             do {
321                 TimeUnit.SECONDS.sleep(5);
322                 agentCount = agents.stream().filter(MobileAgent::isInUse).count();
323             } while (agentCount > 0 && System.currentTimeMillis() < timeout);
324 
325             if (agentCount > 0) {
326                 throw new TimeoutException("Server did not drain in the allotted amount of time");
327             }
328         } catch (InterruptedException e) {
329             Thread.currentThread().interrupt();
330         }
331     }

374     /**
375      * Stop the server with an optional force flag
376      *
377      * @param force force shutdown
378      */
379     public static void stopServer(final boolean force) {
380         // TODO pull these out to methods and test them
381 
382         LOG.info("Beginning shutdown of EmissaryServer");
383         logThreadDump("Thread dump before anything");
384 
385         try {
386             pause();
387             LOG.info("Done pausing server");
388         } catch (Exception ex) {
389             LOG.error("Error pausing server", ex);
390         }
391 
392         try {
393             Sentinel sentinel = Sentinel.lookup();
394             sentinel.quit();
395         } catch (Exception ex) {
396             LOG.warn("No sentinel available");
397         }
398 
399         try {
400             if (force) {
401                 AgentPool.lookup().kill();
402             } else {
403                 AgentPool.lookup().close();
404             }
405         } catch (Exception e) {
406             LOG.warn("Problem stopping AgentPool", e);
407         }
408 
409         logThreadDump("Thread dump after closing agent pool");
410 
411         try {
412             DirectoryPlace.lookup().shutDown();
413         } catch (Exception e) {
414             LOG.warn("Problem shutting down DirectoryPlace", e);
415         }
416 
417         try {
418             MoveSpool.lookup().quit();
419         } catch (Exception e) {
420             LOG.warn("Problem stopping MoveSpool", e);
421         }
422 
423         // Stop the places
424         for (String key : Namespace.keySet()) {
425             try {
426                 Object obj = Namespace.lookup(key);
427                 if (obj instanceof IServiceProviderPlace) {
428                     LOG.info("Stopping {} ", obj);
429                     ((IServiceProviderPlace) obj).shutDown();
430                     // make sure key is removed from namespace
431                     Namespace.unbind(key);
432                     LOG.info("Done stopping place: {}", key);
433                 }
434             } catch (Exception ex) {
435                 LOG.error("Error shutting down " + key, ex);
436             }
437         }
438         LOG.info("Done stopping all places");
439 
440         RollManager.shutdown();
441 
442         // Print the stats
443         try {
444             ResourceWatcher rw = ResourceWatcher.lookup();
445             rw.logStats(LOG);
446             rw.quit();
447         } catch (Exception ex) {
448             LOG.warn("No resource statistics available");
449         }
450 
451         try {
452             MetricsManager.lookup().shutdown();
453         } catch (Exception ex) {
454             LOG.warn("No metrics manager available");
455         }
456 
457         SPILoader.unload();
458 
459         LOG.info("Done stopping all services");
460 
461         // thread dump now
462         logThreadDump("Thread dump before stopping jetty server");
463 
464         try {
465             EmissaryServer.getInstance().getServer().stop();
466         } catch (InterruptedException e) {
467             LOG.warn("Interrupted! Expected?");
468             Thread.currentThread().interrupt();
469         } catch (Exception e) {
470             LOG.warn("Unable to stop EmissaryServer", e);
471         }
472 
473         LOG.debug("Unbinding name: {}", getDefaultNamespaceName());
474         Namespace.unbind(getDefaultNamespaceName());
475         Namespace.clear();
476         LOG.info("EmissaryServer completely stopped");
477     }

486     @SuppressWarnings("unchecked")
487     private static void logThreadDump(String initialLog) {
488         if (LOG.isTraceEnabled()) {
489             ThreadDumpAction tda = new ThreadDumpAction();
490             Map<String, Object> dumps = tda.getThreaddumps();
491             StringBuilder sb = new StringBuilder();
492             sb.append("\n" + initialLog);
493             sb.append("\nThread DUMP");
494             sb.append("\nEmissary Version: " + dumps.get("emissary.version"));
495             sb.append("\nJVM Version: " + dumps.get("java.version"));
496             sb.append("\nJVM Name: " + dumps.get("java.name"));
497             Map<String, Object> threadcount = (Map<String, Object>) dumps.get("threadcount");
498             sb.append("\nThread count: current=" + threadcount.get("current") + " max=" + threadcount.get("max") + " daemon="
499                     + threadcount.get("daemon"));
500             sb.append("\nDeadlocked Threads:");
501             for (ThreadDumpInfo tdi : (Set<ThreadDumpInfo>) dumps.get("deadlocks")) {
502                 sb.append("\n" + tdi.stack);
503             }
504             sb.append("\nThread dump:");
505             for (ThreadDumpInfo tdi : (Set<ThreadDumpInfo>) dumps.get("threads")) {
506                 sb.append("\n" + tdi.stack);
507             }
508             LOG.trace(sb.toString());
509         }
510     }

609     private static ConstraintSecurityHandler buildSecurityHandler() {
610         ConstraintSecurityHandler handler = new ConstraintSecurityHandler();
611 
612         Constraint authConstraint = new Constraint();
613         authConstraint.setName("auth");
614         authConstraint.setAuthenticate(true);
615         authConstraint.setRoles(new String[] {"everyone", "emissary", "admin", "support", "manager"});
616 
617         Constraint noAuthConstraint = new Constraint();
618         noAuthConstraint.setName("no_auth");
619         noAuthConstraint.setAuthenticate(false);
620 
621         ConstraintMapping mapping = new ConstraintMapping();
622         mapping.setPathSpec("/*");
623         mapping.setConstraint(authConstraint);
624 
625         ConstraintMapping health = new ConstraintMapping();
626         health.setPathSpec("/api/health");
627         health.setConstraint(noAuthConstraint);
628 
629         handler.setConstraintMappings(new ConstraintMapping[] {mapping, health});
630         handler.setAuthenticator(new DigestAuthenticator());
631         return handler;
632     }

650     private ContextHandler buildStaticHandler() {
651         // Add pathspec for static assets
652         ServletHolder homeHolder = new ServletHolder("static-holder", DefaultServlet.class);
653         // If no filesytem path was passed in, load assets from the classpath
654         if (null == cmd.getStaticDir()) {
655             LOG.debug("Loading static resources from the classpath");
656             homeHolder.setInitParameter("resourceBase", this.getClass().getClassLoader().getResource("public").toExternalForm());
657         } else {
658             // use --staticDir ${project_loc}/src/main/resources/public as command args
659             LOG.debug("Loading static resources from staticPath: {}", cmd.getStaticDir());
660             homeHolder.setInitParameter("resourceBase", cmd.getStaticDir().toAbsolutePath().toString());
661         }
662         homeHolder.setInitParameter("dirAllowed", "true");
663         homeHolder.setInitParameter("pathInfoOnly", "true");
664         // homeHolder.setInitOrder(0);
665 
666         ServletContextHandler homeContextHandler = new ServletContextHandler(ServletContextHandler.SESSIONS);
667         homeContextHandler.addServlet(homeHolder, "/*");
668 
669         return homeContextHandler;
670     }

726         }

729         emissaryHolderContext.addEventListener(new InitializeContext(emissaryNode));
730 
731         return emissaryHolderContext;

761     /**
762      * Create a server connector, insecure (http) or secure (https) depending on {@link ServerCommand#isSslEnabled()}
763      *
764      * @param server the Jetty HTTP Servlet Server
765      * @return server connector that is the primary connector for the Jetty server over TCP/IP
766      * @throws IOException if there is an error
767      */
768     private ServerConnector createServerConnector(Server server) throws IOException {
769         ServerConnector connector = cmd.isSslEnabled() ? createHttpsConnector(server) : createHttpConnector(server);
770         connector.setHost(cmd.getHost());
771         connector.setPort(cmd.getPort());
772         return connector;

785     /**
786      * Create a secure https connector
787      *
788      * @param server the Jetty HTTP Servlet Server
789      * @return ServerConnector is the primary connector for the Jetty server over TCP/IP
790      * @throws IOException if there is an error
791      */
792     private ServerConnector createHttpsConnector(Server server) throws IOException {
793         HttpConfiguration httpConfig = new HttpConfiguration();
794         httpConfig.setSecureScheme("https");
795         httpConfig.setSecurePort(cmd.getPort());
796 
797         HttpConfiguration httpsConfig = new HttpConfiguration(httpConfig);
798         httpsConfig.addCustomizer(createSecureRequestCustomizer());
799 
800         return new ServerConnector(server,
801                 new SslConnectionFactory(getSslContextFactory(), HttpVersion.HTTP_1_1.asString()),
802                 new HttpConnectionFactory(httpsConfig));
803     }

805     /**
806      * SecureRequestCustomizer extracts the attribute from an SSLContext and sets them on the request according to Servlet
807      * Specification Requirements. Jetty defaults for the SecureRequestCustomizer are:
808      * <ul>
809      * <li>isSniRequired() => false // Server Name Indication (SNI) is required
810      * <li>isSniHostCheck() => true // SNI Host name must match when there is an SNI certificate
811      * <li>getStsMaxAge() => -1 (no max age) // Strict-Transport-Security (STS) max age
812      * <li>isStsIncludeSubDomains() => false // include-subdomain property is sent with any STS header
813      * </ul>
814      *
815      * @return a secure request customizer
816      */
817     private SecureRequestCustomizer createSecureRequestCustomizer() {
818         SecureRequestCustomizer secureRequestCustomizer = new SecureRequestCustomizer();
819         secureRequestCustomizer.setSniHostCheck(cmd.isSniHostCheckEnabled());
820         return secureRequestCustomizer;
821     }

823     /**
824      * Create a {@link SslContextFactory.Server} using keystore and truststore config from HttpConnectionFactory.cfg,
825      * probably flavored with '-SSL'. Note: only jks files are supported, but could be expanded
826      *
827      * @return SslContextFactory that is used to configure SSL parameters to be used by server connectors
828      * @throws IOException if there is an error getting the context factory
829      */
830     private static SslContextFactory.Server getSslContextFactory() throws IOException {
831         Configurator httpConnFactCfg = ConfigUtil.getConfigInfo(HTTPConnectionFactory.class);
832         String keystore = httpConnFactCfg.findStringEntry("javax.net.ssl.keyStore", "no-keystore");
833         System.setProperty("javax.net.ssl.keyStore", keystore);
834         String keystorePass = httpConnFactCfg.findStringEntry("javax.net.ssl.keyStorePassword", "no-keypass");
835         System.setProperty("javax.net.ssl.keyStorePassword", keystorePass);
836         String trustStore = httpConnFactCfg.findStringEntry("javax.net.ssl.trustStore", keystore);
837         System.setProperty("javax.net.ssl.trustStore", trustStore);
838         String trustStorePass = httpConnFactCfg.findStringEntry("javax.net.ssl.trustStorePassword", keystorePass);
839         System.setProperty("javax.net.ssl.trustStorePassword", trustStorePass);
840 
841         SslContextFactory.Server sslContextFactory = new SslContextFactory.Server();
842         sslContextFactory.setKeyStorePath(keystore);
843         sslContextFactory.setKeyStorePassword(keystorePass);
844 
845         KeyStore trustStoreInstance;
846         try (InputStream is = Files.newInputStream(Paths.get(trustStore))) {
847             trustStoreInstance = KeyStore.getInstance("JKS");
848             trustStoreInstance.load(is, trustStorePass.toCharArray());
849         } catch (KeyStoreException | CertificateException | NoSuchAlgorithmException e) {
850             throw new IOException("There was an issue loading the truststore", e);
851         }
852 
853         sslContextFactory.setTrustStore(trustStoreInstance);
854         return sslContextFactory;
855     }

25  /**
26   * Connects to an arbitrary external service that implements {@link emissary.grpc.sample.v1.SampleServiceGrpc}. Results
27   * are written to an alternate view.
28   */
29  public class GrpcSampleServicePlace extends GrpcRoutingPlace {
30      public static final String SERVICE_PROXY = "SAMPLE_FORM";
31      public static final String ALTERNATE_VIEW_NAME = "SAMPLE_ALTERNATE_VIEW";
32      private static final Pattern FORM_PATTERN = Pattern.compile(SERVICE_PROXY + "-(\\w+)");
33      public static final String GRPC_POOL_KILL_AFTER_RETURN = "GRPC_POOL_KILL_AFTER_RETURN";
34  
35      @Nullable
36      private AtomicBoolean connectionValidated;
37      private final boolean passivateConnection;
38  
39      public GrpcSampleServicePlace(Configurator configs) throws IOException {
40          super(configs);
41          connectionValidated = null;
42          passivateConnection = configs.findBooleanEntry(GRPC_POOL_KILL_AFTER_RETURN, false);
43      }
44  
45      @Override
46      public void process(IBaseDataObject o) {
47          processWithMockBehavior(o, SampleServiceBlockingStub::callSampleService);
48      }
49  
50      /**
51       * Performs Emissary processing while simulating server errors.
52       *
53       * @param o the data object to process
54       * @param e the Exception to throw
55       */
56      public void throwExceptionsDuringProcess(IBaseDataObject o, RuntimeException e) {
57          processWithMockBehavior(o, (stub, request) -> {
58              throw e;
59          });
60      }
61  
62      /**
63       * Performs Emissary processing while simulating server errors. Assumes {@link RetryHandler} is configured to retry the
64       * remote procedural call.
65       *
66       * @param o the data object to process
67       * @param e the Exception to throw
68       * @param maxRetries the maximum number of times the handler can attempt to retry a connection, caps at configured
69       *        amount
70       * @param attempt the current number of processing attempts
71       */
72      public void throwExceptionsDuringProcessWithSuccessfulRetry(IBaseDataObject o, RuntimeException e,
73              int maxRetries, AtomicInteger attempt) {
74  
75          processWithMockBehavior(o, (stub, request) -> {
76              if (attempt.incrementAndGet() < maxRetries) {
77                  throw e;
78              }
79              return stub.callSampleService(request);
80          });
81      }
82  
83      /**
84       * Performs Emissary processing with custom call logic to the mock gRPC server. This is unnecessary in a real
85       * {@link GrpcRoutingPlace} subclass, and is only implemented here to simulate errors occurring in the server.
86       *
87       * @param o the data object to process
88       * @param callLogic method for the mock server to perform
89       */
90      private void processWithMockBehavior(
91              IBaseDataObject o, BiFunction<SampleServiceBlockingStub, SampleRequest, SampleResponse> callLogic) {
92  
93          Matcher matcher = FORM_PATTERN.matcher(o.currentForm());
94          if (matcher.matches()) {
95              String targetId = matcher.group(1);
96              SampleRequest request = SampleRequest.newBuilder()
97                      .setQuery(ByteString.copyFrom(o.data()))
98                      .build();
99              SampleResponse response = invokeGrpc(targetId, SampleServiceGrpc::newBlockingStub, callLogic, request);
100             o.addAlternateView(ALTERNATE_VIEW_NAME, response.getResult().toByteArray());
101         }
102     }
103 
104     /**
105      * Calls a health check to the external service before borrowing the channel from the pool.
106      *
107      * @param managedChannel the gRPC channel to validate
108      * @return {@code true} if channel is healthy, otherwise {@code false}
109      */
110     @Override
111     protected boolean validateConnection(ManagedChannel managedChannel) {
112         if (connectionValidated != null) {
113             connectionValidated.set(true);
114         }
115         SampleServiceBlockingStub stub = SampleServiceGrpc.newBlockingStub(managedChannel);
116         return stub.callSampleHealthCheck(Empty.getDefaultInstance()).getOk();
117     }

119     /**
120      * Shuts down a channel after it's been returned to the pool, if the place has been configured to do so.
121      *
122      * @param managedChannel the gRPC channel to clean up
123      */
124     @Override
125     protected void passivateConnection(ManagedChannel managedChannel) {
126         if (passivateConnection) {
127             managedChannel.shutdownNow();
128         }
129     }

33   * Place for processing data using gRPC connections to external services. Supports multiple end-points with
34   * <i>shared</i> configurations, where each endpoint is identified by a given target ID.
35   * <p>
36   * Configuration Keys:
37   * <ul>
38   * <li>{@code GRPC_HOST_{Target-ID}} - gRPC service hostname or DNS target, where {@code Target-ID} is the unique
39   * identifier for the given host:port</li>
40   * <li>{@code GRPC_PORT_{Target-ID}} - gRPC service port, where {@code Target-ID} is the unique identifier for the given
41   * host:port</li>
42   * <li>See {@link ConnectionFactory} for supported pooling and gRPC channel configuration keys and defaults.</li>
43   * <li>See {@link RetryHandler} for supported retry configuration keys and defaults.</li>

104     private void configureGrpc() {
105         if (configG == null) {
106             throw new IllegalStateException("gRPC configurations not found for " + this.getPlaceName());
107         }
108 
109         hostnameTable.putAll(getHostnameConfigs());
110         portNumberTable.putAll(getPortNumberConfigs());
111 
112         if (!hostnameTable.keySet().equals(portNumberTable.keySet())) {
113             throw new IllegalArgumentException("gRPC hostname target-IDs do not match gRPC port number target-IDs");
114         }
115 
116         if (hostnameTable.isEmpty()) {
117             throw new NullPointerException(String.format(
118                     "Missing required arguments: %s${Target-ID} and %s${Target-ID}", GRPC_HOST, GRPC_PORT));
119         }
120 
121         Set<String> targetIds = hostnameTable.keySet();
122         for (String id : targetIds) {
123             channelPoolTable.put(id, newConnectionPool(id));
124         }
125 
126         retryHandler = new RetryHandler(configG, this.getPlaceName());
127     }

158     /**
159      * Executes a unary gRPC call to a given endpoint using a {@code BlockingStub}. If the gRPC connection fails due to a
160      * {@link PoolException} or a {@link ServiceNotAvailableException}, the call will be tried again per the configurations
161      * set using {@link RetryHandler}. All other Exceptions are thrown on the spot. Will also throw an Exception once max
162      * attempts have been reached.
163      *
164      * @param targetId the identifier used in the configs for the given gRPC endpoint
165      * @param stubFactory function that creates the appropriate gRPC stub from a {@link ManagedChannel}
166      * @param callLogic function that performs the actual gRPC call using the stub and request
167      * @param request the protobuf request message to send
168      * @return the response returned by the gRPC call
169      * @param <Q> the protobuf request type
170      * @param <R> the protobuf response type
171      * @param <S> the gRPC stub type
172      */
173     protected <Q extends GeneratedMessageV3, R extends GeneratedMessageV3, S extends AbstractBlockingStub<S>> R invokeGrpc(
174             String targetId, Function<ManagedChannel, S> stubFactory, BiFunction<S, Q, R> callLogic, Q request) {
175 
176         return retryHandler.execute(() -> {
177             ObjectPool<ManagedChannel> channelPool = channelPoolLookup(targetId);
178             ManagedChannel channel = ConnectionFactory.acquireChannel(channelPool);
179             R response = null;
180             try {
181                 S stub = stubFactory.apply(channel);
182                 response = callLogic.apply(stub, request);
183                 ConnectionFactory.returnChannel(channel, channelPool);
184             } catch (StatusRuntimeException e) {
185                 ConnectionFactory.invalidateChannel(channel, channelPool);
186                 ServiceException.handleGrpcStatusRuntimeException(e);
187             } catch (RuntimeException e) {
188                 ConnectionFactory.invalidateChannel(channel, channelPool);
189                 throw e;
190             }
191             return response;
192         });
193     }

229     protected <T> T tableLookup(Map<String, T> table, String targetId) {
230         if (table.containsKey(targetId)) {

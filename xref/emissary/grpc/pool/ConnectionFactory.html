25  /**
26   * Abstract base class for managing a pool of gRPC {@link ManagedChannel} connections.
27   * <p>
28   * Configuration Keys:
29   * <ul>
30   * <li>{@code GRPC_KEEP_ALIVE_MILLIS} - Time to wait before sending a ping on idle, default={@code 60000L}</li>
31   * <li>{@code GRPC_KEEP_ALIVE_TIMEOUT_MILLIS} - Timeout for receiving ping ACKs, default={@code 30000L}</li>
32   * <li>{@code GRPC_KEEP_ALIVE_WITHOUT_CALLS} - Send pings even when no RPCs are active if {@code true},
33   * default={@code false}</li>
34   * <li>{@code GRPC_LOAD_BALANCING_POLICY} - gRPC load balancing policy, default={@code "round_robin"}</li>
35   * <li>{@code GRPC_MAX_INBOUND_MESSAGE_BYTE_SIZE} - Max inbound gRPC message size, default={@code 4194304}</li>
36   * <li>{@code GRPC_MAX_INBOUND_METADATA_BYTE_SIZE} - Max inbound gRPC metadata size, default={@code 8192}</li>
37   * <li>{@code GRPC_POOL_BLOCK_EXHAUSTED} - If {@code true}, threads block when pool is empty, otherwise throws an
38   * Exception, default={@code true}</li>
39   * <li>{@code GRPC_POOL_ERODING_FACTOR} - Optional shrink rate for idle connections, default={@code -1.0f}</li>
40   * <li>{@code GRPC_POOL_MAX_BORROW_WAIT_MILLIS} - Time to wait before failing a borrow attempt,
41   * default={@code 10000L}</li>
42   * <li>{@code GRPC_POOL_MAX_IDLE_CONNECTIONS} - Maximum idle connections in the pool, default={@code 8}</li>
43   * <li>{@code GRPC_POOL_MAX_SIZE} - Maximum total connections allowed, default={@code 8}</li>
44   * <li>{@code GRPC_POOL_MIN_IDLE_CONNECTIONS} - Minimum idle connections in the pool, default={@code 0}</li>
45   * <li>{@code GRPC_POOL_RETRIEVAL_ORDER} - Whether pool behaves LIFO or FIFO, default={@code "LIFO"}</li>
46   * <li>{@code GRPC_POOL_TEST_BEFORE_BORROW} - If {@code true}, validates pooled connections before use with
47   * {@link #validateObject(PooledObject)}, default={@code true}</li>
48   * </ul>
49   * <a href="https://docs.microsoft.com/en-us/aspnet/core/grpc/performance?view=aspnetcore-5.0">Source</a> for default
50   * gRPC configurations.
51   */
52  public class ConnectionFactory extends BasePooledObjectFactory<ManagedChannel> {
53      public static final String GRPC_KEEP_ALIVE_MILLIS = "GRPC_KEEP_ALIVE_MILLIS";
54      public static final String GRPC_KEEP_ALIVE_TIMEOUT_MILLIS = "GRPC_KEEP_ALIVE_TIMEOUT_MILLIS";
55      public static final String GRPC_KEEP_ALIVE_WITHOUT_CALLS = "GRPC_KEEP_ALIVE_WITHOUT_CALLS";
56      public static final String GRPC_LOAD_BALANCING_POLICY = "GRPC_LOAD_BALANCING_POLICY";
57      public static final String GRPC_MAX_INBOUND_MESSAGE_BYTE_SIZE = "GRPC_MAX_INBOUND_MESSAGE_BYTE_SIZE";
58      public static final String GRPC_MAX_INBOUND_METADATA_BYTE_SIZE = "GRPC_MAX_INBOUND_METADATA_BYTE_SIZE";
59      public static final String GRPC_POOL_BLOCK_EXHAUSTED = "GRPC_POOL_BLOCK_EXHAUSTED";
60      public static final String GRPC_POOL_ERODING_FACTOR = "GRPC_POOL_ERODING_FACTOR";
61      public static final String GRPC_POOL_MAX_BORROW_WAIT_MILLIS = "GRPC_POOL_MAX_BORROW_WAIT_MILLIS";
62      public static final String GRPC_POOL_MAX_IDLE_CONNECTIONS = "GRPC_POOL_MAX_IDLE_CONNECTIONS";
63      public static final String GRPC_POOL_MAX_SIZE = "GRPC_POOL_MAX_SIZE";
64      public static final String GRPC_POOL_MIN_IDLE_CONNECTIONS = "GRPC_POOL_MIN_IDLE_CONNECTIONS";
65      public static final String GRPC_POOL_RETRIEVAL_ORDER = "GRPC_POOL_RETRIEVAL_ORDER";
66      public static final String GRPC_POOL_TEST_BEFORE_BORROW = "GRPC_POOL_TEST_BEFORE_BORROW";

85      /**
86       * Constructs a new gRPC connection factory using the provided host, port, and configuration. Initializes pool settings
87       * and gRPC channel properties from the given configuration source.
88       * <p>
89       * See {@link ConnectionFactory} for supported configuration keys and defaults.
90       * 
91       * @param host gRPC service hostname or DNS target
92       * @param port gRPC service port
93       * @param configG configuration provider for channel and pool parameters
94       * @param channelValidator method to determine if channel can successfully communicate with its associated server
95       * @param channelPassivator method that cleans up a channel after a gRPC call
96       */
97      public ConnectionFactory(String host, int port, Configurator configG,
98              Predicate<ManagedChannel> channelValidator, Consumer<ManagedChannel> channelPassivator) {
99          this.host = host;
100         this.port = port;
101         this.target = host + ":" + port; // target may be a host or dns service
102 
103         this.channelValidator = channelValidator;
104         this.channelPassivator = channelPassivator;
105 
106         // How often (in milliseconds) to send pings when the connection is idle
107         this.keepAliveMillis = configG.findLongEntry(GRPC_KEEP_ALIVE_MILLIS, 60000L);
108 
109         // Time to wait (in milliseconds) for a ping ACK before closing the connection
110         this.keepAliveTimeoutMillis = configG.findLongEntry(GRPC_KEEP_ALIVE_TIMEOUT_MILLIS, 30000L);
111 
112         // Whether to send pings when no RPCs are active
113         // Note: Seme gRPC services have this set to false and will be noisy if not adjusted
114         this.keepAliveWithoutCalls = configG.findBooleanEntry(GRPC_KEEP_ALIVE_WITHOUT_CALLS, false);
115 
116         // Specifies how the client chooses between multiple backend addresses
117         // e.g. "pick_first" uses the first address only, "round_robin" cycles through all of them for client-side balancing
118         this.loadBalancingPolicy = LoadBalancingPolicy.getPolicyName(
119                 configG.findStringEntry(GRPC_LOAD_BALANCING_POLICY), LoadBalancingPolicy.ROUND_ROBIN);
120 
121         // Max size (in bytes) for incoming messages and message metadata from the server
122         this.maxInboundMessageByteSize = configG.findIntEntry(GRPC_MAX_INBOUND_MESSAGE_BYTE_SIZE, 4 << 20); // 4 MiB
123         this.maxInboundMetadataByteSize = configG.findIntEntry(GRPC_MAX_INBOUND_METADATA_BYTE_SIZE, 8 << 10); // 8 KiB
124 
125         // Controls how aggressively idle connections are phased out over time
126         // Set to a float between 0.0 and 1.0 to enable erosion (e.g. 0.2 = mild erosion)
127         // Set to -1.0 to disable automatic pool shrinking entirely
128         this.erodingPoolFactor = (float) configG.findDoubleEntry(GRPC_POOL_ERODING_FACTOR, -1.0f);
129 
130         // Enable thread blocking when borrowing from exhausted pool
131         this.poolConfig.setBlockWhenExhausted(configG.findBooleanEntry(GRPC_POOL_BLOCK_EXHAUSTED,
132                 BaseObjectPoolConfig.DEFAULT_BLOCK_WHEN_EXHAUSTED));
133 
134         // Max duration to wait until block is released from exhausted pool
135         this.poolConfig.setMaxWait(Duration.ofMillis(configG.findLongEntry(GRPC_POOL_MAX_BORROW_WAIT_MILLIS, 10000L)));
136 
137         // Min/max number of idle connections in pool
138         this.poolConfig.setMinIdle(configG.findIntEntry(GRPC_POOL_MIN_IDLE_CONNECTIONS, 0));
139         this.poolConfig.setMaxIdle(configG.findIntEntry(GRPC_POOL_MAX_IDLE_CONNECTIONS, 8));
140 
141         // Max number of total connections in pool
142         this.poolConfig.setMaxTotal(configG.findIntEntry(GRPC_POOL_MAX_SIZE, 8));
143 
144         // Order for pool to borrow connections
145         this.poolConfig.setLifo(PoolRetrievalOrdering.isLifo(
146                 configG.findStringEntry(GRPC_POOL_RETRIEVAL_ORDER), PoolRetrievalOrdering.LIFO));
147 
148         // Whether to validate channels when borrowing from the pool
149         this.poolConfig.setTestOnBorrow(configG.findBooleanEntry(GRPC_POOL_TEST_BEFORE_BORROW, true));
150     }

152     /**
153      * Borrows a {@link ManagedChannel} from the provided connection pool.
154      *
155      * @param pool the object pool to borrow from
156      * @return a managed gRPC channel
157      * @throws PoolException if the pool is exhausted or borrowing fails
158      */
159     public static ManagedChannel acquireChannel(ObjectPool<ManagedChannel> pool) {
160         try {
161             return pool.borrowObject();
162         } catch (Exception e) {
163             throw new PoolException(String.format("Unable to borrow channel from pool: %s", e.getMessage()), e);
164         }
165     }
166 
167     /**
168      * Invalidates a {@link ManagedChannel}, removing it from the pool. Used when a channel is considered unhealthy or no
169      * longer usable.
170      *
171      * @param channel the channel to invalidate
172      * @param pool the pool the channel belongs to
173      */
174     public static void invalidateChannel(@Nullable ManagedChannel channel, ObjectPool<ManagedChannel> pool) {
175         if (channel != null) {
176             try {
177                 pool.invalidateObject(channel);
178             } catch (Exception e) {
179                 logger.error("Unable to invalidate existing grpc connection - check for possible resource leaks: {}", e.getMessage());
180                 logger.debug("Stack trace: ", e);
181             }
182         }
183     }
184 
185     /**
186      * Returns a previously acquired {@link ManagedChannel} to the connection pool. If the return fails, attempts to
187      * invalidate the channel to avoid pool corruption.
188      *
189      * @param channel the channel to return
190      * @param pool the pool to return the channel to
191      */
192     public static void returnChannel(@Nullable ManagedChannel channel, ObjectPool<ManagedChannel> pool) {
193         try {
194             if (channel != null) {
195                 pool.returnObject(channel);
196             }
197         } catch (Exception e) {
198             logger.warn("Unable to cleanly return grpc connection channel to the pool: {}", e.getMessage());
199             logger.debug("Stack trace: ", e);
200             invalidateChannel(channel, pool);
201         }
202     }

204     /**
205      * Creates a new Apache Commons object pool for managing {@link ManagedChannel} instances. If the erosion factor is
206      * positive, wraps the pool with eroding behavior to gradually shrink idle connections.
207      *
208      * @return a new configured connection pool
209      */
210     public ObjectPool<ManagedChannel> newConnectionPool() {
211         if (this.erodingPoolFactor > 0) {
212             return PoolUtils.erodingPool(new GenericObjectPool<>(this, this.poolConfig), this.erodingPoolFactor);
213         }
214         return new GenericObjectPool<>(this, this.poolConfig);
215     }

217     /**
218      * Creates a new {@link ManagedChannel} instance configured with the current factory settings. Called internally by the
219      * object pool during channel instantiation.
220      *
221      * @return a new gRPC channel
222      */
223     @Override
224     public ManagedChannel create() {
225         return ManagedChannelBuilder.forTarget(this.target)
226                 .keepAliveTime(this.keepAliveMillis, TimeUnit.MILLISECONDS)
227                 .keepAliveTimeout(this.keepAliveTimeoutMillis, TimeUnit.MILLISECONDS)
228                 .keepAliveWithoutCalls(this.keepAliveWithoutCalls)
229                 .defaultLoadBalancingPolicy(this.loadBalancingPolicy)
230                 .maxInboundMessageSize(this.maxInboundMessageByteSize)
231                 .maxInboundMetadataSize(this.maxInboundMetadataByteSize)
232                 .usePlaintext().build();
233     }

236      * Wraps a {@link ManagedChannel} for use with the object pool. Provides pooling metadata and lifecycle tracking.
237      *
238      * @param channel the gRPC channel to wrap
239      * @return a pooled object wrapper

241     @Override
242     public PooledObject<ManagedChannel> wrap(ManagedChannel channel) {
243         return new DefaultPooledObject<>(channel);
244     }

246     /**
247      * Called when a {@link ManagedChannel} is returned to the pool. Since gRPC channels are designed to remain ready for
248      * reuse without needing to be reset or cleared, a no-op passivator is fine for most use cases. Custom behavior may
249      * become necessary if using a stub or channel wrapper that requires cleanup between uses.
250      *
251      * @param pooledObject the pooled channel being passivated
252      */
253     @Override
254     public void passivateObject(PooledObject<ManagedChannel> pooledObject) {
255         channelPassivator.accept(pooledObject.getObject());
256     }
257 
258     /**
259      * Validates whether the {@link ManagedChannel} is healthy and can be reused. Called by the pool before returning a
260      * channel to a caller. Implementations should check connection state or channel health as needed. For example, you
261      * might verify that the channel is not shutdown or terminated.
262      *
263      * @param pooledObject the pooled gRPC channel to validate
264      * @return true if the channel is valid and safe to reuse, false otherwise
265      */
266     @Override
267     public boolean validateObject(PooledObject<ManagedChannel> pooledObject) {
268         return channelValidator.test(pooledObject.getObject());
269     }
270 
271     /**
272      * Cleans up a pooled {@link ManagedChannel} when it is removed from the pool. Immediately shuts down the channel to
273      * release resources.
274      *
275      * @param pooledObject the pooled gRPC channel to destroy
276      */
277     @Override
278     public void destroyObject(PooledObject<ManagedChannel> pooledObject) {
279         pooledObject.getObject().shutdownNow();
280     }

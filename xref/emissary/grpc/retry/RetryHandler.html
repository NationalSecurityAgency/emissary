18  /**
19   * Config class for handling retries after failed gRPG connections to a remote server. Uses the
20   * {@link io.github.resilience4j resilience4j} package with exponential backoff.
21   * <p>
22   * Configuration Keys:
23   * <ul>
24   * <li>{@code GRPC_RETRY_INITIAL_WAIT_MILLIS} - The amount of time to wait after the first failure before applying
25   * exponential backoff, default={@code 64}</li>
26   * <li>{@code GRPC_RETRY_MAX_ATTEMPTS} - Maximum number of times to attempt execution, <b>including initial attempt
27   * before retries</b>, default={@code 4}</li>
28   * <li>{@code GRPC_RETRY_MULTIPLIER} - Multiplier used to determine wait-time for successive retries,
29   * default={@code 2}</li>
30   * <li>{@code GRPC_RETRY_NUM_FAILS_BEFORE_WARN} - Determines the number of execution failures before logger should start
31   * sending warnings, default={@code 3}</li>
32   * </ul>
33   * Exponential Backoff Equations:
34   * <ul>
35   * <li>The max wait of the {@code i}th retry is equal to:<br>
36   * {@code GRPC_RETRY_INITIAL_WAIT_MILLIS} * ({@code GRPC_RETRY_MULTIPLIER} ^ ({@code GRPC_RETRY_MAX_ATTEMPTS} -
37   * {@code i}))</li>
38   * <li>The upper bound of total wait time is equal to:<br>
39   * {@code GRPC_RETRY_INITIAL_WAIT_MILLIS} * (({@code GRPC_RETRY_MULTIPLIER} ^ ({@code GRPC_RETRY_MAX_ATTEMPTS} - 1)) -
40   * 1) / ({@code GRPC_RETRY_MULTIPLIER} - 1)</li>
41   * </ul>
42   */
43  public final class RetryHandler {
44      public static final String GRPC_RETRY_INITIAL_WAIT_MILLIS = "GRPC_RETRY_INITIAL_WAIT_MILLIS";
45      public static final String GRPC_RETRY_MAX_ATTEMPTS = "GRPC_RETRY_MAX_ATTEMPTS";
46      public static final String GRPC_RETRY_MULTIPLIER = "GRPC_RETRY_MULTIPLIER";
47      public static final String GRPC_RETRY_NUM_FAILS_BEFORE_WARN = "GRPC_RETRY_NUM_FAILS_BEFORE_WARN";
48  
49      private static final Logger logger = LoggerFactory.getLogger(RetryHandler.class);
50  
51      private final String internalName;
52      private final int maxAttempts;
53      private final int numFailsBeforeWarn;
54      private final Retry retry;
55  
56      /**
57       * Constructs the retry policy for a given gRPC service.
58       * <p>
59       * See {@link RetryHandler} for supported configuration keys and defaults.
60       *
61       * @param configG configuration provider for retry logic
62       * @param retryName unique name to internally identify the retry policy
63       */
64      public RetryHandler(Configurator configG, String retryName) {
65          internalName = retryName;
66          maxAttempts = configG.findIntEntry(GRPC_RETRY_MAX_ATTEMPTS, 4);
67          numFailsBeforeWarn = configG.findIntEntry(GRPC_RETRY_NUM_FAILS_BEFORE_WARN, 3);
68  
69          retry = Retry.of(internalName, RetryConfig.custom()
70                  .maxAttempts(maxAttempts)
71                  .intervalFunction(IntervalFunction.ofExponentialBackoff(
72                          configG.findIntEntry(GRPC_RETRY_INITIAL_WAIT_MILLIS, 64),
73                          configG.findIntEntry(GRPC_RETRY_MULTIPLIER, 2)))
74                  .retryExceptions(PoolException.class, ServiceNotAvailableException.class)
75                  .build());
76  
77          retry.getEventPublisher()
78                  .onRetry(this::logMessageOnRetry)
79                  .onError(this::logMessageOnError);
80      }
81  
82      private void logMessageOnRetry(RetryOnRetryEvent event) {
83          int attemptNumber = event.getNumberOfRetryAttempts();
84          Level level = attemptNumber <= numFailsBeforeWarn ? Level.INFO : Level.WARN;
85          logger.atLevel(level).log("{} failed gRPC connection attempt #{} with event error: {}", internalName, attemptNumber, event);
86      }
87  
88      private void logMessageOnError(RetryOnErrorEvent event) {
89          logger.error("{} failed gRPC connection attempt #{} with event error: {}", internalName, maxAttempts, event);
90      }
91  
92      public <T> T execute(Supplier<T> supplier) {
93          return Retry.decorateSupplier(retry, supplier).get();
94      }
95  }

13  /**
14   * Response object from an identification engine. Contains a list of types for the currentForm and a list of properties
15   * for the parameters
16   */
17  public class Identification implements Serializable {
18      // Serializable
19      static final long serialVersionUID = 9212068103720124108L;
20  
21      protected List<String> types = new ArrayList<>();
22      protected Map<String, String> props = new TreeMap<>();
23  
24      /**
25       * Create a new Identification object
26       */
27      public Identification() {}
28  
29      /**
30       * Create a new identification for the specified type string
31       * 
32       * @param type the type string
33       */
34      public Identification(final String type) {
35          this.types.add(type);
36      }
37  
38      /**
39       * Add a type to the list
40       * 
41       * @param type the new type
42       */
43      public void addType(final String type) {
44          this.types.add(type);
45      }
46  
47      /**
48       * Add new types to the list
49       * 
50       * @param additionalTypes the new types
51       */
52      public void addTypes(final Collection<String> additionalTypes) {
53          this.types.addAll(additionalTypes);
54      }
55  
56      /**
57       * Remove any types and set the type to the value specified
58       * 
59       * @param type the new type value
60       */
61      public void setType(final String type) {
62          this.types.clear();
63          addType(type);
64      }
65  
66      /**
67       * Remove any types and set theh values specified
68       * 
69       * @param theTypes the new type values
70       */
71      public void setTypes(final Collection<String> theTypes) {
72          this.types.clear();
73          addTypes(theTypes);
74      }
75  
76      /**
77       * Get a count of types
78       */
79      public int getTypeCount() {
80          return this.types.size();
81      }
82  
83      /**
84       * Get a list of the types present
85       */
86      public List<String> getTypes() {
87          return new ArrayList<>(this.types);
88      }
89  
90      /**
91       * Get the first type or null if none
92       */
93      @Nullable
94      public String getFirstType() {
95          if (this.types.isEmpty()) {
96              return null;
97          } else {
98              return this.types.get(0);
99          }
100     }
101 
102     /**
103      * Get a comma separated string list of all the types
104      */
105     public String getTypeString() {
106         final StringBuilder b = new StringBuilder();
107         for (final String t : this.types) {
108             if (b.length() > 0) {
109                 b.append(',');
110             }
111             b.append(t);
112         }
113         return b.toString();
114     }
115 
116     /**
117      * Add the key value pair to the properties list
118      */
119     public void addProperty(final String key, final String value) {
120         this.props.put(key, value);
121     }
122 
123     /**
124      * Return a map of all the properties
125      */
126     public Map<String, String> getProperties() {
127         return new HashMap<>(this.props);
128     }
129 
130     /**
131      * Get the value for the specified property
132      * 
133      * @param key the property name
134      */
135     public String getProperty(final String key) {
136         return this.props.get(key);
137     }
138 
139     /**
140      * Get the value of the specified property and remove it from this identification
141      * 
142      * @param key the property name
143      */
144     public String popProperty(final String key) {
145         final String val = this.props.get(key);
146         if (val != null) {
147             this.props.remove(key);
148         }
149         return val;
150     }
151 
152     /**
153      * Remove the first type from the list and return it
154      * 
155      * @return the top top or null if none
156      */
157     @Nullable
158     public String popType() {
159         if (this.types.isEmpty()) {
160             return null;
161         } else {
162             return this.types.remove(0);
163         }
164     }
165 
166     /**
167      * Add the type to the front of the list and return it
168      */
169     public void pushType(String s) {
170         this.types.add(0, s);
171     }
172 
173     /**
174      * String representation of this object
175      */
176     @Override
177     public String toString() {
178         return this.types.toString();
179     }
180 }

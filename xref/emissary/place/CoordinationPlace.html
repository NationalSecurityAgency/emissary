27  /**
28   * This place will coordinate service to several lower level service places. We have a list and will execute each place
29   * in turn. This is good for service orchestration and there is an override point for derived classed to determine
30   * whether to continue processing at each step. It is good for these types of places to register as "STUDY" places so
31   * that current forms of the requested type will come here first instead of going to the individual places for ID, or as
32   * "COORDINATE" if coordinating for TRANSFORM places, or as "VERIFY" if coordinating for IO places (OUTPUT, that is).
33   *
34   * We only coordinate among places in the local Namespace. If the place specified is not initially in the local
35   * namespace we attempt to create it. If it cannot be created it is not used.
36   */
37  public class CoordinationPlace extends ServiceProviderPlace {

95      /**
96       * Set up the place specific information Config items read here are:
97       * 
98       * <ul>
99       * <li>OUTPUT_FORM: default=null, output form for final step of coordination</li>
100      * <li>PUSH_OUTPUT_FORM: default=true, calls pushOutputForm when true, setOutputForm otherwise</li>
101      * <li>SERVICE_COORDINATION: place entries to use for this coordination place, an ordered list of places that must
102      * already be constructed</li>
103      * </ul>
104      */
105     protected void configurePlace() {
106         outputForm = configG.findStringEntry(OUTPUT_FORM, null);
107         pushForm = configG.findBooleanEntry("PUSH_OUTPUT_FORM", true);
108         updateTransformHistory = configG.findBooleanEntry("UPDATE_TRANSFORM_HISTORY", false);
109 
110         placeKeys = configG.findEntries("SERVICE_COORDINATION");
111         logger.debug("We got {} entries to coordinate", placeKeys.size());
112 
113         placeRefs = new ArrayList<>();
114         for (String s : placeKeys) {
115             try {
116                 // See if the place already exists
117                 Object ref = Namespace.lookup(s);
118                 if (ref instanceof IServiceProviderPlace) {
119                     placeRefs.add((IServiceProviderPlace) ref);
120                     logger.debug("Added reference for {}:{}", s, ref);
121                 } else {
122                     logger.error("Referenced place {} is of the wrong type: {}", s, ref.getClass().getName());
123                 }
124             } catch (NamespaceException ex) {
125                 // Try creating the place
126                 try {
127                     String skey = KeyManipulator.getServiceHostUrl(keys.get(0)) + s;
128                     logger.debug("No such place {}, creating as {}", s, skey, ex);
129                     String sclz = PlaceStarter.getClassString(skey);
130                     IServiceProviderPlace p = PlaceStarter.createPlace(skey, null, sclz, dirPlace);
131                     if (p != null) {
132                         placeRefs.add(p);
133                         logger.debug("Place created: {}", p);
134                     } else {
135                         failedCoordPlaceCreation.add(s + " in " + configG.findStringEntry("PLACE_NAME"));
136                         logger.error("Place does not exist and cannot be created: {}", s);
137                     }
138                 } catch (RuntimeException e) {
139                     failedCoordPlaceCreation.add(s + " in " + configG.findStringEntry("PLACE_NAME"));
140                     logger.error("Place does not exist and cannot be created: {}", s, e);
141                 }
142             }
143         }
144     }

146     /**
147      * Evaluate whether to continue processing. Classes can override this method to provide any additional logic during
148      * coordination
149      */
150     protected boolean shouldContinue(IBaseDataObject d, IServiceProviderPlace p) {
151         logger.debug("Continuing with currentForm {} to place {}", d.currentForm(), p);
152         return true;
153     }

155     /**
156      * Return whether to continue traversing the list of coordinated places when an error occurs in one of them
157      *
158      * @param p place that is currently processing the ibdo
159      * @param errorOccurred true if an error occurred
160      *
161      * @return false if processing should not continue
162      */
163     protected boolean shouldContinue(IServiceProviderPlace p, boolean errorOccurred) {
164         if (!continueOnError() && errorOccurred) {
165             logger.info("Error terminating coordination step at {}", p);
166             return false;
167         }
168         return true;
169     }

171     /**
172      * Evaluate whether to skip processing. This will allow the coordination place to continue to the next configured place.
173      * Note that shouldContinue method takes precedence. Use one or the other and be cautious when using both Classes can
174      * override this method to provide any additional logic during coordination
175      */
176     protected boolean shouldSkip(IBaseDataObject d, IServiceProviderPlace p) {
177         logger.debug("Skipping with currentForm {} to place {}", d.currentForm(), p);
178         return false;
179     }

195     private void sproutHook(List<IBaseDataObject> sproutCollection, IBaseDataObject d, boolean hd) {
196         // Allow derived classes a shot at the sprouts
197         if (hd) {
198             sproutHook(sproutCollection, d);
199         }
200     }

202     /**
203      * Consume a data object and coordinate its processing
204      *
205      * @param d the payload to process
206      * @param hd true if doing heavy-duty processing
207      * @return the list of sprouted data objects
208      */
209     protected List<IBaseDataObject> coordinate(IBaseDataObject d, boolean hd) {
210         List<IBaseDataObject> sproutCollection = new ArrayList<>();
211 
212         boolean errorOccurred = false;
213 
214         // Iterate over the configured places
215         for (IServiceProviderPlace p : placeRefs) {
216             // Let derived classed decide to quit or continue this loop
217             if (!shouldContinue(d, p)) {
218                 break;
219             } else if (shouldSkip(d, p)) {
220                 continue;
221             }
222 
223             updateTransformHistory(d, p);
224 
225             // Collect attachments for hd processing
226             List<IBaseDataObject> sprouts = null;
227 
228             // Like an agent would do it
229             try (TimedResource tr = resourceWatcherStart(p)) {
230                 assert tr != null; // to silence an unused resource warning
231 
232                 if (hd) {
233                     // Do the normal HD processing
234                     sprouts = p.agentProcessHeavyDuty(d);
235                 } else {
236                     // Do the normal Non-HD processing
237                     p.agentProcessCall(d);
238                 }
239                 errorOccurred = d.currentForm().equals(Form.ERROR);
240             } catch (Exception ex) {
241                 errorOccurred = handlePlaceException(p, hd, ex);
242             } finally {
243                 if (Thread.interrupted()) {
244                     logger.warn("Place {} was interrupted during execution.", p);
245                 }
246             }
247 
248             if (!shouldContinue(p, errorOccurred)) {
249                 break;
250             }
251 
252             // Track any new attachments
253             if (CollectionUtils.isNotEmpty(sprouts)) {
254                 sproutCollection.addAll(sprouts);
255             }
256         }
257 
258         applyForm(d, errorOccurred);
259 
260         sproutHook(sproutCollection, d, hd);
261 
262         // Allow derived classes a shot to clean up the parent
263         cleanUpHook(d);
264 
265         return sproutCollection;
266     }

268     /**
269      * Allow derived classes a shot to handle a place exception
270      *
271      * @param p place that was processing when the exception was thrown
272      * @param hd true if doing heavy-duty processing
273      * @param ex exception thrown by the place
274      *
275      * @return if an error occurred
276      */
277     protected boolean handlePlaceException(IServiceProviderPlace p, boolean hd, Exception ex) {
278         logger.warn("agentProcess{} called from Coordinate problem", hd ? "HeavyDuty" : "Call", ex);
279         return true;
280     }

282     private void updateTransformHistory(IBaseDataObject d, IServiceProviderPlace p) {
283         if (updateTransformHistory) {
284             DirectoryEntry de = p.getDirectoryEntry();
285             de.setDataType(d.currentForm());
286             // append to the transform history, with flag indicating that the visit was coordinated
287             d.appendTransformHistory(de.getKey(), true);
288         }
289     }

291     /**
292      * How to handle applying the output form if an error occurred during processing
293      *
294      * @param d the ibdo to process
295      * @param errorOccurred true if an error occurred
296      */
297     protected void applyForm(IBaseDataObject d, boolean errorOccurred) {
298         if (!errorOccurred || shouldApplyOutputFormOnError()) {
299             applyOutputForm(d);
300 
301             // Clean up my proxies
302             nukeMyProxies(d);
303         }
304     }

306     /**
307      * If true, process the output form the same for the default and error condition

311     protected boolean shouldApplyOutputFormOnError() {
312         return false;
313     }

315     /**
316      * Apply the output form according to configuration
317      *
318      * @param d the ibdo to process
319      */
320     protected void applyOutputForm(IBaseDataObject d) {
321         if (outputForm != null) {
322             if (pushForm) {
323                 d.pushCurrentForm(outputForm);
324             } else {
325                 if (logger.isDebugEnabled()) {
326                     logger.debug("Object has {} number of forms in the Stack. CoordinationPlace Setting top form from {} to {}", d.currentFormSize(),
327                             d.currentForm(), outputForm);
328                 }
329                 d.setCurrentForm(outputForm);
330             }
331         }
332     }

334     /**
335      * If false, do not continue processing other places after an error occurs
336      *
337      * @return boolean to not continue processing after an error

339     protected boolean continueOnError() {
340         return false;
341     }

343     protected TimedResource resourceWatcherStart(final IServiceProviderPlace place) {
344         TimedResource tr = TimedResource.EMPTY;
345         try {
346             tr = ResourceWatcher.lookup().starting(getAgent(), place);
347         } catch (EmissaryException ex) {
348             logger.debug("No resource monitoring enabled");
349         }
350         return (tr == null) ? TimedResource.EMPTY : tr;
351     }

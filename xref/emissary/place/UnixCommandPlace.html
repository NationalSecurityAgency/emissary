85      /**
86       * Configure the instance
87       * <ul>
88       * <li>SYNCHRONIZED_PROCESS: true if you want to synchronize in java, default is false</li>
89       * <li>NEW_FORM: new current form on success, default UNKNOWN</li>
90       * <li>NEW_FORM_ON_ERROR: new current form on error, default ERROR</li>
91       * <li>ADD_AS_ALTERNATE_VIEW: add output as an alt view using name provided as view name</li>
92       * <li>ADD_AS_META_DATA: add output as metadata value using name provided as key</li>
93       * <li>PERL_CHOP: if true chomp data returned, default false</li>
94       * <li>NUKE_ALL_PROXIES: if true remove all of this places proxies from current form stack when done, default false</li>
95       * <li>OUTPUT_CHARSET: charset of the process output, default 8859_1</li>
96       * <li>KEEP_FILES_DEBUG: when true don't clean up after exec is finished, default false</li>
97       * <li>LOG_FILE_NAME: name of output file to translate into logger commands, default: [servicename].log from key</li>
98       * </ul>
99       * Also all of the config values read by emissary.util.shell.Executrix are needed here
100      */
101     protected void configurePlace() {
102         doSynchronized = configG.findBooleanEntry("SYNCHRONIZED_PROCESS", false);
103         newForm = configG.findStringEntry(NEW_FORM, Form.UNKNOWN);
104         if (newForm == null && keys.get(0).indexOf(".ID.") > -1) {
105             newForm = Form.UNKNOWN;
106         }
107         if ("<null>".equals(newForm)) {
108             newForm = null;
109         }
110         newFormOnError = configG.findStringEntry("NEW_FORM_ON_ERROR", Form.ERROR);
111         alternateView = configG.findStringEntry("ADD_AS_ALTERNATE_VIEW", null);
112         metaDataTag = configG.findStringEntry("ADD_AS_META_DATA", null);
113         if (metaDataTag != null) {
114             addAsMetaData = true;
115         }
116         perlChop = configG.findBooleanEntry("PERL_CHOP", false);
117         nukeAllProxies = configG.findBooleanEntry("NUKE_ALL_PROXIES", false);
118         keepFilesDebug = configG.findBooleanEntry("KEEP_FILES_DEBUG", false);
119         charset = configG.findStringEntry("OUTPUT_CHARSET", charset);
120         executrix = new Executrix(configG);
121         logfilename = configG.findStringEntry("LOG_FILE_NAME", KeyManipulator.getServiceName(keys.get(0)) + ".log");
122         logger.debug("Configured {} type process with charset {}", executrix.getOutput(), charset);
123     }

157     /**
158      * Log the messages found in the log file
159      * 
160      * @param tempDir the directory where the command executed
161      */
162     protected void logMessages(String tempDir) {
163         // if there is a log file, read it and log the messages
164         try {
165             String lfn = tempDir + "/" + logfilename;
166             byte[] logdata = Executrix.readDataFromFile(lfn, true);
167             if (logdata != null) {
168                 for (String message : new String(logdata, charset).split("\n")) {
169                     logger.info(message);
170                 }
171             }
172         } catch (Exception ignore) {
173             logger.debug("Error logging messages", ignore);
174         }
175     }

177     /**
178      * Run the file process
179      */
180     @Nullable
181     @SuppressWarnings("AvoidObjectArrays")
182     public byte[] fileProcess(String[] cmd, String outputFile) {
183         logger.debug("fileProcess({})", Arrays.asList(cmd));
184         StringBuilder errbuf = new StringBuilder();
185         int result = executrix.execute(cmd, (StringBuilder) null, errbuf);
186         if (result != 0) {
187             logger.warn("exec error in fileProcess: {} produced STDERR {}", Arrays.asList(cmd), errbuf.toString());
188             return null;
189         }
190         return Executrix.readDataFromFile(outputFile);
191     }

193     /**
194      * Run the stdout process
195      * 
196      * @param cmd command with arguments
197      * @param chop if true chomp CRLF from output
198      * @return bytes of output from command execution
199      */
200     @Nullable
201     @SuppressWarnings("AvoidObjectArrays")
202     public byte[] stdOutProcess(String[] cmd, boolean chop) {
203         logger.debug("stdOutProcess({},{}) with charset {}", Arrays.asList(cmd), chop, charset);
204         StringBuilder outbuf = new StringBuilder();
205         StringBuilder errbuf = new StringBuilder();
206         int result = executrix.execute(cmd, outbuf, errbuf, charset);
207         if (result != 0) {
208             logger.warn("exec error in stdOutProcess: {} produced STDERR {}", Arrays.asList(cmd), errbuf.toString());
209             return null;
210         }
211         if (chop) {
212             while (outbuf.length() > 0) {
213                 char c = outbuf.charAt(outbuf.length() - 1);
214                 if (c == '\n' || c == '\r') {
215                     outbuf.setLength(outbuf.length() - 1);
216                 } else {
217                     break;
218                 }
219             }
220         }
221 
222         try {
223             return outbuf.toString().getBytes(charset);
224         } catch (UnsupportedEncodingException e) {
225             logger.error("UnixCommandPlace.stdOutProcess charset problem", e);
226             return outbuf.toString().getBytes();
227         }
228     }

275     /**
276      * Helper routine to run command on data
277      * 
278      * @param data the bytes to run the command on
279      * @return byte array of output
280      */
281     @SuppressWarnings("CatchingUnchecked")
282     protected byte[] runCommandOn(byte[] data) throws ResourceException {
283         TempFileNames names = executrix.createTempFilenames();
284         String tempDirName = names.getTempDir();
285         String inputFileName = names.getInputFilename();
286         String outputFileName = names.getOutputFilename();
287         File tempDir = new File(tempDirName);
288         byte[] outputData = null;
289 
290         try {
291             if (!tempDir.mkdirs()) {
292                 logger.warn("Could not create temp directory for process {}", tempDirName);
293                 return outputData;
294             }
295 
296             boolean written = Executrix.writeDataToFile(data, inputFileName, true);
297 
298             if (written) {
299                 String[] cmd = executrix.getCommand(names);
300 
301                 if (executrix.getOutput().equals("FILE")) {
302                     outputData = fileProcess(cmd, outputFileName);
303                 } else if (executrix.getOutput().equals("STD")) {
304                     outputData = stdOutProcess(cmd, perlChop);
305                 } else {
306                     logger.error("No output type specified");
307                 }
308 
309                 logMessages(tempDirName);
310             }
311 
312         } catch (Exception ex) {
313             logger.warn("Bad execution of commands", ex);
314             if (ex instanceof InterruptedException) {
315                 throw new ResourceException(ex); // framework notification to stop
316             }
317         } finally {
318             if (!keepFilesDebug) {
319                 // delete all files here!!!
320                 Executrix.cleanupDirectory(tempDir);
321             }
322         }
323 
324         return outputData;
325 
326     }

397     /**
398      * Run the command and process the results
399      * 
400      * @param tData the data object to process
401      */
402     protected void processData(IBaseDataObject tData) throws ResourceException {
403 
404         byte[] outputData = runCommandOn(findPreferredData(tData));
405         String serviceType = KeyManipulator.getServiceType(keys.get(0));
406 
407         if (serviceType.equals("ID") || serviceType.equals("ANALYZE")) {
408             if (outputData == null || outputData.length == 0) {
409                 errorHook(tData);
410             } else if (addAsMetaData) {
411                 asMetaDataHook(tData, metaDataTag, outputData);
412             } else if (alternateView != null) {
413                 asAlternateViewHook(tData, alternateView, outputData);
414             } else {
415                 asCurrentFormHook(tData, outputData);
416             }
417         } else if (serviceType.equals("TRANSFORM")) {
418             if (outputData == null || outputData.length == 0) {
419                 errorHook(tData);
420             } else if (alternateView != null) {
421                 asAlternateViewHook(tData, alternateView, outputData);
422             } else if (addAsMetaData) {
423                 asMetaDataHook(tData, metaDataTag, outputData);
424             } else {
425                 asDataHook(tData, outputData);
426                 if (nukeAllProxies) {
427                     nukeMyProxies(tData);
428                     if (newForm != null) {
429                         tData.pushCurrentForm(newForm);
430                     }
431                 } else if (newForm != null) {
432                     tData.setCurrentForm(newForm);
433                 }
434             }
435         } else {
436             serviceHook(serviceType, tData, outputData);
437         }
438 
439     }

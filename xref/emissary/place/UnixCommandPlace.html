80      /**
81       * Configure the instance
82       * <ul>
83       * <li>SYNCHRONIZED_PROCESS: true if you want to synchronize in java, default is false</li>
84       * <li>NEW_FORM: new current form on success, default UNKNOWN</li>
85       * <li>NEW_FORM_ON_ERROR: new current form on error, default ERROR</li>
86       * <li>ADD_AS_ALTERNATE_VIEW: add output as an alt view using name provided as view name</li>
87       * <li>ADD_AS_META_DATA: add output as metadata value using name provided as key</li>
88       * <li>PERL_CHOP: if true chomp data returned, default false</li>
89       * <li>NUKE_ALL_PROXIES: if true remove all of this places proxies from current form stack when done, default false</li>
90       * <li>OUTPUT_CHARSET: charset of the process output, default 8859_1</li>
91       * <li>KEEP_FILES_DEBUG: when true don't clean up after exec is finished, default false</li>
92       * <li>LOG_FILE_NAME: name of output file to translate into logger commands, default: [servicename].log from key</li>
93       * </ul>
94       * Also all of the config values read by emissary.util.shell.Executrix are needed here
95       */
96      protected void configurePlace() {
97          doSynchronized = configG.findBooleanEntry("SYNCHRONIZED_PROCESS", false);
98          newForm = configG.findStringEntry(NEW_FORM, Form.UNKNOWN);
99          if (newForm == null && keys.get(0).indexOf(".ID.") > -1) {
100             newForm = Form.UNKNOWN;
101         }
102         if ("<null>".equals(newForm)) {
103             newForm = null;
104         }
105         newFormOnError = configG.findStringEntry("NEW_FORM_ON_ERROR", Form.ERROR);
106         alternateView = configG.findStringEntry("ADD_AS_ALTERNATE_VIEW", null);
107         metaDataTag = configG.findStringEntry("ADD_AS_META_DATA", null);
108         if (metaDataTag != null) {
109             addAsMetaData = true;
110         }
111         perlChop = configG.findBooleanEntry("PERL_CHOP", false);
112         NUKE_ALL_PROXIES = configG.findBooleanEntry("NUKE_ALL_PROXIES", false);
113         keepFilesDebug = configG.findBooleanEntry("KEEP_FILES_DEBUG", false);
114         charset = configG.findStringEntry("OUTPUT_CHARSET", charset);
115         executrix = new Executrix(configG);
116         logfilename = configG.findStringEntry("LOG_FILE_NAME", KeyManipulator.getServiceName(keys.get(0)) + ".log");
117         logger.debug("Configured {} type process with charset {}", executrix.getOutput(), charset);
118     }

152     /**
153      * Log the messages found in the log file
154      * 
155      * @param tempDir the directory where the command executed
156      */
157     protected void logMessages(String tempDir) {
158         // if there is a log file, read it and log the messages
159         try {
160             String lfn = tempDir + "/" + logfilename;
161             byte[] logdata = Executrix.readDataFromFile(lfn, true);
162             if (logdata != null) {
163                 for (String message : new String(logdata, charset).split("\n")) {
164                     logger.info(message);
165                 }
166             }
167         } catch (Exception ignore) {
168             logger.debug("Error logging messages", ignore);
169         }
170     }

172     /**
173      * Run the file process
174      */
175     public byte[] fileProcess(String[] cmd, String outputFile) {
176         logger.debug("fileProcess({})", Arrays.asList(cmd));
177         StringBuilder errbuf = new StringBuilder();
178         int result = executrix.execute(cmd, (StringBuilder) null, errbuf);
179         if (result != 0) {
180             logger.warn("exec error in fileProcess: {} produced STDERR {}", Arrays.asList(cmd), errbuf.toString());
181             return null;
182         }
183         return Executrix.readDataFromFile(outputFile);
184     }

186     /**
187      * Run the stdout process
188      * 
189      * @param cmd command with arguments
190      * @param chop if true chomp CRLF from output
191      * @return bytes of output from command execution
192      */
193     public byte[] stdOutProcess(String[] cmd, boolean chop) {
194         logger.debug("stdOutProcess({},{}) with charset {}", Arrays.asList(cmd), chop, charset);
195         StringBuilder outbuf = new StringBuilder();
196         StringBuilder errbuf = new StringBuilder();
197         int result = executrix.execute(cmd, outbuf, errbuf, charset);
198         if (result != 0) {
199             logger.warn("exec error in stdOutProcess: {} produced STDERR {}", Arrays.asList(cmd), errbuf.toString());
200             return null;
201         }
202         if (chop) {
203             while (outbuf.length() > 0) {
204                 char c = outbuf.charAt(outbuf.length() - 1);
205                 if (c == '\n' || c == '\r') {
206                     outbuf.setLength(outbuf.length() - 1);
207                 } else {
208                     break;
209                 }
210             }
211         }
212 
213         try {
214             return (outbuf.toString().getBytes(charset));
215         } catch (UnsupportedEncodingException e) {
216             logger.error("UnixCommandPlace.stdOutProcess charset problem", e);
217             return (outbuf.toString().getBytes());
218         }
219     }

266     /**
267      * Helper routine to run command on data
268      * 
269      * @param data the bytes to run the command on
270      * @return byte array of output
271      */
272     protected byte[] runCommandOn(byte[] data) throws ResourceException {
273         String[] names = executrix.makeTempFilenames();
274         String tempDirName = names[Executrix.DIR];
275         String inputFileName = names[Executrix.INPATH];
276         String outputFileName = names[Executrix.OUTPATH];
277         File tempDir = new File(tempDirName);
278         byte[] outputData = null;
279 
280         try {
281             if (!tempDir.mkdirs()) {
282                 logger.warn("Could not create temp directory for process {}", tempDirName);
283                 return outputData;
284             }
285 
286             boolean written = Executrix.writeDataToFile(data, inputFileName, true);
287 
288             if (written) {
289                 String[] cmd = executrix.getCommand(names);
290 
291                 if (executrix.getOutput().equals("FILE")) {
292                     outputData = fileProcess(cmd, outputFileName);
293                 } else if (executrix.getOutput().equals("STD")) {
294                     outputData = stdOutProcess(cmd, perlChop);
295                 } else {
296                     logger.error("No output type specified");
297                 }
298 
299                 logMessages(tempDirName);
300             }
301 
302         } catch (Exception ex) {
303             logger.warn("Bad execution of commands", ex);
304             if (ex instanceof InterruptedException) {
305                 throw new ResourceException(ex); // framework notification to stop
306             }
307         } finally {
308             if (!keepFilesDebug) {
309                 // delete all files here!!!
310                 Executrix.cleanupDirectory(tempDir);
311             }
312         }
313 
314         return outputData;
315 
316     }

387     /**
388      * Run the command and process the results
389      * 
390      * @param tData the data object to process
391      */
392     protected void processData(IBaseDataObject tData) throws ResourceException {
393 
394         byte[] outputData = runCommandOn(tData.data());
395         String serviceType = KeyManipulator.getServiceType(keys.get(0));
396 
397         if (serviceType.equals("ID") || serviceType.equals("ANALYZE")) {
398             if (outputData == null || outputData.length == 0) {
399                 errorHook(tData);
400             } else if (addAsMetaData) {
401                 asMetaDataHook(tData, metaDataTag, outputData);
402             } else if (alternateView != null) {
403                 asAlternateViewHook(tData, alternateView, outputData);
404             } else {
405                 asCurrentFormHook(tData, outputData);
406             }
407         } else if (serviceType.equals("TRANSFORM")) {
408             if (outputData == null || outputData.length == 0) {
409                 errorHook(tData);
410             } else if (alternateView != null) {
411                 asAlternateViewHook(tData, alternateView, outputData);
412             } else if (addAsMetaData) {
413                 asMetaDataHook(tData, metaDataTag, outputData);
414             } else {
415                 asDataHook(tData, outputData);
416                 if (NUKE_ALL_PROXIES) {
417                     nukeMyProxies(tData);
418                     if (newForm != null) {
419                         tData.pushCurrentForm(newForm);
420                     }
421                 } else if (newForm != null) {
422                     tData.setCurrentForm(newForm);
423                 }
424             }
425         } else {
426             serviceHook(serviceType, tData, outputData);
427         }
428         return;
429     }

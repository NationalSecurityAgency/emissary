206     /**
207      * Load the configurator, figuring out whence automatically
208      */
209     default Configurator loadConfigurator(@Nullable String placeLocation) throws IOException {
210         if (placeLocation == null) {
211             placeLocation = this.getClass().getSimpleName();
212         }
213         setPlaceLocation(placeLocation);
214 
215         // Extract config data stream name from place location
216         // and try finding config info with and without the
217         // package name of this class (in that order)
218         String myPackage = this.getClass().getPackage().getName();
219         List<String> configLocs = new ArrayList<>();
220         // Dont use KeyManipulator for this, only works when hostname/fqdn has dots
221         int pos = placeLocation.lastIndexOf("/");
222         String serviceClass = pos > -1 ? placeLocation.substring(pos + 1) : placeLocation;
223         configLocs.add(myPackage + DOT + serviceClass + ConfigUtil.CONFIG_FILE_ENDING);
224         configLocs.add(serviceClass + ConfigUtil.CONFIG_FILE_ENDING);
225         setConfigLocations(configLocs);
226         return ConfigUtil.getConfigInfo(configLocs);
227     }
228 
229     /**
230      * Reload the {@link Configurator}
231      *
232      * @param configLocations the list of configuration files to load
233      * @throws IOException if there is an issue loading the config
234      */
235     default Configurator loadConfigurator(@Nullable final List<String> configLocations, final String placeLocation) throws IOException {
236         if (CollectionUtils.isNotEmpty(configLocations)) {
237             return ConfigUtil.getConfigInfo(configLocations);
238         }
239         return loadConfigurator(placeLocation);
240     }

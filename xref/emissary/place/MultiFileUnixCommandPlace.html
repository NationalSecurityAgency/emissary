84      /**
85       * Configure local special variables
86       * <ul>
87       * <li>CONTENT_FILE: file name to become new parent</li>
88       * <li>SYNCHRONIZED_PROCESS: decide whether to synchronize, default false</li>
89       * <li>NUKE_MY_PROXIES: remove all place proxies from incoming payload, default true</li>
90       * <li>KEEP_FILES_DEBUG: keep all files resulting from exec, default false</li>
91       * <li>RECURSE_SUBDIRS: when picking up results, default true</li>
92       * <li>AP_BIN_DIR: where to find executables</li>
93       * <li>OUT_DIRS: where output goes, default '.'</li>
94       * <li>BIN_EXTENSIONS: extension of files to skip when picking up results</li>
95       * <li>BIN_FILES: filenames to skip when picking up results</li>
96       * <li>NEW_ERROR_FORM: form for error result, default ERROR</li>
97       * <li>NEW_PARENT_FORM: form on parent for success, default UNKNOWN, can use '&lt;null&gt;'</li>
98       * <li>NEW_CHILD_FORM: new form on extracted data, default UNKNOWN</li>
99       * <li>SINGLE_CHILD_FILETYPE: file type for single child</li>
100      * <li>KEEP_PARENT_HASHES_FOR_SINGLE_CHILD: when single child is promoted determines if original parent hashes are kept
101      * or not, default false</li>
102      * <li>KEEP_PARENT_FILETYPE_FOR_SINGLE_CHILD: when single child is promoted determines if original parent filetype is
103      * kept or not, default false</li>
104      * <li>SET_TITLE_TO_FILENAME: whether to use filename in doc title, defaul true</li>
105      * <li>SERVICE_DISPLAY_NAME: pretty name for sprout message</li>
106      * <li>CUSTOM_FILE_TYPES: special mapping to set type by file extension</li>
107      * <li>LOG_FILE_NAME: name of output file to translate into logger commands, default: [servicename].log from key</li>
108      * <li>OUTPUT_CHARSET: charset of the process output, default UTF-8</li>
109      * <li>SINGLE_OUTPUT_AS_CHILD: If only one output file keep it as a child and do not replace the parent.</li>
110      * <li>PRESERVE_PARENT_DATA: Stops the parent from getting replaced by output data</li>
111      * </ul>
112      */
113     @Override
114     public void configurePlace() {
115         executrix = new Executrix(configG);
116         contentFile = configG.findStringEntry("CONTENT_FILE", null);
117         doSynchronized = configG.findBooleanEntry("SYNCHRONIZED_PROCESS", false);
118         nukeMyProxies = configG.findBooleanEntry("NUKE_MY_PROXIES", true);
119         keepFilesDebug = configG.findBooleanEntry("KEEP_FILES_DEBUG", false);
120         recurseSubDirs = configG.findBooleanEntry("RECURSE_SUBDIRS", true);
121         apBinDir = configG.findStringEntry("AP_BIN_DIR", "");
122         outDirs = configG.findEntries("OUT_DIRS", ".");
123         charset = configG.findStringEntry("OUTPUT_CHARSET", charset);
124 
125         binFileExt = configG.findEntries("BIN_EXTENSIONS");
126         binFiles = configG.findEntries("BIN_FILES");
127 
128         newErrorForm = configG.findStringEntry("NEW_ERROR_FORM", DEFAULT_NEW_ERROR_FORM);
129         newParentForm = configG.findStringEntry("NEW_PARENT_FORM", DEFAULT_NEW_PARENT_FORM);
130         if (newParentForm.equals("<null>")) {
131             newParentForm = null;
132         }
133         newChildForms = configG.findEntries("NEW_CHILD_FORM", DEFAULT_NEW_CHILD_FORM);
134         SINGLE_CHILD_FILETYPE = configG.findStringEntry("SINGLE_CHILD_FILETYPE", SINGLE_CHILD_FILETYPE);
135         KEEP_PARENT_HASHES_FOR_SINGLE_CHILD = configG.findBooleanEntry("KEEP_PARENT_HASHES_FOR_SINGLE_CHILD", KEEP_PARENT_HASHES_FOR_SINGLE_CHILD);
136         KEEP_PARENT_FILETYPE_FOR_SINGLE_CHILD =
137                 configG.findBooleanEntry("KEEP_PARENT_FILETYPE_FOR_SINGLE_CHILD", KEEP_PARENT_FILETYPE_FOR_SINGLE_CHILD);
138 
139         setTitleToFile = configG.findBooleanEntry("SET_TITLE_TO_FILENAME", true);
140         placeDisplayName = configG.findStringEntry("SERVICE_DISPLAY_NAME", placeName);
141         logfilename = configG.findStringEntry("LOG_FILE_NAME", KeyManipulator.getServiceName(keys.get(0)) + ".log");
142         singleOutputAsChild = configG.findBooleanEntry("SINGLE_OUTPUT_AS_CHILD", singleOutputAsChild);
143         preserveParentData = configG.findBooleanEntry("PRESERVE_PARENT_DATA", preserveParentData);
144 
145         for (String name : configG.findEntries("CUSTOM_FILE_TYPES")) {
146             String tmp = configG.findStringEntry(name + "_EXT", null);
147             if (tmp == null) {
148                 logger.warn("Type missing for {}", name);
149                 continue;
150             }
151             fileTypesByExtension.put(tmp, name);
152         }

221     /**
222      * Get list of things that were excreted from the process by recursively walking the directory
223      * 
224      * @param tmpDir the process execution area
225      * @param inputFileName name of input file so it can be skipped
226      * @param outFiles list to which files that are found can be added
227      */
228     protected void getFileList(File tmpDir, String inputFileName, List<File> outFiles) {
229         // Recursive call to walk the subtree of output files/dirs
230         for (int d = 0; d < outDirs.size(); d++) {
231             logger.debug("outDirs[{}]={}", d, outDirs.get(d));
232             File dir;
233             if (outDirs.get(d).equals(".")) {
234                 dir = tmpDir;
235             } else {
236                 dir = new File(tmpDir, outDirs.get(d));
237             }
238             if (!dir.exists()) {
239                 logger.warn("Output directory does not exist:{}", outDirs.get(d));
240                 continue;
241             }
242             File[] files = dir.listFiles();
243             if (files == null) {
244                 continue;
245             }
246             for (File file : files) {
247                 if (!(file.exists() && file.canRead())) {
248                     logger.warn("cannot access child[{}]:{}", outFiles.size(), file.getAbsolutePath());
249                     continue;
250                 }
251                 if (file.isDirectory()) {
252                     if (recurseSubDirs) {
253                         getFileList(file, inputFileName, outFiles);
254                     } else {
255                         logger.debug("skipping directory: {}", file.getPath());
256                     }
257                     continue;
258                 }
259                 String fname = file.getName();
260                 if (binFiles.contains(fname)) {
261                     logger.debug("Ignoring file '{}' because it is in BIN_FILES", fname);
262                     continue;
263                 }
264                 if (fname.equals(getLogFileName())) {
265                     logger.debug("Using file {} as log message source", fname);
266                     logMessages(tmpDir.getPath());
267                     continue;
268                 }
269                 if (fname.contains(".") && binFileExt.contains(fname.substring(fname.lastIndexOf(".") + 1))) {
270                     logger.debug("Ignoring file '{}' because it is in BIN_EXTENSIONS.", fname);
271                     continue;
272                 }
273                 if (contentFile != null && contentFile.equals(fname)) {
274                     logger.debug("Ignoring file '{}' because it is to be the new parent data.", fname);
275                     continue;
276                 }
277                 if (inputFileName.endsWith(fname)) {
278                     logger.debug("Ignoring file '{}' because it is the input file.", fname);
279                     continue;
280                 }
281                 if (file.length() == 0) {
282                     logger.debug("Ignoring file '{}' because it is empty.", fname);
283                     continue;
284                 }
285                 logger.debug("Adding output file '{}' for processing", fname);
286                 outFiles.add(file);
287             }
288         }
289         outFiles.sort(new FileNameComparator());
290     }

292     protected static class FileNameComparator implements Comparator<File> {
293         @Override
294         public int compare(@Nullable File o1, @Nullable File o2) {
295             if (o1 == null) {
296                 return o2 != null ? 1 : 0;
297             }
298             if (o2 == null) {
299                 return -1;
300             }
301             return o1.getName().compareTo(o2.getName());
302         }

305     /**
306      * Process the result files, turning them into attachments Override hooks: initSprout, preSprout,
307      * postSprout,finishSprout
308      * 
309      * @param parent the original payload
310      * @param files the result files
311      * @param tempDirName execution area name
312      * @param newData data for parent accumulator
313      * @return list of attachments
314      */
315     protected List<IBaseDataObject> sproutResults(IBaseDataObject parent, List<File> files, String tempDirName, StringBuilder newData) {
316         List<IBaseDataObject> sprouts = new ArrayList<>();
317 
318         if (files.isEmpty()) {
319             logger.warn("NO OUTPUT FILES FOUND!");
320             return sprouts;
321         }
322 
323         int fileCount = files.size();
324         int actualFileCount = 0;
325         int birthOrder = parent.getNumChildren() + 1;
326 
327         parent.setNumChildren(fileCount);
328         initSprout(parent, files, newData, tempDirName);
329 
330         for (File f : files) {
331             logger.debug("Handling data file {}", f.getName());
332 
333             if (!f.canRead() || !f.isFile()) {
334                 logger.debug("Cannot read from {}", f.getAbsolutePath());
335                 continue;
336             }
337 
338             byte[] theData = Executrix.readDataFromFile(f.getAbsolutePath());
339             if (theData == null) {
340                 logger.debug("Cannot read data from {}", f.getAbsolutePath());
341                 continue;
342             }
343 
344             if (!preSprout(theData, parent, f, birthOrder, fileCount, newData)) {
345                 continue;
346             }
347 
348             Map<String, Object> metaData = new HashMap<>();
349             if (setTitleToFile) {
350                 metaData.put(DOCUMENT_TITLE, f.getName());
351             }
352 
353             List<String> tmpForms = getFormsFromFile(f);
354 
355             IBaseDataObject dObj = DataObjectFactory.getInstance(theData, parent.getFilename() + Family.SEP + birthOrder, tmpForms.get(0));
356 
357             dObj.putParameters(metaData);
358             sprouts.add(dObj);
359 
360             actualFileCount++;
361             birthOrder++;
362             postSprout(theData, parent, f, birthOrder, fileCount, actualFileCount, newData, dObj);
363         }
364 
365         finishSprout(parent, fileCount, actualFileCount, newData);
366 
367         if (!preserveParentData) {
368             try {
369                 parent.setData(newData.toString().getBytes(charset));
370             } catch (UnsupportedEncodingException e) {
371                 logger.debug("SproutResults charset problem", e);
372                 parent.setData(newData.toString().getBytes());
373             }
374         }
375         return sprouts;
376     }

379     /**
380      * Determines the initial forms for a new data object based on the configuration settings and the file name. This method
381      * may be customized in sub-classes, but at least one form value must be returned!
382      */
383     protected List<String> getFormsFromFile(File f) {
384         List<String> tmpForms = new ArrayList<>();
385 
386         for (Map.Entry<String, String> fileTypes : fileTypesByExtension.entrySet()) {
387             if (f.getName().endsWith(fileTypes.getKey())) {
388                 tmpForms.add(fileTypes.getValue());
389             }
390         }
391         if (tmpForms.isEmpty()) {
392             tmpForms.addAll(newChildForms);
393         }
394         return tmpForms;
395     }

521     /**
522      * Process an incoming payload returning a list of attachments
523      * 
524      * @param tData the payload to process
525      */
526     @Override
527     public List<IBaseDataObject> processHeavyDuty(IBaseDataObject tData) throws ResourceException {
528         List<IBaseDataObject> entries;
529 
530         if (doSynchronized) {
531             entries = synchronizedProcess(tData);
532         } else {
533             entries = unSynchronizedProcess(tData);
534         }
535 
536         if (entries == null || entries.isEmpty()) {
537             logger.debug("no messages found in file.");
538             return Collections.emptyList();
539         }
540 
541         addParentInformation(tData, entries);
542 
543         // Replace parent if single child and singleOutputAsChild is false
544         if (executrix.getOutput().equals("FILE") && entries.size() == 1 && contentFile == null && !singleOutputAsChild) {
545             IBaseDataObject d = entries.get(0);
546             tData.setData(d.data());
547             if (KEEP_PARENT_HASHES_FOR_SINGLE_CHILD) {
548                 KffDataObjectHandler.removeHash(d);
549             }
550             tData.putUniqueParameters(d.getParameters());
551             tData.setCurrentForm(d.currentForm());
552             if (!KEEP_PARENT_FILETYPE_FOR_SINGLE_CHILD) {
553                 tData.setFileType(d.getFileType());
554             }
555             return Collections.emptyList(); // so we just continue with current
556         }
557 
558         // Set the parent form after the addParentInformation call
559         if (newParentForm != null) {
560             logger.debug("Pushing new parent form {}", newParentForm);
561             tData.pushCurrentForm(newParentForm);
562         }
563 
564         return entries;
565     }

579     /**
580      * Process an incoming payload returning attachments, using only some of the data This entry point is shared among all
581      * synchronized, unsynchronized, normal and heavy-duty processing entry points.
582      * 
583      * @param tData the payload to process
584      * @param start offset in data to start
585      * @param len length of data to use
586      * @return attachments
587      */
588     protected List<IBaseDataObject> processData(@Nullable IBaseDataObject tData, int start, int len) throws ResourceException {
589         List<IBaseDataObject> sprouts = new ArrayList<>();
590 
591         // Validate parameters
592         if (tData == null) {
593             logger.debug("Received null data object!");
594             return sprouts;
595         }
596 
597         // Validate data
598         if (tData.data() == null) {
599             logger.debug("Received null data: {}", tData);
600             tData.addProcessingError("NULL data in " + placeName + ".process");
601             tData.pushCurrentForm(newErrorForm);
602             return sprouts;
603         }
604 
605         // Validate start and len
606         if (start < 0 || len <= 0 || (start + len) > tData.dataLength()) {
607             logger.debug("Invalid start/len for data {}/{}", start, len);
608             tData.addProcessingError("Invalid data " + start + "/" + len);
609             tData.pushCurrentForm(newErrorForm);
610             return sprouts;
611         }
612 
613         // make the directory and write the input file.
614         String[] names;
615         File f = null;
616         int result = -1;
617         try {
618             names = executrix.writeDataToNewTempDir(tData.data(), start, len);
619             f = new File(names[Executrix.INPATH]);
620             logger.debug("Wrote file out to {}", f.getPath());
621 
622             // Create the command string and run it
623             String[] cmd = executrix.getCommand(names);
624             StringBuilder parentData = new StringBuilder();
625 
626             logger.debug("Generated command {}", Arrays.asList(cmd));
627 
628             if (executrix.getOutput().equals("FILE")) {
629                 result = processCommand(cmd);
630             } else if (executrix.getOutput().equals("STD")) {
631                 StringBuilder errbuf = new StringBuilder();
632                 result = processCommand(cmd, parentData, errbuf);
633                 if (errbuf.length() > 0) {
634                     tData.addProcessingError(errbuf.toString());
635                 }
636             }
637 
638             // Clean out any proxies that would cause an infinite loop.
639             // we will add new forms later.
640             if (nukeMyProxies) {
641                 nukeMyProxies(tData);
642             }
643             logger.debug("Parent forms {}, nuke={}", tData.getAllCurrentForms(), nukeMyProxies);
644 
645             // Generate the list of resulting files in the directory(s)
646             List<File> files = getFileList(f.getParentFile(), f.getName());
647 
648             if (files != null && !files.isEmpty()) {
649                 sprouts = sproutResults(tData, files, f.getParent(), parentData);
650             }
651         } catch (Exception ex) {
652             logger.error("Problem in command execution", ex);
653             if (ex instanceof InterruptedException) {
654                 throw new ResourceException(ex); // framework notification to stop
655             }
656         } finally {
657             // Delete the temporary directory and all of its contents.
658             if (f != null) {
659                 cleanupFiles(f.getParentFile());
660             }
661         }
662 
663         // If there was not result, then report it in 2 places.
664         if (sprouts.isEmpty()) {
665             logger.debug("Command failed. nothing to sprout for file: result={}", result);
666             tData.addProcessingError("ERROR in " + placeName + ". Exec returned errno " + result);
667             tData.pushCurrentForm(newErrorForm);
668         }
669         return sprouts;
670     }

93      /**
94       * Configure local special variables
95       * <ul>
96       * <li>CONTENT_FILE: file name to become new parent</li>
97       * <li>SYNCHRONIZED_PROCESS: decide whether to synchronize, default false</li>
98       * <li>NUKE_MY_PROXIES: remove all place proxies from incoming payload, default true</li>
99       * <li>KEEP_FILES_DEBUG: keep all files resulting from exec, default false</li>
100      * <li>RECURSE_SUBDIRS: when picking up results, default true</li>
101      * <li>AP_BIN_DIR: where to find executables</li>
102      * <li>OUT_DIRS: where output goes, default '.'</li>
103      * <li>BIN_EXTENSIONS: extension of files to skip when picking up results</li>
104      * <li>BIN_FILES: filenames to skip when picking up results</li>
105      * <li>NEW_ERROR_FORM: form for error result, default ERROR</li>
106      * <li>NEW_PARENT_FORM: form on parent for success, default UNKNOWN, can use '&lt;null&gt;'</li>
107      * <li>NEW_CHILD_FORM: new form on extracted data, default UNKNOWN</li>
108      * <li>SINGLE_CHILD_FILETYPE: file type for single child</li>
109      * <li>KEEP_PARENT_HASHES_FOR_SINGLE_CHILD: when single child is promoted determines if original parent hashes are kept
110      * or not, default false</li>
111      * <li>KEEP_PARENT_FILETYPE_FOR_SINGLE_CHILD: when single child is promoted determines if original parent filetype is
112      * kept or not, default false</li>
113      * <li>SET_TITLE_TO_FILENAME: whether to use filename in doc title, defaul true</li>
114      * <li>SERVICE_DISPLAY_NAME: pretty name for sprout message</li>
115      * <li>CUSTOM_FILE_TYPES: special mapping to set type by file extension</li>
116      * <li>LOG_FILE_NAME: name of output file to translate into logger commands, default: [servicename].log from key</li>
117      * <li>OUTPUT_CHARSET: charset of the process output, default UTF-8</li>
118      * <li>SINGLE_OUTPUT_AS_CHILD: If only one output file keep it as a child and do not replace the parent.</li>
119      * <li>PRESERVE_PARENT_DATA: Stops the parent from getting replaced by output data</li>
120      * <li>IGNORE_EMPTY_FILE: gives the option to save 0 length files, default true</li>
121      * </ul>
122      */

126         contentFile = configG.findStringEntry("CONTENT_FILE", null);
127         doSynchronized = configG.findBooleanEntry("SYNCHRONIZED_PROCESS", false);
128         nukeMyProxies = configG.findBooleanEntry("NUKE_MY_PROXIES", true);
129         keepFilesDebug = configG.findBooleanEntry("KEEP_FILES_DEBUG", false);
130         recurseSubDirs = configG.findBooleanEntry("RECURSE_SUBDIRS", true);
131         apBinDir = configG.findStringEntry("AP_BIN_DIR", "");
132         outDirs = configG.findEntries("OUT_DIRS", ".");
133         charset = configG.findStringEntry("OUTPUT_CHARSET", charset);
134 
135         binFileExt = configG.findEntries("BIN_EXTENSIONS");
136         binFiles = configG.findEntries("BIN_FILES");
137 
138         newErrorForm = configG.findStringEntry("NEW_ERROR_FORM", DEFAULT_NEW_ERROR_FORM);
139         newParentForm = configG.findStringEntry("NEW_PARENT_FORM", DEFAULT_NEW_PARENT_FORM);
140         if (newParentForm.equals("<null>")) {
141             newParentForm = null;
142         }
143         newChildForms = configG.findEntries("NEW_CHILD_FORM", DEFAULT_NEW_CHILD_FORM);
144         singleChildFiletype = configG.findStringEntry("SINGLE_CHILD_FILETYPE", Form.UNKNOWN);
145         keepParentHashesForSingleChild = configG.findBooleanEntry("KEEP_PARENT_HASHES_FOR_SINGLE_CHILD", false);
146         keepParentFiletypeForSingleChild = configG.findBooleanEntry("KEEP_PARENT_FILETYPE_FOR_SINGLE_CHILD", false);
147 
148         setTitleToFile = configG.findBooleanEntry("SET_TITLE_TO_FILENAME", true);
149         placeDisplayName = configG.findStringEntry("SERVICE_DISPLAY_NAME", placeName);
150         logfilename = configG.findStringEntry("LOG_FILE_NAME", KeyManipulator.getServiceName(keys.get(0)) + ".log");
151         singleOutputAsChild = configG.findBooleanEntry("SINGLE_OUTPUT_AS_CHILD", singleOutputAsChild);
152         preserveParentData = configG.findBooleanEntry("PRESERVE_PARENT_DATA", preserveParentData);
153         ignoreEmptyFile = configG.findBooleanEntry("IGNORE_EMPTY_FILE", true);
154 
155         for (String name : configG.findEntries("CUSTOM_FILE_TYPES")) {
156             String tmp = configG.findStringEntry(name + "_EXT", null);
157             if (tmp == null) {
158                 logger.warn("Type missing for {}", name);
159                 continue;
160             }
161             fileTypesByExtension.put(tmp, name);
162         }

231     /**
232      * Get list of things that were excreted from the process by recursively walking the directory
233      * 
234      * @param tmpDir the process execution area
235      * @param inputFileName name of input file so it can be skipped
236      * @param outFiles list to which files that are found can be added
237      */
238     protected void getFileList(File tmpDir, String inputFileName, List<File> outFiles) {
239         // Recursive call to walk the subtree of output files/dirs
240         for (int d = 0; d < outDirs.size(); d++) {
241             logger.debug("outDirs[{}]={}", d, outDirs.get(d));
242             File dir;
243             if (outDirs.get(d).equals(".")) {
244                 dir = tmpDir;
245             } else {
246                 dir = new File(tmpDir, outDirs.get(d));
247             }
248             if (!dir.exists()) {
249                 logger.warn("Output directory does not exist:{}", outDirs.get(d));
250                 continue;
251             }
252             File[] files = dir.listFiles();
253             if (files == null) {
254                 continue;
255             }
256             for (File file : files) {
257                 if (!(file.exists() && file.canRead())) {
258                     logger.warn("cannot access child[{}]:{}", outFiles.size(), file.getAbsolutePath());
259                     continue;
260                 }
261                 if (file.isDirectory()) {
262                     if (recurseSubDirs) {
263                         getFileList(file, inputFileName, outFiles);
264                     } else {
265                         logger.debug("skipping directory: {}", file.getPath());
266                     }
267                     continue;
268                 }
269                 String fname = file.getName();
270                 if (binFiles.contains(fname)) {
271                     logger.debug("Ignoring file '{}' because it is in BIN_FILES", fname);
272                     continue;
273                 }
274                 if (fname.equals(getLogFileName())) {
275                     logger.debug("Using file {} as log message source", fname);
276                     logMessages(tmpDir.getPath());
277                     continue;
278                 }
279                 if (fname.contains(".") && binFileExt.contains(fname.substring(fname.lastIndexOf(".") + 1))) {
280                     logger.debug("Ignoring file '{}' because it is in BIN_EXTENSIONS.", fname);
281                     continue;
282                 }
283                 if (contentFile != null && contentFile.equals(fname)) {
284                     logger.debug("Ignoring file '{}' because it is to be the new parent data.", fname);
285                     continue;
286                 }
287                 if (inputFileName.endsWith(fname)) {
288                     logger.debug("Ignoring file '{}' because it is the input file.", fname);
289                     continue;
290                 }
291                 if (ignoreEmptyFile && file.length() == 0) {
292                     logger.debug("Ignoring file '{}' because it is empty.", fname);
293                     continue;
294                 }
295                 logger.debug("Adding output file '{}' for processing", fname);
296                 outFiles.add(file);
297             }
298         }
299         outFiles.sort(new FileNameComparator());
300     }

302     protected static class FileNameComparator implements Comparator<File> {
303         @Override
304         public int compare(@Nullable File o1, @Nullable File o2) {
305             if (o1 == null) {
306                 return o2 != null ? 1 : 0;
307             }
308             if (o2 == null) {
309                 return -1;
310             }
311             return o1.getName().compareTo(o2.getName());
312         }

315     /**
316      * Process the result files, turning them into attachments Override hooks: initSprout, preSprout,
317      * postSprout,finishSprout
318      * 
319      * @param parent the original payload
320      * @param files the result files
321      * @param tempDirName execution area name
322      * @param newData data for parent accumulator
323      * @return list of attachments
324      */
325     protected List<IBaseDataObject> sproutResults(IBaseDataObject parent, List<File> files, String tempDirName, StringBuilder newData) {
326         List<IBaseDataObject> sprouts = new ArrayList<>();
327 
328         if (files.isEmpty()) {
329             logger.warn("NO OUTPUT FILES FOUND!");
330             return sprouts;
331         }
332 
333         int fileCount = files.size();
334         int actualFileCount = 0;
335         int birthOrder = parent.getNumChildren() + 1;
336 
337         parent.setNumChildren(fileCount);
338         initSprout(parent, files, newData, tempDirName);
339 
340         for (File f : files) {
341             logger.debug("Handling data file {}", f.getName());
342 
343             if (!f.canRead() || !f.isFile()) {
344                 logger.debug("Cannot read from {}", f.getAbsolutePath());
345                 continue;
346             }
347 
348             byte[] theData = Executrix.readDataFromFile(f.getAbsolutePath());
349             if (theData == null) {
350                 logger.debug("Cannot read data from {}", f.getAbsolutePath());
351                 continue;
352             }
353 
354             if (!preSprout(theData, parent, f, birthOrder, fileCount, newData)) {
355                 continue;
356             }
357 
358             Map<String, Object> metaData = new HashMap<>();
359             if (setTitleToFile) {
360                 metaData.put(DOCUMENT_TITLE, f.getName());
361             }
362 
363             List<String> tmpForms = getFormsFromFile(f);
364 
365             IBaseDataObject dObj = DataObjectFactory.getInstance(theData, parent.getFilename() + Family.SEP + birthOrder, tmpForms.get(0));
366 
367             dObj.putParameters(metaData);
368             sprouts.add(dObj);
369 
370             actualFileCount++;
371             birthOrder++;
372             postSprout(theData, parent, f, birthOrder, fileCount, actualFileCount, newData, dObj);
373         }
374 
375         finishSprout(parent, fileCount, actualFileCount, newData);
376 
377         if (!preserveParentData) {
378             try {
379                 parent.setData(newData.toString().getBytes(charset));
380             } catch (UnsupportedEncodingException e) {
381                 logger.debug("SproutResults charset problem", e);
382                 parent.setData(newData.toString().getBytes());
383             }
384         }
385         return sprouts;
386     }

389     /**
390      * Determines the initial forms for a new data object based on the configuration settings and the file name. This method
391      * may be customized in sub-classes, but at least one form value must be returned!
392      */
393     protected List<String> getFormsFromFile(File f) {
394         List<String> tmpForms = new ArrayList<>();
395 
396         for (Map.Entry<String, String> fileTypes : fileTypesByExtension.entrySet()) {
397             if (f.getName().endsWith(fileTypes.getKey())) {
398                 tmpForms.add(fileTypes.getValue());
399             }
400         }
401         if (tmpForms.isEmpty()) {
402             tmpForms.addAll(newChildForms);
403         }
404         return tmpForms;
405     }

546     /**
547      * Process an incoming payload returning a list of attachments
548      * 
549      * @param tData the payload to process
550      */
551     @Override
552     public List<IBaseDataObject> processHeavyDuty(IBaseDataObject tData) throws ResourceException {
553         List<IBaseDataObject> entries;
554 
555         if (doSynchronized) {
556             entries = synchronizedProcess(tData);
557         } else {
558             entries = unSynchronizedProcess(tData);
559         }
560 
561         if (entries == null || entries.isEmpty()) {
562             logger.debug("no messages found in file.");
563             return Collections.emptyList();
564         }
565 
566         addParentInformation(tData, entries);
567 
568         // Replace parent if single child and singleOutputAsChild is false
569         if (executrix.getOutput().equals("FILE") && entries.size() == 1 && contentFile == null && !singleOutputAsChild) {
570             IBaseDataObject d = entries.get(0);
571             tData.setData(d.data());
572             if (keepParentHashesForSingleChild) {
573                 KffDataObjectHandler.removeHash(d);
574             }
575             tData.putUniqueParameters(d.getParameters());
576             tData.setCurrentForm(d.currentForm());
577             if (!keepParentFiletypeForSingleChild) {
578                 tData.setFileType(d.getFileType());
579             }
580             return Collections.emptyList(); // so we just continue with current
581         }
582 
583         // Set the parent form after the addParentInformation call
584         if (newParentForm != null) {
585             logger.debug("Pushing new parent form {}", newParentForm);
586             tData.pushCurrentForm(newParentForm);
587         }
588 
589         return entries;
590     }

604     /**
605      * Process an incoming payload returning attachments, using only some of the data This entry point is shared among all
606      * synchronized, unsynchronized, normal and heavy-duty processing entry points.
607      * 
608      * @param tData the payload to process
609      * @param start offset in data to start
610      * @param len length of data to use
611      * @return attachments
612      */
613     protected List<IBaseDataObject> processData(@Nullable IBaseDataObject tData, int start, int len) throws ResourceException {
614         List<IBaseDataObject> sprouts = new ArrayList<>();
615 
616         // Validate parameters
617         if (tData == null) {
618             logger.debug("Received null data object!");
619             return sprouts;
620         }
621 
622         // Validate data
623         if (tData.data() == null) {
624             logger.debug("Received null data: {}", tData);
625             tData.addProcessingError("NULL data in " + placeName + ".process");
626             tData.pushCurrentForm(newErrorForm);
627             return sprouts;
628         }
629 
630         // Validate start and len
631         if (start < 0 || len <= 0 || (start + len) > tData.dataLength()) {
632             logger.debug("Invalid start/len for data {}/{}", start, len);
633             tData.addProcessingError("Invalid data " + start + "/" + len);
634             tData.pushCurrentForm(newErrorForm);
635             return sprouts;
636         }
637 
638         File f = null;
639         int result = -1;
640         try {
641             // make the directory and write the input file.
642             TempFileNames names = executrix.writeInputDataToNewTempDir(tData.data(), start, len);
643             f = new File(names.getInputFilename());
644             logger.debug("Wrote file out to {}", f.getPath());
645 
646             // Create the command string and run it
647             String[] cmd = executrix.getCommand(names);
648             StringBuilder parentData = new StringBuilder();
649 
650             logger.debug("Generated command {}", Arrays.asList(cmd));
651 
652             if (executrix.getOutput().equals("FILE")) {
653                 result = processCommand(cmd);
654             } else if (executrix.getOutput().equals("STD")) {
655                 StringBuilder errbuf = new StringBuilder();
656                 result = processCommand(cmd, parentData, errbuf);
657                 if (errbuf.length() > 0) {
658                     tData.addProcessingError(errbuf.toString());
659                 }
660             }
661 
662             // Clean out any proxies that would cause an infinite loop.
663             // we will add new forms later.
664             if (nukeMyProxies) {
665                 nukeMyProxies(tData);
666             }
667             logger.debug("Parent forms {}, nuke={}", tData.getAllCurrentForms(), nukeMyProxies);
668 
669             // Generate the list of resulting files in the directory(s)
670             List<File> files = getFileList(f.getParentFile(), f.getName());
671 
672             if (files != null && !files.isEmpty()) {
673                 sprouts = sproutResults(tData, files, f.getParent(), parentData);
674             }
675         } catch (RuntimeException ex) {
676             logger.error("Problem in command execution", ex);
677         } finally {
678             // Delete the temporary directory and all of its contents.
679             if (f != null) {
680                 cleanupFiles(f.getParentFile());
681             }
682         }
683 
684         // If there was no result, then report it in 2 places.
685         if (sprouts.isEmpty()) {
686             logger.debug("Command failed. nothing to sprout for file: result={}", result);
687             tData.addProcessingError("ERROR in " + placeName + ". Exec returned errno " + result);
688             tData.pushCurrentForm(newErrorForm);
689         }
690         return sprouts;
691     }

693     /**
694      * Execute the command and args in the array
695      * 
696      * @param cmd the command and args to execute
697      * @return the process errno status value
698      */
699     protected int processCommand(String[] cmd) {
700         return processCommand(cmd, new StringBuilder(), new StringBuilder());
701     }

703     /**
704      * Execute the command and args in the array
705      * 
706      * @param cmd the command and args to execute
707      * @param stdout builder to append stdout from process
708      * @param stderr builder to append stderr from process
709      * @return the process errno status value
710      */
711     protected int processCommand(String[] cmd, StringBuilder stdout, StringBuilder stderr) {
712         return executrix.execute(cmd, stdout, stderr, charset);
713     }

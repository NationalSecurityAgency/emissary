3   import emissary.core.IBaseDataObject;
4   import emissary.place.ServiceProviderPlace;
5   
6   import java.io.IOException;
7   
8   /**
9    * This place performs no action other than holding everything it processes for a configurable amount of milliseconds
10   */
11  public class DelayPlace extends ServiceProviderPlace {
12  
13      protected long delayTimeMillis = 2000L;
14  
15      /**
16       * Create and register
17       */
18      public DelayPlace(String configInfo, String dir, String placeLoc) throws IOException {
19          super(configInfo, dir, placeLoc);
20          configurePlace();
21      }
22  
23      /**
24       * Create for test
25       */
26      public DelayPlace(String configInfo) throws IOException {
27          super(configInfo, "DelayPlace.www.example.com:8001");
28          configurePlace();
29      }
30  
31      protected void configurePlace() {
32          delayTimeMillis = configG.findLongEntry("DELAY_TIME_MILLIS", delayTimeMillis);
33      }
34  
35      /**
36       * Consume the data object
37       */
38      @Override
39      public void process(IBaseDataObject tData) {
40          if (logger.isDebugEnabled()) {
41              logger.debug("Delay starting {}", tData.getAllCurrentForms());
42          }
43          try {
44              Thread.sleep(delayTimeMillis);
45          } catch (InterruptedException e) {
46              logger.warn("Interrupted before delay expired", e);
47              Thread.currentThread().interrupt();
48          }
49          if (logger.isDebugEnabled()) {
50              logger.debug("Delay ended {}", tData.getAllCurrentForms());
51          }
52      }
53  }

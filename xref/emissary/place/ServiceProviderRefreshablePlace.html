66      /**
67       * Refresh specific constructor, basically clones a place and swaps out the Namespace ref
68       *
69       * @param place the ServiceProviderRefreshablePlace to clone
70       * @param register set true to register with directory place, false otherwise
71       * @throws IOException if there is an issue trying to create the place
72       */
73      public ServiceProviderRefreshablePlace(final ServiceProviderRefreshablePlace place, final boolean register) throws IOException {
74          super(place);
75  
76          if (!register) {
77              // not registering so reuse some stuff
78              this.placeLocation = place.placeLocation;
79              this.serviceDescription = place.serviceDescription;
80              this.keys.addAll(place.keys);
81              this.denyList.addAll(place.denyList);
82          }
83  
84          // reload the config files and get the place updates
85          this.configLocs.addAll(place.configLocs);
86          this.configG = loadConfigurator(configLocs, place.placeLocation);
87          setupPlace(null, place.placeLocation, register);
88      }

90      @Override
91      protected void setupPlace(@Nullable final String theDir, final String placeLocation, final boolean register) throws IOException {
92          super.setupPlace(theDir, placeLocation, register);
93  
94          try {
95              Preconditions.checkNotNull(this.configG, "The configurator is null");
96              final var path = configG.findStringEntry("MONITORING_PATH");
97              if (StringUtils.isNotBlank(path)) {
98                  final var intervalMinutes = configG.findLongEntry("MONITORING_INTERVAL_MINUTES", 15);
99                  this.monitor = new Monitor(this, path, intervalMinutes);
100                 logger.info("Monitoring [{}] for changes every {} minutes", path, intervalMinutes);
101             }
102         } catch (RuntimeException e) {
103             logger.warn("Could not create file monitor, skipping!", e);
104         }
105     }

107     /**
108      * Get the invalid flag of the place. An invalidated place may indicate that the place has changes, such as new
109      * configuration, and may trigger a follow-on process to reconfigure, reinitialize, or re-create the place.
110      *
111      * @return true if the place has been invalidated, false otherwise
112      */
113     public final boolean isInvalidated() {
114         if (!this.invalidated.get() && this.monitor != null) {
115             this.monitor.run();
116         }
117         return this.invalidated.get();
118     }

156     /**
157      * Reinitialize the place by reloading the configurator and reconfiguring the place. Must call {@link #invalidate()}
158      * before attempting to refresh the place.
159      *
160      * @param full true unbind from the namespace and rebuild all directory keys, false to reuse existing keys and just
161      *        refresh place configs
162      * @param silent true to just log any issues that arise, false to throw runtime exceptions
163      */
164     public final void refresh(final boolean full, final boolean silent) {
165         logger.trace("Waiting for lock in refresh()");
166         synchronized (this.allocatorLock) {
167             logger.debug("Attempting to refresh place...");
168             if (!this.defunct.get()) {
169                 if (isInvalidated()) {
170                     if (full) {
171                         unbindFromNamespace();
172                         new ArrayList<>(this.keys).forEach(this::removeKey);
173                     }
174                     Factory.create(this.getClass().getName(), this, full);
175                     this.defunct.set(true);
176                     logger.info("Place refresh performed successfully");
177                 } else {
178                     if (!silent) {
179                         throw new IllegalStateException("Cannot refresh place without first calling invalidate; no refresh performed");
180                     }
181                     logger.warn("Cannot refresh place without first calling invalidate; no refresh performed");
182                 }
183             } else {
184                 if (!silent) {
185                     throw new IllegalStateException("Error refreshing defunct place");
186                 }
187                 logger.warn("Error refreshing defunct place");
188             }
189         }
190     }

192     /**
193      * Similar logic to a {@link org.apache.commons.io.monitor.FileAlterationMonitor} to trigger any registered
194      * {@link FileAlterationObserver} at a specified interval, except that this does not spawn a monitoring thread
195      */
196     class Monitor {
197 
198         private final List<FileAlterationObserver> observers = new CopyOnWriteArrayList<>();
199         private final long intervalMinutes;
200         private Instant lastCheck;
201 
202         protected Monitor(final ServiceProviderRefreshablePlace place, final String path, final long intervalMinutes) throws IOException {
203             Preconditions.checkNotNull(place, "Refreshable place cannot be null");
204             Preconditions.checkArgument(StringUtils.isNotBlank(path), "Path cannot be blank");
205             Preconditions.checkArgument(intervalMinutes > 0, "Monitoring interval is not greater than 0");
206 
207             final Path file = Paths.get(path);
208             Preconditions.checkArgument(Files.exists(file), "Path does not exist");
209 
210             final FileAlterationObserver observer;
211             if (Files.isDirectory(file)) {
212                 logger.debug("Monitoring directory {} for changes", file);
213                 observer = FileAlterationObserver.builder().setFile(file.toFile()).get();
214             } else {
215                 final Path parent = file.getParent();
216                 final String fileName = file.getFileName().toString();
217                 logger.debug("Monitoring file {} in directory {} for changes", fileName, parent);
218                 observer = FileAlterationObserver.builder().setFile(parent.toFile()).setFileFilter(new NameFileFilter(fileName)).get();
219             }
220 
221             try {
222                 final var listener = new RefreshListener(place);
223                 observer.addListener(listener);
224                 observer.initialize();
225                 this.observers.add(observer);
226             } catch (Exception e) {
227                 throw new IOException("Error initializing observer", e);
228             }
229 
230             this.intervalMinutes = intervalMinutes;
231             this.lastCheck = Instant.now();
232         }
233 
234         synchronized void run() {
235             if (Duration.between(this.lastCheck, Instant.now()).toMinutes() >= this.intervalMinutes) {
236                 logger.debug("Last file check was at {}, checking for changed files.", this.lastCheck);
237                 this.observers.forEach(FileAlterationObserver::checkAndNotify);
238                 this.lastCheck = Instant.now();
239             }
240         }

242         class RefreshListener extends FileAlterationListenerAdaptor {
243 
244             private final ServiceProviderRefreshablePlace place;
245             private long lastModified = -1L;
246 
247             RefreshListener(final ServiceProviderRefreshablePlace place) {
248                 this.place = place;
249             }
250 
251             @Override
252             public void onFileChange(final File file) {
253                 logger.trace("File change detected: {} on {}", file, file.lastModified());
254                 handleChangeEvent(file);
255             }
256 
257             @Override
258             public void onFileCreate(final File file) {
259                 logger.trace("File creation detected: {} on {}", file, file.lastModified());
260                 handleChangeEvent(file);
261             }
262 
263             @Override
264             public void onFileDelete(final File file) {
265                 logger.trace("File delete detected: {} on {}", file, file.lastModified());
266                 handleChangeEvent(file);
267             }
268 
269             @Override
270             public void onStop(final FileAlterationObserver observer) {
271                 if (hasChanges()) {
272                     this.place.invalidate("due to a config change");
273                     this.lastModified = -1L;
274                 }
275             }
276 
277             private void handleChangeEvent(final File file) {
278                 logger.debug("Change event observed for {}", file);
279                 this.lastModified = Long.max(this.lastModified, file.lastModified());
280             }
281 
282             private boolean hasChanges() {
283                 logger.trace("LastModified time {}", this.lastModified);
284                 if (this.lastModified > 0L) {
285                     final var lastMod = Instant.ofEpochMilli(this.lastModified);
286 
287                     // make sure no edits have occurred recently
288                     final boolean aboveThreshold = Duration.between(lastMod, Instant.now()).toMinutes() > 2;
289                     if (!aboveThreshold) {
290                         logger.debug("Files have been recently modified, waiting till next cycle.");
291                     }
292                     return aboveThreshold;
293                 }
294                 return false;
295             }
296         }
297     }

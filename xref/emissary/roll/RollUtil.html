3   import emissary.config.Configurator;
4   import emissary.core.Factory;
5   
6   import jakarta.annotation.Nullable;
7   
8   import java.util.Map;
9   import java.util.concurrent.TimeUnit;
10  
11  /**
12   * Util class to grab known parameters from configs to help configure Rollable objects.
13   */
14  public class RollUtil {
15      /**
16       * returns the time period from the current Configurator using the value from ROLLABLE_TIME_PERIOD. Defaults to 10L
17       * 
18       * @return the configured rollable time period
19       */
20      public static long getPeriod(Configurator configG) {
21          return configG.findLongEntry("ROLLABLE_TIME_PERIOD", 10L);
22      }
23  
24      /**
25       * Returns the TimeUnit by extracting the value from ROLLABLE_TIME_PERIOD. The time unit value must be one that exactly
26       * matches those defined in the TimeUnit enum. Defaults to MINUTES.
27       * 
28       * @return the configured rollable TimeUnit
29       */
30      public static TimeUnit getTimeUnit(Configurator configG) {
31          return getUnit(configG.findStringEntry("ROLLABLE_TIME_UNIT", "MINUTES"));
32      }
33  
34      /**
35       * Builds a Roller from a map of options. Specifically, this method looks for the keys<br>
36       * CLASS - the fully qualified Rollable class name to instantiate <br>
37       * TIME_UNIT - The time unit to roll on (MINUTES, SECONDS, etc) <br>
38       * TIME_PERIOD - A long value for the TimeUnit multiplier. <br>
39       * PROGRESS_MAX - A long value for progress based rolls
40       * 
41       * @param config a map containing expected configuration values
42       * @return a newly configured Roller
43       */
44      public static Roller buildRoller(Map<String, String> config) {
45          String clz = config.get("CLASS");
46          // assumes a no-arg constructor and a Rollable instance
47          Rollable r = (Rollable) Factory.create(clz);
48          TimeUnit t = getUnit(config.get("TIME_UNIT"));
49          long period = getLong(config.get("TIME_PERIOD"));
50          long max = getLong(config.get("PROGRESS_MAX"));
51          return new Roller(max, t, period, r);
52      }
53  
54      // convert a long from a string or return 0
55      private static long getLong(@Nullable String l) {
56          return l == null ? 0L : Long.parseLong(l);
57      }
58  
59      // attempt to parse time unit.
60      @Nullable
61      private static TimeUnit getUnit(@Nullable String unit) {
62          return unit == null ? null : TimeUnit.valueOf(unit);
63      }
64  
65      /** This class is not meant to be instantiated. */
66      private RollUtil() {}
67  }

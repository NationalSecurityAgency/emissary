41  /**

43   * processing files of input, including files of sessions that must be identified and parsed using a ParserFactory. It
44   * knows nothing about where the data comes from, though. The method of input, either a directory or set of directories
45   * to monitor, a socket, a WorkSpace provider, or something else, comes from classes that extend this one.
46   */

144     /**
145      * Configure the place specific items
146      * <ul>
147      * <li>MINIMUM_DATA_SIZE: min size in bytes of a file</li>
148      * <li>MAXIMUM_DATA_SIZE: max size in bytes of a file, -1 for unlimited</li>
149      * <li>OVERSIZE_DATA_HOLDING_AREA: where to put oversize data</li>
150      * <li>HOLDING_AREA: where to put data while inprocess</li>
151      * <li>ERROR_DATA: where to put things that have errors</li>
152      * <li>DONE_DATA: where it goes when done</li>
153      * <li>SIMPLE_MODE: boolean when true turns off DataIdentification engine</li>
154      * <li>INITIAL_FORM: one or more forms for new payloads</li>
155      * </ul>
156      */
157     protected void configurePickUpPlace() {
158         minimumContentLength = configG.findIntEntry("MINIMUM_DATA_SIZE", minimumContentLength);
159         maximumContentLength = configG.findSizeEntry("MAXIMUM_DATA_SIZE", maximumContentLength);
160         oversizeArea = configG.findStringEntry("OVERSIZE_DATA_HOLDING_AREA", oversizeArea);
161 
162         simpleMode = configG.findBooleanEntry("SIMPLE_MODE", false);
163         holdingArea = configG.findCanonicalFileNameEntry("HOLDING_AREA", null);
164         doneArea = configG.findCanonicalFileNameEntry("DONE_DATA", doneArea);
165         errorArea = configG.findCanonicalFileNameEntry("ERROR_DATA", "errorArea");
166 
167         if (doneArea != null && doneArea.equals("")) {
168             doneArea = null;
169             logger.info("Alert: Completed data will be deleted from the system due to DONE_AREA setting");
170         }
171 
172         logger.debug("Pickup Canonical HOLD => {}, Pickup Canonical DONE => {}, Pickup Canonical ERROR => {}", holdingArea, doneArea, errorArea);
173 
174         initialFormValues = configG.findEntries("INITIAL_FORM");
175         if (initialFormValues.isEmpty()) {
176             initialFormValues.add(Form.UNKNOWN);
177         }
178 
179         // Grab the default pool
180         try {
181             agentPool = AgentPool.lookup();
182         } catch (NamespaceException e) {
183             logger.warn("Cannot find agent pool!");
184         }
185 
186         alwaysCopyMetadataVals = configG.findEntriesAsSet("ALWAYS_COPY_METADATA");
187 
188         // Whether or not to use the objectTrace logger
189         useObjectTraceLogger = configG.findBooleanEntry("USE_OBJECT_TRACE_LOGGER", useObjectTraceLogger);
190     }

264     /**
265      * Call back from a data server or queue server when a new file is ready to process. This method is called for raw
266      * files, not work bundles, so the simpleMode determination is made by this Place configuration.
267      * 
268      * @param f file to process
269      * @return true if it worked
270      * @throws IOException If there is some I/O problem.
271      */
272     public boolean processDataFile(File f) throws IOException, EmissaryException {
273         boolean isOversize = false;
274         if (maximumContentLength != -1 && f.length() > maximumContentLength) {
275             logger.warn("Sorry, This file is too large ({} < {}): {}", f.length(), maximumContentLength, f.getPath());
276             isOversize = true;
277             // Let it continue on knowing it is too big
278             // as we may need a record of the file
279         }
280         String fixedName = fixFileName(f.getName());
281         return processDataFile(f, fixedName, isOversize, simpleMode, getDoneArea());
282     }

284     /**
285      * Call back from a data server or queue server when a new file is ready to process
286      * 
287      * @param theFile file to process
288      * @param fixedName the good short name of the file
289      * @param isOversize true if the content is too big by configuration
290      * @param simpleMode true if no session parsing is desired
291      * @param outputRoot the done area
292      * @return true if it worked
293      * @throws IOException If there is some I/O problem.
294      */
295     public boolean processDataFile(File theFile, String fixedName, boolean isOversize, boolean simpleMode, String outputRoot) throws IOException,
296             EmissaryException {
297         boolean success = true;
298         logger.debug("Starting processDataFile in PickUpPlace for {}", theFile);
299 
300         ObjectTracingService.emitLifecycleEvent(null, fixedName, ObjectTracing.Stage.PICK_UP, useObjectTraceLogger);
301 
302         // Handle oversize data quickly without reading the file
303         if (isOversize) {
304             handleOversizePayload(theFile, fixedName, simpleMode);
305         }
306 
307         // Handle it without session parsing if simple mode is on
308         else if (simpleMode) {
309             handleSimplePayload(theFile, fixedName);
310         }
311 
312         // Parse sessions out of the file
313         else {
314             try {
315                 logger.debug("Starting processSessions on {}", theFile);
316                 processSessions(theFile, fixedName);
317                 logger.debug("Finished with processSessions on {}", theFile);
318             } catch (ParserException ex) {
319                 logger.error("Cannot parse {}", theFile.getName(), ex);
320                 success = false;
321             }
322         }
323 
324         if (success) {
325             handleFileSuccess(theFile, outputRoot);
326         } else {
327             handleFileError(theFile);
328         }
329 
330         logger.debug("Ending processDataFile {} {} {}", theFile, success ? "success" : "failure", simpleMode ? "simple" : "");
331         return success;
332     }

340      * @return true

352         return true;
353     }

378      * Action to move th file to the done area when successfully processed using the specified outut area
379      * 
380      * @param theFile the file that was processed
381      * @param outputRoot a specified output root
382      * @return true if the file was renamed
383      */
384     protected boolean renameFileToDoneArea(File theFile, @Nullable String outputRoot) {
385         String base = theFile.getPath();
386         boolean renamed = false;
387         if (holdingArea != null) {
388             base = base.substring(holdingArea.length());
389         }
390 
391         if (outputRoot != null) {
392             File dest = new File(outputRoot + "/" + base);
393             dest.getParentFile().mkdirs();
394             renamed = theFile.renameTo(dest);
395             if (renamed) {
396                 logger.info("{} processed and moved to done area {}", theFile.getName(), outputRoot);
397             } else {
398                 logger.warn("{} processed but could not be moved to done area as {}", theFile.getName(), dest);
399             }
400         }
401         return renamed;
402     }

426      * Action to move the file to inProcess area when taking ownership
427      * 
428      * @param source the file to be renamed
429      * @param dest where it should end up, or use the holdingArea if nil
430      * @return true if renamed, false if not
431      */
432     protected boolean renameToInProcessAreaAs(File source, @Nullable File dest) {
433         if (holdingArea == null && dest == null) {
434             logger.warn("Holding area not configured, cannot rename {}", source);
435             return false;
436         }
437 
438         if (dest == null) {
439             dest = getInProcessFileNameFor(source, null);
440         } else {
441             dest.getParentFile().mkdirs();
442         }
443 
444         boolean renamed = source.renameTo(dest);
445         if (renamed) {
446             logger.debug("{} moved to inProcess area as {}", source.getName(), dest);
447         } else {
448             logger.warn("{} could not be moved to inProcess area {}", source.getName(), dest);
449         }
450         return renamed;
451     }

539     /**
540      * Set up the dataobject and send it on the way
541      * 
542      * @param d the nascent data object
543      * @param fixedName the short name of it
544      * @param theFile where it came from
545      * @param simpleMode simple flag from the input
546      * @return true if it works
547      */
548     protected boolean processDataObject(IBaseDataObject d, String fixedName, File theFile, boolean simpleMode) throws EmissaryException {
549         String currentForm = d.popCurrentForm();
550         if (currentForm == null) {
551             // Add our stuff to the form stack if none is set (e.g from DecomposedSession)
552             for (int j = initialFormValues.size() - 1; j >= 0; j--) {
553                 d.pushCurrentForm(initialFormValues.get(j));
554             }
555         } else {
556             d.pushCurrentForm(currentForm);
557         }
558 
559         d.setParameter("SIMPLE_MODE", Boolean.toString(simpleMode));
560         dataObjectCreated(d, theFile);
561         logger.info("**Deploying an agent for {} and object {} forms={} simple={}", fixedName, d.getInternalId(), d.getAllCurrentForms(),
562                 simpleMode ? "simple" : "");
563 
564         assignToPooledAgent(d, -1L);
565         return true;
566     }

568     /**
569      * Parse out sessions and process data from a file
570      * 
571      * @param theFile file to process
572      * @param fixedName the good short name of the file
573      * @return count of sessions parsed
574      * @throws IOException If there is some I/O problem.
575      */
576     public int processSessions(File theFile, String fixedName) throws IOException, ParserException {
577         // We are going to prefer a RAF parser if one
578         // is available so start by getting the file opened
579         logger.debug("PickUpPlace: Starting on {}", theFile.getName());
580         int sessionNum = 0;
581         try (RandomAccessFile raf = new RandomAccessFile(theFile, "r")) {
582 
583             // Get the right type of session parser
584             SessionParser sp = parserFactory.makeSessionParser(raf.getChannel());
585             logger.debug("Using session parser from raf ident {}", sp.getClass().getName());
586 
587             // ... and a session producer to crank out the data objects...
588             SessionProducer dof = new SessionProducer(sp, myKey, null);
589 
590             long fileStart = System.currentTimeMillis();
591             long totalSize = 0;
592 
593             // For each session get a data object from the producer
594             boolean isParserComplete = false;
595             while (!isParserComplete) {
596                 long sessionStart = System.currentTimeMillis();
597                 try {
598                     // Use filename-xx for default name
599                     String sessionName = fixedName + "-" + (sessionNum + 1);
600 
601                     IBaseDataObject dataObject = dof.getNextSession(sessionName);
602                     logger.debug("Pulled session {} from {} shortName={}", sessionName, theFile.getName(), dataObject.shortName());
603                     sessionNum++;
604                     long sessionEnd = System.currentTimeMillis();
605                     totalSize += dataObject.data().length;
606                     logger.info("sessionParseMetric:{},{},{},{},{},{}", sessionEnd - sessionStart, sp.getClass().getName(), theFile, sessionName,
607                             sessionNum, dataObject.data().length);
608                     processDataObject(dataObject, sessionName, theFile, false);
609                 } catch (ParserEOFException eof) {
610                     // expected at end of file
611                     isParserComplete = true;
612                     long fileEnd = System.currentTimeMillis();
613                     logger.info("fileParseMetric:{},{},{},{},{}", fileEnd - fileStart, sp.getClass().getName(), theFile, sessionNum, totalSize);
614                 } catch (EmissaryException ex) {
615                     logger.error("Could not dispatch {}", theFile.getName(), ex);
616                     throw new ParserException("Could not process" + theFile.getName(), ex);
617                 }
618             }
619         }
620 
621         logger.debug("Done processing {} sessions from {}", sessionNum, theFile.getName());
622         return sessionNum;
623     }

625     /**
626      * Parse out sessions and process data from a byte array
627      * 
628      * @param data the bytes to process
629      * @param fixedName the good short name of the file
630      * @param theFile file object representing path data belongs to
631      * @return count of sessions parsed
632      */
633     public int processSessions(byte[] data, String fixedName, File theFile) {
634         // We are going to prefer a byte array parser since
635         // the data is already in memory
636         logger.debug("PickUpPlace: Starting on {}", theFile.getName());
637         int sessionNum = 0;
638 
639         // Get the right type of session parser
640         SessionParser sp = parserFactory.makeSessionParser(InMemoryChannelFactory.create(data).create());
641 
642         // .. and a session producer to crank out the data objects...
643         SessionProducer dof = new SessionProducer(sp, myKey, null);
644 
645         // For each session get a data object from the producer
646         boolean isParserComplete = false;
647         while (!isParserComplete) {
648             try {
649                 // Use filename-xx for default name
650                 String sessionName = fixedName + "-" + (sessionNum + 1);
651 
652                 IBaseDataObject dataObject = dof.getNextSession(sessionName);
653                 sessionNum++;
654                 processDataObject(dataObject, sessionName, theFile, false);
655             } catch (ParserEOFException eof) {
656                 // expected at end of file
657                 isParserComplete = true;
658             } catch (EmissaryException ex) {
659                 logger.error("Could not dispatch {}", fixedName, ex);
660             }
661         }
662 
663         logger.debug("Done processing {} sessions from {}", sessionNum, theFile.getName());
664         return sessionNum;
665     }

667     /**
668      * Produce a legal tracking filename from the disk filename
669      * 
670      * @return fixed filename
671      */
672     @Nullable
673     protected String fixFileName(@Nullable String v) {
674         if (v == null) {
675             return null;
676         }
677         String s = v.replace(' ', '_');
678         s = s.replace('\t', '_');
679         s = s.replace('\n', '_');
680         s = s.replace('\r', '_');
681         s = s.replace('\f', '_');
682         s = s.replace(':', '_');
683 
684         if (s.startsWith(".")) {
685             s = "_dot_" + s.substring(1);
686         }
687 
688         return s;
689     }

696      *        time will not be strictly observed because the pool itself blocks for a configurable amount of time when
697      *        requesting an agent. We will wait no more than the specified timeoutMs + the configured pool timeout value.
698      * @throws EmissaryException when an agent cannot be obtained
699      */
700     public void assignToPooledAgent(IBaseDataObject payload, long timeoutMs) throws EmissaryException {
701         assignToPooledAgent(payload, agentPool, this, timeoutMs);
702     }

704     /**
705      * Retrieve and agent from the specified pool and assign the payload to it
706      * 
707      * @param payload the payload for the agent
708      * @param agentPool the pool of agents
709      * @param startingLocation the agent launch point
710      * @param timeoutMs maximum time in millis to wait for an agent from the pool. Set to -1 to wait forever. The specified
711      *        time will not be strictly observed because the pool itself blocks for a configurable amount of time when
712      *        requesting an agent. We will wait no more than the specified timeoutMs + the configured pool timeout value.
713      * @return mobile agent assigned to pool
714      * @throws EmissaryException when an agent cannot be obtained
715      */
716     @SuppressWarnings("ThreadPriorityCheck")
717     public static IMobileAgent assignToPooledAgent(IBaseDataObject payload, @Nullable AgentPool agentPool, IServiceProviderPlace startingLocation,
718             long timeoutMs) throws EmissaryException {
719         IMobileAgent agent = null;
720         long startTime = System.currentTimeMillis();
721         boolean warningGiven = false;
722         int loopCount = 0;
723 
724         MDC.put(MDCConstants.SHORT_NAME, payload.shortName());
725         try {
726 
727             if (agentPool == null) {
728                 agentPool = AgentPool.lookup();
729             }
730 
731             do {
732                 loopCount++;
733                 try {
734                     agent = agentPool.borrowAgent();
735                 } catch (Exception e) {
736                     if (!warningGiven) {
737                         slogger.debug("Cannot get agent from pool, trying again ", e);
738                         warningGiven = true;
739                     }
740                 }
741             } while (agent == null && (timeoutMs < 0 || (startTime + timeoutMs) < System.currentTimeMillis()));
742 
743             if (agent == null) {
744                 throw new EmissaryException("No agent found for " + payload.shortName() + " after " + loopCount + " tries.");
745             } else if (loopCount > 1) {
746                 slogger.info("Found agent after {} tries", loopCount);
747             }
748 
749             agent.go(payload, startingLocation);
750             Thread.yield();
751         } finally {
752             MDC.remove(MDCConstants.SHORT_NAME);
753         }
754 
755         return agent;
756     }

1   package emissary.pickup;
2   
3   import jakarta.annotation.Nullable;
4   import org.slf4j.Logger;
5   import org.slf4j.LoggerFactory;
6   
7   import java.util.ArrayDeque;
8   import java.util.Deque;
9   
10  /**
11   * A size limited queue for holding data to process.
12   */
13  public class PickupQueue {
14      // Data structure for the bundles
15      protected final Deque<WorkBundle> queue = new ArrayDeque<>();
16  
17      // Our logger
18      private static final Logger logger = LoggerFactory.getLogger(PickupQueue.class);
19  
20      /**
21       * These parameters determine the enqueing behavior. The desire is to minimize the number of remote calls from main to
22       * and instance of this class with the getQueSize method, and at the same keep all of the places busy. We do this by
23       * making the MAX_QUE_SIZE large enough to hold enough files to be processed in SLEEP_TIME. BUT we don't just make the
24       * MAX_QUE_SIZE huge because then we use too much memory. Some feeds put stuff on the Que in blocks. If our que is a
25       * prime numbered size they cannot fill it completely, which will help prevent blocking maybe.
26       */
27      private int maxQueSize = 19;
28  
29      /**
30       * Normal pickup queue creation
31       */
32      public PickupQueue() {}
33  
34      /**
35       * Create a queue with the given max size
36       *
37       * @param maxSize the maximum size the queue can grow to
38       */
39      public PickupQueue(int maxSize) {
40          maxQueSize = maxSize;
41      }
42  
43      /**
44       * Add incoming information to the queue of file names to process and notify anyone waiting on the queue
45       *
46       * @param paths the WorkBundle object containing files to queue up
47       * @return true if it was enqueued, false if we are too busy to handle it
48       */
49      public boolean enque(@Nullable WorkBundle paths) {
50          if (paths == null || paths.size() == 0) {
51              logger.warn("Enque of a null or empty WorkBundle structure!");
52              return true;
53          }
54  
55          synchronized (queue) {
56              // Add it to the queue
57              queue.addFirst(paths);
58              queue.notifyAll();
59          }
60  
61          synchronized (this) {
62              // notify waiters every time paths added
63              this.notifyAll();
64          }
65  
66          return true;
67      }
68  
69      /**
70       * Return size of queue
71       */
72      public int size() {
73          return getQueSize();
74      }
75  
76      /**
77       * Getter for the size of the queue
78       *
79       * @return the size of the queue
80       */
81      public int getQueSize() {
82          int size = -1;
83          synchronized (queue) {
84              size = queue.size();
85          }
86          return size;
87      }
88  
89      /**
90       * Get one data object from the queue.
91       *
92       * @return the dequeued WorkBundle or null if none
93       */
94      public synchronized WorkBundle deque() {
95          WorkBundle nextFile = null;
96          int size = -1;
97          synchronized (queue) {
98              size = queue.size();
99              if (size > 0) {
100                 nextFile = queue.removeLast();
101             }
102         }
103         return nextFile;
104     }
105 
106     /**
107      * Tell caller if we can hold this many more items
108      *
109      * @return true iff there is room for num items
110      */
111     public boolean canHold(int num) {
112         return getQueSize() + num <= maxQueSize;
113     }
114 }

28  /**
29   * Pull bundles of file info from a WorkSpace and process as a normal FilePickUp. Monitors a queue rather than a
30   * directory, but reads files from disk as specified in the received WorkBundle objects. Whether workBundles are
31   * processed in simpleMode or not is controlled by the bundle settings not by the inherited configuration of this
32   * client.
33   */
34  public class FilePickUpClient extends PickUpSpace implements IPickUp {
35      /**
36       * These parameters determine the enqueing behavior. The desire is to minimize the number of remote calls from WorkSpace
37       * or Distributor to an instance of this class with the getQueSize method, and at the same keep all of the places busy.
38       * We do this by making the MAX_QUE_SIZE large enough to hold enough files to be processed in pollingInterval. BUT we
39       * don't just make the MAX_QUE_SIZE huge because then we use too much memory. Some feeds put stuff on the Que in blocks.
40       * If our que is a prime numbered size they cannot fill it completely, which will help prevent blocking maybe.
41       */
42      protected int pollingInterval = 500;
43      protected int MAX_QUE_SIZE = 5;

166     /**
167      * Find a file in the holding area that matches our guy.
168      */
169     @Nullable
170     protected File findFileInHoldingArea(File f, @Nullable String eatPrefix) {
171         if (holdingArea != null) {
172             String fpart = f.getName();
173             if (eatPrefix != null) {
174                 fpart = f.getPath().substring(eatPrefix.length());
175             }
176 
177             // See if it is sitting at the InProcess level
178             File hf = new File(holdingArea + fpart);
179             if (hf.exists()) {
180                 logger.debug("Data recovered from holding area " + hf);
181                 return hf;
182             }
183             logger.debug("File did not exist in InProcess area directoy as " + hf);
184 
185 
186             // Or if it is one level down due to an emissary.node.name-emissary.node.port dir
187             for (File subdir : new File(holdingArea).listFiles(new FileFilter() {
188                 @Override
189                 public boolean accept(File d) {
190                     return d.isDirectory();
191                 }
192             })) {
193                 File hdf = new File(subdir + fpart);
194                 if (hdf.exists()) {
195                     logger.debug("Data recovered from holding subdir " + hdf);
196                     return hdf;
197                 }
198                 logger.debug("File did not exist in nested InProcess area as " + hdf);
199             }
200         }
201         return null;
202     }

204     /**
205      * Call back from queue server when a new bundle is dequeued for processing.
206      *
207      * @param paths the dequeued item
208      * @return true if the files in the WorkBundle were handled
209      */
210     protected boolean processBundle(WorkBundle paths) {
211         boolean success = true;
212         currentBundle = paths; // for use by callbacks
213         String outputRoot = fixFilePath(paths.getOutputRoot());
214         String prefix = "";
215         if (null != paths.getEatPrefix()) {
216             prefix = fixFilePath(paths.getEatPrefix());
217         }
218 
219         for (String path : paths.getFileNameList()) {
220             boolean wasInHoldingArea = false;
221             String opath = path;
222             path = fixFilePath(path);
223             File f = new File(path);
224             String fixedName = fixFileName(f.getName());
225             // Ensure it exists
226             if (!f.exists()) {
227                 // If the errorCount is > 0 look in the holding area
228                 if (paths.getErrorCount() > 0) {
229                     logger.debug("Looking for " + f + " in holding area using eatPrefix of " + prefix);
230                     File holdFile = findFileInHoldingArea(f, prefix);
231                     if (holdFile != null) {
232                         logger.info("Switching to found holdArea file " + holdFile);
233                         wasInHoldingArea = true;
234                         f = holdFile;
235                     } else {
236                         logger.debug("File was not found in holding area " + f + " using eatPrefix of " + prefix);
237                     }
238                 } else {
239                     logger.debug("File does not exist but had errorCount of 0 so not looking in holding area");
240                 }
241             }
242 
243             if (!f.exists()) {
244                 // Try to get the data from the workspace
245                 logger.debug("Non-existent file " + opath);
246                 continue;
247             }
248 
249             // Ensure it can be read
250             if (!f.canRead()) {
251                 logger.warn("Sorry, Cannot read file: " + f.getPath());
252                 continue;
253             }
254 
255             // Only process files here, but give a hook
256             // for subclasses to handle other things
257             if (!f.isFile()) {
258                 processDirectoryEntry(outputRoot, prefix, paths.getCaseId(), f, paths.getSimpleMode());
259                 continue;
260             }
261 
262             // Make sure it is big enough to process
263             if (f.length() <= minimumContentLength) {
264                 logger.warn("Sorry, This file is too small (" + f.length() + " <" + minimumContentLength + "): " + path);
265                 // No record is made of too small items
266                 continue;
267             }
268 
269             // Make sure it is not too big to process
270             boolean isOversize = false;
271             if (maximumContentLength != -1 && f.length() > maximumContentLength) {
272                 logger.warn("Sorry, This file is too large (" + f.length() + " <" + maximumContentLength + "): " + path);
273                 isOversize = true;
274                 // Let it continue on knowing it is too big
275                 // as we may need a record of the file
276             }
277 
278             // Possibly rename the file to a holding area
279             // if one is defined
280             File toProcess = getInProcessFileNameFor(f, wasInHoldingArea ? holdingArea : prefix);
281             if (holdingArea != null && toProcess != null && !wasInHoldingArea) {
282                 if (!renameToInProcessAreaAs(f, toProcess)) {
283                     logger.error("File: " + f.getPath() + " Could not be renamed to: " + toProcess.getPath());
284                     continue;
285                 }
286             } else {
287                 toProcess = f;
288             }
289 
290             // Start the processing. The file may be in the original
291             // location or may be in the holding area
292             try {
293                 success = processDataFile(toProcess, fixedName, isOversize, paths.getSimpleMode(), outputRoot);
294                 logger.debug("Finished with processDataFile on " + toProcess + " as " + fixedName);
295             } catch (Exception e) {
296                 // Return false and let another
297                 // processor have a try at this work bundle
298                 // TODO: What is some files work and some fail?
299                 handleErrorInBundledFile(toProcess, fixedName, isOversize, simpleMode, e);
300                 success = false;
301                 break;
302             }
303         }
304         logger.debug("Finished processBundle " + paths.getBundleId() + " " + (success ? "success" : "failure"));
305         return success;
306     }

308     protected void handleErrorInBundledFile(File toProcess, String fixedName, boolean isOversize, boolean simpleMode, Exception e) {
309         // Error either way but louder if debug is on
310         if (logger.isDebugEnabled()) {
311             logger.error("Cannot complete " + toProcess.getPath() + " as " + fixedName + " [isOversize=" + isOversize + ", simpleMode="
312                     + simpleMode + "]", e);
313         } else {
314             logger.error("Cannot complete " + toProcess.getPath() + " as " + fixedName, e);
315         }
316 
317         String errDir = getErrorArea();
318         // Move the problem file to the error area if there is one
319         if (errDir != null) {
320             if (!toProcess.renameTo(new File(errDir, toProcess.getName()))) {
321                 logger.error("Cannot rename " + toProcess.getName() + " to the error location " + errDir);
322             } else {
323                 logger.error("Moved " + toProcess + " to the errorArea " + errDir);
324             }
325         } else {
326             logger.error("There is no configured errorArea in which to drop failed input files like " + toProcess);
327         }
328     }

370     /**
371      * Add in a target bin parameter with user and date Override point for subclasses
372      *
373      * @param d the nascent data object from the SessionProducer
374      * @param f the file it came from
375      */
376     @Override
377     protected void dataObjectCreated(IBaseDataObject d, File f) {
378         super.dataObjectCreated(d, f);
379         String fixedDirName = fixFileName(f.getParent()).replace('\\', '/');
380         String eatPrefix = currentBundle.getEatPrefix();
381         currentBundle.getOutputRoot();
382         boolean simpleParam = Boolean.parseBoolean(d.getStringParameter("SIMPLE_MODE"));
383         if (eatPrefix != null && eatPrefix.length() > 0 && fixedDirName.startsWith(eatPrefix)) {
384             fixedDirName = fixedDirName.substring(eatPrefix.length());
385         }
386 
387 
388         // payloadHandler.setup(d);
389         d.putParameter("TARGETBIN", fixedDirName);
390         d.putParameter(SessionParser.ORIG_DOC_SIZE_KEY, Integer.valueOf(d.dataLength()));
391         d.setPriority(currentBundle.getPriority());
392 
393         // Fix up the complete path
394         String ep = currentBundle.getEatPrefix();
395         String fn = f.getAbsolutePath();
396         if (ep != null && fn.startsWith(ep)) {
397             fn = fn.substring(ep.length());
398         }
399 
400         if (simpleParam) {
401             d.putParameter(ORIGINAL_FILENAME, fn);
402         }
403 
404         d.putParameter(INPUT_FILEDATE, TimeUtil.getDateAsISO8601(f.lastModified()));
405         d.putParameter(INPUT_FILENAME, f.getName());
406 
407         // Fix up the case/project metadata, e.g. PROJECT:GERONIMO22
408         String cid = currentBundle.getCaseId();
409         if (cid != null && cid.indexOf(":") > 0) {
410             String[] parts = cid.split(":");
411             if (d.getParameter(parts[0]) == null) {
412                 d.putParameter(parts[0], parts[1]);
413             }
414 
415             if (simpleParam && fn != null && digest != null) {
416                 final MessageDigest theDigest = this.digest;
417                 synchronized (theDigest) {
418                     theDigest.reset();
419                     byte[] hash = theDigest.digest(fn.getBytes());
420                     d.setFilename(parts[1] + "-" + Hexl.toUnformattedHexString(hash));
421                 }
422             }
423         } else {
424             // Take care of the caseid
425             String fixedCaseId = caseIdHook(cid, d.shortName(), f.toString(), d.getParameters());
426             if (fixedCaseId == null) {
427                 // current yyyyjjj
428                 fixedCaseId = TimeUtil.getCurrentDateOrdinal();
429             }
430             d.putParameter("DATABASE_CASE_ID", fixedCaseId);
431         }
432     }

434     /**
435      * Generate a filename using the file's path and a prefix
436      * 
437      * @param filePath the path of the file
438      * @param prefix a prefix to prepend to the resultant filename
439      * @return the generated filename
440      */
441     protected String createFilename(String filePath, String prefix) {
442         final MessageDigest theDigest = this.digest;
443         synchronized (theDigest) {
444             theDigest.reset();
445             byte[] hash = theDigest.digest(filePath.getBytes());
446             return new File(prefix + "-" + Hexl.toUnformattedHexString(hash)).getName();
447         }
448     }

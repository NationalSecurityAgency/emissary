29  /**
30   * Pull bundles of file info from a WorkSpace and process as a normal FilePickUp. Monitors a queue rather than a
31   * directory, but reads files from disk as specified in the received WorkBundle objects. Whether workBundles are
32   * processed in simpleMode or not is controlled by the bundle settings not by the inherited configuration of this
33   * client.
34   */
35  public class FilePickUpClient extends PickUpSpace implements IPickUp {
36      /**
37       * These parameters determine the enqueing behavior. The desire is to minimize the number of remote calls from WorkSpace
38       * or Distributor to an instance of this class with the getQueSize method, and at the same keep all of the places busy.
39       * We do this by making the MAX_QUE_SIZE large enough to hold enough files to be processed in pollingInterval. BUT we
40       * don't just make the MAX_QUE_SIZE huge because then we use too much memory. Some feeds put stuff on the Que in blocks.
41       * If our que is a prime numbered size they cannot fill it completely, which will help prevent blocking maybe.
42       */
43      protected int pollingInterval = 500;
44      protected int maxQueSize = 5;

167     /**
168      * Find a file in the holding area that matches our guy.
169      */
170     @Nullable
171     protected File findFileInHoldingArea(File f, @Nullable String eatPrefix) {
172         if (holdingArea != null) {
173             String fpart = f.getName();
174             if (eatPrefix != null) {
175                 fpart = f.getPath().substring(eatPrefix.length());
176             }
177 
178             // See if it is sitting at the InProcess level
179             File hf = new File(holdingArea + fpart);
180             if (hf.exists()) {
181                 logger.debug("Data recovered from holding area " + hf);
182                 return hf;
183             }
184             logger.debug("File did not exist in InProcess area directoy as " + hf);
185 
186 
187             // Or if it is one level down due to an emissary.node.name-emissary.node.port dir
188             for (File subdir : new File(holdingArea).listFiles(new FileFilter() {
189                 @Override
190                 public boolean accept(File d) {
191                     return d.isDirectory();
192                 }
193             })) {
194                 File hdf = new File(subdir + fpart);
195                 if (hdf.exists()) {
196                     logger.debug("Data recovered from holding subdir " + hdf);
197                     return hdf;
198                 }
199                 logger.debug("File did not exist in nested InProcess area as " + hdf);
200             }
201         }
202         return null;
203     }

205     /**
206      * Call back from queue server when a new bundle is dequeued for processing.
207      *
208      * @param paths the dequeued item
209      * @return true if the files in the WorkBundle were handled
210      */
211     protected boolean processBundle(WorkBundle paths) {
212         boolean success = true;
213         currentBundle = paths; // for use by callbacks
214         String outputRoot = fixFilePath(paths.getOutputRoot());
215         String prefix = "";
216         if (null != paths.getEatPrefix()) {
217             prefix = fixFilePath(paths.getEatPrefix());
218         }
219 
220         for (String path : paths.getFileNameList()) {
221             boolean wasInHoldingArea = false;
222             String opath = path;
223             path = fixFilePath(path);
224             File f = new File(path);
225             String fixedName = fixFileName(f.getName());
226             // Ensure it exists
227             if (!f.exists()) {
228                 // If the errorCount is > 0 look in the holding area
229                 if (paths.getErrorCount() > 0) {
230                     logger.debug("Looking for " + f + " in holding area using eatPrefix of " + prefix);
231                     File holdFile = findFileInHoldingArea(f, prefix);
232                     if (holdFile != null) {
233                         logger.info("Switching to found holdArea file " + holdFile);
234                         wasInHoldingArea = true;
235                         f = holdFile;
236                     } else {
237                         logger.debug("File was not found in holding area " + f + " using eatPrefix of " + prefix);
238                     }
239                 } else {
240                     logger.debug("File does not exist but had errorCount of 0 so not looking in holding area");
241                 }
242             }
243 
244             if (!f.exists()) {
245                 // Try to get the data from the workspace
246                 logger.debug("Non-existent file " + opath);
247                 continue;
248             }
249 
250             // Ensure it can be read
251             if (!f.canRead()) {
252                 logger.warn("Sorry, Cannot read file: " + f.getPath());
253                 continue;
254             }
255 
256             // Only process files here, but give a hook
257             // for subclasses to handle other things
258             if (!f.isFile()) {
259                 processDirectoryEntry(outputRoot, prefix, paths.getCaseId(), f, paths.getSimpleMode());
260                 continue;
261             }
262 
263             // Make sure it is big enough to process
264             if (f.length() <= minimumContentLength) {
265                 logger.warn("Sorry, This file is too small (" + f.length() + " <" + minimumContentLength + "): " + path);
266                 // No record is made of too small items
267                 continue;
268             }
269 
270             // Make sure it is not too big to process
271             boolean isOversize = false;
272             if (maximumContentLength != -1 && f.length() > maximumContentLength) {
273                 logger.warn("Sorry, This file is too large (" + f.length() + " <" + maximumContentLength + "): " + path);
274                 isOversize = true;
275                 // Let it continue on knowing it is too big
276                 // as we may need a record of the file
277             }
278 
279             // Possibly rename the file to a holding area
280             // if one is defined
281             File toProcess = getInProcessFileNameFor(f, wasInHoldingArea ? holdingArea : prefix);
282             if (holdingArea != null && toProcess != null && !wasInHoldingArea) {
283                 if (!renameToInProcessAreaAs(f, toProcess)) {
284                     logger.error("File: " + f.getPath() + " Could not be renamed to: " + toProcess.getPath());
285                     continue;
286                 }
287             } else {
288                 toProcess = f;
289             }
290 
291             // Start the processing. The file may be in the original
292             // location or may be in the holding area
293             try {
294                 success = processDataFile(toProcess, fixedName, isOversize, paths.getSimpleMode(), outputRoot);
295                 logger.debug("Finished with processDataFile on " + toProcess + " as " + fixedName);
296             } catch (Exception e) {
297                 // Return false and let another
298                 // processor have a try at this work bundle
299                 // TODO: What is some files work and some fail?
300                 handleErrorInBundledFile(toProcess, fixedName, isOversize, simpleMode, e);
301                 success = false;
302                 break;
303             }
304         }
305         logger.debug("Finished processBundle " + paths.getBundleId() + " " + (success ? "success" : "failure"));
306         return success;
307     }

309     protected void handleErrorInBundledFile(File toProcess, String fixedName, boolean isOversize, boolean simpleMode, Exception e) {
310         // Error either way but louder if debug is on
311         if (logger.isDebugEnabled()) {
312             logger.error("Cannot complete " + toProcess.getPath() + " as " + fixedName + " [isOversize=" + isOversize + ", simpleMode="
313                     + simpleMode + "]", e);
314         } else {
315             logger.error("Cannot complete " + toProcess.getPath() + " as " + fixedName, e);
316         }
317 
318         String errDir = getErrorArea();
319         // Move the problem file to the error area if there is one
320         if (errDir != null) {
321             if (!toProcess.renameTo(new File(errDir, toProcess.getName()))) {
322                 logger.error("Cannot rename " + toProcess.getName() + " to the error location " + errDir);
323             } else {
324                 logger.error("Moved " + toProcess + " to the errorArea " + errDir);
325             }
326         } else {
327             logger.error("There is no configured errorArea in which to drop failed input files like " + toProcess);
328         }
329     }

371     /**
372      * Add in a target bin parameter with user and date Override point for subclasses
373      *
374      * @param d the nascent data object from the SessionProducer
375      * @param f the file it came from
376      */
377     @Override
378     protected void dataObjectCreated(IBaseDataObject d, File f) {
379         super.dataObjectCreated(d, f);
380         String fixedDirName = fixFileName(f.getParent()).replace('\\', '/');
381         String eatPrefix = currentBundle.getEatPrefix();
382         currentBundle.getOutputRoot();
383         boolean simpleParam = Boolean.parseBoolean(d.getStringParameter("SIMPLE_MODE"));
384         if (eatPrefix != null && eatPrefix.length() > 0 && fixedDirName.startsWith(eatPrefix)) {
385             fixedDirName = fixedDirName.substring(eatPrefix.length());
386         }
387 
388 
389         // payloadHandler.setup(d);
390         d.putParameter("TARGETBIN", fixedDirName);
391         d.putParameter(SessionParser.ORIG_DOC_SIZE_KEY, Integer.valueOf(d.dataLength()));
392         d.setPriority(currentBundle.getPriority());
393 
394         // Fix up the complete path
395         String ep = currentBundle.getEatPrefix();
396         String fn = f.getAbsolutePath();
397         if (ep != null && fn.startsWith(ep)) {
398             fn = fn.substring(ep.length());
399         }
400 
401         if (simpleParam) {
402             d.putParameter(ORIGINAL_FILENAME, fn);
403         }
404 
405         d.putParameter(INPUT_FILEDATE, TimeUtil.getDateAsISO8601(f.lastModified()));
406         d.putParameter(INPUT_FILENAME, f.getName());
407 
408         // Fix up the case/project metadata, e.g. PROJECT:GERONIMO22
409         String cid = currentBundle.getCaseId();
410         if (cid != null && cid.indexOf(":") > 0) {
411             String[] parts = cid.split(":");
412             if (d.getParameter(parts[0]) == null) {
413                 d.putParameter(parts[0], parts[1]);
414             }
415 
416             if (simpleParam && fn != null && digest != null) {
417                 final MessageDigest theDigest = this.digest;
418                 synchronized (theDigest) {
419                     theDigest.reset();
420                     byte[] hash = theDigest.digest(fn.getBytes());
421                     d.setFilename(parts[1] + "-" + Hexl.toUnformattedHexString(hash));
422                 }
423             }
424         } else {
425             // Take care of the caseid
426             String fixedCaseId = caseIdHook(cid, d.shortName(), f.toString(), d.getParameters());
427             if (fixedCaseId == null) {
428                 // current yyyyjjj
429                 fixedCaseId = TimeUtil.getCurrentDateOrdinal();
430             }
431             d.putParameter("DATABASE_CASE_ID", fixedCaseId);
432         }
433     }

435     /**
436      * Generate a filename using the file's path and a prefix
437      * 
438      * @param filePath the path of the file
439      * @param prefix a prefix to prepend to the resultant filename
440      * @return the generated filename
441      */
442     protected String createFilename(String filePath, String prefix) {
443         final MessageDigest theDigest = this.digest;
444         synchronized (theDigest) {
445             theDigest.reset();
446             byte[] hash = theDigest.digest(filePath.getBytes());
447             return new File(prefix + "-" + Hexl.toUnformattedHexString(hash)).getName();
448         }
449     }

47  /**
48   * Recursively process input and distribute files to one or more remote PickUp client instances when they ask for a
49   * bundle of work to do.
50   * <p>
51   * The work bundles, emissary.pickup.WorkBundle objects, are placed on a queue for sending to consumers by the producer
52   * here. Once a WorkBundle has been requested and sent to a consumer (i.e. FilePickUpClient) it is placed on a pending
53   * queue tagged with the consumer id until the consumer notifies this WorkSpace that the work has been completed. If the
54   * consumer goes away without notification, then the work is moved back to the outbound queue and given to another
55   * consumer.

257         // Must be before we start processing files and directories
258         initializeCase();
259     }

514     /**
515      * Set the pattern for finding pickup clients
516      *
517      * @param thePattern the new pattern
518      * @see emissary.directory.KeyManipulator#gmatch(String,String)
519      */
520     public void setPattern(@Nullable final String thePattern) throws Exception {
521 
522         if ((this.pattern != null) && this.pattern.equals(thePattern)) {
523             logger.debug("The pattern is already set to {}", thePattern);
524             return;
525         }
526 
527         this.pattern = thePattern;
528 
529         // Clear out old pick up clients
530         logger.warn("Clearing client list so we can look for new pattern {} in {}", thePattern, getKey());
531         this.pups.clear();
532 
533         // Find new ones
534         this.pups.addAll(getPickUpClients(this.pattern));
535 
536         // Set up a new observer on the directory
537         if (this.watcher != null) {
538             DirectoryAdapter.remove(this.watcher);
539         }
540         this.watcher = new WorkSpaceDirectoryWatcher(this.pattern);
541         DirectoryAdapter.register(this.watcher);
542     }

649     /**
650      * Notify one pickup
651      *
652      * @param pup the key of the one to notify
653      */
654     protected boolean notifyPickUp(final String pup) {
655         final WorkSpaceAdapterpter.html#WorkSpaceAdapter">WorkSpaceAdapter tpa = new WorkSpaceAdapter();
656         logger.debug("Sending notice to {}", pup);
657 
658         boolean notified = false;
659         int tryCount = 0;
660 
661         while (!notified && tryCount < 5) {
662             final EmissaryResponse status = tpa.outboundOpenWorkSpace(pup, this.workSpaceKey);
663 
664             // TODO Consider putting this method in the response
665             if (status.getStatus() != HttpStatus.SC_OK) {
666                 logger.warn("Failed to notify {} on try {}: {}", pup, tryCount, status.getContentString());
667                 try {
668                     Thread.sleep((tryCount + 1) * 100L);
669                 } catch (InterruptedException ignore) {
670                     Thread.currentThread().interrupt();
671                 }
672             } else {
673                 notified = true;
674             }
675             tryCount++;
676         }
677 
678         if (logger.isInfoEnabled()) {
679             logger.info("Notified {} in {} attempts: {}", pup, tryCount, (notified ? "SUCCESS" : "FAILED"));
680         }
681 
682         return notified;
683     }

699     /**
700      * Remove a pickup, if it had work bundles pending completion transfer them back to the outbound queue
701      *
702      * @param remoteKey the directory observer string key that was removed
703      */
704     protected void removePickUp(final String remoteKey) {
705         this.pups.remove(remoteKey);
706         if (logger.isDebugEnabled()) {
707             logger.debug("Removed pickup {}, size={}: {}", remoteKey, this.pups.size(), this.pups);
708         }
709         int pendCount = 0;
710         final String remoteName = KeyManipulator.getServiceHost(remoteKey);
711         synchronized (this.QLOCK) {
712             // NB: no enhanced for loop with Iterator.remove()
713             for (Iterator<String> i = this.pending.keySet().iterator(); i.hasNext();) {
714                 final String id = i.next();
715                 final WorkBundle wb = this.pending.get(id);
716                 if (remoteName.equals(wb.getSentTo())) {
717                     i.remove(); // remove from pending
718                     wb.setSentTo(null); // clear in progress indicator
719                     this.retryCount++;
720                     if (wb.incrementErrorCount() <= this.MAX_BUNDLE_RETRIES) {
721                         logger.debug("Removing pending bundle {} from pending pool, re-adding to outbound with errorCount={}", wb.getBundleId(),

940     /**
941      * Monitoring progress of the WorkSpace. Indicate some stats once in a while and do not let the foreground thread
942      * terminate while there is still work on the outbound queue or the pending lists.
943      */
944     protected void monitorProgress() {
945         long outboundEmptyTimestamp = -1L;
946 
947         // Do while outbound or pending work exists or collector is
948         // still running
949         while (true) {
950             final int outboundSize = getOutboundQueueSize();
951             int pendingSize = getPendingQueueSize();
952             final boolean reallyQuit = this.timeToQuit && (outboundSize == 0) && (pendingSize == 0);
953 
954             // Rmember when outbound becomes empty
955             if (outboundSize == 0 && outboundEmptyTimestamp == -1L) {
956                 outboundEmptyTimestamp = System.currentTimeMillis();
957             } else if (outboundSize > 0 && outboundEmptyTimestamp > 0L) {
958                 outboundEmptyTimestamp = -1L;
959             }
960 
961             // See if it is time to give up on pending items
962             if ((outboundSize == 0) && !this.loop && ((outboundEmptyTimestamp + this.PENDING_HANG_TIME) < System.currentTimeMillis())) {
963                 if (logger.isInfoEnabled()) {
964                     logger.info("Giving up on {} items due to timeout", pendingSize);
965                     for (final Map.Entry<String, WorkBundle> entry : this.pending.entrySet()) {
966                         logger.info("Pending item {}: {}", entry.getKey(), entry.getValue());
967                     }
968                 }
969                 clearPendingQueue();
970                 pendingSize = 0;
971             }
972 
973             // All work is done and collector has finished
974             if (outboundSize + pendingSize == 0) {
975                 if (reallyQuit) {
976                     break;
977                 }
978                 publishStats();
979             }
980 
981             // Else sleep a while
982             try {
983                 for (int si = 0; si < 3000; si++) {
984                     Thread.sleep(10L);
985                     if (reallyQuit) {
986                         break;
987                     }
988                 }
989             } catch (InterruptedException ex) {
990                 Thread.currentThread().interrupt();
991             }
992 
993             if (!this.timeToQuit) {
994                 publishStats();
995             }
996         }
997 
998         // Case closing actions
999         closeCase();
1000     }

1397             Integer count = this.remoteMap.get(machine);
1398             if (count == null) {
1399                 count = 1;
1400             } else {
1401                 count = count + 1;
1402             }
1403             this.remoteMap.put(machine, count);
1404         }

1438     /**
1439      * Watch the directory for changes to pickup up client places
1440      */
1441     public class WorkSpaceDirectoryWatcher extends DirectoryAdapter {
1442         /**
1443          * Watch the directory for registrations that match pattern
1444          *
1445          * @param pattern the pattern to match
1446          */
1447         public WorkSpaceDirectoryWatcher(final String pattern) {
1448             super(pattern);
1449             logger.debug("PickupClient pattern is {}", pattern);
1450         }
1451 
1452         /**
1453          * Accept registration notices that match our pattern
1454          *
1455          * @param observableKey the reporting directory
1456          * @param placeKey the key of the matching registered place
1457          */
1458         @Override
1459         public void placeRegistered(final String observableKey, final String placeKey) {
1460             final String k = KeyManipulator.removeExpense(placeKey);
1461             logger.debug("Registration message from {}", k);
1462             if (WorkSpace.this.pups.contains(k) && WorkSpace.this.useRetryStrategy) {
1463                 // This covers the case where the pickup dies and restarts
1464                 // before the Heartbeat mechanism figures out there was
1465                 // a problem.
1466                 logger.info("Already known pickup {} must be reinitialized to clear pending work.", k);
1467                 removePickUp(k);
1468             }
1469 
1470             if (!WorkSpace.this.pups.contains(k)) {
1471                 logger.info("New pickup place {}", k);
1472             }
1473 
1474             // add to list and maybe send open msg. Dup places
1475             // will not be added but might be re-notified
1476             addPickUp(k);
1477 
1478         }
1479 
1480         /**
1481          * Accept deregistration notices that match our pattern
1482          *
1483          * @param observableKey the reporting directory
1484          * @param placeKey the key of the matching deregistered place
1485          */
1486         @Override
1487         public void placeDeregistered(final String observableKey, final String placeKey) {
1488             final String k = KeyManipulator.removeExpense(placeKey);
1489             logger.debug("DeRegistration message from {}", k);
1490             if (!WorkSpace.this.pups.contains(k)) {
1491                 logger.info("Unknown pickup deregistered {}", k);
1492             } else {
1493                 logger.info("Pickup place {} is gone", k);
1494                 if (WorkSpace.this.useRetryStrategy) {
1495                     removePickUp(k);
1496                 }
1497             }
1498         }
1499     }

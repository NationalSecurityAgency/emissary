4   import org.apache.commons.io.input.BoundedInputStream;
5   import org.apache.commons.lang3.Validate;

7   import java.io.IOException;
8   import java.io.InputStream;
9   import java.nio.ByteBuffer;
10  import java.nio.channels.SeekableByteChannel;
11  
12  public class InputStreamChannelFactory {
13      private InputStreamChannelFactory() {}
14  
15      public static final int SIZE_IS_UNKNOWN = -1;
16  
17      /**
18       * Creates a factory implementation based on an {@link InputStreamFactory}
19       * 
20       * @param size if known, else provide a negative value to allow the factory to work out the size upon first create
21       * @param inputStreamFactory for the data
22       * @return an InputStreamChannelFactory instance of the data
23       */
24      public static SeekableByteChannelFactory create(final long size, final InputStreamFactory inputStreamFactory) {
25          return new InputStreamChannelFactoryImpl(size, inputStreamFactory);
26      }
27  
28      private static class InputStreamChannelFactoryImpl implements SeekableByteChannelFactory {
29          private final long size;
30          @Nullable
31          private final IOException ioException;
32          private final InputStreamFactory inputStreamFactory;
33  
34          public InputStreamChannelFactoryImpl(final long size, final InputStreamFactory inputStreamFactory) {
35              Validate.notNull(inputStreamFactory, "Required: inputStream not null");
36  
37              // If the size is unknown then calculate it and save any IOException that occurs.
38              if (size < 0) {
39                  long tempSize = SIZE_IS_UNKNOWN;
40                  IOException tempIoException = null;
41  
42                  try (InputStream is = inputStreamFactory.create()) {
43                      tempSize = SeekableByteChannelHelper.available(is);
44                  } catch (IOException e) {
45                      tempIoException = e;
46                  }
47  
48                  this.size = tempSize;
49                  ioException = tempIoException;
50              } else {
51                  this.size = size;
52                  this.ioException = null;
53              }
54  
55              this.inputStreamFactory = inputStreamFactory;
56          }
57  
58          @Override
59          public SeekableByteChannel create() {
60              return new InputStreamChannel(size, inputStreamFactory, ioException);
61          }
62      }
63  
64      private static class InputStreamChannel extends AbstractSeekableByteChannel {
65          /**
66           * The InputStreamFactory used to get InputStream instances.
67           */
68          private final InputStreamFactory inputStreamFactory;
69          private final long size;
70          private final IOException ioException;
71  
72          /**
73           * The current InputStream instance.
74           */
75          @Nullable
76          private BoundedInputStream inputStream;
77  
78          /**
79           * Create a new InputStreamChannel instance with a fixed size and data source
80           * 
81           * @param size of the InputStreamChannel
82           * @param inputStreamFactory data source
83           */
84          public InputStreamChannel(final long size, final InputStreamFactory inputStreamFactory, final IOException ioException) {
85              Validate.notNull(inputStreamFactory, "Required: inputStreamFactory not null!");
86  
87              this.size = size;
88              this.inputStreamFactory = inputStreamFactory;
89              this.ioException = ioException;
90          }
91  
92          @Override
93          protected final int readImpl(final ByteBuffer byteBuffer) throws IOException {
94              if (inputStream != null && position() < inputStream.getCount()) {
95                  inputStream.close();
96                  inputStream = null;
97              }
98  
99              if (inputStream == null) {
100                 inputStream = BoundedInputStream.builder()
101                         .setInputStream(inputStreamFactory.create())
102                         .get();
103             }
104 
105             // Actually perform the read
106             return SeekableByteChannelHelper.getFromInputStream(inputStream, byteBuffer, position() - inputStream.getCount());
107         }
108 
109 
110         @Override
111         protected long sizeImpl() throws IOException {
112             if (ioException != null) {
113                 throw ioException;
114             }
115 
116             return size;
117         }
118 
119         @Override
120         protected final void closeImpl() throws IOException {
121             if (inputStream != null) {
122                 inputStream.close();
123                 inputStream = null;
124             }
125         }
126     }

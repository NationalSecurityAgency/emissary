8   import java.io.IOException;

12  /**
13   * This class creates a InputStreamFactory for logging accesses to every method on the InputStream. It is intended only
14   * for development/debugging purposes.
15   */
16  public final class LoggingInputStreamFactory {
17      private LoggingInputStreamFactory() {}
18  
19      /**
20       * Creates an InputStreamFactory for logging accesses to every method on the InputStream.
21       * 
22       * @param inputStreamFactory that creates the InputStream to wrap.
23       * @param identifier that is added to the log message.
24       * @param logger to use for logging.
25       * @param logStackTrace specifies whether a stack trace should be added to the log message.
26       * @return the logging InputStreamFactory
27       */
28      public static InputStreamFactory create(final InputStreamFactory inputStreamFactory, final String identifier,
29              final Logger logger, final boolean logStackTrace) {
30          return new LoggingInputStreamFactoryImpl(inputStreamFactory, identifier, logger, logStackTrace);
31      }
32  
33      /**
34       * The InputStreamFactory for creating logging InputStreams.
35       */
36      private static class LoggingInputStreamFactoryImpl implements InputStreamFactory {
37          /**
38           * The InputStream that is being wrapped.
39           */
40          private final InputStreamFactory inputStreamFactory;
41          /**
42           * An identifier add to the log message.
43           */
44          private final String identifier;
45          /**
46           * The logger to be used for logging.
47           */
48          private final Logger logger;
49          /**
50           * Determines whether a stack trace should also be logged.
51           */
52          private final boolean logStackTrace;
53          /**
54           * A one-up counter for identifying each instance.
55           */
56          private final AtomicLong currentInstance = new AtomicLong(0);
57  
58          private LoggingInputStreamFactoryImpl(final InputStreamFactory inputStreamFactory, final String identifier,
59                  final Logger logger, final boolean logStackTrace) {
60              Validate.notNull(inputStreamFactory, "Required: inputStreamFactory not null!");
61              Validate.notNull(identifier, "Required: identifier not null!");
62              Validate.notNull(logger, "Required: logger not null!");
63  
64              this.inputStreamFactory = inputStreamFactory;
65              this.identifier = identifier;
66              this.logger = logger;
67              this.logStackTrace = logStackTrace;
68          }
69  
70          @Override
71          public InputStream create() throws IOException {
72              return new LoggingInputStream(inputStreamFactory.create(),
73                      identifier + " : " + currentInstance.getAndIncrement(), logger, logStackTrace);
74          }
75      }
76  }

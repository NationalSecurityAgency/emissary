3   import org.slf4j.Logger;

6   import java.util.concurrent.TimeUnit;

8   public abstract class Pausable extends Thread implements IPausable {
9   
10      private static final Logger logger = LoggerFactory.getLogger(Pausable.class);
11  
12      public static final long DEFAULT_PAUSE_INTERVAL = TimeUnit.MINUTES.toMillis(1);
13      protected long pauseInterval;
14  
15      private boolean paused = false;
16  
17      public Pausable(String name) {
18          this(name, DEFAULT_PAUSE_INTERVAL);
19      }
20  
21      public Pausable(String name, long pauseInterval) {
22          super(name);
23          this.pauseInterval = pauseInterval;
24      }
25  
26      /**
27       * Stop taking work
28       */
29      @Override
30      public void pause() {
31          paused = true;
32      }
33  
34      /**
35       * Resume taking work
36       */
37      @Override
38      public void unpause() {
39          paused = false;
40      }
41  
42      /**
43       * Check to see if the current thread is paused
44       *
45       * @return true if work is paused, false otherwise
46       */
47      @Override
48      public boolean isPaused() {
49          return paused;
50      }
51  
52      /**
53       * Get the time to sleep before checking if the thread has been unpaused
54       *
55       * @return the pause check interval
56       */
57      public long getPauseInterval() {
58          return pauseInterval;
59      }
60  
61      /**
62       * Check to see if we are currently paused. If paused, sleep for the configured interval and return true.
63       *
64       * @return true if paused, false otherwise
65       */
66      public boolean checkPaused() {
67          // check to see if we want to stop taking work
68          if (isPaused()) {
69              try {
70                  logger.info("{} currently paused, sleeping for {}", getClass().getName(), getPauseInterval());
71                  sleep(getPauseInterval());
72              } catch (InterruptedException ignore) {
73                  Thread.currentThread().interrupt();
74              }
75              return true;
76          }
77          return false;
78      }
79  
80      @Override
81      public abstract void run();
82  }

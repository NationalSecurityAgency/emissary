493                     try {
494                         final byte[] bytes = SeekableByteChannelHelper.getByteArrayFromChannel(value,

496                         final Element childElement = preserve(protectedElementBase64(childElementName, bytes));
497 
498                         childElement.setAttribute(LENGTH_ATTRIBUTE_NAME, Integer.toString(bytes.length));
499 
500                         parentElement.addContent(childElement);
501                     } catch (final IOException e) {
502                         LOGGER.error("Could not get bytes from SeekableByteChannel!", e);
503                     }
504                 }
505             }
506         }

522                                 BaseDataObject.MAX_BYTE_ARRAY_SIZE);
523                         final Element childElement = preserve(protectedElementHash(childElementName, bytes));
524 
525                         childElement.setAttribute(LENGTH_ATTRIBUTE_NAME, Integer.toString(bytes.length));
526 
527                         parentElement.addContent(childElement);

531                 }
532             }

535         private static Element protectedElementHash(final String name, final byte[] bytes) {
536             final Element element = new Element(name);
537 
538             if (ByteUtil.hasNonPrintableValues(bytes)) {
539                 element.setAttribute(ENCODING_ATTRIBUTE_NAME, SHA256);
540                 element.addContent(ByteUtil.sha256Bytes(bytes));
541             } else {
542                 element.addContent(new String(bytes, StandardCharsets.ISO_8859_1));
543             }
544 
545             return element;
546         }

570     private static class StringEncoder implements ElementEncoder<String> {
571         @Override
572         public void encode(final List<String> values, final Element parentElement, final String childElementName) {
573             for (int i = values.size() - 1; i >= 0; i--) {
574                 String value = values.get(i);
575 
576                 if (PROCESSING_ERROR.equals(childElementName) && StringUtils.isNotEmpty(value)) {
577                     value = value.substring(0, value.length() - 1);
578                 }
579 
580                 if (value != null) {
581                     parentElement.addContent(preserve(protectedElement(childElementName, value)));

728     /**
729      * Return UTF8 bytes from an XML value, decoding base64 if required
730      * 
731      * @param encoding e.g. 'base64', otherwise it returns the bytes as they are presented
732      * @param elementValue containing the data
733      * @return the data from elementValue, decoded from base64 if required
734      */
735     public static byte[] extractBytes(final String encoding, final String elementValue) {
736         if (BASE64.equalsIgnoreCase(encoding)) {
737             final String newElementValue = elementValue.replace("\n", "");
738             final byte[] bytes = newElementValue.getBytes(StandardCharsets.UTF_8);
739             return BASE64_DECODER.decode(bytes);
740         }
741 
742         return elementValue.getBytes(StandardCharsets.UTF_8);
743     }

769      * Creates a 'protected' element which can be encoded with base64 if it contains non-printable characters
770      * 
771      * See method source for specific definition of 'non-printable'.
772      * 
773      * @param name of the element
774      * @param bytes to wrap, if they contain non-printable characters
775      * @return the created element
776      */
777     public static Element protectedElementBase64(final String name, final byte[] bytes) {
778         final Element element = new Element(name);
779 
780         if (ByteUtil.hasNonPrintableValues(bytes)) {
781             String base64String = BASE64_NEW_LINE_STRING +
782                     BASE64_ENCODER.encodeToString(bytes) +
783                     BASE64_NEW_LINE_STRING;
784 
785             element.setAttribute(ENCODING_ATTRIBUTE_NAME, BASE64);
786             element.addContent(base64String);
787         } else {
788             element.addContent(new String(bytes, StandardCharsets.ISO_8859_1));
789         }
790 
791         return element;
792     }

794     /**
795      * Creates a 'protected' element which can be hashed with sha256 if it contains non-printable characters
796      * 
797      * See method source for specific definition of 'non-printable'.
798      * 
799      * @param name of the element
800      * @param bytes to wrap, if they contain non-printable characters.
801      * @return the created element
802      */
803     public static Element protectedElementSha256(final String name, final byte[] bytes) {
804         final Element element = new Element(name);
805 
806         if (ByteUtil.hasNonPrintableValues(bytes)) {
807             element.setAttribute(IBaseDataObjectXmlCodecs.ENCODING_ATTRIBUTE_NAME, IBaseDataObjectXmlCodecs.SHA256);

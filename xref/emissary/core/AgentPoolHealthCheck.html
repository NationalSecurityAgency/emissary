10  public class AgentPoolHealthCheck extends HealthCheck {
11  
12      @Override
13      protected Result check() {
14          int active = 0;
15          int idle = 0;
16          try {
17              for (int i = 0; i < AgentPool.lookup().getMaxTotal(); i++) {
18                  String agentKey = MobileAgentFactory.AGENT_NAME + "-" + String.format(Locale.getDefault(), "%02d", i);
19                  if (Namespace.exists(agentKey)) {
20                      if (Namespace.lookup(agentKey).toString().startsWith("Idle")) {
21                          idle++;
22                      } else {
23                          active++;
24                      }
25                  } else {
26                      return Result.unhealthy("Missing an agent in the Namespace: " + agentKey);
27                  }
28              }
29              return Result.healthy("Pool size active/idle: " + active + "/" + idle);
30          } catch (EmissaryException e) {
31              return Result.unhealthy("Problem when looking up the pool: " + e.getMessage());
32          }
33      }

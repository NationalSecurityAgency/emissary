78          if (p == null) {

81              return;

87          addPayload(p);

127     /**
128      * The arrive method that takes in a list of payloads arriving on the new machine
129      * 
130      * @param payload the real payload, exisitng if any will be cleared
131      * @param arrivalPlace the place we start at
132      * @param moveErrorCount state transfer from sending agent
133      * @param queuedItineraryItems state transfer from sending agent
134      */
135     @Override
136     @SuppressWarnings("unchecked")
137     public synchronized void arrive(final Object payload, final IServiceProviderPlace arrivalPlace, final int moveErrorCount,
138             final List<DirectoryEntry> queuedItineraryItems) throws Exception {
139 
140         logger.debug("Arrived at {}", arrivalPlace.toString());
141 
142         clear();
143         moveErrorsOccurred = moveErrorCount;
144         nextKeyQueue.addAll(queuedItineraryItems);
145 
146         if (payload instanceof IBaseDataObject) {
147             go(payload, arrivalPlace, true);
148         } else if (payload instanceof Collection) {
149             addPayload((Collection<IBaseDataObject>) payload);
150             setAgentId(getPayload().shortName());
151             go(null, arrivalPlace, true);
152         } else {
153             throw new Exception("Illegal payload sent to HDMobileAgent, cannot handle " + payload.getClass().getName());
154         }
155     }
156 
157     @Override
158     @SuppressWarnings("unchecked")
159     public synchronized void go(final Object payload, final IServiceProviderPlace arrivalPlace) {
160         if (payload instanceof IBaseDataObject) {
161             super.go(payload, arrivalPlace);
162         } else if (payload instanceof Collection) {
163             addPayload((Collection<IBaseDataObject>) payload);
164             setAgentId(getPayload().shortName());
165             go(null, arrivalPlace, false);
166         } else {
167             logger.error("Illegal payload sent to HDMobileAgent, cannot handle {}", payload.getClass().getName());
168         }
169     }

171     /**
172      * The main control loop to determine and go through an itinerary Since we have a list of payload object that can
173      * potentially grow at every place we visit, we need to iterate over them until we get done but we cannot use a normal
174      * iterator which will throw an exception if the underlying collection mutates while iterating. So we make the selection
175      * of the first payload that needs work, find the place for it, process all other payloads that have the same form and
176      * lastPlace, then move on.
177      */
178     @Override
179     protected void agentControl(final IServiceProviderPlace currentPlaceArg) {
180         DirectoryEntry newEntry = currentPlaceArg.getDirectoryEntry();
181         logger.debug("In agentControlHD {} for {}", currentPlaceArg, agentId);
182 
183         // Set into the super classes payload member...
184         IBaseDataObject mypayload = getPayload();
185 
186         // Go until all the payloads disappear or get done
187         IServiceProviderPlace currentPlace = currentPlaceArg;
188         int loopCount = 0;
189         boolean nextKeyRecorded = true;
190         boolean controlError = false;
191 
192         while (currentPlace != null && newEntry != null && mypayload != null) {
193             // One based loop counter
194             loopCount++;
195 
196             // Remember the payload's form and last place before
197             // doing the processing
198             final String primaryCurrentForm = mypayload.currentForm();
199             final DirectoryEntry primaryLastEntry = mypayload.getLastPlaceVisited();
200 
201             if (logger.isDebugEnabled()) {
202                 logger.debug("Starting control loop for {}, currentPlace={}, newEntry= {}, loopCount={}", mypayload.shortName(),
203                         currentPlace.getKey(), newEntry.getFullKey(), loopCount);
204             }
205 
206             // First time in, we just have the pickup place where we started
207             // our mission. We dont process there, just use it to call through
208             // to the directory, so skip the processing if this is true
209             if ((loopCount > 1 || getProcessFirstPlace()) && !controlError) {
210                 // If we are at IO phase, add them all since the deferrals
211                 // below should make everyone ready to drop off at the same time
212                 if ("IO".equals(currentPlace.getDirectoryEntry().getServiceType())) {
213                     // Drop off doesn't sprout so ignore return value
214                     if (!nextKeyRecorded) {
215                         logger.debug("Recording history drop off case");
216                         recordHistory(newEntry, this.payloadList);
217                         nextKeyRecorded = true;
218                     }
219                     atPlaceHD(currentPlace, this.payloadList);
220                 } else {
221                     // Add the primary payload object to a list
222                     final List<IBaseDataObject> toBeProcessed = new ArrayList<>();
223                     toBeProcessed.add(mypayload);
224 
225                     // Add any other payload that has the same current form
226                     // and last place visited as this one while we are here...
227                     for (final IBaseDataObject slug : this.payloadList) {
228                         final DirectoryEntry slugLastPlaceVisited = slug.getLastPlaceVisited();
229 
230                         if (slug != mypayload
231                                 && slug.searchCurrentForm(primaryCurrentForm) > -1
232                                 && ((primaryLastEntry == null && slugLastPlaceVisited == null) || (primaryLastEntry != null
233                                         && slugLastPlaceVisited != null && slugLastPlaceVisited.getKey().equals(primaryLastEntry.getKey())))) {
234                             // We don't need to call getNextKey but do
235                             // need to simulate this side effect of it...
236                             slug.pullFormToTop(primaryCurrentForm);
237 
238                             toBeProcessed.add(slug);
239 
240                             if (logger.isDebugEnabled()) {
241                                 logger.debug("Adding slug {} with key {} to ride with {} having key {} current form {}", slug.shortName(),
242                                         slugLastPlaceVisited == null ? "null" : slugLastPlaceVisited.getKey(), mypayload.shortName(),
243                                         primaryLastEntry == null ? "null" : primaryLastEntry.getKey(), primaryCurrentForm);
244                             }
245                         }
246                     }
247 
248                     // Process everything on the list
249                     if (!nextKeyRecorded) {
250                         if (loopCount == 1 && !getProcessFirstPlace()) {
251                             // ArrivalPlace for MobileAgent.send()
252                             logger.debug("Recording history two normal loop-1 case");
253                             recordHistory(currentPlace, toBeProcessed);
254                         } else {
255                             logger.debug("Recording history two non-loop-1 case");
256                             recordHistory(newEntry, toBeProcessed);
257                         }
258                         nextKeyRecorded = true;
259                     }
260                     final List<IBaseDataObject> sprouts = atPlaceHD(currentPlace, toBeProcessed);
261 
262                     // Add any sprouts collected from the payloads
263                     if (!sprouts.isEmpty()) {
264                         addPayload(sprouts);
265                     }
266                 }
267             }
268 
269             // Where to go next...
270             controlError = false;
271             newEntry = getNextKey(currentPlace, mypayload);
272             nextKeyRecorded = false;
273 
274             // Defer IO phase for now if there are attachments to process
275             // and we aren't already in the io phase
276             if ((newEntry != null) && (payloadCount() > 1) && "IO".equals(newEntry.getServiceType())
277                     && !"IO".equals(currentPlace.getDirectoryEntry().getServiceType())) {
278                 logger.debug("Deferring IO Phase place for {}", newEntry);
279                 newEntry = null;
280             } else {
281                 if (newEntry != null) {
282                     logger.debug("Continuing with place {}", newEntry);
283                 }
284             }
285 
286             // Choose the first place on the list that
287             // doesn't have a null nextKey when we run out
288             // of key for the one we were working on intitially
289             DirectoryEntry dropOffEntry = null;
290             int haveDropOffFor = -1;
291             if (newEntry == null) {
292                 logger.debug("Got null newEntry for {} looking for a better payload...", mypayload.shortName());
293                 for (int i = 0; i < payloadCount(); i++) {
294                     final IBaseDataObject p = getPayload(i);
295                     if (p == mypayload) {
296                         continue;
297                     }
298                     setParallelTrackingInfoFor(p);
299                     newEntry = getNextKey(currentPlace, p);
300                     if (newEntry != null) {
301                         // Defer IO Phase until sure we are all done
302                         if ("IO".equals(newEntry.getServiceType())) {
303                             logger.debug("Found IO service for part {}, {} deferring that and continuing to look", i, p.shortName());
304                             if (haveDropOffFor == -1) {
305                                 haveDropOffFor = i;
306                                 dropOffEntry = newEntry;
307                             }
308                             newEntry = null;
309                             continue;
310                         }
311 
312                         if (logger.isDebugEnabled()) {
313                             logger.debug("Found good key {} for new payload {}, serviceName={}", newEntry, p.shortName(), newEntry.getServiceName());
314                         }
315 
316                         // Found a new top dog to process
317                         // Pull it to the top of the list in case we have to move
318                         if (i != 0) {
319                             switchPrimaryPayload(i);
320                         }
321                         // Remember it for processing
322                         mypayload = p;
323                         break; // out of the for loop
324                     }
325                 }
326             }
327 
328             // Reset drop off if we deferred it above and found nothing better
329             if (newEntry == null && haveDropOffFor > -1) {
330                 // Pull entry to top
331                 if (haveDropOffFor != 0) {
332                     switchPrimaryPayload(haveDropOffFor);
333                     mypayload = getPayload(0);
334                     setParallelTrackingInfoFor(mypayload);
335                     logger.debug("Pulling payload {} to top before IO reinstatement", haveDropOffFor);
336                 }
337 
338                 // Set newEntry and go to drop off, deferred as long as possible
339                 newEntry = dropOffEntry;
340                 logger.debug("Resetting newEntry to IO phase");
341             }
342 
343             // Null entry at this point means we are all done
344             // with all the payloads, normal processing termination
345             if (newEntry == null) {
346                 break;
347             }
348 
349             // Local processing, go around the loop and process there
350             if (newEntry.isLocal()) {
351                 logger.debug("Choosing local place {}", newEntry.getFullKey());
352                 currentPlace = newEntry.getLocalPlace();
353                 continue;
354             }
355 
356             if (logger.isDebugEnabled()) {
357                 logger.debug("Recording history one move case loopCount={} getProcessFirstPlace={} currentPlace={} newEntry={}", loopCount,
358                         getProcessFirstPlace(), currentPlace, newEntry.getFullKey());
359             }
360 
361             // Time to move, entry is remote, record the history and go
362             recordHistory(newEntry, mypayload);
363             nextKeyRecorded = true;
364 
365             controlError = true;
366             if (!KeyManipulator.isKeyComplete(mypayload.currentForm())) {
367                 mypayload.replaceCurrentForm(ERROR_FORM);
368             } else {
369                 mypayload.popCurrentForm();
370             }
371         }
372 
373         // If null we are completely finished, otherwise we
374         // should just be moving to another machine
375         if (newEntry == null) {
376             logAgentCompletion();
377         }
378     }

380     /**
381      * Make the payload at the specified index the new primary one and reset the logger context to the new value
382      * 
383      * @param i the index of the desired payload
384      */
385     protected void switchPrimaryPayload(final int i) {
386         synchronized (this) {
387             // Pull them both
388             final IBaseDataObject oldTop = getPayload(0);
389             final IBaseDataObject p = getPayload(i);
390             // Swap them
391             this.payloadList.set(0, p);
392             this.payloadList.set(i, oldTop);
393             // switch logger context
394             MDC.put(MDCConstants.SHORT_NAME, p.shortName());
395         }
396     }

398     /**
399      * Do work now that we have arrived at the specified place
400      * 
401      * @param place the place we are asking to work for us
402      * @param payloadListArg list of IBaseDataObject for the place to operate on
403      * @return list of &quot;sprouted&quot; payloads
404      */
405     @SuppressWarnings("MemberName")
406     protected List<IBaseDataObject> atPlaceHD(final IServiceProviderPlace place, final List<IBaseDataObject> payloadListArg) {
407         MDC.put(MDCConstants.SERVICE_LOCATION, place.toString());
408         logger.debug("In atPlaceHD {} with {} payload items", place, payloadListArg.size());
409 
410         List<IBaseDataObject> ret = Collections.emptyList();
411 
412         try (TimedResource tr = resourceWatcherStart(place)) {
413             assert tr != null; // to silence an unused resource warning
414 
415             // Process and get back a list of sprouted payloads
416             lastPlaceProcessed = place.getDirectoryEntry().getKey();
417 
418             if (moveErrorsOccurred > 0) {
419                 addMoveErrorCount(payloadListArg);
420             }
421 
422             ret = place.agentProcessHeavyDuty(payloadListArg);
423 
424             for (Iterator<IBaseDataObject> it = ret.iterator(); it.hasNext();) {
425                 final IBaseDataObject ibdo = it.next();
426                 if (ibdo == null) {
427                     logger.error("{} violated contract and returned null IBaseDataObject. Child counts and IDs may be inconsistent.", place);
428                     it.remove();
429                 }
430             }
431 
432             if (moveErrorsOccurred > 0) {
433                 deleteMoveErrorCount(payloadListArg);
434             }
435 
436             logger.debug("done with agentProcessHD for {} with {} sprouted results along for the ride", place, ret.size());
437         } catch (Throwable problem) {
438             logger.warn("**{} caught {} with {} payloads", place, problem, payloadListArg.size(), problem);
439             // We don't know here which one of the items on the list
440             // caused the exception so we are going to error them all
441             // If place providers don't catch their own exceptions
442             // we are conservative in what we think is safe to do here
443             for (final IBaseDataObject p : payloadListArg) {
444                 p.addProcessingError("agentProcessHeavyDury(" + place + "): " + problem);
445                 p.replaceCurrentForm(MobileAgent.ERROR_FORM);
446             }
447         } finally {
448             if (!(place instanceof EmptyFormPlace)) {
449                 for (final IBaseDataObject p : payloadListArg) {
450                     if (p.currentFormSize() == 0) {
451                         logger.error("Place {} left an empty form stack, changing it to ERROR", place);
452                         p.addProcessingError(place + " left an empty form stack");
453                         p.pushCurrentForm(ERROR_FORM);
454                     }
455                 }
456             }
457             MDC.remove(MDCConstants.SERVICE_LOCATION);
458             checkInterrupt(place);
459         }
460 
461         return ret;
462     }

520     /**
521      * Setup the parallel type set tracking variable for a possible new payload
522      */
523     protected void setParallelTrackingInfoFor(final IBaseDataObject d) {
524 
525         // Clear out current value
526         clearParallelTrackingInfo();
527 
528         // Look at history from the tail backwards in time and
529         // while looking at the same parallelType serviceType
530         // add values to the visitedPlaces tracking variable.
531         // We need to do this because this data object may have
532         // been processed through some of them as a slug and it
533         // will not have filled in the visitedPlace properly.
534         final List<String> history = d.transformHistory();
535         int lastParallelType = -1;
536         for (int i = history.size() - 1; i >= 0; i--) {
537             final String key = history.get(i);
538             final int typeSet = typeLookup(KeyManipulator.getServiceType(key));
539             if (lastParallelType == -1 && isParallelServiceType(typeSet)) {
540                 lastParallelType = typeSet;
541             }
542             if (typeSet != lastParallelType) {
543                 break;
544             }
545             addParallelTrackingInfo(KeyManipulator.getServiceName(key));
546         }
547     }

549     /**
550      * To string method useful from the Namespace when bound there
551      */
552     @Override
553     public String toString() {
554         if (isZombie()) {
555             return "Closed";
556         } else if (!isInUse()) {
557             return "Idle";
558         }
559 
560         String sn = null;
561         int sz = 0;
562         if (this.payloadList != null && !this.payloadList.isEmpty()) {
563             // Avoid synchronizing this [don't call getPayload()]
564             try {
565                 sn = this.payloadList.get(0).shortName();
566                 sz = this.payloadList.size();
567             } catch (Throwable ignored) {
568                 // empty catch block
569             }
570         }
571         if (sn == null) {
572             sn = "Missing payload";
573         }
574         return sn + "(" + sz + ") - " + lastPlaceProcessed;
575     }

1   package emissary.core;

3   import jakarta.annotation.Nullable;
4   
5   /**
6    * Information class about processing stages in the workflow
7    */
8   enum Stage {
9   
10      /*
11       * The order of these stage declarations is important, as the order of the declarations directly corresponds to the
12       * stage sequence order
13       */
14  
15      /** Prepare, coordinate idents */
16      STUDY(false),
17      /** Identification phase */
18      ID(false),
19      /** Coordinate processing */
20      COORDINATE(false),
21      /** Before transform hook */
22      PRETRANSFORM(true),
23      /** Transformation phase */
24      TRANSFORM(false),
25      /** After transform hook */
26      POSTTRANSFORM(true),
27      /** Analysis/metadata generation */
28      ANALYZE(true),
29      /** Verify prior to emitting output */
30      VERIFY(false),
31      /** Emit output */
32      IO(false),
33      /** Finish */
34      REVIEW(false);
35  
36      final boolean isParallel;
37  
38      /**
39       * Constructor, which specifies whether the specific stage support parallel operations
40       *
41       * @param isParallel flag indicating parallel processing support
42       */
43      Stage(boolean isParallel) {
44          this.isParallel = isParallel;
45      }
46  
47      /**
48       * Indicates whether the specific stage support parallel operations
49       *
50       * @return isParallel indicator
51       */
52      public boolean isParallel() {
53          return isParallel;
54      }
55  
56      /**
57       * Attempts to resolve the stage value by name
58       *
59       * @param name value to look up
60       * @return resolved Stage name or null if no matching value was found
61       */
62      @Nullable
63      public static Stage getByName(String name) {
64          try {
65              return Stage.valueOf(name);
66          } catch (IllegalArgumentException | NullPointerException e) {
67              return null;
68          }
69      }
70  
71      /**
72       * Attempts to resolve the stage value by ordinal position
73       *
74       * @param index value to look up
75       * @return resolved Stage, or null if no matching value was found
76       */
77      @Nullable
78      public static Stage getByOrdinal(final int index) {
79          if (index < 0 || index >= Stage.values().length) {
80              return null;
81          }
82          return Stage.values()[index];
83      }
84  
85      /**
86       * Attempts to resolve the stage value by ordinal position
87       *
88       * @param index value to look up
89       * @return resolved Stage name or null if no matching value was found
90       */
91  
92      public static String getStageName(final int index) {
93          Stage stage = getByOrdinal(index);
94          return stage == null ? "UNDEFINED" : stage.name();
95      }
96  
97      /**
98       * Return the stage following the provided stage
99       *
100      * @param current current stage
101      * @return the Stage following the current stage or null there is no next value
102      */
103     public static Stage nextStageAfter(final Stage current) {
104         final int nextIndex = current.ordinal() + 1;
105         return getByOrdinal(nextIndex);
106     }
107 
108     /**
109      * Return the stage following the named stage
110      *
111      * @param name stage to look up
112      * @return the Stage following the name stage or null there is no next value
113      */
114     @Nullable
115     public static Stage nextStageAfter(final String name) {
116         Stage current = Stage.getByName(name);
117         if (current == null) {
118             return null;
119         }
120         final int nextIndex = current.ordinal() + 1;
121         return getByOrdinal(nextIndex);
122     }
123 
124     /**
125      * Resolves a Stage from a given ordinal position and indicates whether is supports parallel operations
126      *
127      * @param index ordinal position for the Stage
128      * @return value indicating whether the Stage supports parallel processing. Returns false if the specified index is
129      *         invalid.
130      */
131     public static boolean isParallelStage(final int index) {
132         Stage stage = getByOrdinal(index);
133         return stage != null && stage.isParallel();
134     }
135 
136 }

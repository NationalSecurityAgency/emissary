20  /**
21   * Track mobile agents and take action on suspicious behavior
22   */
23  public class Sentinel implements Runnable {
24  
25      protected static final Logger logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
26  
27      public static final String DEFAULT_NAMESPACE_NAME = "Sentinel";
28  
29      // protocols contain an action to perform when the set of rule conditions are met
30      protected final Set<Protocol> protocols = new LinkedHashSet<>();
31  
32      // the default configuration Sentinel.cfg
33      protected Configurator config;
34  
35      // how many minutes to sleep before checking protocols
36      protected long pollingInterval = 5;
37  
38      // Loop control
39      protected boolean timeToQuit = false;
40  
41      // turn on/off sentinel
42      protected boolean enabled = false;
43  
44      /**
45       * Create a Sentinel - set it running and bind into the {@link Namespace}
46       */
47      @SuppressWarnings("ThreadPriorityCheck")
48      public Sentinel() {
49          configure();
50          if (this.enabled) {
51              final Thread thread = new Thread(this, DEFAULT_NAMESPACE_NAME);
52              thread.setPriority(Thread.NORM_PRIORITY);
53              thread.setDaemon(true);
54              Namespace.bind(DEFAULT_NAMESPACE_NAME, this);
55              thread.start();
56          } else {
57              logger.info("Sentinel is disabled");
58          }
59      }
60  
61      /**
62       * Start up the Sentinel thread
63       */
64      public static void start() {
65          new Sentinel();
66      }
67  
68      /**
69       * Lookup the default Sentinel in the {@link Namespace}
70       *
71       * @return The registered Sentinel
72       */
73      public static Sentinel lookup() throws NamespaceException {
74          return (Sentinel) Namespace.lookup(DEFAULT_NAMESPACE_NAME);
75      }
76  
77      /**
78       * Get the currently configured polling interval
79       *
80       * @return the currently configured polling interval
81       */
82      public long getPollingInterval() {
83          return pollingInterval;
84      }
85  
86      /**
87       * Safely stop the monitoring Thread
88       */
89      public void quit() {
90          logger.info("Stopping Sentinel...");
91          this.timeToQuit = true;
92          ThreadUtils.findThreadsByName(DEFAULT_NAMESPACE_NAME).forEach(Thread::interrupt);
93      }
94  
95      /**
96       * Runnable interface where we get to monitor stuff
97       */
98      @Override
99      public void run() {
100         logger.info("Sentinel is watching");
101         while (!this.timeToQuit) {
102             // Delay this loop
103             try {
104                 Thread.sleep(TimeUnit.MINUTES.toMillis(pollingInterval));
105                 logger.debug("Sentinel is still watching");
106                 protocols.forEach(this::run);
107             } catch (InterruptedException ignore) {
108                 Thread.currentThread().interrupt();
109             }
110         }
111         Namespace.unbind(DEFAULT_NAMESPACE_NAME);
112         logger.info("Sentinel stopped");
113     }
114 
115     protected void run(final Protocol protocol) {
116         try {
117             protocol.run();
118         } catch (IOException e) {
119             logger.error("There was an error running protocol", e);
120         }
121     }
122 
123     @Override
124     public String toString() {
125         return "Watching agents with " + protocols;
126     }
127 
128     /**
129      * Get the Configurator
130      */
131     protected void configure() {
132         try {
133             this.config = ConfigUtil.getConfigInfo(Sentinel.class);
134             init();
135         } catch (IOException e) {
136             logger.warn("Cannot read Sentinel.cfg, taking default values");
137         }
138     }
139 
140     /**
141      * Initialize Protocols
142      */
143     protected void init() {
144         this.enabled = config.findBooleanEntry("ENABLED", false);
145         if (this.enabled) {
146             this.pollingInterval = this.config.findIntEntry("POLLING_INTERVAL_MINUTES", 5);
147 
148             logger.trace("Sentinel protocols initializing...");
149             for (String protocolConfig : this.config.findEntries("PROTOCOL")) {
150                 try {
151                     Protocol protocol = ProtocolFactory.get(protocolConfig);
152                     if (protocol.isEnabled()) {
153                         logger.debug("Sentinel protocol initialized {}", protocol);
154                         this.protocols.add(protocol);
155                     } else {
156                         logger.debug("Sentinel protocol disabled {}", protocol);
157                     }
158                 } catch (RuntimeException e) {
159                     logger.warn("Unable to configure Sentinel Protocol[{}]: {}", protocolConfig, e.getMessage());
160                 }
161             }
162             if (this.protocols.isEmpty()) {
163                 logger.warn("Sentinel initialization failed due to no protocols found, disabling");
164                 this.enabled = false;
165             } else {
166                 logger.info("Sentinel initialized protocols {}", protocols);
167             }
168         }
169     }

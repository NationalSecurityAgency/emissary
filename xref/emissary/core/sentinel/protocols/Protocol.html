26  /**
27   * This protocol buckets places that are running in mobile agents and then looks at max and min time in place and the
28   * number of agents that are potentially "stuck." After places are bucketed, the place stats are run against the
29   * configured rules to determine if all conditions are met. Once all rule conditions are met, then the configured action
30   * will be triggered, i.e. log/notify.
31   */
32  public class Protocol {
33  
34      private static final Logger logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
35  
36      protected Configurator config;
37      protected boolean enabled = false;
38      protected final Map<String, Rule> rules = new ConcurrentHashMap<>();
39      protected Action action;
40  
41      Protocol() {}
42  
43      public Protocol(String conf) {
44          configure(conf);
45      }
46  
47      public boolean isEnabled() {
48          return enabled;
49      }
50  
51      /**
52       * Run the configured rules over the watched mobile-agents
53       */
54      public void run(Map<String, Sentinel.Tracker> trackers) {
55  
56          Map<String, PlaceAgentStats> placeAgentStats = generatePlaceAgentStats(trackers);
57          if (!placeAgentStats.isEmpty()) {
58              logger.debug("Running rules on agents {}", placeAgentStats);
59              if (rules.values().stream().allMatch(rule -> rule.condition(placeAgentStats.values()))) {
60                  logger.warn("Sentinel rules matched -- {}", rules.values());
61                  action.trigger(trackers);
62              }
63          }
64      }
65  
66      /**
67       * Get the Configurator
68       *
69       * @param conf the location of the configuration file
70       */
71      protected void configure(String conf) {
72          try {
73              this.config = ConfigUtil.getConfigInfo(conf);
74              init(this.config);
75          } catch (IOException e) {
76              logger.warn("Cannot read {}, skipping!!", conf);
77          }
78      }
79  
80      /**
81       * Initialize rule set and action
82       */
83      protected void init(Configurator config) {
84          this.enabled = config.findBooleanEntry("ENABLED", false);
85          if (enabled) {
86  
87              this.action = (Action) Factory.create(config.findStringEntry("ACTION", Notify.class.getName()));
88  
89              logger.trace("Loading rules...");
90              for (String ruleId : config.findEntries("RULE_ID")) {
91                  try {
92                      if (this.rules.containsKey(ruleId)) {
93                          logger.warn("Sentinel rule with ID[{}] already exists, this may result in unexpected behavior", ruleId);
94                      }
95                      Map<String, String> map = config.findStringMatchMap(ruleId + "_");
96                      String rule = map.getOrDefault("RULE", AllMaxTime.class.getName());
97                      Rule../../../../emissary/core/sentinel/protocols/rules/Rule.html#Rule">Rule ruleImpl = (Rule) Factory.create(rule, ruleId, validate(map.get("PLACE_MATCHER")), map.get("TIME_LIMIT_MINUTES"),
98                              map.get("PLACE_THRESHOLD"));
99                      logger.debug("Sentinel loaded rule[{}] - {}", ruleId, ruleImpl);
100                     this.rules.put(ruleId, ruleImpl);
101                 } catch (Exception e) {
102                     logger.warn("Sentinel rule[{}] is invalid: {}", ruleId, e.getMessage());
103                 }
104             }
105 
106             // if no rules then disable protocol
107             if (this.rules.isEmpty()) {
108                 this.enabled = false;
109             }
110         }
111     }

113     /**
114      * Validate the place exists in the {@link Namespace}
115      *
116      * @param place the name of the place
117      * @throws NamespaceException if the directory place does not exist
118      * @throws IllegalStateException if the place cannot be found
119      */
120     protected String validate(String place) throws NamespaceException {
121         // validate that the place exists
122         DirectoryPlace directoryPlace = Namespace.lookup(DirectoryPlace.class).iterator().next();
123         if (directoryPlace.getEntries().stream()
124                 .noneMatch(entry -> KeyManipulator.getServiceClassname(entry.getFullKey()).matches(place))) {
125             throw new IllegalStateException("Place not found in the directory");
126         }
127         return place;
128     }
129 
130     protected Map<String, PlaceAgentStats> generatePlaceAgentStats(Map<String, Sentinel.Tracker> trackers) {
131         Map<String, PlaceAgentStats> placeAgentStats = new ConcurrentHashMap<>();
132         for (Sentinel.Tracker tracker : trackers.values()) {
133             String placeKey = tracker.getPlaceName();
134             if (StringUtils.isNotBlank(placeKey)) {
135                 placeAgentStats.put(placeKey, placeAgentStats.getOrDefault(placeKey, new PlaceAgentStats(placeKey)).update(tracker.getTimer()));
136             }
137         }
138         return placeAgentStats;
139     }

141     @Override
142     public String toString() {
143         return new StringJoiner(", ", "{", "}")
144                 .add("\"rules\":" + rules.values())
145                 .add("\"action\":" + action)
146                 .toString();
147     }

149     public static class PlaceAgentStats {
150 
151         private final String place;
152         private int count;
153         private long maxTimeInPlace = -1;
154         private long minTimeInPlace = -1;
155 
156         public PlaceAgentStats(String place) {
157             this.place = place;
158         }
159 
160         public String getPlace() {
161             return place;
162         }
163 
164         public int getCount() {
165             return count;
166         }
167 
168         public long getMaxTimeInPlace() {
169             return maxTimeInPlace;
170         }
171 
172         public long getMinTimeInPlace() {
173             return minTimeInPlace;
174         }
175 
176         public PlaceAgentStats update(long timer) {
177             this.count++;
178             this.minTimeInPlace = this.minTimeInPlace < 0 ? timer : Math.min(this.minTimeInPlace, timer);
179             this.maxTimeInPlace = Math.max(this.maxTimeInPlace, timer);
180             return this;
181         }
182     }

14  public class AnyMaxTime extends Rule {
15  
16      protected static final Logger logger = LoggerFactory.getLogger(MethodHandles.lookup().lookupClass());
17  
18      public AnyMaxTime(String name, String place, long timeLimit, double threshold) {
19          super(name, place, timeLimit, threshold);
20      }
21  
22      public AnyMaxTime(String name, String place, String timeLimit, String threshold) {
23          super(name, place, timeLimit, threshold);
24      }
25  
26      /**
27       * Check to see if ANY places in mobile agents are over the configured time limit
28       *
29       * @param placeAgentStats the stats of a place that is currently processing
30       * @return true if any places in mobile agents are over the configured time limit, false otherwise
31       */
32      @Override
33      protected boolean overTimeLimit(Collection<Protocol.PlaceAgentStats> placeAgentStats) {
34          long maxTimeInPlace = placeAgentStats.stream().mapToLong(Protocol.PlaceAgentStats::getMaxTimeInPlace).max().orElse(0);
35          logger.debug("Testing timeLimit for place={}, maxTime={}, timeLimit={}", place, maxTimeInPlace, timeLimit);
36          return maxTimeInPlace >= this.timeLimit;
37      }

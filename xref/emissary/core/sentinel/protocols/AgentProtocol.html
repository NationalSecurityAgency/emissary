26  /**
27   * This protocol buckets places that are running in mobile agents and then looks at max and min time in place and the
28   * number of agents that are potentially "stuck." After places are bucketed, the place stats are run against the
29   * configured rules to determine if all conditions are met. Once all rule conditions are met, then the configured action
30   * will be triggered, i.e. log/notify.
31   */
32  public class AgentProtocol extends Protocol {
33  
34      // key: agent name, value: how long Sentinel has observed the mobile agent
35      protected final Map<String, AgentTracker> trackers = new ConcurrentHashMap<>();
36  
37      public AgentProtocol() {}
38  
39      public AgentProtocol(Configurator config) {
40          super(config);
41      }
42  
43      @Override
44      protected Rule<PlaceAgentStats> getRule(final String ruleId) throws IOException {
45          try {
46              Map<String, String> map = config.findStringMatchMap(ruleId + "_");
47              String rule = map.getOrDefault("RULE", AllMaxTime.class.getName());
48              return (Rule) Factory.create(rule, ruleId, validate(map.get("PLACE_MATCHER")), map.get("TIME_LIMIT_MINUTES"),
49                      map.get("PLACE_THRESHOLD"));
50          } catch (NamespaceException e) {
51              throw new IOException(e);
52          }
53      }
54  
55      /**
56       * Validate the place exists in the {@link Namespace}
57       *
58       * @param place the name of the place
59       * @throws NamespaceException if the directory place does not exist
60       * @throws IllegalStateException if the place cannot be found
61       */
62      protected String validate(final String place) throws NamespaceException {
63          // validate that the place exists
64          DirectoryPlace directoryPlace = Namespace.lookup(DirectoryPlace.class).iterator().next();
65          if (directoryPlace.getEntries().stream()
66                  .noneMatch(entry -> KeyManipulator.getServiceClassname(entry.getFullKey()).matches(place))) {
67              throw new IllegalStateException("Place not found in the directory");
68          }
69          return place;
70      }
71  
72      /**
73       * Checks to see if the mobile agents are processing the same data since the last polling event
74       *
75       * @throws IOException if there is a problem looking up resources in the {@link Namespace}
76       */
77      @Override
78      public void run() throws IOException {
79          try {
80              List<String> agentKeys = Namespace.keySet().stream()
81                      .filter(k -> k.startsWith(MobileAgentFactory.AGENT_NAME))
82                      .sorted()
83                      .collect(Collectors.toList());
84              for (String agentKey : agentKeys) {
85                  updateTracker(agentKey);
86              }
87              runRules(trackers);
88          } catch (NamespaceException e) {
89              throw new IOException("There was an issue running protocol", e);
90          }
91      }
92  
93      /**
94       * Checks to see if the mobile agent is processing the same data since the last polling event
95       *
96       * @param agentKey the agent key, i.e. MobileAgent-01
97       * @throws NamespaceException if there is a problem looking up resources in the {@link Namespace}
98       */
99      protected void updateTracker(final String agentKey) throws NamespaceException {
100         logger.trace("Searching for agent [{}]", agentKey);
101         IMobileAgent mobileAgent = (IMobileAgent) Namespace.lookup(agentKey);
102         AgentTracker trackedAgent = trackers.computeIfAbsent(mobileAgent.getName(), AgentTracker::new);
103         if (mobileAgent.isInUse()) {
104             if (!Objects.equals(mobileAgent.agentId(), trackedAgent.getAgentId())
105                     || !Objects.equals(mobileAgent.getLastPlaceProcessed(), trackedAgent.getDirectoryEntryKey())) {
106                 trackedAgent.clear();
107                 trackedAgent.setAgentId(mobileAgent.agentId());
108                 trackedAgent.setDirectoryEntryKey(mobileAgent.getLastPlaceProcessed());
109             }
110             trackedAgent.incrementTimer(Sentinel.lookup().getPollingInterval());
111             logger.trace("Agent acquired {}", trackedAgent);
112         } else {
113             trackedAgent.clear();
114             logger.trace("Agent not in use [{}]", agentKey);
115         }
116     }
117 
118     /**
119      * Run the configured rules over the watched mobile-agents
120      */
121     public void runRules(Map<String, AgentTracker> trackers) {
122 
123         Map<String, PlaceAgentStats> placeAgentStats = generatePlaceAgentStats(trackers);
124         if (!placeAgentStats.isEmpty()) {
125             logger.debug("Running rules on agents {}", placeAgentStats);
126             if (rules.values().stream().allMatch(rule -> rule.condition(placeAgentStats.values()))) {
127                 logger.warn("Sentinel rules matched -- {}", rules.values());
128                 action.trigger((Map) trackers);
129             }
130         }
131     }
132 
133     protected Map<String, PlaceAgentStats> generatePlaceAgentStats(Map<String, AgentTracker> trackers) {
134         Map<String, PlaceAgentStats> placeAgentStats = new ConcurrentHashMap<>();
135         for (AgentTracker tracker : trackers.values()) {
136             String placeKey = tracker.getPlaceName();
137             if (StringUtils.isNotBlank(placeKey)) {
138                 placeAgentStats.put(placeKey, placeAgentStats.getOrDefault(placeKey, new PlaceAgentStats(placeKey)).update(tracker.getTimer()));
139             }
140         }
141         return placeAgentStats;
142     }
143 
144     @Override
145     public String toString() {
146         return new StringJoiner(", ", "{", "}")
147                 .add("\"rules\":" + rules.values())
148                 .add("\"action\":" + action)
149                 .toString();
150     }
151 
152     public static class PlaceAgentStats {
153 
154         private final String place;
155         private int count;
156         private long maxTimeInPlace = -1;
157         private long minTimeInPlace = -1;
158 
159         public PlaceAgentStats(String place) {
160             this.place = place;
161         }
162 
163         public String getPlace() {
164             return place;
165         }
166 
167         public int getCount() {
168             return count;
169         }
170 
171         public long getMaxTimeInPlace() {
172             return maxTimeInPlace;
173         }
174 
175         public long getMinTimeInPlace() {
176             return minTimeInPlace;
177         }
178 
179         public PlaceAgentStats update(long timer) {
180             this.count++;
181             this.minTimeInPlace = this.minTimeInPlace < 0 ? timer : Math.min(this.minTimeInPlace, timer);
182             this.maxTimeInPlace = Math.max(this.maxTimeInPlace, timer);
183             return this;
184         }
185     }

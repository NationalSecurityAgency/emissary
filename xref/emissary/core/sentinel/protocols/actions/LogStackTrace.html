19  /**
20   * Log the problem agents/threads
21   */
22  public class LogStackTrace extends Action {
23  
24      private static final Logger SENTINEL_LOG = LoggerFactory.getLogger("sentinel");
25  
26      @Override
27      public void trigger(final Map<String, Tracker> trackers) {
28          SENTINEL_LOG.warn("Sentinel detected possible locked agents -- {}", format(trackers));
29          for (AgentTracker agent : toAgentTrackers(trackers)) {
30              try {
31                  final IMobileAgent mobileAgent = (IMobileAgent) Namespace.lookup(agent.getAgentName());
32                  final Thread t = ThreadUtils.findThreadById(mobileAgent.getThreadId());
33                  MDC.put(MDCConstants.SHORT_NAME, mobileAgent.getShortName());
34                  MDC.put(MDCConstants.SERVICE_LOCATION, agent.getDirectoryEntryKey());
35                  /*
36                   * mobileAgent.currentForm() is not necessarily the form at the time the payload arrived at this place, since some
37                   * places change the form even before processing, but it's the best we have to identify the payload being processed
38                   */
39                  SENTINEL_LOG.info("In agent {} with currentForm {} for {} minute(s)\n\t{}", agent.getAgentName(), mobileAgent.getPayloadCurrentForm(),
40                          agent.getTimer(),
41                          Arrays.stream(t.getStackTrace())
42                                  .map(StackTraceElement::toString)
43                                  .collect(Collectors.joining("\n\t")));
44                  MDC.remove(MDCConstants.SHORT_NAME);
45                  MDC.remove(MDCConstants.SERVICE_LOCATION);
46              } catch (NamespaceException e) {
47                  logger.error("Could not fetch agent {}", agent, e);
48              }
49          }
50      }
51  }

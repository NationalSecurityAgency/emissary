10  import java.util.stream.Collectors;

12  /**
13   * Attempts to recover the mobile agents by interrupting the thread
14   */
15  public class Recover extends Action {
16  
17      @Override
18      public void trigger(Map<String, Tracker> tracker) {
19  
20          logger.warn("Sentinel detected locked agents, attempting recovery...");
21          List<String> agentNames = tracker.values().stream()
22                  .filter(AgentTracker.class::isInstance)
23                  .map(t -> ((AgentTracker) t).getAgentName())
24                  .sorted()
25                  .collect(Collectors.toList());
26  
27          for (String agentName : agentNames) {
28              try {
29                  IMobileAgent mobileAgent = (IMobileAgent) Namespace.lookup(agentName);
30                  logger.warn("Sentinel attempting recovery for {}", agentName);
31                  mobileAgent.interrupt();
32              } catch (Exception e) {
33                  throw new IllegalStateException("Recovery unavailable", e);
34              }
35          }
36      }
37  
38  }

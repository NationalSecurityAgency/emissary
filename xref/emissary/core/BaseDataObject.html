204     /**
205      * <p>
206      * Determine what state we're in with respect to the byte[] of data vs a channel.
207      * </p>
208      * 
209      * <p>
210      * Not exposed publicly as consumers should be moving to channels, meaning ultimately the states will be simply either a
211      * channel factory exists or does not exist.
212      * </p>
213      * 
214      * <p>
215      * Consumers should not modify their behaviour based on the state of the BDO, if they're being modified to handle
216      * channels, they should only handle channels, not both channels and byte[].
217      * </p>
218      * 
219      * @return the {@link DataState} of this BDO
220      */
221     protected DataState getDataState() {
222         if (theData == null) {
223             if (seekableByteChannelFactory == null) {
224                 return DataState.NO_DATA;
225             } else {
226                 return DataState.CHANNEL_ONLY;
227             }
228         } else {
229             if (seekableByteChannelFactory == null) {
230                 return DataState.BYTE_ARRAY_ONLY;
231             } else {
232                 return DataState.BYTE_ARRAY_AND_CHANNEL;
233             }
234         }
235     }

399     /**
400      * <p>
401      * Return BaseDataObjects byte array OR as much as we can from the reference to the data up to MAX_BYTE_ARRAY_SIZE.
402      * </p>
403      * 
404      * <p>
405      * Data returned from a backing Channel will be truncated at {@link BaseDataObject#MAX_BYTE_ARRAY_SIZE}. Using
406      * channel-related methods is now preferred to allow handling of larger objects
407      * </p>
408      * 
409      * <p>
410      * <b>WARNING</b>: There is no way for the caller to know whether the data being returned is the direct array held in
411      * memory, or a copy of the data from a byte channel factory, so the returned byte array should be treated as live and
412      * not be modified.
413      * </p>
414      * 
415      * @see #getChannelFactory()
416      * @return the data as a byte array
417      */
418     @Nullable
419     @Override
420     @SuppressWarnings("UnnecessaryDefaultInEnumSwitch")
421     public byte[] data() {
422         switch (getDataState()) {
423             case BYTE_ARRAY_AND_CHANNEL:
424                 throw new IllegalStateException(String.format(INVALID_STATE_MSG, shortName()));
425             case BYTE_ARRAY_ONLY:
426                 return theData;
427             case CHANNEL_ONLY:
428                 // Max size here is slightly less than the true max size to avoid memory issues
429                 return SeekableByteChannelHelper.getByteArrayFromBdo(this, MAX_BYTE_ARRAY_SIZE);
430             case NO_DATA:
431             default:
432                 return null; // NOSONAR maintains backwards compatibility
433         }
434     }

445     /**
446      * <p>
447      * Set new data on the BDO, using a range of the provided byte array. This will remove the reference to any byte channel
448      * factory that backs this BDO so be careful!
449      * </p>
450      * 
451      * <p>
452      * Limited in size to 2^31. Use channel-based methods for larger data.
453      * </p>
454      * 
455      * @param newData containing the source of the new data
456      * @param offset where to start copying from
457      * @param length how much to copy
458      * @see #setChannelFactory(SeekableByteChannelFactory)
459      */
460     @Override
461     public void setData(@Nullable final byte[] newData, final int offset, final int length) {
462         this.seekableByteChannelFactory = null;
463         if (length <= 0 || newData == null) {
464             this.theData = new byte[0];
465         } else {
466             this.theData = new byte[length];
467             System.arraycopy(newData, offset, this.theData, 0, length);
468         }
469     }

504     /**
505      * Fetch the size of the payload. Prefer to use: {@link #getChannelSize}
506      * 
507      * @return the length of theData, or the size of the seekable byte channel up to
508      *         {@link BaseDataObject#MAX_BYTE_ARRAY_SIZE}.
509      */
510     @Override
511     @SuppressWarnings("UnnecessaryDefaultInEnumSwitch")
512     public int dataLength() {
513         switch (getDataState()) {
514             case BYTE_ARRAY_AND_CHANNEL:
515                 throw new IllegalStateException(String.format(INVALID_STATE_MSG, shortName()));
516             case BYTE_ARRAY_ONLY:
517                 return ArrayUtils.getLength(theData);
518             case CHANNEL_ONLY:
519                 try {
520                     return (int) Math.min(getChannelSize(), MAX_BYTE_ARRAY_SIZE);
521                 } catch (final IOException ioe) {
522                     logger.error("Couldn't get size of channel on object {}", shortName(), ioe);
523                     return 0;
524                 }
525             case NO_DATA:
526             default:
527                 return 0;
528         }
529     }

536     /**
537      * Construct the shortname
538      */
539     private String makeShortName() {
540         /*
541          * using the file object works for most cases. It fails on the unix side if it is given a valid Windows path.
542          */
543         // File file = new File( theFileName );
544         // return file.getName();
545         // so..... we'll have to perform the check ourselves ARRRRRRRRRGH!!!!
546         final int unixPathIndex = this.theFileName.lastIndexOf("/");
547         if (unixPathIndex >= 0) {
548             return this.theFileName.substring(unixPathIndex + 1);
549         }
550         // check for windows path
551         final int windowsPathIndex = this.theFileName.lastIndexOf("\\");
552         if (windowsPathIndex >= 0) {
553             return this.theFileName.substring(windowsPathIndex + 1);
554         }
555 
556         return this.theFileName;
557     }

621     @Override
622     public int deleteCurrentForm(final String form) {
623         int count = 0;
624 
625         if (this.currentForm == null || this.currentForm.isEmpty()) {
626             return count;
627         }
628 
629         // Remove all matching
630         for (final Iterator<String> i = this.currentForm.iterator(); i.hasNext();) {
631             final String val = i.next();
632             if (val.equals(form)) {
633                 i.remove();
634                 count++;
635             }
636         }
637         return count;
638     }

675     @Override
676     public int pushCurrentForm(final String newForm) {
677         if (newForm == null) {
678             throw new IllegalArgumentException("caller attempted to push a null form value");
679         } else if (!PayloadUtil.isValidForm(newForm)) {
680             // If there is a key separator in the form, then throw an error log as this will cause issues in routing
681             logger.error("INVALID FORM: The form can only contain a-z, A-Z, 0-9, '-', '_', '()', '/', '+'. Given form: {}", newForm);
682         }
683 
684         checkForAndLogDuplicates(newForm, "pushCurrentForm");
685         return addCurrentFormAt(0, newForm);
686     }

693     @Override

879     /**
880      * Merge in new parameters using the specified policy to determine whether to keep all values, unique values, or prefer
881      * existing values
882      *
883      * @param m map of new parameters
884      * @param policy the merge policy
885      */
886     @Override
887     public void putParameters(final Map<? extends String, ? extends Object> m, final MergePolicy policy) {
888         for (final Map.Entry<? extends String, ? extends Object> entry : m.entrySet()) {
889             final String name = entry.getKey();
890 
891             if ((policy == MergePolicy.KEEP_EXISTING) && this.parameters.containsKey(name)) {
892                 continue;
893             }
894 
895             final Object value = entry.getValue();
896 
897             if ((policy == MergePolicy.DROP_EXISTING)) {
898                 // store the provided value for this key, discarding any previously-stored value
899                 setParameter(name, value);
900                 continue;
901             }
902 
903             if (value instanceof Iterable) {
904                 for (final Object v : (Iterable<?>) value) {
905                     if (policy == MergePolicy.KEEP_ALL || policy == MergePolicy.KEEP_EXISTING) {
906                         this.parameters.put(name, v);
907                     } else if (policy == MergePolicy.DISTINCT) {
908                         if (!this.parameters.containsEntry(name, v)) {
909                             this.parameters.put(name, v);
910                         }
911                     } else {
912                         throw new IllegalStateException("Unhandled parameter merge policy " + policy + " for " + name);
913                     }
914                 }
915             } else {
916                 if (policy == MergePolicy.KEEP_ALL || policy == MergePolicy.KEEP_EXISTING) {
917                     this.parameters.put(name, value);
918                 } else if (policy == MergePolicy.DISTINCT) {
919                     if (!this.parameters.containsEntry(name, value)) {
920                         this.parameters.put(name, value);
921                     }
922                 } else {
923                     throw new IllegalStateException("Unhandled parameter merge policy " + policy + " for " + name);
924                 }
925             }
926         }
927     }

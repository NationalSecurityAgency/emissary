86      /**
87       * This method compares two lists of IBaseDataObject's and adds any differences to the provided string list.
88       *
89       * @param ibdoList1 the first list of IBaseDataObjects to compare.
90       * @param ibdoList2 the second list of IBaseDataObjects to compare.
91       * @param identifier a string that helps identify the context of comparing these two list of IBaseDataObjects.
92       * @param differences the string list differences are to be added to.
93       * @param options {@link DiffCheckConfiguration} containing config to specify whether to check data etc.
94       */
95      public static void diff(final List<IBaseDataObject> ibdoList1, final List<IBaseDataObject> ibdoList2,
96              final String identifier, final List<String> differences, final DiffCheckConfiguration options) {
97          Validate.notNull(identifier, ID_NOT_NULL_MSG);
98          Validate.notNull(differences, DIFF_NOT_NULL_MSG);
99  
100         final int ibdoList1Size = (ibdoList1 == null) ? 0 : ibdoList1.size();
101         final int ibdoList2Size = (ibdoList2 == null) ? 0 : ibdoList2.size();
102 
103         if (ibdoList1Size != ibdoList2Size) {
104             differences.add(String.format("%s%s: 1.s=%s 2.s=%s", identifier, ARE_NOT_EQUAL, ibdoList1Size, ibdoList2Size));
105         } else if (ibdoList1 != null && ibdoList2 != null) {
106             final List<String> childDifferences = new ArrayList<>();
107             for (int i = 0; i < ibdoList1.size(); i++) {
108                 childDifferences.clear();
109 
110                 diff(ibdoList1.get(i), ibdoList2.get(i), childDifferences, options);
111 
112                 final String prefix = identifier + " : " + i + " : ";
113                 while (!childDifferences.isEmpty()) {
114                     differences.add(prefix + childDifferences.remove(0)); // NOSONAR Used correctly
115                 }
116             }
117         }
118     }

120     /**
121      * This method compares two {@link SeekableByteChannelFactory} (SBCF) objects and adds any differences to the provided
122      * string list.
123      *
124      * @param sbcf1 the first SBCF to compare.
125      * @param sbcf2 the second SBCF to compare.
126      * @param identifier an identifier to describe the context of this SBCF comparison.
127      * @param differences the string list differences are to be added to.
128      */
129     public static void diff(final SeekableByteChannelFactory#SeekableByteChannelFactory">SeekableByteChannelFactory sbcf1, final SeekableByteChannelFactory sbcf2,
130             final String identifier, final List<String> differences) {
131         Validate.notNull(identifier, ID_NOT_NULL_MSG);
132         Validate.notNull(differences, DIFF_NOT_NULL_MSG);
133 
134         if (sbcf1 != null && sbcf2 != null) {
135             try (SeekableByteChannel sbc1 = sbcf1.create();
136                     SeekableByteChannel sbc2 = sbcf2.create();
137                     InputStream is1 = Channels.newInputStream(sbc1);
138                     InputStream is2 = Channels.newInputStream(sbc2)) {
139                 if (!IOUtils.contentEquals(is1, is2)) {
140                     differences.add(String.format("%s not equal. 1.cs=%s 2.cs=%s",
141                             identifier, sbc1.size(), sbc2.size()));
142                 }
143             } catch (IOException e) {
144                 differences.add(String.format("Failed to compare %s: %s", identifier, e.getMessage()));
145             }
146         } else if (sbcf1 == null && sbcf2 == null) {
147             // Do nothing as they are considered equal.
148         } else {
149             differences.add(String.format("%s not equal. sbcf1=%s sbcf2=%s", identifier, sbcf1, sbcf2));
150         }
151     }

153     /**
154      * This method compares two Objects and adds any differences to the provided string list.
155      *
156      * @param object1 the first Object to compare.
157      * @param object2 the second Object to compare.
158      * @param identifier an identifier to describe the context of this Object comparison.
159      * @param differences the string list differences are to be added to.
160      */
161     public static void diff(final Object object1, final Object object2, final String identifier,
162             final List<String> differences) {
163         Validate.notNull(identifier, ID_NOT_NULL_MSG);
164         Validate.notNull(differences, DIFF_NOT_NULL_MSG);
165 
166         if (!Objects.deepEquals(object1, object2)) {
167             differences.add(String.format("%s%s: %s : %s", identifier, ARE_NOT_EQUAL, object1, object2));
168         }
169     }

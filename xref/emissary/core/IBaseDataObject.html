365     /**
366      * Retrieve a specified metadata element as a string value
367      * 
368      * @param key name of the metadata element
369      * @param sep the separator for multivalued fields
370      * @return the string value or null if no such element
371      * @deprecated use {@link #getParameterAsConcatString(String, String)}
372      */
373     @Deprecated
374     @Nullable
375     default String getStringParameter(final String key, final String sep) {
376         final List<Object> obj = getParameter(key);
377         if (obj == null) {
378             return null;
379         } else if (obj.isEmpty()) {
380             return null;
381         } else if ((obj.size() == 1) && (obj.get(0) instanceof String)) {
382             return (String) obj.get(0);
383         } else if ((obj.size() == 1) && (obj.get(0) == null)) {
384             return null;
385         } else {
386             final StringBuilder sb = new StringBuilder();
387             for (final Object item : obj) {
388                 if (sb.length() > 0) {
389                     sb.append(sep);
390                 }
391                 sb.append(item);
392             }
393             return sb.toString();
394         }
395     }

397     /**
398      * Retrieve the Collection of metadata values identified by key where each element is converted to a string
399      *
400      * @param key name of the metadata element collection
401      * @return Collection of elements converted to strings
402      */
403     default Collection<String> getParameterAsStrings(final String key) {
404         final var obj = getParameter(key);
405         if (CollectionUtils.isEmpty(obj) || ((obj.size() == 1) && (obj.get(0) == null))) {
406             return Collections.emptyList();
407         } else if ((obj.size() == 1) && (obj.get(0) instanceof String)) {
408             return Collections.singletonList((String) obj.get(0));
409         } else {
410             return obj.stream().map(String::valueOf).collect(Collectors.toList());
411         }
412     }

432     /**
433      * Retrieve a specified metadata element as a string of concatenated values
434      *
435      * @param key name of the metadata element
436      * @param sep the separator for multivalued fields
437      * @return the string value or null if no such element
438      */
439     @Nullable
440     default String getParameterAsConcatString(final String key, final String sep) {
441         Collection<String> strings = getParameterAsStrings(key);
442         if (strings.stream().anyMatch(Objects::nonNull)) {
443             return String.join(sep, strings);
444         }
445         return null;
446     }

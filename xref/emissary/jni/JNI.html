82      public JNI(@Nullable final String theDir, final Configurator configG) {
83          if (theDir != null) {
84              try {
85                  this.theDir = (IDirectoryPlace) Namespace.lookup(theDir);
86              } catch (NamespaceException ne) {
87                  logger.debug("Cannot get directory using {}: {}", theDir, ne.getLocalizedMessage());
88              }
89          }
90  
91          if (this.theDir == null) {
92              try {
93                  this.theDir = DirectoryPlace.lookup();
94              } catch (EmissaryException ex) {
95                  logger.debug("Unable to lookup default directory", ex);
96              }
97          }
98  
99          this.configG = configG;
100         configurePlace();
101     }

103     /**
104      * Configure the place based on the current {@link #configG} setting.
105      */
106     private void configurePlace() {
107 
108         if (this.configG == null) {
109             return;
110         }
111 
112         List<String> parms = this.configG.findEntries("SHARED_PREFIX");
113         for (final String entry : parms) {
114             final int ndx = entry.indexOf(':');
115             if (ndx == -1) {
116                 logger.warn("Invalid SHARED_PREFIX: {}", entry);
117                 continue;
118             }
119 
120             final String arch = entry.substring(0, ndx);
121             String prefix = "";
122             if (ndx < entry.length() - 1) {
123                 prefix = entry.substring(ndx + 1);
124             }
125 
126             this.sharedPrefix.put(arch, prefix);
127 
128             // Get the osname-dependent SAVE_PATH
129             final List<String> iparms = this.configG.findEntries(arch + "_LIBRARY_SAVE_PATH");
130             if (CollectionUtils.isNotEmpty(iparms)) {
131                 this.savePath.put(arch, iparms.get(0));
132             }
133         }
134 
135         parms = this.configG.findEntries("SHARED_SUFFIX");
136         for (final String entry : parms) {
137             final int ndx = entry.indexOf(':');
138             if (ndx == -1) {
139                 logger.warn("Invalid SHARED_SUFFIX: {}", entry);
140                 continue;
141             }
142 
143             final String arch = entry.substring(0, ndx);
144             String suffix = "";
145             if (ndx < entry.length() - 1) {
146                 suffix = entry.substring(ndx + 1);
147             }
148             this.sharedSuffix.put(arch, suffix);
149         }
150 
151         parms = this.configG.findEntries("LIBRARY_VERSION");
152         for (final String entry : parms) {
153             final int ndx = entry.indexOf(':');
154             if (ndx == -1) {
155                 logger.warn("Invalid LIBRARY_VERSION: {}", entry);
156                 continue;
157             }
158 
159             final String lib = entry.substring(0, ndx);
160             String ver = "";
161             if (ndx < entry.length() - 1) {
162                 ver = entry.substring(ndx + 1);
163             }
164             this.libVersions.put(lib, ver);
165         }
166 
167         if (logger.isDebugEnabled()) {
168             logger.debug("JNI config save paths {}", this.savePath);
169             logger.debug("JNI config prefixes {}", this.sharedPrefix);
170             logger.debug("JNI config suffixes {}", this.sharedSuffix);
171             logger.debug("JNI config versions {}", this.libVersions);
172         }
173     }

175     /**
176      * Expand the root name of the library by the architecture and version constants.
177      */
178     private String expandLibraryName(final String name) {
179         final String osarch = System.getProperty("os.arch").replace(' ', '_');
180         final String osname = System.getProperty("os.name").replace(' ', '_');
181         // String osver = System.getProperty("os.version").replace(' ','_');
182         String libver = this.libVersions.get(name);
183         final String sep = "-";
184 
185         if (libver == null) {
186             libver = "1.0";
187         }
188 
189         return name + sep + // foo-
190                 osname + sep + // Solaris-
191                 osarch + sep + // sparc-
192                 // osver+sep+ // 2.x-
193                 libver; // v3.2
194     }

196     /**
197      * Expand the full library name with the way it is stored in the file system. This method expects something like
198      * foo-Solaris-sparc-2.x-v3.2 and returns libfoo-Solaris-sparc-2.x-v3.2.so
199      */
200     private String filesystemLibraryName(final String name) {
201         final String osname = System.getProperty("os.name").replace(' ', '_');
202         String prefix = this.sharedPrefix.get(osname);
203         String suffix = this.sharedSuffix.get(osname);
204 
205         if (prefix == null) {
206             prefix = "";
207         }
208 
209         if (suffix == null) {
210             suffix = "";
211         }
212 
213         return prefix + name + suffix;
214     }

216     /**
217      * Load a native library, finding it if it isn't already here. First try to load the library and catch any exception.
218      * Try to ask the directory place where the JniRepositoryPlace is and see if the repository has the library we are
219      * looking for. If it comes back, save it to disk in the agreed upon location and then load it in the regular way. Throw
220      * an UnsatisfiedLinkError if this doesn't work.
221      * <p>
222      * Note that System.loadLibrary expects the name as it comes back from expandLibraryName while the repository will
223      * respond to the name as it comes from filesystemLibraryName.
224      *
225      * @throws UnsatisfiedLinkError If it fails to load.
226      */
227     public void loadLibrary(final String lib) {
228 
229         final String libname = expandLibraryName(lib);
230         final String filename = filesystemLibraryName(libname);
231         final String osname = System.getProperty("os.name").replace(' ', '_');
232         final String theLocation = this.savePath.get(osname);
233         final String fullPathName = theLocation + File.separator + filename;
234 
235         logger.debug("In JNI.loadLibrary({})", lib);
236         logger.debug("loading library: {}", libname);
237 
238         // Try it on the LD_LIBRARY_PATH or equivalent
239         try {
240             System.loadLibrary(libname);
241             return;
242         } catch (UnsatisfiedLinkError e) {
243             final String syspath = System.getProperty("java.library.path", "<none>");
244             logger.debug("Unable to link local {} from incoming {} using system path {}", libname, lib, syspath, e);
245         }
246 
247 
248         // Try it in the save area, in case it's not on the LD_LIBRARY_PATH
249         try {
250             System.load(fullPathName);
251             return;
252         } catch (UnsatisfiedLinkError e) {
253             logger.debug("Unable to link abs path {} from incoming {}", fullPathName, lib, e);
254         }
255 
256         // Retrieve the File and dependencies.
257         // errorMsg valid only when return is false
258         final String[] errorMsg = new String[1];
259 
260         // Retrieve everything listed as a dependency of the requested library.
261         // If the library was obtained via System.loadLibrary() or System.load()
262         // above then we assume this has already been done (by a previous
263         // incantation of this routine) and not needed again. Note there is
264         // therefor NO versioning capability of the dependent libraries without
265         // jacking the version number for the actual library being requested.
266         if (!retrieveDependencies(lib, errorMsg)) {
267             logger.debug("Unable to retrieve dependencies:{}", errorMsg[0]);
268             throw new UnsatisfiedLinkError("Unable to retrieve dependencies for " + filename + " : " + errorMsg[0]);
269         }
270 
271         if (!retrieveFile(filename, errorMsg)) {
272             logger.debug("Unable to retrieve:{}", errorMsg[0]);
273             throw new UnsatisfiedLinkError("Unable to retrieve " + filename + " : " + errorMsg[0]);
274         }
275 
276         // Loadlib the file we just retrieved and saved
277         try {
278             System.load(fullPathName);
279             logger.debug("LINK SUCCESS for {}", fullPathName);
280         } catch (UnsatisfiedLinkError e) {
281             logger.debug("Unable to link retrieved {}:{}", fullPathName, e.getLocalizedMessage());
282 
283             // We have done all we can. Throw an exception and return
284             throw new UnsatisfiedLinkError("Cannot link with retrieved library " + fullPathName + ":" + e);
285         }
286     }

288     /**
289      * Retrieve all the dependencies of a given library. For example if the incoming libname is foo, then the full path of
290      * the lib might be libfoo-Solaris-:-2.x-v1.2.so and the dependencies would be listed in the config file as DEP_foo =
291      * "libdep1.so" DEP_foo = "libdep2.so"
292      */
293     private boolean retrieveDependencies(final String libname, final String[] errmsg) {
294         final List<String> deps = this.configG.findEntries("DEP_" + libname);
295 
296         // See if anything needs to be retrieved
297         if ((deps == null) || deps.isEmpty()) {
298             return true;
299         }
300 
301         // Use this string on error
302         final String myErr = libname + " dependencies are " + deps;
303 
304         for (final String file : deps) {
305             logger.debug("JNI: Retrieving dependent file {}", file);
306             if (!retrieveFile(file, errmsg)) {
307                 errmsg[0] = myErr + ": failed on " + file;
308                 return false;
309             }
310         }
311         return true;
312     }

314     /**
315      * Actually retrieve the byteStream of the native library over the network and save it to a diskfile.
316      */
317     public boolean retrieveFile(final String filename, final String[] errmsg) {
318 
319         final String osname = System.getProperty("os.name").replace(' ', '_');
320         final String theLocation = this.savePath.get(osname);
321         final String fullPathName = theLocation + File.separator + filename;
322         final byte[] libContents = returnFile(filename, errmsg);
323 
324         if (libContents == null) {
325             return false;
326         }
327 
328         // Save the contents into the disk file
329         try (FileOutputStream fos = new FileOutputStream(fullPathName);
330                 BufferedOutputStream bos = new BufferedOutputStream(fos)) {
331             bos.write(libContents, 0, libContents.length);
332         } catch (IOException ioe) {
333             errmsg[0] = "Cannot write retrieved JNI library to " + fullPathName + ": " + ioe;
334             return false;
335         }
336 
337         return true;
338     }

340     /**
341      * Look up the repository using our directory and retrieve the bytestream over the network with a synchronous call.
342      */
343     @Nullable
344     public byte[] returnFile(final String filename, final String[] errmsg) {
345 
346         if (this.theDir == null) {
347             // Probably instantiated from a static main standalone constructor
348             // which should have called returnFile(String,String[],String)
349             // instead and specified the key
350             errmsg[0] = "No DirectoryPlace available";
351             return null;
352         }
353 
354         try {
355             final List<DirectoryEntry> entries = this.theDir.nextKeys("JNI", null, null);
356             if (CollectionUtils.isEmpty(entries)) {
357                 errmsg[0] = "No JNI place in directory for:" + filename;
358                 return null;
359             }
360 
361             // Just use the first one
362             final DirectoryEntry entry = entries.get(0);
363             String repositoryKey = entry.getKey();
364 
365             // No related place, try a bootstrapping repository
366             if (repositoryKey == null) {
367                 repositoryKey = System.getProperty("emissary.repository");
368             }
369 
370             // No repository at all. Bail out
371             if (repositoryKey == null) {
372                 errmsg[0] = "JNI.returnFile: cannot retrieve files " + "without a repository.";
373                 return null;
374             }
375 
376             // We have a repository of some sort, try using it
377             return returnFile(filename, errmsg, repositoryKey);
378 
379         } catch (RuntimeException ve) {
380             errmsg[0] = "JNI.returnFile: " + ve;
381             return null;
382         }
383     }

385     /**
386      * Retrieve from the Repository with the specified Key. Can be used for bootstrapping when config files for directories
387      * and Repositories might not exist. A non-related repository can be specified in this case and the system will
388      * bootstrap from it.
389      */
390     @Nullable
391     public byte[] returnFile(final String filename, final String[] errmsg, final String repositoryKey) {
392 
393         final String repositoryAddrString = KeyManipulator.getServiceLocation(repositoryKey);
394 
395         IJniRepositoryPlace repositoryProxy;
396 
397         try {
398             final String look = repositoryAddrString.substring(repositoryAddrString.indexOf("//"));
399 
400             repositoryProxy = (IJniRepositoryPlace) Namespace.lookup(look);
401         } catch (NamespaceException | RuntimeException e) {
402             errmsg[0] = "JNI.returnFile: " + e;
403             return null;
404         }
405 
406         // Ask the repository to send the byte stream of the native
407         // library contents
408         final byte[] libContents;
409         try {
410             libContents = repositoryProxy.nativeLibraryDeliver(filename);
411         } catch (RemoteException | RuntimeException e) {
412             errmsg[0] = "Error calling nativeLibraryDeliver: " + e;
413             return null;
414         }
415 
416         if (libContents == null) {
417             errmsg[0] = "Unsuccessful request to repository: " + "got zero bytes";
418         }
419 
420         return libContents;
421     }

431     /**
432      * Provide access to the remote file timestamp
433      */
434     public long lastModified(final String filename, final String[] errmsg, final String repositoryKey) {
435 
436         long stamp = 0L;
437 
438         String repositoryAddrString = KeyManipulator.getServiceLocation(repositoryKey);
439 
440         if (StringUtils.contains(repositoryAddrString, "//")) {
441             repositoryAddrString = repositoryAddrString.substring(repositoryAddrString.indexOf("//"));
442         }
443 
444         IJniRepositoryPlace repositoryProxy;
445 
446         try {
447             repositoryProxy = (IJniRepositoryPlace) Namespace.lookup(repositoryAddrString);
448         } catch (NamespaceException ne) {
449             errmsg[0] = "JNI.returnFile: " + ne;
450             return stamp;
451         }
452 
453         // Ask the repository to send the timestamp
454 
455         try {
456             stamp = repositoryProxy.lastModified(filename);
457         } catch (RuntimeException e) {
458             errmsg[0] = "Error calling lastModified: " + e;
459         }
460 
461         return stamp;
462     }

464     /**
465      * Static main used from Makefiles to produce the correct expanded name of the library. Guaranteed to match when the
466      * calling program asks for it via the repository mechanism
467      */
468     public static void main(final String[] args) throws IOException {
469 
470         if (args.length < 1) {
471             logger.info("usage: java JNI [-L LEVEL] libname");
472             return;
473         }
474 
475         int argpos = 0;
476         while ((argpos < args.length) && args[argpos].startsWith("-")) {
477             if ("--".equals(args[argpos])) {
478                 argpos++;
479                 break;
480             }
481         }
482 
483         final JNI jni = new JNI();
484         String path = jni.getSavePath();
485         if (path != null && !path.endsWith("/")) {
486             path += "/";
487         }
488         for (int i = argpos; i < args.length; i++) {
489             logger.info("{}{}", path, jni.filesystemLibraryName(jni.expandLibraryName(args[i])));
490         }
491     }

15  public class ObjectTracingService {
16  
17      @Nullable
18      @SuppressWarnings("NonFinalStaticField")
19      private static ServiceLoader<ObjectTracing> loader = null;
20      private static final Logger objectTraceLogger = LoggerFactory.getLogger("objectTrace");
21  
22      private ObjectTracingService() {}
23  
24      public static synchronized void emitLifecycleEvent(IBaseDataObject d, String filename, ObjectTracing.Stage stage, boolean useObjectTracing) {
25          if (useObjectTracing) {
26              if (loader == null) {
27                  loader = ServiceLoader.load(ObjectTracing.class);
28              }
29  
30              // have the appropriate providers add fields
31              Map<String, String> jsonFieldMap = new HashMap<>();
32              for (ObjectTracing tracing : loader) {
33                  tracing.getObjectTraceFields(d, filename, stage, jsonFieldMap);
34              }
35              // once we have added fields from all providers, perform remapping of field names as appropriate
36              for (ObjectTracing tracing : loader) {
37                  tracing.mapFieldNames(jsonFieldMap);
38              }
39              objectTraceLogger.info(appendEntries(jsonFieldMap), "");
40          }
41      }

10  import java.io.IOException;
11  import java.util.HashMap;

13  public class JavaCharSetInitializationProvider implements InitializationProvider {
14      protected static Logger logger = LoggerFactory.getLogger(JavaCharSetInitializationProvider.class);
15  
16      @Override
17      public void initialize() {
18          if (!JavaCharSet.isInitialized()) {
19              try {
20                  final Configurator config = ConfigUtil.getConfigInfo(JavaCharSet.class);
21                  JavaCharSet.initialize(config.findStringMatchMap("CHARSET_", true));
22              } catch (IOException e) {
23                  logger.error("Error initializing JavaCharSet: ", e);
24                  JavaCharSet.initialize(new HashMap<>());
25              }
26          }
27      }
28  }

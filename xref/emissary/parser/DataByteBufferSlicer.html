7   import java.io.ByteArrayOutputStream;
8   import java.io.IOException;
9   import java.nio.BufferUnderflowException;
10  import java.nio.ByteBuffer;
11  import java.util.List;
12  
13  public class DataByteBufferSlicer {
14      private static final Logger logger = LoggerFactory.getLogger(DataByteBufferSlicer.class);
15  
16      /**
17       * Slice data from a buffer based on a single position record
18       * 
19       * @param data the data to pull from
20       * @param r the position record indicating absolute offsets
21       */
22      public static byte[] makeDataSlice(ByteBuffer data, PositionRecord r) {
23          if (r.getLength() > Integer.MAX_VALUE) {
24              throw new IllegalStateException("Implementation currently only handles up to Intger.MAX_VALUE lengths");
25          }
26          int len = (int) r.getLength();
27          byte[] n = new byte[len];
28          try {
29              data.position((int) r.getPosition());
30              data.get(n);
31          } catch (BufferUnderflowException ex) {
32              logger.warn("Underflow getting {} bytes at {}", n.length, r.getPosition());
33          }
34          return n;
35      }
36  
37      /**
38       * Slice a ByteBuffer based on a list of position record
39       * 
40       * @param data the data to pull from
41       * @param list the list of position records indicating absolute offsets
42       */
43      @Nullable
44      public static byte[] makeDataSlice(ByteBuffer data, @Nullable List<PositionRecord> list) {
45          // Nothing to do
46          if (list == null || list.isEmpty()) {
47              return null;
48          }
49  
50          // Use higher performing impl when only one record
51          if (list.size() == 1) {
52              return makeDataSlice(data, list.get(0));
53          }
54  
55          // Aggregate all the pieces using the baos
56          byte[] ret = null;
57          try (ByteArrayOutputStream out = new ByteArrayOutputStream()) {
58              for (PositionRecord r : list) {
59                  int start = (int) r.getPosition();
60                  int len = (int) r.getLength();
61  
62                  if (len <= 0) {
63                      continue;
64                  }
65                  if (start < 0) {
66                      continue;
67                  }
68  
69                  data.position(start);
70                  try {
71                      byte[] n = new byte[len];
72                      data.get(n);
73                      out.write(n);
74                  } catch (BufferUnderflowException ex) {
75                      logger.error("Underflow getting {} bytes at {}", len, start);
76                  }
77              }
78              ret = out.toByteArray();
79          } catch (IOException iox) {
80              logger.warn("io error on bytearray stream cant happen", iox);
81          }
82  
83          return ret;
84      }
85  
86      /** This class is not meant to be instantiated. */
87      private DataByteBufferSlicer() {}
88  }

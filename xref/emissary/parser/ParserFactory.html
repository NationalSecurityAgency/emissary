17  /**
18   * Provide a factory for getting the proper type of input parser Provide the implementing classes for that match the
19   * configured Data Identifier Engine in PARSER_NIO_IMPL_[type] variants if available. All configured parsers must
20   * implement emissary.parser.SessionParser.
21   *
22   * When no proper mappings are found or the specified parser cannot be instantiated, the SimpleNioParser is used
23   * instead. If these cannot be instantiated, then something is likely seriously wrong.
24   *
25   * If an NIO parser is requested, see makeSessionParser(FileChannel), but cannot be found for the data type, the Channel
26   * is evaluated and if under MAX_NIO_FALLBACK_SIZE, then the bytes are consumed and a standard parser is produced if one
27   * is available.
28   *
29   * The ID engine is configured with the ID_ENGINE_CLASS in the configuration file and must be an instance of
30   * emissary.parser.DataIdentifier.
31   */
32  public class ParserFactory {
33      // Logger
34      protected static final Logger logger = LoggerFactory.getLogger(ParserFactory.class.getName());
35  
36      // Map of dataType to FileChannel parser implementation class name
37      // Read from config file
38      protected Map<String, String> nioTypeMap = new HashMap<>();
39  
40      // For channel sizes larger than this no fallback to a byte[]
41      // parser is attempted.
42      protected long nioFallbackMax = 1024L * 1024L * 100L; // 100 Mb
43  
44      protected static final String DEFAULT_NIO_PARSER = "emissary.parser.SimpleNioParser";
45      protected String nioParser = DEFAULT_NIO_PARSER;
46  
47  
48      // Data type identification engine
49      @Nullable
50      DataIdentifier idEngine = null;

54       */

56          reconfigure();
57      }

69      /**
70       * Configure this factory with default config (for keeping the API backward compatible)
71       */
72      public void reconfigure() {
73          reconfigure(null);
74      }
75  
76      /**
77       * Reconfigure the factory causes the configuration to be reloaded It is not threadsafe to call this while data is being
78       * identified or parsers are being instantiated.
79       *
80       * @param config the configuration to use or null for the default
81       * @since 3.7.1
82       */
83      public void reconfigure(@Nullable Configurator config) {
84          try {
85              if (config == null) {
86                  config = ConfigUtil.getConfigInfo(ParserFactory.class);
87              }
88  
89              Map<String, String> m = config.findStringMatchMap("PARSER_NIO_IMPL_", Configurator.PRESERVE_CASE);
90  
91              nioFallbackMax = config.findSizeEntry("MAX_NIO_FALLBACK_SIZE", nioFallbackMax);
92  
93              nioTypeMap.clear();
94              nioTypeMap.putAll(m);
95  
96              logger.debug("Loaded {} nio parsers with fallback size {}", nioTypeMap.size(), nioFallbackMax);
97  
98              // change this to "DEFAULT_PARSER"
99              nioParser = config.findStringEntry("DEFAULT_NIO_PARSER", DEFAULT_NIO_PARSER);
100 
101             String idEngineClass = config.findStringEntry("ID_ENGINE_CLASS", null);
102 
103             if (idEngineClass != null) {
104                 makeIdEngine(idEngineClass);
105             }
106         } catch (IOException ex) {
107             logger.error("Unable to read configuration", ex);
108         }
109     }

111     /**
112      * Make a session parser with the data in channel. If no NIO parser is configured for the type of this data, a standard
113      * byte[] parser will be produced if there is one available and the size of the data in the channel is less than the
114      * configured MAX_NIO_FALLBACK_SIZE. Otherwise the default NIO parser will be used.
115      *
116      * @param channel the data to be parsed
117      * @return SessionParser implementation
118      */
119     public SessionParser makeSessionParser(SeekableByteChannel channel) {
120         String id = identify(channel);
121         return makeSessionParser(id, channel);
122     }

140     /**
141      * Make a session parser with the data in channel. If no NIO parser is configured for the type of this data, a standard
142      * byte[] parser will be produced if there is one available and the size of the data in the channel is less than the
143      * configured MAX_NIO_FALLBACK_SIZE. Otherwise the default NIO parser will be used.
144      *
145      * @param channel the data to be parsed
146      * @param type the type of data
147      * @return SessionParser implementation
148      */
149     public SessionParser makeSessionParser(SeekableByteChannel channel, String type) {
150         SessionParser sp;
151 
152         if (nioTypeMap.containsKey(type)) {
153             sp = makeSessionParserClass(nioTypeMap.get(type), channel);
154         } else {
155             sp = makeSessionParserClass(nioParser, channel);
156         }
157         return sp;
158     }
159 
160     /**
161      * Make a session parser for the specified data type with the args
162      *
163      * @param clazz the class name of the parser to create
164      * @param args arguments to the parser constructor
165      * @return SessionParser implementation
166      */
167     @Nullable
168     protected SessionParser makeSessionParserClass(@Nullable String clazz, Object... args) {
169         // Choose implementation class based on data type
170         if (clazz == null) {
171             logger.warn("Cannot make a session parser for a null class");
172             return null;
173         }
174 
175         SessionParser sp = null;
176 
177         try {
178             sp = (SessionParser) Factory.create(clazz, args);
179         } catch (RuntimeException e) {
180             logger.error("Unable to instantiate {}", clazz, e);
181         }
182 
183         return sp;
184     }
185 
186     /**
187      * Instantiate the specified DataIdentifier class for typing the data
188      */
189     protected void makeIdEngine(String clazz) {
190         try {
191             DataIdentifier d = (DataIdentifier) Factory.create(clazz);
192             idEngine = d;
193         } catch (RuntimeException ex) {
194             logger.warn("Cannot make data identifier from " + clazz, ex);
195         }
196     }

199     /**
200      * Return the key identification type fo the data in the channel
201      *
202      * @param channel the channel containing bytes to identify
203      * @return string matching the keys in ParserFactory.cfg
204      */
205     public String identify(SeekableByteChannel channel) {
206         if (idEngine != null) {
207             try {
208                 long pos = channel.position();
209                 byte[] buf = Executrix.readDataFromChannel(channel, pos, idEngine.DATA_ID_STR_SZ);
210                 channel.position(pos);
211                 return idEngine.identify(buf);
212             } catch (IOException e) {
213                 logger.warn("Unable to reposition file channel", e);
214             }
215         }
216         return DataIdentifier.UNKNOWN_TYPE;
217     }

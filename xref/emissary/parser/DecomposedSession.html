87      /**
88       * Set the data entry using the passed in byte array or a copy Previously existing data is lost
89       *
90       * @param d bytes to set
91       * @param copy make a copy when true
92       */
93      public void setData(@Nullable byte[] d, boolean copy) {
94          if (d == null || !copy) {
95              data = d;
96          } else {
97              setData(d, 0, d.length);
98          }
99      }

101     /**
102      * Set the data from the specified portion of the array Always makes a copy. Previously existing data is lost.
103      *
104      * @param d bytes to set
105      * @param start start offset
106      * @param end ending offset
107      */
108     public void setData(byte[] d, int start, int end) {
109         data = new byte[end - start];
110         System.arraycopy(d, start, data, 0, data.length);
111     }

223     /**
224      * Add a map of metadata to the existing map MetaData map will be created if not existing
225      *
226      * @param m map of items to add
227      */
228     public void addMetaData(@Nullable Map<String, ?> m) {
229         if (m != null && m.size() > 0) {
230             for (Map.Entry<String, ?> entry : m.entrySet()) {
231                 String key = entry.getKey();
232                 Object v = entry.getValue();
233                 if (v instanceof Iterable) {
234                     metadata.putAll(key, (Iterable<?>) v);
235                 } else {
236                     metadata.put(key, v);
237                 }
238             }
239         }
240     }

260     /**
261      * Get a simplified form of a single metadata item
262      *
263      * @param key the name
264      * @param sep the separator
265      */
266     @Nullable
267     public String getStringMetadataItem(String key, String sep) {
268         List<Object> o = metadata.get(key);
269         if (o.isEmpty()) {
270             return null;
271         }
272         StringBuilder sb = new StringBuilder();
273         for (Object item : o) {
274             if (sb.length() > 0) {
275                 sb.append(sep);
276             }
277             sb.append(item);
278         }
279         return sb.toString();
280     }

5   import java.io.ByteArrayOutputStream;
6   import java.io.IOException;
7   
8   public class JsonEscape {
9   
10      private static final String ESCAPES = "ntr\"'/\\";
11  
12      /**
13       * Unescape a bunch of JSON bytes that might have \\uxxxx character values. Should already be UTF-8 since JSON is
14       * specified as UTF-8 by RFC 4627
15       */
16      public static byte[] unescape(byte[] data) {
17          ByteArrayOutputStream out = new ByteArrayOutputStream();
18          for (int i = 0; i < data.length; i++) {
19              if (data[i] == '\\' && (i + 5) < data.length && (data[i + 1] == 'u' || data[i + 1] == 'U')) {
20                  // process unicode escape
21                  try {
22                      String s = new String(data, i + 2, 4);
23                      char[] c = HtmlEscape.unescapeHtmlChar(s, true);
24                      if (c != null && c.length > 0) {
25                          out.write(new String(c).getBytes());
26                          // logger.debug("Unicode '{}' ==> '{}'", s, new String(c));
27                          i += 5;
28                      } else {
29                          out.write(data[i]);
30                      }
31                  } catch (IOException iox) {
32                      out.write(data[i]);
33                  }
34              } else if (data[i] == '\\' && (i + 1) < data.length && isOctalDigit(data[i + 1])) {
35                  // Process octal escape
36                  int end = i + 1;
37                  if ((i + 2) < data.length && isOctalDigit(data[i + 2])) {
38                      end++;
39                  }
40                  if ((i + 3) < data.length && isOctalDigit(data[i + 3])) {
41                      end++;
42                  }
43                  String s = new String(data, i + 1, (end - i));
44                  try {
45                      int num = Integer.parseInt(s, 8);
46                      char[] ch = Character.toChars(num);
47                      out.write(new String(ch).getBytes());
48                      // logger.debug("Octal '{}' ==> '{}'", s, new String(ch));
49                      i += s.length();
50                  } catch (Exception ex) {
51                      out.write(data[i]);
52                  }
53              } else if (data[i] == '\\' && (i + 1) < data.length && ESCAPES.indexOf(data[i + 1]) != -1) {
54                  byte b = data[i + 1];
55                  if (b == 'n') {
56                      out.write('\n');
57                  } else if (b == 't') {
58                      out.write('\t');
59                  } else if (b == 'r') {
60                      out.write('\r');
61                  } else {
62                      out.write(b);
63                  }
64                  i++;
65              } else {
66                  out.write(data[i]);
67              }
68          }
69          return out.toByteArray();
70      }
71  
72      protected static boolean isOctalDigit(byte b) {
73          return (b >= '0' && b <= '7');
74      }
75  
76      /** This class is not meant to be instantiated. */
77      private JsonEscape() {}
78  
79      @SuppressWarnings("SystemOut")
80      public static void main(String[] args) throws Exception {
81          for (int i = 0; i < args.length; i++) {
82              byte[] content = Executrix.readDataFromFile(args[i]);
83              if (content == null) {
84                  System.out.println(args[i] + ": Unreadable");
85                  continue;
86              }
87  
88              System.out.println(args[i]);
89              byte[] escaped = JsonEscape.unescape(content);
90              System.out.write(escaped, 0, escaped.length);
91              System.out.println();
92          }
93      }
94  }

6   import java.io.IOException;

8   @Deprecated
9   public class JsonEscape {
10  
11      private static final String ESCAPES = "ntr\"'/\\";
12  
13      /**
14       * Unescape a bunch of JSON bytes that might have \\uxxxx character values. Should already be UTF-8 since JSON is
15       * specified as UTF-8 by RFC 4627
16       */
17      public static byte[] unescape(byte[] data) {
18          ByteArrayOutputStream out = new ByteArrayOutputStream();
19          for (int i = 0; i < data.length; i++) {
20              if (data[i] == '\\' && (i + 5) < data.length && (data[i + 1] == 'u' || data[i + 1] == 'U')) {
21                  // process unicode escape
22                  try {
23                      String s = new String(data, i + 2, 4);
24                      char[] c = HtmlEscape.unescapeHtmlChar(s, true);
25                      if (c != null && c.length > 0) {
26                          out.write(new String(c).getBytes());
27                          // logger.debug("Unicode '{}' ==> '{}'", s, new String(c));
28                          i += 5;
29                      } else {
30                          out.write(data[i]);
31                      }
32                  } catch (IOException iox) {
33                      out.write(data[i]);
34                  }
35              } else if (data[i] == '\\' && (i + 1) < data.length && isOctalDigit(data[i + 1])) {
36                  // Process octal escape
37                  int end = i + 1;
38                  if ((i + 2) < data.length && isOctalDigit(data[i + 2])) {
39                      end++;
40                  }
41                  if ((i + 3) < data.length && isOctalDigit(data[i + 3])) {
42                      end++;
43                  }
44                  String s = new String(data, i + 1, end - i);
45                  try {
46                      int num = Integer.parseInt(s, 8);
47                      char[] ch = Character.toChars(num);
48                      out.write(new String(ch).getBytes());
49                      // logger.debug("Octal '{}' ==> '{}'", s, new String(ch));
50                      i += s.length();
51                  } catch (Exception ex) {
52                      out.write(data[i]);
53                  }
54              } else if (data[i] == '\\' && (i + 1) < data.length && ESCAPES.indexOf(data[i + 1]) != -1) {
55                  byte b = data[i + 1];
56                  if (b == 'n') {
57                      out.write('\n');
58                  } else if (b == 't') {
59                      out.write('\t');
60                  } else if (b == 'r') {
61                      out.write('\r');
62                  } else {
63                      out.write(b);
64                  }
65                  i++;
66              } else {
67                  out.write(data[i]);
68              }
69          }
70          return out.toByteArray();
71      }
72  
73      protected static boolean isOctalDigit(byte b) {
74          return b >= '0' && b <= '7';
75      }
76  
77      /** This class is not meant to be instantiated. */
78      private JsonEscape() {}
79  
80      @SuppressWarnings("SystemOut")
81      public static void main(String[] args) {
82          for (int i = 0; i < args.length; i++) {
83              byte[] content = Executrix.readDataFromFile(args[i]);
84              if (content == null) {
85                  System.out.println(args[i] + ": Unreadable");
86                  continue;
87              }
88  
89              System.out.println(args[i]);
90              byte[] escaped = JsonEscape.unescape(content);
91              System.out.write(escaped, 0, escaped.length);
92              System.out.println();
93          }
94      }

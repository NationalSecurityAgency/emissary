44      /**
45       * Unescape some HTML data, turning <code>&amp;#xxxx;</code> into UNICODE characters Because this operation inserts java
46       * Character objects into the byte array, it probably only makes sense to send in data that already matches the platform
47       * encoding (i.e. UTF-8 for normal usage). Otherwise the result will be a mixed up mess of multiple character sets that
48       * cannot possibly be understood or displayed properly.
49       *
50       * @param data the array of bytes containing HTML escaped characters
51       * @param counters to measure what is changed
52       * @return modified byte array
53       */
54      public static byte[] unescapeHtml(@Nullable byte[] data, @Nullable CharacterCounterSet counters) {
55  
56          ByteArrayOutputStream baos = null;
57          byte[] returnBytes = null;
58          if (data == null || data.length == 0) {
59              return new byte[0];
60          }
61  
62          try {
63              baos = new ByteArrayOutputStream();
64              for (int i = 0; i < data.length; i++) {
65                  // Grab one encoded character
66                  if (data[i] == '&' && i + 3 < data.length && data[i + 1] == '#') {
67                      int j = i + 2;
68                      boolean isHex = false;
69  
70                      // Determine if &#xnnnn; or &#nnnn;
71                      if (data[j] == 'X' || data[j] == 'x') {
72                          j++;
73                          isHex = true;
74                      }
75  
76                      int startPos = j;
77  
78                      // Jump to end of digits, find a semi-colon
79                      while (j < data.length && ByteUtil.isHexadecimal(data[j]) && j < startPos + 5) {
80                          j++;
81                      }
82  
83                      if (j < data.length && data[j] == ';') {
84                          // Try to convert it
85                          char[] c = unescapeHtmlChar(new String(data, startPos, j - startPos), isHex);
86                          if (c != null) {
87                              // write a codepoint
88                              String s = new String(c);
89                              baos.write(s.getBytes());
90                              if (counters != null) {
91                                  counters.count(s);
92                              }
93                              i = j;
94                          } else {
95                              // Do no harm if the conversion fails
96                              baos.write(data[i]);
97                          }
98                      } else {
99                          baos.write(data[i]);
100                     }
101                 } else {
102                     baos.write(data[i]);
103                 }
104             }
105             returnBytes = baos.toByteArray();
106             baos.close();
107         } catch (IOException e) {
108             logger.debug("Cannot decode HTML bytes", e);
109         }
110         return returnBytes;
111     }

123     /**
124      * Unescape some HTML data, turning <code>$#xxxx;</code> into UNICODE characters
125      *
126      * @param s the string of characters possibly containing escaped HTML
127      * @param counters to measure what is changed
128      * @return the new String without escaped HTML
129      */
130     public static String unescapeHtml(@Nullable String s, @Nullable CharacterCounterSet counters) {
131         if (s == null || s.length() == 0) {
132             return "";
133         }
134 
135         StringBuffer sb = new StringBuffer(s.length());
136         Matcher m = HESC_PATTERN.matcher(s);
137 
138         // Match each occurrence
139         while (m.find()) {
140             // Grab digits from first match group
141             String hexModifier = m.group(1);
142             boolean isHex = hexModifier != null && hexModifier.length() > 0;
143             String encodedChar = m.group(2);
144             char[] c = unescapeHtmlChar(encodedChar, isHex);
145             if (c != null) {
146                 // Append non-matching portion plus decoded char
147                 m.appendReplacement(sb, "");
148                 sb.append(c);
149                 if (counters != null) {
150                     counters.count(new String(c));
151                 }
152             } else {
153                 // It failed, append non-match plus original match
154                 m.appendReplacement(sb, "$0");
155             }
156         }
157 
158         // Append terminal portion
159         m.appendTail(sb);
160 
161         // Return the new string
162         return sb.toString();
163     }

166     /**
167      * Unescape one HTML character or null if we cannot convert
168      *
169      * @param s the four digit number from the HTML encoding
170      * @param isHex true if the digits are in hex
171      * @return the Unicode codepoint in a[] char or null
172      */
173     @Nullable
174     public static char[] unescapeHtmlChar(String s, boolean isHex) {
175         int num = -1;
176         try {
177             num = Integer.parseInt(s, isHex ? 16 : 10);
178         } catch (NumberFormatException ex) {
179             logger.debug("Failed to parse char", ex);
180             return null;
181         }
182         // Turn the number into a Unicode codepoint
183         try {
184             return Character.toChars(num);
185         } catch (Exception ex) {
186             logger.debug("Cannot toChars: not a valid Unicode code point", ex);
187             return null;
188         }
189     }

200     /**
201      * Unescape HTML entities like &amp;nbsp; into normal characters Also handle broken entities like &amp;;nbsp; and
202      * &amp;nbsp (extra semi-colon and missing semi-colon respectively)
203      *
204      * @param s the string to find entities in
205      * @param counters to measure what was changed
206      */
207     public static String unescapeEntities(String s, @Nullable CharacterCounterSet counters) {
208         int slen = s.length();
209         StringBuilder sb = new StringBuilder(s.length());
210 
211         for (int i = 0; i < slen; i++) {
212             char c = s.charAt(i);
213             if (c != '&') {
214                 sb.append(c);
215             } else {
216                 int spos = i;
217                 int j = spos + 1;
218                 while (j < slen && j < (spos + LONGEST_ENTITY_NAME) && s.charAt(j) != ';' && s.charAt(j) != ' ') {
219                     j++;
220                 }
221 
222                 if (j < slen && j == spos + 1) // broken case with extra semicolon
223                 {
224                     spos++;
225                     j = spos + 1;
226                     while (j < slen && j < (spos + LONGEST_ENTITY_NAME) && s.charAt(j) != ';' && s.charAt(j) != ' ') {
227                         j++;
228                     }
229                 }
230 
231                 if (j < slen + 1) {
232                     String ent = null;
233                     if (j < slen) {
234                         if (s.charAt(j) == ';' || s.charAt(j) == ' ') {
235                             ent = s.substring(spos + 1, j);
236                         }
237                     } else {
238                         // all the rest
239                         ent = s.substring(spos + 1);
240                     }
241 
242                     if (ent != null) {
243                         String val = getValueForHtmlEntity(ent);
244                         if (val != null) {
245                             sb.append(val);
246                             if (counters != null) {
247                                 counters.count(val);
248                             }
249                             if (j >= slen) {
250                                 i = slen; // all done
251                             } else {
252                                 i = s.charAt(j) == ' ' ? (j - 1) : j;
253                             }
254                         } else {
255                             sb.append(c);
256                         }
257                     } else {
258                         sb.append(c);
259                     }
260                 } else {
261                     sb.append(c);
262                 }
263             }
264         }
265         return sb.toString();
266     }

276     /**
277      * Unescape HTML Entities like &amp;nbsp; into normal characters Also handle broken entities like &amp;;nbsp; and
278      * &amp;nbsp (extra semi-colon and missing semi-colon respectively)
279      */
280     public static byte[] unescapeEntities(byte[] s, @Nullable CharacterCounterSet counters) {
281         ByteArrayOutputStream baos = new ByteArrayOutputStream();
282         int slen = s.length;
283 
284         for (int i = 0; i < slen; i++) {
285             if (i + 4 < slen && s[i] == '&') {
286                 int spos = i;
287                 int epos = spos + 1;
288                 while (epos < slen && epos < spos + LONGEST_ENTITY_NAME && s[epos] != ';' && s[epos] != ' ') {
289                     epos++;
290                 }
291 
292                 if (epos == spos + 1) // broken case with extra semi-colon
293                 {
294                     spos++;
295                     epos = spos + 1;
296                     while (epos < slen && epos < spos + LONGEST_ENTITY_NAME && s[epos] != ';' && s[epos] != ' ') {
297                         epos++;
298                     }
299                 }
300 
301                 String val = HTML_ENTITY_MAP.getValueForHTMLEntity(new String(s, spos + 1, epos - (spos + 1)));
302                 if (val != null) {
303                     try {
304                         baos.write(val.getBytes());
305                         if (counters != null) {
306                             counters.count(val);
307                         }
308                         // if we used the space as a terminator, keep the
309                         // space in the output, even though we consumed it
310                         if (epos < slen) {
311                             i = s[epos] == ' ' ? (epos - 1) : epos;
312                         } else {
313                             i = slen;
314                         }
315                     } catch (IOException iox) {
316                         logger.debug("Error writing unescaped bytes", iox);
317                         baos.write(s[i]);
318                     }
319                 } else {
320                     baos.write(s[i]);
321                 }
322             } else {
323                 baos.write(s[i]);
324             }
325         }
326         return baos.toByteArray();
327     }

329     @Nullable
330     private static String getValueForHtmlEntity(String entity) {
331         String s = HTML_ENTITY_MAP.getValueForHTMLEntity(entity);
332         if (s != null) {
333             return s;
334         }
335         return null;

341     public static void main(String[] args) throws Exception {
342         boolean useString = false;
343         int i = 0;
344         if (args.length > 0 && args[i].equals("-s")) {
345             System.out.println("Switching to string mode");
346             useString = true;
347             i++;
348         }
349 
350         for (; i < args.length; i++) {
351             byte[] content = Executrix.readDataFromFile(args[i]);
352             if (content == null) {
353                 System.out.println(args[i] + ": Unreadable");
354                 continue;
355             }
356 
357             System.out.println(args[i]);
358             if (useString) {
359                 String escaped = HtmlEscape.unescapeHtml(new String(content));
360                 escaped = HtmlEscape.unescapeEntities(escaped);
361                 System.out.println(escaped);
362             } else {
363                 byte[] escaped = HtmlEscape.unescapeHtml(content);
364                 escaped = HtmlEscape.unescapeEntities(escaped);
365                 System.out.write(escaped, 0, escaped.length);
366                 System.out.println();
367             }
368         }

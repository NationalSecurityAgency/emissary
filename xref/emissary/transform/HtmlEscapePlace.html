64      /**
65       * Take care of special place configuration
66       */
67      protected void configurePlace() {
68          outputForm = configG.findStringEntry(OUTPUT_FORM, null);
69  
70          // Force statics to load
71          HtmlEscape.unescapeHtml(new byte[0]);
72      }

74      /**
75       * Consume a dataObject and return a modified one.
76       */
77      @Override
78      public void process(IBaseDataObject d) {
79          if (DataUtil.isEmpty(d)) {
80              logger.debug("empty data");
81              return;
82          }
83          String incomingForm = d.currentForm();
84          CharacterCounterSet counters = new CharacterCounterSet();
85  
86          logger.debug("Just got a payload with form {}", incomingForm);
87  
88          byte[] newData = HtmlEscape.unescapeHtml(d.data(), counters);
89  
90          if (newData != null && newData.length > 0) {
91              newData = HtmlEscape.unescapeEntities(newData, counters);
92              if (outputForm != null) {
93                  d.setCurrentForm(outputForm);
94              }
95              // Track how much change in size there was
96              int variance = d.dataLength() - newData.length;
97              if (variance < 0) {
98                  variance *= -1;
99              }
100             d.setParameter("HTML_Entity_Decode_Variance", Integer.toString(variance));
101             d.setData(newData);
102             d.setFileTypeIfEmpty(HTML);
103 
104             for (String key : counters.getKeys()) {
105                 d.putParameter(key + SUFFIXES_HTML_ESCAPE, Integer.toString(counters.get(key)));
106             }
107 
108         } else {
109             logger.warn("error doing HtmlEscape, unable to decode");
110             d.pushCurrentForm(Form.ERROR);
111         }
112 
113         unescapeAltViews(d);
114         unescapeSummary(d);
115         unescapeDocTitle(d);
116         processEncoding(d);
117         processCurrentForms(d);
118         nukeMyProxies(d);
119     }
120 
121     protected void unescapeAltViews(IBaseDataObject d) {
122         // Unescape any TEXT alt views we may have
123         d.getAlternateViewNames().stream().filter(v -> v.startsWith(TEXT)).forEach(viewName -> {
124             byte[] textView = d.getAlternateView(viewName);
125             if (ArrayUtils.isNotEmpty(textView)) {
126                 byte[] s = HtmlEscape.unescapeHtml(textView);
127                 if (ArrayUtils.isNotEmpty(s)) {
128                     s = HtmlEscape.unescapeEntities(s);
129                     if (ArrayUtils.isNotEmpty(s)) {
130                         d.addAlternateView(viewName, s);
131                     }
132                 }
133             }
134         });
135     }
136 
137     protected void unescapeSummary(IBaseDataObject d) {
138         // Unescape the Summary if present
139         String summary = d.getStringParameter(SUMMARY);
140         if (StringUtils.contains(summary, "&#")) {
141             logger.debug("Working on summary "/* + summary */);
142             String s = makeString(HtmlEscape.unescapeHtml(summary.getBytes()));
143             if (StringUtils.isNotBlank(s)) {
144                 s = HtmlEscape.unescapeEntities(s);
145                 d.deleteParameter(SUMMARY);
146                 d.putParameter(SUMMARY, s);
147             }
148         }
149     }
150 
151     protected void unescapeDocTitle(IBaseDataObject d) {
152         // Unescape the Document Title
153         String title = d.getStringParameter(DOCUMENT_TITLE);
154         if (StringUtils.contains(title, "&#")) {
155             logger.debug("Working on title "/* + title */);
156             String s = makeString(HtmlEscape.unescapeHtml(title.getBytes()));
157             if (StringUtils.isNotBlank(s)) {
158                 d.deleteParameter(DOCUMENT_TITLE);
159                 s = HtmlEscape.unescapeEntities(s);
160                 d.putParameter(DOCUMENT_TITLE, s);
161             }
162         }
163         logger.debug("Retrieved new title ");
164     }
165 
166     protected void processEncoding(IBaseDataObject d) {
167         // If the encoding or the LANG- form has -HTMLESC remove it
168         String enc = d.getFontEncoding();
169         if (StringUtils.contains(enc, SUFFIXES_HTMLESC)) {
170             d.setFontEncoding(enc.replaceFirst(SUFFIXES_HTMLESC, ""));
171         }
172     }
173 
174     protected void processCurrentForms(IBaseDataObject d) {
175         for (String cf : d.getAllCurrentForms()) {
176             if (cf.contains(PREFIXES_LANG) && cf.contains(SUFFIXES_HTMLESC)) {
177                 // Get the old pos
178                 int pos = d.searchCurrentForm(cf);
179                 d.deleteCurrentForm(cf);
180                 cf = cf.replaceFirst(SUFFIXES_HTMLESC, "");
181                 d.addCurrentFormAt(pos, cf);
182                 break;
183             }
184         }
185     }
186 
187     public static String makeString(byte[] s) {
188         return new String(s, StandardCharsets.UTF_8);
189     }

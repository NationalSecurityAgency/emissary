18  /**
19   * ChecksumCalculator is a utility class which computes checksums and message digests.
20   * 
21   * @see java.util.zip.CRC32 java.util.zip.CRC32
22   * @see java.security.MessageDigest java.security.MessageDigest
23   */
24  public class ChecksumCalculator {
25      /** Used for CRC32 calculations */
26      @Nullable
27      private CRC32 crc = null;
28      /** Used for SSDEEP calculations */
29      @Nullable
30      private Ssdeep ssdeep = null;
31  
32      /** Used for hash calculations */
33      private final List<MessageDigest> digest = new ArrayList<>();

35      /**

38       * @throws NoSuchAlgorithmException if the SHA algorithm isn't available
39       */
40      public ChecksumCalculator() throws NoSuchAlgorithmException {
41          this("SHA-1", true);
42      }

44      /**
45       * Constructor initializes specified algorithm
46       * 
47       * @param alg string name of algorightm, e.g. SHA

49       * @throws NoSuchAlgorithmException if the algorithm isn't available

51      public ChecksumCalculator(String alg, boolean useCrc) throws NoSuchAlgorithmException {
52          this(List.of(alg));

56      /**
57       * Constructor initializes specified set of algorithms
58       * 
59       * @param algs array of String algorithm names, put CRC32 on list to enable
60       * @throws NoSuchAlgorithmException if an algorithm isn't available
61       */
62      @Deprecated
63      @SuppressWarnings("AvoidObjectArrays")
64      public ChecksumCalculator(@Nullable String[] algs) throws NoSuchAlgorithmException {
65          if (algs != null && algs.length > 0) {
66              for (String alg : algs) {
67                  if (alg.equals("CRC32")) {
68                      setUseCrc(true);
69                  } else if (alg.equals("SSDEEP")) {
70                      setUseSsdeep(true);
71                  } else {
72                      digest.add(MessageDigest.getInstance(alg));
73                  }
74              }
75          }
76      }

84      public ChecksumCalculator(@Nullable Collection<String> algs) throws NoSuchAlgorithmException {
85          if (CollectionUtils.isNotEmpty(algs)) {

139     /**
140      * Calculates a CRC32 and a digest on a byte array.
141      * 
142      * @param buffer Data to compute results for
143      * @return results of computing the requested hashes on the data
144      */
145     public ChecksumResults digest(byte[] buffer) {
146         // return object to hold results
147         ChecksumResults res = new ChecksumResults();
148 
149         // Reset and compute
150         for (MessageDigest d : digest) {
151             d.reset();
152             d.update(buffer, 0, buffer.length);
153             res.setHash(d.getAlgorithm(), d.digest());
154         }
155 
156         // Only use CRC if non-null
157         if (crc != null) {
158             crc.reset();
159             crc.update(buffer, 0, buffer.length);
160             res.setCrc(crc.getValue());
161         }
162 
163         // Only use ssdeep if non-null
164         if (ssdeep != null) {
165             res.setSsdeep(ssdeep.fuzzyHash(buffer));
166         }
167 
168         return res;
169     }

171     /**
172      * Calculates a CRC32 and a digest on a {@link java.nio.channels.SeekableByteChannel} of data.
173      *
174      * @param sbcf Provider of data to compute results for
175      * @return results of computing the requested hashes on the data
176      */
177     public ChecksumResults digest(final SeekableByteChannelFactory sbcf) {
178         final ChecksumResults res = new ChecksumResults();
179         final byte[] b = new byte[1024];
180 
181         for (final MessageDigest d : digest) {
182             try (InputStream is = Channels.newInputStream(sbcf.create())) {
183                 d.reset();
184 
185                 int bytesRead;
186                 while ((bytesRead = is.read(b)) != -1) {
187                     d.update(b, 0, bytesRead);
188                 }
189 
190                 res.setHash(d.getAlgorithm(), d.digest());
191             } catch (final IOException ioe) {
192                 // Ignore
193             }
194         }
195 
196         if (crc != null) {
197             try (InputStream is = Channels.newInputStream(sbcf.create())) {
198                 crc.reset();
199 
200                 int bytesRead;
201                 while ((bytesRead = is.read(b)) != -1) {
202                     crc.update(b, 0, bytesRead);
203                 }
204 
205                 res.setCrc(crc.getValue());
206             } catch (final IOException ioe) {
207                 // Ignore
208             }
209         }
210 
211         if (ssdeep != null) {
212             res.setSsdeep(ssdeep.fuzzyHash(sbcf));
213         }
214 
215         return res;
216     }

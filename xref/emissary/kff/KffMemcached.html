23  /**
24   * KffMemcached checks Emissary hashes against a set of external memcached servers. If a given Emissary hash does not
25   * hit in memcached, it is added to the memcached. The value stored is however the input is identified (notionally, some
26   * type of id). Only one type of hash can be checked against memcached (i.e. SHA-1...not SHA-1 <i>and</i> SHA-256). See
27   * the PREF_ALG configuration option.
28   *
29   * If the option MEMCACHED_STORE_ID_DUPE is set to true, if an Emissary hash already exists in memcached, the id will
30   * also be loaded into memcached as a <i>key</i>. The purpose of this is for other follow-on processes (non-Emissary) to
31   * query memcached and determine if a given id is a duplicate (i.e. if it is present).
32   *
33   * If a server goes down, this code will wait and try to reconnect with increasingly long intervals between retries.
34   * Under this mode of operation, it is expected someone will bring the server back online.
35   *
36   * Configuration file options are:
37   *
38   * MEMCACHED_SERVER: one line for each server, optional : separated port MEMCACHED_AGEOFF: how long to hold an object in
39   * memcached before expiring MEMCACHED_OP_TIMEOUT_MILLIS: how long to wait before timing out a memcached operation
40   * MEMCACHED_IGNORE_VALUE_PATTERN: do not store values that contain this pattern (non-regexO MEMCACHED_FAILURE_MODE:
41   * what to do in case of server failure MEMCACHED_STORE_ID_DUPE: boolean to store the id if it's hash is already
42   * contained in memcached PREF_ALG: Which Emissary hash to use as the key stored in memcached
43   */

116     /**
117      *
118      * @param testIdWithSpaces Unused
119      * @param filterName Name of the filter (typically sent in by KffChainLoader)
120      * @param duplicate Filter type (again, sent in by KffChainLoader)
121      * @param testClient Memcached client to be used if specified (will instantiate a client if null)
122      * @throws IOException is thrown if either the file cannot be read of memcached cannot be contacted
123      */
124     public KffMemcached(String testIdWithSpaces, String filterName, FilterType duplicate, @Nullable MemcachedClient testClient) throws IOException {
125         // Set logger to run time class
126         logger = LoggerFactory.getLogger(this.getClass().getName());
127         // Set the logger impl to use log4j
128         System.setProperty("net.spy.log.LoggerImpl", "net.spy.memcached.compat.log.Log4JLogger");
129 
130         // testIdWithSpaces is not used
131         this.ftype = duplicate;
132         this.filterName = filterName;
133 
134         Configurator configG = ConfigUtil.getConfigInfo(KffMemcached.class);
135 
136         // Load up the list of servers
137         Set<String> serversFromConfig = configG.findEntriesAsSet("MEMCACHED_SERVER");
138         List<InetSocketAddress> servers = new ArrayList<>();
139         for (String serverFromConfig : serversFromConfig) {
140             // Transform to an InetSocketAddress
141             if (serverFromConfig.contains(":")) {
142                 String[] serverTokens = serverFromConfig.split(":");
143                 String host = serverTokens[0];
144                 int port = Integer.parseInt(serverTokens[1]);
145                 servers.add(new InetSocketAddress(host, port));
146             } else {
147                 // In this case, assume port is 11211
148                 servers.add(new InetSocketAddress(serverFromConfig, 11211));
149             }
150         }
151 
152         logger.debug("The following memcached servers are configured:");
153         for (InetSocketAddress server : servers) {
154             logger.debug("Server configured: {}", server);
155         }
156 
157         // Default to 24 hours timeout
158         ageoff = configG.findIntEntry("MEMCACHED_AGEOFF", 86400);
159 
160         // Set the preferred algorithm
161         preferredAlgorithm = configG.findStringEntry("PREF_ALG");
162 
163         // Set the preferred algorithm
164         ignorePatterns = configG.findEntriesAsSet("MEMCACHED_IGNORE_VALUE_PATTERN");
165 
166         // Whether or not to keep track of dupe IDs in memcached
167         storeIdDupe = configG.findBooleanEntry("MEMCACHED_STORE_ID_DUPE", false);
168 
169         // Set whether to use the binary protocol or not
170         useBinaryProtocol = configG.findBooleanEntry("MEMCACHED_USE_BINARY_PROTOCOL", useBinaryProtocol);
171 
172         // Set the operation timeout
173         opTimeoutMillis = configG.findLongEntry("MEMCACHED_OP_TIMEOUT_MILLIS", opTimeoutMillis);
174 
175         String failModeAsString = configG.findStringEntry("MEMCACHED_FAILURE_MODE", "Cancel");
176         if (failModeAsString.equalsIgnoreCase("cancel")) {
177             failMode = FailureMode.Cancel;
178         } else if (failModeAsString.equalsIgnoreCase("retry")) {
179             failMode = FailureMode.Retry;
180         }
181 
182         // Finally, setup the client. ConnectionFactoryBuilder ultimately
183         // creates a DefaultConnectionFactory with the values set below
184         ConnectionFactoryBuilder cfb = new ConnectionFactoryBuilder();
185         cfb.setDaemon(true); // Just to keep the process from hanging
186         cfb.setFailureMode(failMode); // How to handle operations when they fail
187         cfb.setMaxReconnectDelay(60); // At most, wait 1 minute for attempting to reconnect to a server
188         cfb.setOpTimeout(opTimeoutMillis); // Use the same for the connection as the concurrent Future object
189 
190         if (useBinaryProtocol) {
191             cfb.setProtocol(Protocol.BINARY);
192         }
193 
194         if (testClient == null) {
195             client = new MemcachedClient(cfb.build(), servers);
196         } else {
197             client = testClient;
198         }
199 
200         // logger.debug(client.toString());
201     }
202 
203     /**
204      * Contact the memcached server and lookup the hash. If it is found, then return true. If it is not found, store it and
205      * return false. If it matches a special ignore pattern, return false. If the server is down or any other problems throw
206      * an exception
207      */
208     @Override
209     public boolean check(String id, ChecksumResults sums) throws Exception {
210 
211         if (sums == null) {
212             throw new Exception("Poorly formed input to check() in sums");
213         }
214 
215         if ((id == null) || (id.length() == 0)) {
216             throw new Exception("Poorly formed input to check() in fname");
217         }
218 
219         // Ignore any IDs that contain the ignorePatterns string verbatim
220         if (ignorePatterns != null) {
221             // Loop through all patterns to see if any match
222             for (String ignorePattern : ignorePatterns) {
223                 if (id.contains(ignorePattern)) {
224                     return false;
225                 }
226             }
227         }
228 
229         byte[] hash = sums.getHash(preferredAlgorithm);
230 
231         if ((hash == null) || (hash.length == 0)) {
232             throw new Exception("Poorly formed input to check() in hash");
233         }
234 
235         String key = Hexl.toUnformattedHexString(hash);
236 
237         // Send the query
238         Future<Object> future = client.asyncGet(key);
239 
240 
241         // Let the TimeoutException propagate up
242         Object result = future.get(opTimeoutMillis, TimeUnit.MILLISECONDS);
243 
244         if (result != null) {
245             if (storeIdDupe && !result.equals(id)) {
246                 // As long as the id is not the same as what was already stored, then
247                 // store it on its own
248                 var unused = client.set(id, ageoff, key);
249                 // logger.debug("Storing duplicate Id: {} with value (hash) {}", id, key);
250             }
251             // logger.debug("Found key: {} with value {}", key, (String) result);
252             // Found the key
253             return true;
254         }
255         // logger.debug("Did not find key: {}", key);
256         // Did not find the key...store it and move on
257         var unused = client.set(key, ageoff, id);
258         return false;
259 
260 
261     }

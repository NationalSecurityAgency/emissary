69          /** A buffer for the main hash output. */
70          private final byte[] fuzzHash1 = new byte[SPAMSUM_LENGTH + 1];
71  
72          /** A buffer for the secondary hash output. */
73          private final byte[] fuzzHash2 = new byte[SPAMSUM_LENGTH / 2 + 1];
74  
75          /** The count of output bytes currently stored in {@link #fuzzHash1}, initially zero. */
76          private int fuzzLen1;
77  
78          /** The count of output bytes currently stored in {@link #fuzzHash2}, initially zero. */

85          /**
86           * Estimate the block size to use.
87           *
88           * @param expectedInputLength The expected amount of data to be processed, in bytes. A 0 value can be used if the length
89           *        is unknown, in which case a default block size is returned.
90           * @return The estimated block size to use.
91           */
92          private static long estimateBlockSize(final long expectedInputLength) {
93              long blockSize = MIN_BLOCKSIZE;
94              while ((blockSize * SPAMSUM_LENGTH) < expectedInputLength) {
95                  blockSize *= 2;
96              }
97              return blockSize;
98          }

140         /**
141          * A simple non-rolling hash, based on the FNV hash
142          * 
143          * @param b The next byte value, assumed to be in the range 0..255.
144          * @param h The existing hash value, 32-bit unsigned.
145          * @return The updated hash value, 32-bit unsigned.
146          */
147         private static long updateSumHash(final int b, final long h) {
148             return ((h * HASH_PRIME) ^ b) & MASK32;
149         }

151         /**
152          * Apply some bytes to a SpamSum context.
153          *
154          * @param rollState The rolling hash state to use.
155          * @param buffer A buffer containing the input bytes.
156          * @param start The starting offset in {@code buffer}, inclusive.
157          * @param end The ending offset in {@code buffer}, exclusive.
158          */
159         private void applyBytes(final RollingState rollState, final byte[] buffer, final int start, final int end) {
160             // At each byte we update the rolling hash and the normal
161             // hash. When the rolling hash hits the reset value, we
162             // emit the normal hash as an element of the signature and
163             // reset both hashes.
164             for (int i = start; i < end; i++) {
165                 // Get the next input byte and normalize to 0..255.
166                 final int nextByte = ((int) buffer[i]) & 0xff;
167 
168                 // Apply the next byte to the hashes.
169                 this.sumHash1 = updateSumHash(nextByte, this.sumHash1);
170                 this.sumHash2 = updateSumHash(nextByte, this.sumHash2);
171                 final long rollingHash = rollState.roll(nextByte);
172 
173                 if ((rollingHash % this.blockSize) == (this.blockSize - 1)) {
174                     // We have hit a reset point. We now emit a hash
175                     // which is based on all bytes in the input
176                     // between the last reset point and this one.
177                     if (this.fuzzLen1 < (SPAMSUM_LENGTH - 1)) {
178                         // We can have a problem with the tail
179                         // overflowing. The easiest way to cope with
180                         // this is to only reset the second hash if we
181                         // have room for more characters in our
182                         // signature. This has the effect of combining
183                         // the last few pieces of the message into a
184                         // single piece
185                         this.fuzzHash1[this.fuzzLen1++] = b64EncodeLowBits(this.sumHash1);
186                         this.sumHash1 = HASH_INIT;
187                     }
188 
189                     // This produces a second signature with a block size
190                     // of blockSize*2. By producing dual signatures in
191                     // this way the effect of small changes in the message
192                     // size near a block size boundary is greatly reduced.
193                     //
194                     // NOTE: we only have to check this when the main
195                     // signature has hit a reset point, because
196                     // mathematically:
197                     //
198                     // [ h === -1 (mod 2*bs) ] --implies--> [ h === -1 (mod bs) ]
199                     //
200                     // In other words, if this condition is true then the
201                     // main signature condition must always also be true.
202                     // Therefore this secondary signature condition can
203                     // only potentially be true if the main signature
204                     // condition (which we've already checked) is true.
205                     if ((rollingHash % (this.blockSize * 2)) == ((this.blockSize * 2) - 1)) {
206                         if (this.fuzzLen2 < (SPAMSUM_LENGTH / 2 - 1)) {
207                             this.fuzzHash2[this.fuzzLen2++] = b64EncodeLowBits(this.sumHash2);
208                             this.sumHash2 = HASH_INIT;
209                         }
210                     }
211                 }
212             }
213         }

215         /**
216          * Discard any existing hash state and prepare to compute a new hash. This should be followed by calls to
217          * {@link #applyBytes(RollingState, byte[], int, int)} to provide the data, and then
218          * {@link #finishHashing(RollingState)} to complete the computations.
219          */
220         private void beginHashing() {
221             this.fuzzLen1 = 0;
222             this.fuzzLen2 = 0;
223             this.sumHash1 = HASH_INIT;
224             this.sumHash2 = HASH_INIT;
225         }

227         /**
228          * Truncate an array if larger than the given length.
229          *
230          * @param input The input array.
231          * @param maxLength The desired maximum array length.
232          * @return If {@code input} is no larger than {@code maxLength}, this just returns {@code input}. Otherwise this returns
233          *         a new array with the same content as {@code input} but with the length truncated to {@code maxLength}.
234          */
235         private static byte[] truncateArray(final byte[] input, final int maxLength) {
236             if (input.length == maxLength) {
237                 return input;
238             } else {
239                 return Arrays.copyOf(input, maxLength);
240             }
241         }

243         /**
244          * Finish hashing and generate the final signature. This should be done after all bytes have been applied with
245          * {@link #applyBytes(RollingState, byte[], int, int)}.
246          *
247          * @param rollState The rolling hash state used during hashing.
248          * @return The final signature.
249          */
250         private SpamSumSignature finishHashing(final RollingState rollState) {
251             if (rollState.getHash() != 0) {
252                 this.fuzzHash1[this.fuzzLen1++] = b64EncodeLowBits(this.sumHash1);
253                 this.fuzzHash2[this.fuzzLen2++] = b64EncodeLowBits(this.sumHash2);
254             }
255 
256             final byte[] finalHash1 = truncateArray(this.fuzzHash1, this.fuzzLen1);
257             final byte[] finalHash2 = truncateArray(this.fuzzHash2, this.fuzzLen2);
258             return new SpamSumSignature(this.blockSize, finalHash1, finalHash2);
259         }

261         /**
262          * Generate the hash for some input.
263          *
264          * <p>
265          * The computations will use the current block size from the context, but any other existing hash state will be
266          * discarded.
267          *
268          * @param data The bytes to hash.
269          * @return The signature for the given data.
270          */
271         public SpamSumSignature generateHash(@Nullable final byte[] data) {
272             beginHashing();
273             final RollingState rollState = new RollingState();
274 
275             if (data != null) {
276                 applyBytes(rollState, data, 0, data.length);
277             }
278 
279             return finishHashing(rollState);
280         }

282         public SpamSumSignature generateHash(final SeekableByteChannelFactory sbcf) {
283             beginHashing();
284             final RollingState rollState = new RollingState();
285 
286             try (InputStream is = Channels.newInputStream(sbcf.create())) {
287                 final byte[] b = new byte[1024];
288 
289                 int bytesRead;
290                 while ((bytesRead = is.read(b)) != -1) {
291                     applyBytes(rollState, b, 0, bytesRead);
292                 }
293             } catch (final IOException e) {
294                 // Ignore
295             }
296 
297             return finishHashing(rollState);
298         }

300         /**
301          * Generate the hash for some input.
302          *
303          * <p>
304          * The computations will use the current block size from the context, but any other existing hash state will be
305          * discarded.
306          *
307          * @param stream A file containing the bytes to hash. Assumed non-{@code null}. The processing will start reading at the
308          *        current file position and hash all of the data from there to the end of the file. The file position when this
309          *        returns is unspecified. The file is not closed by this operation.
310          * @return The signature for the given stream content.
311          * @throws IOException If there is some I/O problem while reading the stream.
312          */
313         public SpamSumSignature generateHash(final RandomAccessFile stream) throws IOException {
314             beginHashing();
315             final RollingState rollState = new RollingState();
316 
317             final byte[] buffer = new byte[BUFFER_SIZE];
318             while (true) {
319                 final int bytesRead = stream.read(buffer, 0, buffer.length);
320                 if (bytesRead <= 0) {
321                     break; // No more input.
322                 }
323                 applyBytes(rollState, buffer, 0, bytesRead);
324             }
325 
326             return finishHashing(rollState);
327         }

330     /**
331      * A rolling hash, based on the Adler checksum. By using a rolling hash we can perform auto resynchronisation after
332      * inserts/deletes.
333      */
334     private static final class RollingState {
335 
336         /** Rolling window. Each value is in the range 0..255. Initially all 0. */
337         private final int[] window = new int[ROLLING_WINDOW_SIZE];
338 
339         /** An index into {@link #window}. Initially 0. */
340         private int windowPosition;
341 
342         /** The sum of the values in {@link #window}. Initially 0. */
343         private long h1;
344 
345         /**
346          * The original documentation says this is the sum of the bytes times the index, but I'm not sure about that. 32-bit
347          * unsigned, initially 0.
348          */
349         private long h2;
350 
351         /**
352          * A shift/xor based rolling hash, mostly needed to ensure that we can cope with large blocksize values. 32-bit
353          * unsigned, initially 0.
354          */
355         private long h3;
356 
357         /**
358          * Construct a new rolling hash state.
359          */
360         public RollingState() {}
361 
362         /**
363          * Get the current hash value.
364          *
365          * @return The current 32-bit unsigned hash value.
366          */
367         public long getHash() {
368             return (this.h1 + this.h2 + this.h3) & MASK32;
369         }
370 
371         /**
372          * Update the rolling hash state with another input byte.
373          *
374          * @param b The byte value to apply. Assumed to be in the range 0..255.
375          * @return The state is updated and the resulting unsigned 32-bit hash value is returned.
376          */
377         public long roll(final int b) {
378             this.h2 = (this.h2 - this.h1 + (ROLLING_WINDOW_SIZE * ((long) b))) & MASK32;
379             this.h1 = (this.h1 + b - this.window[this.windowPosition]) & MASK32;
380             this.window[this.windowPosition] = b;
381 
382             // Advance the window position, wrappping around at the end.
383             if (this.windowPosition == (ROLLING_WINDOW_SIZE - 1)) {
384                 this.windowPosition = 0;
385             } else {
386                 this.windowPosition++;
387             }
388 
389             this.h3 = ((this.h3 << 5) & MASK32) ^ b;
390 
391             return (this.h1 + this.h2 + this.h3) & MASK32;
392         }
393     }

403     public String fuzzyHash(final byte[] data) {
404         final SsContext ctx = new SsContext(data);
405         while (true) {
406             final SpamSumSignature signature = ctx.generateHash(data);
407 
408             // Our blocksize guess may have been way off, repeat with
409             // a smaller block size if necessary.
410             if ((ctx.blockSize > MIN_BLOCKSIZE) && (ctx.fuzzLen1 < (SPAMSUM_LENGTH / 2))) {
411                 ctx.blockSize = ctx.blockSize / 2;
412             } else {
413                 return signature.toString();
414             }
415         }

469     /**
470      * Search an array for a subsequence of another array.
471      *
472      * @param haystack The array to search.
473      * @param needle The array containing the sequence to search for.
474      * @param needleStart The starting offset of the sequence to search for, inclusive. Assumed to be in range for
475      *        {@code needle}.
476      * @param length The length of the sequence to search for. Assumed to be in range for {@code needle}. Assumed greater
477      *        than zero.
478      * @return If the subsequence of {@code needle} is present in {@code haystack}, this returns the least offset where the
479      *         subsequence occurs. Otherwise -1.
480      */
481     private static int indexOfSubSequence(final byte[] haystack, final byte[] needle, final int needleStart, final int length) {
482         final int lastCandidatePos = haystack.length - length;
483         final byte firstNeedleByte = needle[needleStart];
484         NEXT_CANDIDATE: for (int candidatePos = 0; candidatePos <= lastCandidatePos; candidatePos++) {
485             if (haystack[candidatePos] == firstNeedleByte) {
486                 // The first needle byte matches at this candidate
487                 // position, so look for the rest of the needle
488                 // following it.
489                 for (int needlePos = 1; needlePos < length; needlePos++) {
490                     if (haystack[candidatePos + needlePos] != needle[needleStart + needlePos]) {
491                         continue NEXT_CANDIDATE; // Needle mismatch.
492                     }
493                 }
494                 // If we reach here, the entire needle subsequence
495                 // matched in the haystack at the candidate position.
496                 return candidatePos;
497             }
498         }
499         return -1;
500     }

502     /**
503      * Search two arrays for a common subsequence of the given length.
504      *
505      * @param s1 The first byte array for comparison.
506      * @param s2 The second byte array for comparison.
507      * @param length The substring length to look for. Assumed greater than zero.
508      * @return {@code true} iff {@code s1} and {@code s2} have at least one byte sequence of length {@code length} in common
509      *         at arbitrary offsets.
510      */
511     private static boolean hasCommonSequence(final byte[] s1, final byte[] s2, final int length) {
512         if ((s1.length < length) || (s2.length < length)) {
513             return false; // The strings are not large enough.
514         }
515 
516         // This is just a brute-force approach. We move a window of
517         // the specified length through s1 and check whether it exists
518         // anywhere in s2.
519         final int lastS1Pos = s1.length - length;
520         for (int s1Pos = 0; s1Pos <= lastS1Pos; s1Pos++) {
521             final int s2Pos = indexOfSubSequence(s2, s1, s1Pos, length);
522             if (s2Pos != -1) {
523                 return true;
524             }
525         }
526         return false;
527     }

529     /**
530      * Truncate sequences of longer than 3 identical bytes. These sequences contain very little information so they tend to
531      * just bias the result unfairly.
532      *
533      * @param in The input bytes.
534      * @return An array containing the same content as {@code in}, except that any sequences of more than 3 identical bytes
535      *         are truncated to 3 bytes. For example "aaabbbbcddddd" becomes "aaabbbcddd".
536      */
537     private static byte[] eliminateLongSequences(final byte[] in) {
538         if (in.length < 4) {
539             return in; // There is not enough input to require any change.
540         }
541 
542         // We just need to initialize prev to something other than in[0].
543         byte prev = (in[0] != 0) ? 0 : (byte) 1;
544         int repeatCount = 0;
545 
546         // Scan the input looking for the index of the first byte that
547         // will need to be removed.
548         int inPos = 0;
549         while (true) {
550             if (inPos == in.length) {
551                 // We didn't find anything that needed to be removed.
552                 return in;
553             } else if (in[inPos] == prev) {
554                 // This is a repeat of the previous byte.
555                 repeatCount++;
556                 if (repeatCount == 3) {
557                     break; // inPos needs to be removed.
558                 }
559             } else {
560                 // This is not a repeat of the previous byte.
561                 prev = in[inPos];
562                 repeatCount = 0;
563             }
564             inPos++;
565         }
566 
567         // At this point inPos is the first index that needs to be
568         // removed, prev is set to its byte value, and repeatCount is
569         // set to 3. Start an output array and copy everything up to
570         // but not including inPos.
571         final byte[] out = new byte[in.length - 1];
572         System.arraycopy(in, 0, out, 0, inPos);
573         int outPos = inPos;
574 
575         // Continue scanning and copying to output.
576         while (++inPos < in.length) {
577             if (in[inPos] == prev) {
578                 repeatCount++;
579             } else {
580                 prev = in[inPos];
581                 repeatCount = 0;
582             }
583             if (repeatCount < 3) {
584                 out[outPos++] = in[inPos];
585             }
586         }
587 
588         return (outPos == out.length) ? out : Arrays.copyOf(out, outPos);
589     }

591     /**
592      * This is the low level string scoring algorithm. It takes two strings and scores them on a scale of 0-100 where 0 is a
593      * terrible match and 100 is a great match. The blockSize is used to cope with very small messages.
594      */
595     private static long scoreStrings(final byte[] s1, final byte[] s2, final long blockSize) {
596         final int len1 = s1.length;
597         final int len2 = s2.length;
598 
599         if ((len1 > SPAMSUM_LENGTH) || (len2 > SPAMSUM_LENGTH)) {
600             // not a real spamsum signature?
601             return 0;
602         }
603 
604         // The two strings must have a common substring of length
605         // ROLLING_WINDOW_SIZE to be candidates.
606         if (!hasCommonSequence(s1, s2, ROLLING_WINDOW_SIZE)) {
607             return 0;
608         }
609 
610         // Compute the edit distance between the two strings. The edit
611         // distance gives us a pretty good idea of how closely related
612         // the two strings are.
613         long score = EditDistance.calculateEditDistance(s1, len1, s2, len2);
614         if (logger.isDebugEnabled()) {
615             logger.debug("edit_dist: {}", score);
616         }
617 
618         // Scale the edit distance by the lengths of the two
619         // strings. This changes the score to be a measure of the
620         // proportion of the message that has changed rather than an
621         // absolute quantity. It also copes with the variability of
622         // the string lengths.
623         score = (score * SPAMSUM_LENGTH) / (len1 + len2);
624 
625         // At this stage the score occurs roughly on a 0-64 scale,
626         // with 0 being a good match and 64 being a complete mismatch.
627 
628         // Rescale to a 0-100 scale (friendlier to humans).
629         score = (100 * score) / 64;
630 
631         // It is possible to get a score above 100 here, but it is a
632         // really terrible match.
633         if (score >= 100) {
634             return 0;
635         }
636 
637         // Now re-scale on a 0-100 scale with 0 being a poor match and
638         // 100 being a excellent match.
639         score = 100 - score;
640 
641         // When the blocksize is small we don't want to exaggerate the
642         // match size.
643         if (score > (blockSize / MIN_BLOCKSIZE * Math.min(len1, len2))) {
644             score = blockSize / MIN_BLOCKSIZE * Math.min(len1, len2);
645         }
646         return score;
647     }

650      * Given two spamsum signature return a value indicating the degree to which they match.
651      *
652      * @param signature1 The first signature.
653      * @param signature2 The second signature.
654      * @return The score for the two signatures. The value is in the range 0..100, where 0 is a terrible match and 100 is a
655      *         great match.
656      */
657     public int compare(@Nullable final SpamSumSignature signature1, @Nullable final SpamSumSignature signature2) {
658         if ((null == signature1) || (null == signature2)) {
659             return -1;
660         }
661         final long blockSize1 = signature1.getBlockSize();
662         final long blockSize2 = signature2.getBlockSize();
663 
664         // We require the block sizes to either be equal, or for one
665         // to be twice the other. If the blocksizes don't match then
666         // we are comparing apples to oranges. This isn't an 'error'
667         // per se. We could have two valid signatures, but they can't
668         // be compared.
669         if ((blockSize1 != blockSize2) && (blockSize1 != (blockSize2 * 2)) && (blockSize2 != (blockSize1 * 2))) {
670             if (logger.isDebugEnabled()) {
671                 logger.debug("block sizes too different: {} {}", blockSize1, blockSize2);
672             }
673             return 0;
674         }
675 
676         // There is very little information content is sequences of
677         // the same character like 'LLLLL'. Eliminate any sequences
678         // longer than 3. This is especially important when combined
679         // with the hasCommonSequence() test.
680         final byte[] s1First = eliminateLongSequences(signature1.getHashPart1());
681         final byte[] s1Second = eliminateLongSequences(signature1.getHashPart2());
682         final byte[] s2First = eliminateLongSequences(signature2.getHashPart1());
683         final byte[] s2Second = eliminateLongSequences(signature2.getHashPart2());
684 
685         // Each signature has a string for two block sizes. We now
686         // choose how to combine the two block sizes. We checked above
687         // that they have at least one block size in common.
688         final long score;
689         if (blockSize1 == blockSize2) {
690             // The signature block sizes are equal.
691             final long score1 = scoreStrings(s1First, s2First, blockSize1);
692             final long score2 = scoreStrings(s1Second, s2Second, blockSize2);
693             score = Math.max(score1, score2);
694         } else if (blockSize1 == (blockSize2 * 2)) {
695             // The first signature has twice the block size of the second.
696             score = scoreStrings(s1First, s2Second, blockSize1);
697         } else {
698             // The second signature has twice the block size of the first.
699             score = scoreStrings(s1Second, s2First, blockSize2);
700         }
701 
702         return (int) score;
703     }

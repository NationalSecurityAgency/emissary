15  /**
16   * KffFile provides access to the known file filter data. The NIST/NSRL data is a CSV file with other information. It
17   * must be preprocessed in order for this class to access it. The input file for this class must consist of a sorted
18   * list of known values, where a known value is the CRC32 appended to the SHA-1. This file must be a binary file, so
19   * each record will be 24 bytes long (20-byte SHA + 4-byte CRC). The CRC should be big endian.
20   *
21   * Implementation notes: The binary input file is too big to read into memory, so we implement a binary search on the
22   * file itself. This is why the records must be sorted, and it will improve performance if only unique records are
23   * generated as well. This class assumes JDK1.4+ and memory maps the file. For earlier versions of the JDK, we can seek
24   * through the RandomAccessFile instead but performance isn't as good. The file cannot be larger than 2^31 (2 GB)
25   * because that is the maximum length of the mapped ByteBuffer.
26   *
27   * Update: 11/27/2006 Having out-of-memory issues with all the memory mapping we're trying to do, (multiple KFFs, plus
28   * HotSpot), so try using just RandomAccessFile instead.
29   */

66      /**
67       * Creates a new instance of KffFile
68       *
69       * @param filename Name of binary file containing sorted RECORD_LENGTH records that are the hash codes possibly
70       *        concatenated with the CRC-32
71       * @param filterName the name of this filter
72       * @param ftype type of this filter
73       * @param recordLength fixed record length in file
74       * @throws IOException if thrown by file I/O
75       */
76      public KffFile(String filename, String filterName, FilterType ftype, int recordLength) throws IOException {
77          this.ftype = ftype;
78          this.filterName = filterName;
79          this.recordLength = recordLength;
80  
81          // Set logger to run time class
82          logger = LoggerFactory.getLogger(this.getClass());
83  
84          // Open file in read-only mode
85          knownFile = new RandomAccessFile(filename, "r");
86  
87          // Initial high value for binary search is the largest index
88          bSearchInitHigh = ((int) knownFile.length() / recordLength) - 1;
89  
90          logger.debug("KFF File {} has {} records", filename, (bSearchInitHigh + 1));
91      }

135     /**
136      * Performs a binary search on the file to see if a given HASH/CRC is in the list.
137      *
138      * @param hash Result of HASH calculation
139      * @param crc Result of CRC calculation
140      * @return true if the record is in the list, false if it isn't
141      */
142     private boolean binaryFileSearch(@Nonnull byte[] hash, long crc) {
143 
144         // Initialize indexes for binary search
145         int low = 0;
146         int high = bSearchInitHigh;
147 
148         /* Buffer to hold a record */
149         byte[] rec = new byte[recordLength];
150 
151         // Search until the indexes cross
152         while (low <= high) {
153             // Calculate the midpoint
154             int mid = (low + high) >> 1;
155 
156             try {
157                 knownFile.seek(rec.length * (long) mid);
158                 int count = knownFile.read(rec);
159                 if (count != rec.length) {
160                     logger.warn("Short read on KffFile at {} read {} expected {}", (rec.length * mid), count, rec.length);
161                     return false;
162                 }
163             } catch (IOException x) {
164                 logger.warn("Exception reading KffFile: {}", x.getMessage());
165                 return false;
166             }
167 
168             // Compare the record with the target. Adjust the indexes accordingly.
169             int c = compare(rec, hash, crc);
170             if (c < 0) {
171                 high = mid - 1;
172             } else if (c > 0) {
173                 low = mid + 1;
174             } else {
175                 return true;
176             }
177         }
178 
179         // not found
180         return false;
181     }

183     /**
184      * Compares the given hash/crc to the one in the record.
185      *
186      * @param rec bytes from the kff binary file, one record long
187      * @param hash HASH to compare to record
188      * @param crc CRC to compare to record
189      * @return &lt;0 if given value is less than record, &gt;0 if given value is greater than record, 0 if they match
190      */
191     private int compare(@Nonnull byte[] rec, @Nonnull byte[] hash, long crc) {
192         int i;
193 
194         // Compare the HASHs first. We can't compare the bytes directly
195         // because a Java byte is signed and may generate the wrong
196         // result. We must convert to integers and then mask off the
197         // sign bits to get proper results.
198         for (i = 0; i < hash.length; i++) {
199             int ihash = hash[i] & 0xff;
200             int irec = rec[i] & 0xff;
201             if (ihash < irec) {
202                 return -1;
203             } else if (ihash > irec) {
204                 return 1;
205             }
206         }
207 
208         // If the HASHs match, check the CRCs.
209         if (crc != -1L) {
210             for (int j = 24; i < rec.length; i++, j -= 8) {
211                 int icrc = ((int) crc >> j) & 0xff;
212                 int irec = rec[i] & 0xff;
213                 if (icrc < irec) {
214                     return -1;
215                 } else if (icrc > irec) {
216                     return 1;
217                 }
218             }
219         }
220         return 0;
221     }

223     @Override
224     public boolean check(String fname, ChecksumResults csum) throws Exception {
225         byte[] hash = csum.getHash(myPreferredAlgorithm);
226         if (hash == null) {
227             logger.warn("Filter cannot be used, {} not computed on {}", myPreferredAlgorithm, fname);
228             return false;
229         }
230         return binaryFileSearch(hash, csum.getCrc());
231     }

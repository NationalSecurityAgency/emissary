20  /**
21   * Loads a chain of file filter specified by the configuration subsystem Expects to find a configuration file with a
22   * list containing class names that implement known or dupe type filter working either against file stores of hashes or
23   * database tables of hashes, plus a list of the algorithms that are desired to use. This should be a superset of the
24   * algorithms required by all of the filter in the chain and those desired as end product in their own right. This is a
25   * singleton implementation.
26   */
27  public class KffChainLoader {
28  
29      private static final Logger logger = LoggerFactory.getLogger(KffChainLoader.class);
30  
31      @Nullable
32      @SuppressWarnings("NonFinalStaticField")
33      private static KffChain theInstance = null;
34  
35      @SuppressWarnings("NonFinalStaticField")
36      private static Map<String, String> classes;
37  
38      /**
39       * Take away the public constructor
40       */
41      private KffChainLoader() {}
42  
43      /**
44       * Construct KFF chain elements from the specified configG or return the already constructed instance.
45       */
46      public static synchronized KffChain getChainInstance() {
47          if (theInstance == null) {
48              KffChain chain = new KffChain();
49              try {
50                  Configurator configG = ConfigUtil.getConfigInfo(KffChain.class);
51                  classes = configG.findStringMatchMap("KFF_IMPL_");
52                  loadFrom(chain, configG.findStringMatchMap("KFF_FILE_KNOWN_"), FilterType.IGNORE);
53                  loadFrom(chain, configG.findStringMatchMap("KFF_FILE_DUPE_"), FilterType.DUPLICATE);
54  
55                  chain.setMinDataSize(configG.findIntEntry("KFF_MIN_SIZE", 0));
56                  Set<String> algs = configG.findEntriesAsSet("KFF_ALG");
57                  chain.setAlgorithms(algs);
58              } catch (IOException iox) {
59                  logger.debug("No configuration for Known File Filter. Continuing...");
60              }
61              theInstance = chain;
62              logger.debug("KFF Chain loaded with {} filter using algorithms {}", theInstance.size(), theInstance.getAlgorithms());
63          }
64          return theInstance;
65      }
66  
67  
68      /**
69       * Load a set from one of the keys into the chain
70       *
71       * @param chain the chain we are loading
72       * @param m map of config entries items
73       * @param filterType either IGNORE, KNOWN, or DUPE filter
74       * @return number of filter loaded onto chain
75       */
76      private static int loadFrom(KffChain chain, Map<String, String> m, FilterType filterType) {
77          int countLoaded = 0;
78  
79          // Load KFF File filter
80          for (Map.Entry<String, String> entry : m.entrySet()) {
81              String key = entry.getKey();
82              String name = entry.getValue();
83              try {
84                  String clazz = classes.get(key);
85                  if (clazz == null || clazz.length() == 0) {
86                      // cannot construct a null class for key
87                      continue;
88                  }
89  
90                  // see if known KffType
91                  KffFilter k;
92                  try {
93                      k = (KffFilter) Factory.create(clazz, name, key, filterType);
94                  } catch (RuntimeException x) {
95                      logger.warn("Cannot create KffFilter, using default", x);
96                      k = new KffFile(name, key, filterType);
97                  }
98  
99                  chain.addFilter(k);
100                 countLoaded++;
101             } catch (IOException e) {
102                 logger.error("Exception creating KFF chain element", e);
103             }
104         }
105         return countLoaded;
106     }
107 
108     /**
109      * Load the configured chain and run some data
110      */
111     @SuppressWarnings("SystemOut")
112     public static void main(String[] args) throws Exception {
113         KffChain kff = getChainInstance();
114 
115         for (int i = 0; i < args.length; i++) {
116             try (InputStream is = Files.newInputStream(Paths.get(args[i]))) {
117                 byte[] buffer = IOUtils.toByteArray(is);
118 
119                 KffResult r = kff.check(args[i], buffer);
120                 System.out.println(args[i] + ": known=" + r.isKnown());
121                 System.out.println("   CRC32: " + r.getCrc32());
122                 for (String s : r.getResultNames()) {
123                     System.out.println("   " + s + ": " + r.getResultString(s));
124                 }
125             }
126         }
127     }

20  /**
21   * Loads a chain of file filter specified by the configuration subsystem Expects to find a configuration file with a
22   * list containing class names that implement known or dupe type filter working either against file stores of hashes or
23   * database tables of hashes, plus a list of the algorithms that are desired to use. This should be a superset of the
24   * algorithms required by all of the filter in the chain and those desired as end product in their own right. This is a
25   * singleton implementation.
26   */
27  public class KffChainLoader {
28  
29      private static final Logger logger = LoggerFactory.getLogger(KffChainLoader.class);
30  
31      @Nullable
32      private static KffChain theInstance = null;
33      private static Map<String, String> classes;
34  
35      /**
36       * Take away the public constructor
37       */
38      private KffChainLoader() {}
39  
40      /**
41       * Construct KFF chain elements from the specified configG or return the already constructed instance.
42       */
43      public static synchronized KffChain getChainInstance() {
44          if (theInstance == null) {
45              KffChain chain = new KffChain();
46              try {
47                  Configurator configG = ConfigUtil.getConfigInfo(KffChain.class);
48                  classes = configG.findStringMatchMap("KFF_IMPL_");
49                  loadFrom(chain, configG.findStringMatchMap("KFF_FILE_KNOWN_"), FilterType.IGNORE);
50                  loadFrom(chain, configG.findStringMatchMap("KFF_FILE_DUPE_"), FilterType.DUPLICATE);
51  
52                  chain.setMinDataSize(configG.findIntEntry("KFF_MIN_SIZE", 0));
53                  Set<String> algs = configG.findEntriesAsSet("KFF_ALG");
54                  chain.setAlgorithms(algs);
55              } catch (IOException iox) {
56                  logger.debug("No configuration for Known File Filter. Continuing...");
57              }
58              theInstance = chain;
59              logger.debug("KFF Chain loaded with {} filter using algorithms {}", theInstance.size(), theInstance.getAlgorithms());
60          }
61          return theInstance;
62      }
63  
64  
65      /**
66       * Load a set from one of the keys into the chain
67       *
68       * @param chain the chain we are loading
69       * @param m map of config entries items
70       * @param filterType either IGNORE, KNOWN, or DUPE filter
71       * @return number of filter loaded onto chain
72       */
73      private static int loadFrom(KffChain chain, Map<String, String> m, FilterType filterType) {
74          int countLoaded = 0;
75  
76          // Load KFF File filter
77          for (Map.Entry<String, String> entry : m.entrySet()) {
78              String key = entry.getKey();
79              String name = entry.getValue();
80              try {
81                  String clazz = classes.get(key);
82                  if (clazz == null || clazz.length() == 0) {
83                      // cannot construct a null class for key
84                      continue;
85                  }
86  
87                  // see if known KffType
88                  KffFilter k;
89                  try {
90                      k = (KffFilter) Factory.create(clazz, name, key, filterType);
91                  } catch (Exception x) {
92                      logger.warn("Cannot create KffFilter, using default", x);
93                      k = new KffFile(name, key, filterType);
94                  }
95  
96                  chain.addFilter(k);
97                  countLoaded++;
98              } catch (IOException e) {
99                  logger.error("Exception creating KFF chain element", e);
100             }
101         }
102         return countLoaded;
103     }
104 
105     /**
106      * Load the configured chain and run some data
107      */
108     @SuppressWarnings("SystemOut")
109     public static void main(String[] args) throws Exception {
110         KffChain kff = getChainInstance();
111 
112         for (int i = 0; i < args.length; i++) {
113             try (InputStream is = Files.newInputStream(Paths.get(args[i]))) {
114                 byte[] buffer = IOUtils.toByteArray(is);
115 
116                 KffResult r = kff.check(args[i], buffer);
117                 System.out.println(args[i] + ": known=" + r.isKnown());
118                 System.out.println("   CRC32: " + r.getCrc32());
119                 for (String s : r.getResultNames()) {
120                     System.out.println("   " + s + ": " + r.getResultString(s));
121                 }
122             }
123         }
124     }

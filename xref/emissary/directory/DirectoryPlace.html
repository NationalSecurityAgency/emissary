25  /**
26   * The DirectoryPlace class is used to store information relating to Places/Services in the Emissary Agent-Based
27   * architecture. When a Place comes up it calls the method addPlace passing in all the relevant information to store in
28   * the Directory. Agents query the directory by calling the method nextKeys which requires a query String search
29   * pattern.
30   *
31   * <p>
32   * We try to support some network topographic constructions by providing a set of peer directories. Peers are monitored
33   * and checked automatically by HeartbeatManager and the peer network is assumed to be fully connected. Peer directories
34   * are a fairly-static list of peer directories read from a config file. At least one host must be listed in order to
35   * bootstrap the network.
36   *
37   * <p>
38   * Emissary directory instances are also observable with respect to Peer activities, and Place activities. Peer
39   * observers will be called with a list of current members of the peer group (including this directory) whenever the
40   * peer group loses or gains members. Place observers will be called with a key that matches the pattern supplied on
41   * their subscription and an indication of whether it is a register or deregister or cost change.
42   *
43   */
44  public class DirectoryPlace extends ServiceProviderPlace implements IRemoteDirectory {

144     /**
145      * Shared code for all the constructors to take advantage of in initializing directory services Configuration items read
146      * here are
147      * <ul>
148      * <li>HEARTBEAT_DELAY_SECONDS, default is 30</li>
149      * <li>HEARTBEAT_INTERVAL_SECONDS, default is 30</li>
150      * <li>HEARTBEAT_FAILURE_THRESHOLD, set transient failure count, default owned by HeartbeatManager</li>
151      * <li>HEARTBEAT_PERMANENT_FAILURE_THRESHOLD, set permanent failure count, default owned by HeartbeatManager</li>
152      * </ul>
153      */
154     private void setupDirectory() {
155         if (this.emissaryNode.isValid() && !this.emissaryNode.isStandalone()) {
156             // Start a heart beat manager with initial and interval seconds
157             final int initialSeconds = configG.findIntEntry("HEARTBEAT_DELAY_SECONDS", 30);
158             final int intervalSeconds = configG.findIntEntry("HEARTBEAT_INTERVAL_SECONDS", 30);
159 
160             this.heartbeat = new HeartbeatManager(myKey, initialSeconds, intervalSeconds);
161 
162             final int heartbeatFailureThreshold = configG.findIntEntry("HEARTBEAT_FAILURE_THRESHOLD", -1);
163             if (heartbeatFailureThreshold > 0) {
164                 this.heartbeat.setFailThreshold(heartbeatFailureThreshold);
165             }
166 
167             final int heartbeatPermanentFailure = configG.findIntEntry("HEARTBEAT_PERMANENT_FAILURE_THRESHOLD", -1);
168             if (heartbeatPermanentFailure > 0) {
169                 this.heartbeat.setPermanentFailThreshold(heartbeatPermanentFailure);
170             }
171         }
172 
173         // Set up deferred stuff from ServiceProviderPlace
174         // for directories only we are our own localDirPlace
175         // and the key is our own key
176         localDirPlace = this;
177         dirPlace = myKey;
178 
179         // Start an observer manager
180         this.observerManager = new DirectoryObserverManager(myKey);
181 
182         // Configure my initial rendezvous peers
183         configureNetworkTopology();
184 
185         // Add an entry representing myself into the
186         // local entry map. This allows observers to
187         // work for this case, and allows Jetty instances
188         // with just a DirectoryPlace and some bunches
189         // of other non-Place code to function well and trigger
190         // the peer discovery mechanism when they zone transfer
191         // this entry
192         final List<String> list = new ArrayList<>();
193         list.add(keys.get(0));
194         addPlaces(list);
195         this.running = true;
196     }

198     /**
199      * Find an optional peer config stream or file and initialize tracking of the peers found there.
200      * <p>
201      * We don't actually contact any of the remote directories here, so we can get the heck out of the constructor code and
202      * get this place registered in the namespace quick! so other directories can find us in a timely fashion.
203      */
204     private void configureNetworkTopology() {
205         if (!this.emissaryNode.isValid()) {
206             if (this.emissaryNode.isStandalone()) {
207                 logger.debug("Running as a standalone emissary node");
208             } else {
209                 logger.debug("Not configured as an emissary node");
210             }
211             return;
212         }
213 
214         logger.debug("Emissary node info: {}", this.emissaryNode);
215 
216         try {
217             // Peer network configuration is from peer.cfg
218             final Configurator peerConfig = this.emissaryNode.getPeerConfigurator();
219             final Set<String> peers = peerConfig.findEntriesAsSet("RENDEZVOUS_PEER");
220             this.staticPeers.addAll(peers);
221             addPeerDirectories(peers, true);
222 
223             logger.debug("Configured {} rendezvous peers from {} config entries.", this.peerDirectories.size(), peers.size());
224             logger.debug("This directory is {}a rendezvous peer.", this.rdvPeer ? "" : "NOT (yet) ");
225         } catch (IOException iox) {
226             logger.debug("There is no peer.cfg data available");
227         }
228     }

273     /**
274      * Add a Set of peer directory to this one
275      *
276      * @param keys set of string key for peer directories
277      * @param initPhase true if during place initialization
278      */
279     // TODO Look at DirectoryPlaceTest at the cases where spied methods are used
280     public void addPeerDirectories(final Set<String> keys, final boolean initPhase) {
281 
282         if (this.shutdownInitiated) {
283             logger.error("Shutdown has been initiated. Cannot add peer directories in this state.");
284             return;
285         }
286 
287         boolean changeMade = false;
288 
289         for (final String key : keys) {
290             if (isLocal(key)) {
291                 // I am listed as a rendezvous for someone
292                 this.rdvPeer = true;
293                 continue;
294             }
295 
296             if (this.emissaryNode.isStandalone()) {
297                 logger.debug("Not adding peers in standalone nodes");
298                 continue;
299             }
300 
301             if (!isStaticPeer(key)) {
302                 logger.warn("Unknown peer requesting to be added: {}", key);
303                 continue;
304             }
305 
306             if (!isKnownPeer(key)) {
307                 this.peerDirectories.add(new DirectoryEntry(key));
308                 logger.debug("Added peer directory {}", key);
309 
310                 // Setup heartbeat to new peer directory
311                 if (initPhase) {
312                     // not contacted yet
313                     this.heartbeat.addRemoteDirectory(key, HeartbeatManager.NO_CONTACT);
314                 } else {
315                     // already contacted
316                     this.heartbeat.addRemoteDirectory(key, HeartbeatManager.IS_ALIVE);
317 
318                     // Initial transfer of remote directory info here
319                     // It may not be up yet, so be resilient
320                     loadPeerEntries(key);
321                 }
322 
323 
324                 changeMade = true;
325             } else {
326                 logger.debug("We already knew about peer {}", key);
327                 if (!this.heartbeat.isAlive(key)) {
328                     logger.debug("Forcing peer {} alive due to arriving registration", key);
329                     this.heartbeat.setHealthStatus(key, HeartbeatManager.IS_ALIVE, "Received peer registration");
330                     loadPeerEntries(key);
331                 }
332             }
333         }
334 
335         // Notify all observers
336         if (changeMade) {
337             this.observerManager.peerUpdate(new HashSet<>(this.peerDirectories));
338         }
339     }

341     /**
342      * Retrieve and load (zone transfer) all the entries from the specified peer directory. Zone transfers do not trigger
343      * observables like addPlaces does
344      *
345      * @param peerKey the key of the peer directory
346      */
347     protected void loadPeerEntries(final String peerKey) {
348 
349         if (this.emissaryNode.isStandalone()) {
350             logger.debug("Cannot load peer entries in standalone nodes");
351             return;
352         }
353 
354         logger.debug("Doing zone transfer with peer {}", peerKey);
355         // TODO See DirectoryPlace for spy example which needs to be addressed
356         final DirectoryEntryMap newEntries = loadRemoteEntries(peerKey, this.entryMap);
357         if ((newEntries == null) || newEntries.isEmpty()) {
358             logger.debug("We got nothing back from the peer zone xfer");
359             return;
360         }
361 
362         // We just did this guy remove his stuff
363         newEntries.removeAllOnDirectory(peerKey);
364 
365         // Remove local stuff
366         newEntries.removeAllOnDirectory(myKey);
367 
368         // Make note of any possible new peer directory
369         // We should only be seeing peers here
370         final Set<String> newPeers = new HashSet<>();
371         for (final DirectoryEntry newEntry : newEntries.allEntries()) {
372             if (!isLocal(newEntry)) {
373                 final String possiblePeer = KeyManipulator.getDefaultDirectoryKey(newEntry.getKey());
374                 if (!isKnownPeer(possiblePeer) && !newPeers.contains(possiblePeer)) {
375                     logger.debug("Discovered new peer {} from {} during zt with {}", possiblePeer, newEntry.getKey(), peerKey);
376                     newPeers.add(possiblePeer);
377                 }
378             }
379         }
380         if (!newPeers.isEmpty()) {
381             logger.debug("Adding {} new peers from zt with {}", newPeers.size(), peerKey);
382             addPeerDirectories(newPeers, false);
383         }
384     }

386     /**
387      * Retrieve and load (zone transfer) all the entries from specified remote directory into the specified map. Remove any
388      * stale entries from the destination map if one is specified and merge in the new entries. Zone transfers do not
389      * trigger observables like addPlaces does
390      *
391      * @param key key of the remote directory to transfer from
392      * @param loadMap the map to load into or null for no load. Observers are notified if loadMap is not null
393      * @return the new entries
394      */
395     @Nullable
396     private DirectoryEntryMap loadRemoteEntries(final String key, @Nullable final DirectoryEntryMap loadMap) {
397 
398         if (this.emissaryNode.isStandalone()) {
399             logger.debug("Cannot load remote entries in standalone nodes");
400             return null;
401         }
402 
403         if (!isStaticPeer(key)) {
404             logger.debug("Ignoring non-configured peer {}", key);
405             return null;
406         }
407 
408         // Track how long the zone transfer takes and use that
409         // info along with the slop window to help determine if
410         // there are stale entries and what they might be.
411 
412         final long startZone = System.currentTimeMillis();
413         DirectoryEntryMap map = null;
414         try {
415             // Also registers as a peer with them
416             // TODO should we need to get the current EmissaryClient to ensure parameters are set correctly
417             final DirectoryAdapter da = new DirectoryAdapter();
418             map = da.outboundRegisterPeer(key, myKey);
419 
420             if (logger.isDebugEnabled()) {
421                 logger.debug("Retrieved {} entries in zone transfer from {} in {} millis", map.entryCount(), key,
422                         System.currentTimeMillis() - startZone);
423             }
424 
425             // No entries mean we got the remote message,
426             // and they just don't have any places registered yet
427             if (map.isEmpty()) {
428                 return map;
429             }
430 
431             if (loadMap != null) {
432 
433                 // Remove and notify of any stale entries in loadMap
434                 removeStaleEntries(loadMap, key, startZone - this.zoneSlopWindowMillis, map, true);
435 
436                 // Remove any duplicate entries from map
437                 // so that they don't get double notified to observers
438                 // do the load and notify all observers
439                 cleanLoadNotifyEntries(map, loadMap, myKey, REMOTE_COST_OVERHEAD);
440             } else {
441                 logger.debug("Skipping load of {} new entries from {} returning list to caller", map.entryCount(), key);
442             }
443         } catch (Exception ex) {
444             if (logger.isDebugEnabled()) {
445                 logger.debug("Unable to zone transfer with {}", key, ex);
446             } else {
447                 logger.info("Unable to zone transfer with {}", key);
448             }
449             // Failure condition. Trigger state change in heartbeat manager
450             this.heartbeat.setHealthStatus(key, HeartbeatManager.NO_CONTACT, "Remote directory failed zone transfer");
451         }
452 
453         return map;
454     }

456     /**
457      * Remove stale entries from the specified map and notify any observers Nothing older than checkpoint time can be
458      * considered stale and nothing that is on the incming newEntries list can be considered stale since we would just be
459      * adding it back again. Duplicates (non-stale entries) are removed from the newEntries map to avoid further confusion
460      * but only if the cost is the same. Otherwise, we leave it so that a cost-change event can propagete from later code
461      * but still avoid triggering a place removed event.
462      *
463      * @param loadMap the map we are removing from
464      * @param key the key of the directory whose entries might be stale
465      * @param checkpoint the time window to determine possible staleness
466      * @param newEntries the new map arriving
467      * @param performNotification only use observerManager if true
468      * @return list of entries that were removed
469      */
470     private List<DirectoryEntry> removeStaleEntries(final DirectoryEntryMap loadMap, final String key, final long checkpoint,
471             @Nullable final DirectoryEntryMap newEntries, final boolean performNotification) {
472 
473         final List<DirectoryEntry> staleEntries = new ArrayList<>();
474 
475         // Nothing newer than the checkpoint time can be stale
476         // Nothing that is in teh loadMap but also duplicated in
477         // the newEntries map can be stale either. This helps eliminate
478         // the problem of removing it just so we can add it back.
479         // This uses a mark and sweep to prevent concurrent mod exceptions
480         for (final DirectoryEntry d : loadMap.collectAllMatching(key)) {
481             // is it old enough to be possibly stale
482             if (d.getAge() < checkpoint) {
483                 // is it missing from the new list
484                 if (newEntries != null) {
485                     final List<DirectoryEntry> matches = newEntries.collectAllMatching(d.getKey());
486                     if (matches.isEmpty()) {
487                         logger.debug("Marking stale entry {}", d.getKey());
488                         staleEntries.add(d);
489                     } else if (matches.size() == 1) {
490                         // remove from newEntries if exact dup
491                         final DirectoryEntry me = matches.get(0);
492                         if (me.getFullKey().equals(d.getFullKey())) {
493                             logger.debug("Removing duplcate key from incoming map {}", me.getKey());
494                             newEntries.removeEntry(me.getKey());
495                         }
496                     }
497                 } else {
498                     // must be stale if no newEntries
499                     logger.debug("Marking stale entry (no new entries){}", d.getKey());
500                     staleEntries.add(d);
501                 }
502             }
503         }
504 
505         // Remove and notify
506         if (!staleEntries.isEmpty()) {
507             for (final DirectoryEntry stale : staleEntries) {
508                 logger.debug("Removing stale entry {}", stale.getKey());
509                 loadMap.removeEntry(stale.getKey());
510             }
511 
512             if (performNotification) {
513                 logger.debug("Notifying observers of {} stale entry removals", staleEntries.size());
514                 this.observerManager.placeRemoveEntries(staleEntries);
515             }
516         } else {
517             logger.debug("There were no stale entries to remove");
518         }
519 
520         return staleEntries;
521     }

523     /**
524      * Grok the details of a new entry list and figure out which observers need to be notified. Remove any entries that are
525      * not going to end up being added anyway.
526      *
527      * @param map the new entries to understand
528      * @param loadMap the map the entries will be loaded into
529      * @param purgeKey remove any keys matching
530      * @param costBump add cost to incoming
531      */
532     private void cleanLoadNotifyEntries(final DirectoryEntryMap map, @Nullable final DirectoryEntryMap loadMap, @Nullable final String purgeKey,
533             final int costBump) {
534         // Remove local entries from the new map
535         // We already know about our local stuff.
536         if (purgeKey != null) {
537             final List<DirectoryEntry> removed = map.removeAllOnDirectory(purgeKey);
538             logger.debug("Clean/load removed {} entries based on {} remaining = {}", removed.size(), purgeKey, map.entryCount());
539         }
540 
541         // Add remote overhead to remaining
542         if (costBump > 0) {
543             map.addCostToMatching("*.*.*.*", costBump);
544             logger.debug("Clean/load did cost-bump of {} on {} entries", costBump, map.entryCount());
545         }
546 
547         if (loadMap != null) {
548             final DirectoryEntryMap newEntries = new DirectoryEntryMap();
549             final DirectoryEntryMap costChangeEntries = new DirectoryEntryMap();
550             for (final DirectoryEntry e : map.allEntries()) {
551                 final List<DirectoryEntry> matches = loadMap.collectAllMatching(e.getKey());
552                 if (matches.isEmpty()) {
553                     newEntries.addEntry(e);
554                 } else if ((matches.size() == 1) && e.isBetterThan(matches.get(0))) {
555                     costChangeEntries.addEntry(e);
556                 }
557             }
558 
559             // Merge remaining truly new entries and notify observers
560             final int newCount = newEntries.entryCount();
561             final int cceCount = costChangeEntries.entryCount();
562             if (newCount > 0) {
563                 logger.debug("Loading {} new entries", newCount);
564                 loadMap.addEntries(newEntries);
565                 this.observerManager.placeAdd(newEntries.allEntryKeys());
566             } else {
567                 logger.debug("Nothing truly new from {} entries", map.entryCount());
568             }
569 
570             // .. and cost change entries
571             if (cceCount > 0) {
572                 logger.debug("Loading {} better cost entries", cceCount);
573                 loadMap.addEntries(costChangeEntries);
574                 this.observerManager.placeCostChange(costChangeEntries.allEntryKeys());
575             } else {
576                 logger.debug("No cost change entries from {} entries", map.entryCount());
577             }
578 
579             // Now let the map that gets returned have just the new
580             // and cost changed entries, no already known stuff
581             if ((newCount + cceCount) < map.entryCount()) {
582                 map.clear();
583                 map.addEntries(costChangeEntries);
584                 map.addEntries(newEntries);
585                 map.sort();
586             }
587         } else {
588             logger.debug("Clean/load got a null loadMap so skipping the load for {} entries", map.entryCount());
589         }
590     }

607     /**
608      * Add a list of entries to the directory Entries are kept in a Hash by "datatype::serviceType" Each entry is a List of
609      * sorted DirectoryEntries sorted order on cost and then quality, held in a DirectoryEntryList object
610      *
611      * @param entryList the new entries to add
612      */
613     protected void addEntries(final List<DirectoryEntry> entryList) {
614         logger.debug("Adding {} new entries", entryList.size());
615 
616         // add them
617         this.entryMap.addEntries(entryList);
618 
619         // notify all observers
620         this.observerManager.placeAddEntries(entryList);
621 
622         final Set<String> peerSet = new HashSet<>();
623         for (final DirectoryEntry newEntry : entryList) {
624             // Make a note of any possible new peer directory
625             if (!isLocal(newEntry)) {
626                 final String peerKey = KeyManipulator.getDefaultDirectoryKey(newEntry.getKey());
627                 if (!isKnownPeer(peerKey) && !peerSet.contains(peerKey)) {
628                     logger.debug("Discovered new peer {} from  addEntries {}", peerKey, newEntry.getKey());
629                     peerSet.add(peerKey);
630                 } else {
631                     logger.debug("No new peer implications to {} from {}", peerKey, newEntry.getKey());
632                 }
633             }
634         }
635 
636         if (!peerSet.isEmpty()) {
637             logger.debug("Adding {} newly discovered peer entries", peerSet.size());
638             addPeerDirectories(peerSet, false);
639         }
640     }

662     /**
663      * Determine if key represents a known peer
664      */
665     private boolean isKnownPeer(final String key) {
666         for (final DirectoryEntry sde : this.peerDirectories) {
667             if (KeyManipulator.isLocalTo(sde.getKey(), key)) {
668                 return true;
669             }
670         }
671         return false;
672     }

674     /**
675      * Remove a peer from the peer list
676      *
677      * @param key the peer to remove
678      */
679     @Nullable
680     private DirectoryEntry removePeer(final String key) {
681         if (this.emissaryNode.isStandalone()) {
682             logger.debug("Cannot remove peers from standalone nodes");
683             return null;
684         }
685 
686         DirectoryEntry expeer = null;
687 
688         // Find it
689         for (final DirectoryEntry sde : this.peerDirectories) {
690             if (KeyManipulator.isLocalTo(sde.getKey(), key)) {
691                 // nb. COW Set does not support iterator.remove
692                 expeer = sde;
693                 break;
694             }
695         }
696 
697         // Nuke it
698         if (expeer != null) {
699             // Remove from COW set
700             this.peerDirectories.remove(expeer);
701 
702             // Remove from heartbeat manager
703             this.heartbeat.removeRemoteDirectory(expeer.getKey());
704 
705             // Notify all observers, but don't give them
706             // access to our own Set object
707             this.observerManager.peerUpdate(new HashSet<>(this.peerDirectories));
708 
709             // Remove the entries if any remain
710             removePlaces(Collections.singletonList(KeyManipulator.getHostMatchKey(expeer.getKey())));
711         }
712 
713         return expeer;
714     }

716     /**
717      * Remove directory. Called from the heartbeat manager and from the EmissaryClient
718      *
719      * @param key string key of failed directory
720      * @param permanent true if from a normal shutdown rather than a transient error
721      * @return count of how many places were removed locally
722      */
723     @Override
724     public int irdFailDirectory(final String key, final boolean permanent) {
725 
726         if (this.emissaryNode.isStandalone()) {
727             logger.debug("Cannot fail remotes in standalone nodes");
728             return 0;
729         }
730 
731         // Validate remote input
732         if (!KeyManipulator.isValid(key)) {
733             logger.warn("Ignoring, called with invalid key {}", key);
734             return 0;
735         }
736 
737         if (this.shutdownInitiated) {
738             logger.debug("Remote {} reported as failed, in shutdown", key);
739             return 0;
740         }
741 
742         // Reports of my demise are premature...
743         if (isLocal(key)) {
744             logger.warn(
745                     "Someone reported me as failed, but I appear to be still running. Refusing to remove my own entries and propagate this filthy lie.");
746             return 0;
747         }
748 
749         final String dirKey = KeyManipulator.getDefaultDirectoryKey(key);
750         final String hmKey = KeyManipulator.getHostMatchKey(key);
751         int count = 0;
752 
753         logger.debug("irdFailDirectory {} {} permanent", key, permanent ? "is" : "is not");
754 
755         // Modify local entries for the failed remote directory
756         // Permanent failure removes entries on failed directory.
757         // Transient failure adjusts weight of entries on failed directory.
758         if (permanent) {
759             logger.debug("Permanent failure of remote {}", key);
760             count += removePlaces(Collections.singletonList(hmKey));
761         } else {
762             // Change the cost for all places matching the
763             // failed directory. This has the effect of causing them
764             // not to be chosen as much.
765             final List<DirectoryEntry> list = this.entryMap.collectAllMatching(hmKey);
766             this.observerManager.placeCostChangeEntries(list);
767         }
768 
769         // Handle permanent removal of remote directory
770         if (permanent) {
771             // Notify my heartbeat manager so that a normal deregistration
772             // followed by a restart will trigger a state transition even
773             // if under the timer check time
774             this.heartbeat.setHealthStatus(key, HeartbeatManager.NO_CONTACT, "Permanent deregistration");
775 
776             // Remove from peer list
777             if (isKnownPeer(dirKey)) {
778                 if (!isStaticPeer(dirKey)) {
779                     logger.debug("Removing non-static peer {}", dirKey);
780                     removePeer(dirKey);
781                 } else {
782                     logger.debug("Static peer {} is deregistered but monitoring continues", dirKey);
783                 }
784             } else {
785                 logger.warn("Directory {} failed but it isn't a peer??", dirKey);
786             }
787         }
788 
789         return count;
790     }

854     /**
855      * Register a list of entries. This signature only meant to be called from within EmissaryClient code. Each entry will
856      * have a separate key, cost and quality but should all be local to each other.
857      *
858      * @param entryList list of directoryEntry to add
859      * @param propagating true if going back down the directory chain
860      */
861     @Override
862     public void irdAddPlaces(@Nullable final List<DirectoryEntry> entryList, final boolean propagating) {
863 
864         if ((entryList == null) || entryList.isEmpty()) {
865             logger.debug("irdAddPlaces called with null or empty entryList!");
866             return;
867         }
868 
869         // Validate remote input
870         for (final DirectoryEntry d : entryList) {
871             if (d == null || !d.isValid()) {
872                 logger.warn("Ignoring irdAddPlaces called with {} DirectoryEntry objects due to invalid key in {}", entryList.size(), d);
873                 return;
874             }
875         }
876 
877         // These keys better all be from the same emissary node
878         // We should check that they are and throw if not
879         final String place = entryList.get(0).getKey(); // !!
880         final boolean isLocal = isLocal(place);
881 
882         if (logger.isDebugEnabled()) {
883             logger.debug("Starting irdAddPlaces with {} entries for {} place  - place={}, myKey={}", entryList.size(),
884                     isLocal ? "local" : "non-local", place, myKey);
885         }
886 
887         // make a defensive deep copy of the incoming list, so we
888         // can safely proxy and adjust cost as needed
889         final List<DirectoryEntry> entries = new ArrayList<>();
890         for (final DirectoryEntry d : entryList) {
891             entries.add(new DirectoryEntry(d, DirectoryEntry.PRESERVE_TIME));
892         }
893 
894         // Let each directory add this non-local component to the cost
895         // based on the place locality. This should be enough to
896         // dwarf any cost variants among truly local places
897         if (!isLocal) {
898             for (final DirectoryEntry d : entries) {
899                 d.addCost(REMOTE_COST_OVERHEAD);
900             }
901         }
902 
903         logger.debug("Doing addEntries for {} new entries", entries.size());
904         addEntries(entries);
905 
906         // Notify peers if entries are being added locally
907         if (isLocal && !this.peerDirectories.isEmpty()) {
908             // This may fail if the peer is not up yet. That is normal.
909             for (final DirectoryEntry peer : this.peerDirectories) {
910                 if (this.heartbeat.isAlive(peer.getKey())) {
911                     registerWith(peer, entries, false);
912                 } else if (logger.isDebugEnabled()) {
913                     logger.debug("Not registering {} with peer {}, not alive right now", entries.size(), peer.getKey());
914                 }
915             }
916         }
917     }

960     /**
961      * Get tne next logical entry based on current dataId and last place visited
962      *
963      * @param dataId key to entryMap, dataType::serviceType, e.g. UNKNOWN::ID
964      * @param payload the payload being routed
965      * @param lastPlace place agent visited last, this is not stateless
966      * @param entries map of DirectoryEntry stored in this directory
967      * @return List of DirectoryEntry with next place to go or empty list if none
968      */
969     protected List<DirectoryEntry> nextKeys(final String dataId, final IBaseDataObject payload, @Nullable final DirectoryEntry lastPlace,
970             final DirectoryEntryMap entries) {
971         // Find the entry list for the type being requested
972         final DirectoryEntryList currentList = getWildcardedEntryList(dataId, entries);
973 
974         // Nothing for the dataId or any wildcarded versions, we are done
975         if ((currentList == null) || currentList.isEmpty()) {
976             logger.debug("nextKey - nothing found here for {}", dataId);
977             return List.of();
978         }
979 
980         // remove denied entries
981         currentList.removeIf(de -> de.getLocalPlace() != null && de.getLocalPlace().isDenied(payload.currentForm()));
982 
983         // test for invalidated places
984         currentList.stream().filter(e -> e.getLocalPlace() instanceof ServiceProviderRefreshablePlace
985                 && ((ServiceProviderRefreshablePlace) e.getLocalPlace()).isInvalidated())
986                 .forEach(this::handleEntryRefresh);
987 
988         if (currentList.isEmpty()) {
989             logger.debug("nextKeys - no non-DENIED entries found here for {}", dataId);
990             return List.of();
991         }
992         // The list we are building for return to the caller
993         final List<DirectoryEntry> keyList = new ArrayList<>();
994 
995         // The dataId this time is different from the last place
996         // visited, so we can just choose from the list of the lowest
997         // expense places and get on with it
998         DirectoryEntry trialEntry = currentList.getEntry(0);
999         if (lastPlace == null || (!lastPlace.getDataId().equals(dataId) && !trialEntry.getServiceLocation().equals(lastPlace.getServiceLocation()))) {
1000             logger.debug("doing first in list for {}", trialEntry);
1001             keyList.add(currentList.pickOneOf(trialEntry.getExpense()));
1002         } else {
1003             // Trying a particular "dataType::serviceType" pair again
1004             for (int i = 0; i < currentList.size(); i++) {
1005                 trialEntry = currentList.getEntry(i);
1006 
1007                 // Skip entry if less/same expensive. Includes the obvious
1008                 // test, plus evaluation of whether we would choose a
1009                 // particular non-local place if it was here. If we wouldn't
1010                 // choose it if it was here, we certainly aren't willing
1011                 // to move to get it.
1012                 final int te = trialEntry.getExpense() % REMOTE_EXPENSE_OVERHEAD;
1013                 final int le = lastPlace.getExpense() % REMOTE_EXPENSE_OVERHEAD;
1014 
1015                 // Always skip service cheaper than what we already did
1016                 if (te < le) {
1017                     logger.debug("nextKey skip lower cost {}", trialEntry.getFullKey());
1018                     continue;
1019                 }
1020 
1021                 // If relaying, we want to be hopping closer to the target
1022                 if ((te == le) && (trialEntry.getExpense() >= lastPlace.getExpense())
1023                         && !trialEntry.getServiceHostUrl().equals(lastPlace.getServiceHostUrl())) {
1024                     logger.debug("nextKey skip equal cost {}", trialEntry.getFullKey());
1025                     continue;
1026                 }
1027 
1028                 // If equal or lower cost, no point in using the entry
1029                 if ((trialEntry.getExpense() <= lastPlace.getExpense()) && trialEntry.getServiceHostUrl().equals(lastPlace.getServiceHostUrl())) {
1030                     logger.debug("nextKey skip lower cost not relaying {}", trialEntry.getFullKey());
1031                     continue;
1032                 }
1033 
1034                 // Entry is more expense and different service
1035                 logger.debug("nextKey - doing next in list");
1036                 keyList.add(currentList.pickOneOf(trialEntry.getExpense()));
1037                 break;
1038             }
1039 
1040         }
1041 
1042         return keyList;
1043     }

1045     protected synchronized void handleEntryRefresh(final DirectoryEntry entry) {
1046         // attempt to see if the place has been refreshed already
1047         entry.clearLocalPlace();
1048 
1049         final ServiceProviderRefreshablePlace place = (ServiceProviderRefreshablePlace) entry.getLocalPlace();
1050         if (place.isInvalidated()) {
1051             logger.debug("{} has been invalidated, attempting refresh ....", entry);
1052             place.refresh();
1053             entry.clearLocalPlace();
1054             logger.debug("{} has been refreshed", entry);
1055         }
1056     }

1076      * Payloads that need to traverse the relay gateway can visit here to be forwarded on to the correct destination
1077      * <p>
1078      * The payload will have the simple current form that caused this relay point to be selected replace with the full
1079      * four-tupled key of the place matching the request on the proper side of this relay point.
1080      *
1081      * @param d the payload to be inspected
1082      */

1085         if (d.currentForm().equals(this.myKey)) {
1086             logger.debug("Probe routing has been removed");

1093     /**
1094      * Handle the routing for a payload
1095      *
1096      * @param d the visiting payload
1097      */
1098     protected void handleRouting(final IBaseDataObject d) {
1099         // The source entry we are interested in is the one that got us
1100         // here. The "lastPlaceVisited" should be my own key, so we want
1101         // the one before that.
1102         DirectoryEntry sourceEntry = d.getPenultimatePlaceVisited();
1103 
1104         // Source entry still null?
1105         if (sourceEntry == null) {
1106             logger.debug("Payload had no source entry and no places visited. " + "Using my own directory key, which is probably wrong.");
1107             sourceEntry = this.getDirectoryEntry();
1108         }
1109 
1110         // Last place visited shows the key that cause the payload
1111         // to arrive at this place since it is logged into the history
1112         // just before calling this method. The dataId on this entry
1113         // is the entry that finally was selected for use, so we reuse
1114         // it here in the proper entry map.
1115         final DirectoryEntry thisEntry = d.getLastPlaceVisited();
1116         final String dataId = thisEntry.getDataId();
1117 
1118         if (logger.isDebugEnabled()) {
1119             logger.debug("Relay payload '{}' arrived with form {} coming from {} arrival entry {} arrival dataId={}", d.shortName(), d.currentForm(),
1120                     sourceEntry.getKey(), thisEntry.getKey(), dataId);
1121         }
1122 
1123         // Where we want to go from here
1124         List<DirectoryEntry> destination = nextKeys(dataId, d, sourceEntry);
1125 
1126         if (logger.isDebugEnabled()) {
1127             logger.debug("Selected {} entries {} from incoming {} and data id {} current form={}", destination.size(), destination,
1128                     sourceEntry.getKey(), dataId, d.currentForm());
1129         }
1130 
1131         // Replace the current form with the full key version of same
1132         d.popCurrentForm();
1133         for (final DirectoryEntry destEntry : destination) {
1134             d.pushCurrentForm(destEntry.getKey());
1135         }
1136 
1137         if (logger.isDebugEnabled()) {
1138             logger.debug("Leaving relay gateway with current form {}", d.getAllCurrentForms());
1139         }
1140     }

1198     /**
1199      * Deregister places removing all keys for the specified places. Should only be called externally from EmissaryClient
1200      *
1201      * @see #removePlaces(List)
1202      * @param keys four-tuple key for the place
1203      * @param propagating true if going down the line
1204      * @return count of how many keys removed
1205      */
1206     @Override
1207     public int irdRemovePlaces(@Nullable final List<String> keys, final boolean propagating) {
1208 
1209         if ((keys == null) || keys.isEmpty()) {
1210             logger.warn("Ignoring null or empty key list for irdRemovePlaces");
1211             return 0;
1212         }
1213 
1214         // Validate remote input
1215         for (final String key : keys) {
1216             if (!KeyManipulator.isValid(key)) {
1217                 logger.warn("Ignoring irdRemovePlaces called with {} keys due to invalid key {}", keys.size(), key);
1218                 return 0;
1219             }
1220         }
1221 
1222         // Note we don't just pull the dataId from the map
1223         // because we anticipate the incoming keys will be
1224         // wildcarded as places go away rather than individual
1225         // service proxy keys
1226         final List<DirectoryEntry> matches = new ArrayList<>();
1227         for (final String key : keys) {
1228             final List<DirectoryEntry> m = this.entryMap.removeAllMatching(key);
1229             matches.addAll(m);
1230         }
1231 
1232         final int count = matches.size();
1233         if (logger.isDebugEnabled()) {
1234             logger.debug("Found {} entries for removal matching {} keys={}", count, keys.size(), keys);
1235         }
1236 
1237         // Nothing do to, nothing to propagate
1238         if (count == 0) {
1239             return 0;
1240         }
1241 
1242         int localCount = 0;
1243 
1244         // Count the locals and hit the observers
1245         for (final DirectoryEntry e : matches) {
1246             if (isLocal(e.getKey())) {
1247                 localCount++;
1248             }
1249             // Notify observers of entry removal
1250             this.observerManager.placeRemove(e.getFullKey());
1251         }
1252 
1253         // Notify peers if local entries are being removed
1254         if (!this.peerDirectories.isEmpty() && (localCount > 0)) {
1255             // This may fail if the peer is not up. That is normal.
1256             for (final DirectoryEntry peer : this.peerDirectories) {
1257                 if (this.heartbeat.isAlive(peer.getKey())) {
1258                     logger.debug("Deregistering {} keys from peer {}", keys.size(), peer);
1259                     deregisterFrom(peer, keys, false);
1260                 }
1261             }
1262         }
1263 
1264         final List<String> localKeys = new ArrayList<>();
1265         for (final DirectoryEntry match : matches) {
1266             final String key = match.getKey();
1267             if (isLocal(key)) {
1268                 logger.debug("Removed {} putting in local bucket", key);
1269                 localKeys.add(key);
1270             }
1271         }
1272 
1273         return localKeys.size();
1274     }

1316         }
1317 
1318         // Remove all entries and notify all observers
1319         final List<DirectoryEntry> matches = this.entryMap.collectAllMatching("*.*.*.*");
1320         for (final DirectoryEntry e : matches) {
1321             this.observerManager.placeRemove(e.getFullKey());
1322         }
1323 
1324         // Nuke em
1325         this.entryMap.clear();
1326 
1327         // Remove peers and Notify all observers that we are leaving the group
1328         this.peerDirectories.clear();
1329         this.observerManager.peerUpdate(this.peerDirectories);
1330 
1331         unbindFromNamespace();
1332         logger.info("Done shutting down DirectoryPlace");
1333 
1334     }

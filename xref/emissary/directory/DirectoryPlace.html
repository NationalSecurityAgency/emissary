24  /**
25   * The DirectoryPlace class is used to store information relating to Places/Services in the Emissary Agent-Based
26   * architecture. When a Place comes up it calls the method addPlace passing in all the relevant information to store in
27   * the Directory. Agents query the directory by calling the method nextKeys which requires a query String search
28   * pattern.
29   *
30   * <p>
31   * We try to support some network topographic constructions by providing a set of peer directories. Peers are monitored
32   * and checked automatically by HeartbeatManager and the peer network is assumed to be fully connected. Peer directories
33   * are a fairly-static list of peer directories read from a config file. At least one host must be listed in order to
34   * bootstrap the network.
35   *
36   * <p>
37   * Emissary directory instances are also observable with respect to Peer activities, and Place activities. Peer
38   * observers will be called with a list of current members of the peer group (including this directory) whenever the
39   * peer group loses or gains members. Place observers will be called with a key that matches the pattern supplied on
40   * their subscription and an indication of whether it is a register or deregister or cost change.
41   *
42   */
43  public class DirectoryPlace extends ServiceProviderPlace implements IRemoteDirectory {

220             final int initialSeconds = configG.findIntEntry("HEARTBEAT_DELAY_SECONDS", 30);
221             final int intervalSeconds = configG.findIntEntry("HEARTBEAT_INTERVAL_SECONDS", 30);
222

224 
225             final int heartbeatFailureThreshold = configG.findIntEntry("HEARTBEAT_FAILURE_THRESHOLD", -1);
226             if (heartbeatFailureThreshold > 0) {
227                 this.heartbeat.setFailThreshold(heartbeatFailureThreshold);
228             }
229 
230             final int heartbeatPermanentFailure = configG.findIntEntry("HEARTBEAT_PERMANENT_FAILURE_THRESHOLD", -1);
231             if (heartbeatPermanentFailure > 0) {
232                 this.heartbeat.setPermanentFailThreshold(heartbeatPermanentFailure);
233             }
234         }

236         // Set up deferred stuff from ServiceProviderPlace
237         // for directories only we are our own localDirPlace
238         // and the key is our own key
239         localDirPlace = this;
240         dirPlace = myKey;

242         // Start an observer manager
243         this.observerManager = new DirectoryObserverManager(myKey);

245         // Configure my initial rendezvous peers
246         configureNetworkTopology();

249         // local entry map. This allows observers to

336     /**
337      * Add a Set of peer directory to this one
338      *
339      * @param keys set of string key for peer directories
340      * @param initPhase true if during place initialization
341      */
342     // TODO Look at DirectoryPlaceTest at the cases where spied methods are used
343     public void addPeerDirectories(final Set<String> keys, final boolean initPhase) {
344 
345         if (this.shutdownInitiated) {
346             logger.error("Shutdown has been initiated. Cannot add peer directories in this state.");
347             return;
348         }
349 
350         boolean changeMade = false;
351 
352         for (final String key : keys) {
353             if (isLocal(key)) {
354                 // I am listed as a rendezvous for someone
355                 this.rdvPeer = true;
356                 continue;
357             }
358 
359             if (this.emissaryNode.isStandalone()) {
360                 logger.debug("Not adding peers in standalone nodes");
361                 continue;
362             }
363 
364             if (!isStaticPeer(key)) {
365                 logger.warn("Unknown peer requesting to be added: {}", key);
366                 continue;
367             }
368 
369             if (!isKnownPeer(key)) {
370                 this.peerDirectories.add(new DirectoryEntry(key));
371                 logger.debug("Added peer directory {}", key);
372 
373                 // Setup heartbeat to new peer directory
374                 if (initPhase) {
375                     // not contacted yet
376                     this.heartbeat.addRemoteDirectory(key, HeartbeatManager.NO_CONTACT);
377                 } else {
378                     // already contacted
379                     this.heartbeat.addRemoteDirectory(key, HeartbeatManager.IS_ALIVE);
380 
381                     // Initial transfer of remote directory info here
382                     // It may not be up yet, so be resilient
383                     loadPeerEntries(key);
384                 }
385 
386 
387                 changeMade = true;
388             } else {
389                 logger.debug("We already knew about peer {}", key);
390                 if (!this.heartbeat.isAlive(key)) {
391                     logger.debug("Forcing peer {} alive due to arriving registration", key);
392                     this.heartbeat.setHealthStatus(key, HeartbeatManager.IS_ALIVE, "Received peer registration");
393                     loadPeerEntries(key);
394                 }
395             }
396         }
397 
398         // Notify all observers
399         if (changeMade) {
400             this.observerManager.peerUpdate(new HashSet<>(this.peerDirectories));
401         }
402     }

404     /**
405      * Retrieve and load (zone transfer) all the entries from the specified peer directory. Zone transfers do not trigger
406      * observables like addPlaces does
407      *
408      * @param peerKey the key of the peer directory
409      */
410     protected void loadPeerEntries(final String peerKey) {
411 
412         if (this.emissaryNode.isStandalone()) {
413             logger.debug("Cannot load peer entries in standalone nodes");
414             return;
415         }
416 
417         logger.debug("Doing zone transfer with peer {}", peerKey);
418         // TODO See DirectoryPlace for spy example which needs to be addressed
419         final DirectoryEntryMap newEntries = loadRemoteEntries(peerKey, this.entryMap);
420         if ((newEntries == null) || newEntries.isEmpty()) {
421             logger.debug("We got nothing back from the peer zone xfer");
422             return;
423         }
424 
425         // We just did this guy remove his stuff
426         newEntries.removeAllOnDirectory(peerKey);
427 
428         // Remove local stuff
429         newEntries.removeAllOnDirectory(myKey);
430 
431         // Make note of any possible new peer directory
432         // We should only be seeing peers here
433         final Set<String> newPeers = new HashSet<>();
434         for (final DirectoryEntry newEntry : newEntries.allEntries()) {
435             if (!isLocal(newEntry)) {
436                 final String possiblePeer = KeyManipulator.getDefaultDirectoryKey(newEntry.getKey());
437                 if (!isKnownPeer(possiblePeer) && !newPeers.contains(possiblePeer)) {
438                     logger.debug("Discovered new peer {} from {} during zt with {}", possiblePeer, newEntry.getKey(), peerKey);
439                     newPeers.add(possiblePeer);
440                 }
441             }
442         }
443         if (!newPeers.isEmpty()) {
444             logger.debug("Adding {} new peers from zt with {}", newPeers.size(), peerKey);
445             addPeerDirectories(newPeers, false);
446         }
447     }

518     /**
519      * Remove stale entries from the specified map and notify any observers Nothing older than checkpoint time can be
520      * considered stale and nothing that is on the incming newEntries list can be considered stale since we would just be
521      * adding it back again. Duplicates (non-stale entries) are removed from the newEntries map to avoid further confusion
522      * but only if the cost is the same. Otherwise, we leave it so that a cost-change event can propagete from later code
523      * but still avoid triggering a place removed event.
524      *
525      * @param loadMap the map we are removing from
526      * @param key the key of the directory whose entries might be stale
527      * @param checkpoint the time window to determine possible staleness
528      * @param newEntries the new map arriving
529      * @param performNotification only use observerManager if true
530      * @return list of entries that were removed
531      */
532     private List<DirectoryEntry> removeStaleEntries(final DirectoryEntryMap loadMap, final String key, final long checkpoint,
533             @Nullable final DirectoryEntryMap newEntries, final boolean performNotification) {
534 
535         final List<DirectoryEntry> staleEntries = new ArrayList<>();
536 
537         // Nothing newer than the checkpoint time can be stale
538         // Nothing that is in teh loadMap but also duplicated in
539         // the newEntries map can be stale either. This helps eliminate
540         // the problem of removing it just so we can add it back.
541         // This uses a mark and sweep to prevent concurrent mod exceptions
542         for (final DirectoryEntry d : loadMap.collectAllMatching(key)) {
543             // is it old enough to be possibly stale
544             if (d.getAge() < checkpoint) {
545                 // is it missing from the new list
546                 if (newEntries != null) {
547                     final List<DirectoryEntry> matches = newEntries.collectAllMatching(d.getKey());
548                     if (matches.isEmpty()) {
549                         logger.debug("Marking stale entry {}", d.getKey());
550                         staleEntries.add(d);
551                     } else if (matches.size() == 1) {
552                         // remove from newEntries if exact dup
553                         final DirectoryEntry me = matches.get(0);
554                         if (me.getFullKey().equals(d.getFullKey())) {
555                             logger.debug("Removing duplcate key from incoming map {}", me.getKey());
556                             newEntries.removeEntry(me.getKey());
557                         }
558                     }
559                 } else {
560                     // must be stale if no newEntries
561                     logger.debug("Marking stale entry (no new entries){}", d.getKey());
562                     staleEntries.add(d);
563                 }
564             }
565         }
566 
567         // Remove and notify
568         if (!staleEntries.isEmpty()) {
569             for (final DirectoryEntry stale : staleEntries) {
570                 logger.debug("Removing stale entry {}", stale.getKey());
571                 loadMap.removeEntry(stale.getKey());
572             }
573 
574             if (performNotification) {
575                 logger.debug("Notifying observers of {} stale entry removals", staleEntries.size());
576                 this.observerManager.placeRemoveEntries(staleEntries);
577             }
578         } else {
579             logger.debug("There were no stale entries to remove");
580         }
581 
582         return staleEntries;
583     }

587      * not going to end up being added anyway.

607         }
608

610             final DirectoryEntryMapryEntryMap">DirectoryEntryMap newEntries = new DirectoryEntryMap();
611             final DirectoryEntryMapMap">DirectoryEntryMap costChangeEntries = new DirectoryEntryMap();
612             for (final DirectoryEntry e : map.allEntries()) {
613                 final List<DirectoryEntry> matches = loadMap.collectAllMatching(e.getKey());
614                 if (matches.isEmpty()) {
615                     newEntries.addEntry(e);
616                 } else if ((matches.size() == 1) && e.isBetterThan(matches.get(0))) {
617                     costChangeEntries.addEntry(e);
618                 }
619             }
620 
621             // Merge remaining truly new entries and notify observers
622             final int newCount = newEntries.entryCount();
623             final int cceCount = costChangeEntries.entryCount();
624             if (newCount > 0) {
625                 logger.debug("Loading {} new entries", newCount);
626                 loadMap.addEntries(newEntries);
627                 this.observerManager.placeAdd(newEntries.allEntryKeys());
628             } else {
629                 logger.debug("Nothing truly new from {} entries", map.entryCount());
630             }
631 
632             // .. and cost change entries
633             if (cceCount > 0) {
634                 logger.debug("Loading {} better cost entries", cceCount);
635                 loadMap.addEntries(costChangeEntries);
636                 this.observerManager.placeCostChange(costChangeEntries.allEntryKeys());
637             } else {
638                 logger.debug("No cost change entries from {} entries", map.entryCount());
639             }
640 
641             // Now let the map that gets returned have just the new
642             // and cost changed entries, no already known stuff
643             if ((newCount + cceCount) < map.entryCount()) {
644                 map.clear();
645                 map.addEntries(costChangeEntries);
646                 map.addEntries(newEntries);
647                 map.sort();
648             }
649         } else {
650             logger.debug("Clean/load got a null loadMap so skipping the load for {} entries", map.entryCount());
651         }

736     /**
737      * Remove a peer from the peer list
738      *
739      * @param key the peer to remove
740      */
741     private DirectoryEntry removePeer(final String key) {
742         if (this.emissaryNode.isStandalone()) {
743             logger.debug("Cannot remove peers from standalone nodes");
744             return null;
745         }
746 
747         DirectoryEntry expeer = null;
748 
749         // Find it
750         for (final DirectoryEntry sde : this.peerDirectories) {
751             if (KeyManipulator.isLocalTo(sde.getKey(), key)) {
752                 // nb. COW Set does not support iterator.remove
753                 expeer = sde;
754                 break;
755             }
756         }
757 
758         // Nuke it
759         if (expeer != null) {
760             // Remove from COW set
761             this.peerDirectories.remove(expeer);
762 
763             // Remove from heartbeat manager
764             this.heartbeat.removeRemoteDirectory(expeer.getKey());
765 
766             // Notify all observers, but don't give them
767             // access to our own Set object
768             this.observerManager.peerUpdate(new HashSet<>(this.peerDirectories));
769 
770             // Remove the entries if any remain
771             removePlaces(Collections.singletonList(KeyManipulator.getHostMatchKey(expeer.getKey())));
772         }
773 
774         return expeer;
775     }

777     /**
778      * Remove directory. Called from the heartbeat manager and from the EmissaryClient
779      *
780      * @param key string key of failed directory
781      * @param permanent true if from a normal shutdown rather than a transient error
782      * @return count of how many places were removed locally
783      */
784     @Override
785     public int irdFailDirectory(final String key, final boolean permanent) {
786 
787         if (this.emissaryNode.isStandalone()) {
788             logger.debug("Cannot fail remotes in standalone nodes");
789             return 0;
790         }
791 
792         // Validate remote input
793         if (!KeyManipulator.isValid(key)) {
794             logger.warn("Ignoring, called with invalid key {}", key);
795             return 0;
796         }
797 
798         if (this.shutdownInitiated) {
799             logger.debug("Remote {} reported as failed, in shutdown", key);
800             return 0;
801         }
802 
803         // Reports of my demise are premature...
804         if (isLocal(key)) {
805             logger.warn(
806                     "Someone reported me as failed, but I appear to be still running. Refusing to remove my own entries and propagate this filthy lie.");
807             return 0;
808         }
809 
810         final String dirKey = KeyManipulator.getDefaultDirectoryKey(key);
811         final String hmKey = KeyManipulator.getHostMatchKey(key);
812         int count = 0;
813 
814         logger.debug("irdFailDirectory {} {} permanent", key, (permanent ? "is" : "is not"));
815 
816         // Modify local entries for the failed remote directory
817         // Permanent failure removes entries on failed directory.
818         // Transient failure adjusts weight of entries on failed directory.
819         if (permanent) {
820             logger.debug("Permanent failure of remote {}", key);
821             count += removePlaces(Collections.singletonList(hmKey));
822         } else {
823             // Change the cost for all places matching the
824             // failed directory. This has the effect of causing them
825             // not to be chosen as much.
826             final List<DirectoryEntry> list = this.entryMap.collectAllMatching(hmKey);
827             this.observerManager.placeCostChangeEntries(list);
828         }
829 
830         // Handle permanent removal of remote directory
831         if (permanent) {
832             // Notify my heartbeat manager so that a normal deregistration
833             // followed by a restart will trigger a state transition even
834             // if under the timer check time
835             this.heartbeat.setHealthStatus(key, HeartbeatManager.NO_CONTACT, "Permanent deregistration");
836 
837             // Remove from peer list
838             if (isKnownPeer(dirKey)) {
839                 if (!isStaticPeer(dirKey)) {
840                     logger.debug("Removing non-static peer {}", dirKey);
841                     removePeer(dirKey);
842                 } else {
843                     logger.debug("Static peer {} is deregistered but monitoring continues", dirKey);
844                 }
845             } else {
846                 logger.warn("Directory {} failed but it isn't a peer??", dirKey);
847             }
848         }
849 
850         return count;
851     }

915     /**
916      * Register a list of entries. This signature only meant to be called from within EmissaryClient code. Each entry will
917      * have a separate key, cost and quality but should all be local to each other.
918      *
919      * @param entryList list of directoryEntry to add
920      * @param propagating true if going back down the directory chain
921      */
922     @Override
923     public void irdAddPlaces(@Nullable final List<DirectoryEntry> entryList, final boolean propagating) {
924 
925         if ((entryList == null) || entryList.isEmpty()) {
926             logger.debug("irdAddPlaces called with null or empty entryList!");
927             return;
928         }
929 
930         // Validate remote input
931         for (final DirectoryEntry d : entryList) {
932             if (d == null || !d.isValid()) {
933                 logger.warn("Ignoring irdAddPlaces called with {} DirectoryEntry objects due to invalid key in {}", entryList.size(), d);
934                 return;
935             }
936         }
937 
938         // These keys better all be from the same emissary node
939         // We should check that they are and throw if not
940         final String place = entryList.get(0).getKey(); // !!
941         final boolean isLocal = isLocal(place);
942 
943         if (logger.isDebugEnabled()) {
944             logger.debug("Starting irdAddPlaces with {} entries for {} place  - place={}, myKey={}", entryList.size(),
945                     (isLocal ? "local" : "non-local"), place, myKey);
946         }
947 
948         // make a defensive deep copy of the incoming list, so we
949         // can safely proxy and adjust cost as needed
950         final List<DirectoryEntry> entries = new ArrayList<>();
951         for (final DirectoryEntry d : entryList) {
952             entries.add(new DirectoryEntry(d, DirectoryEntry.PRESERVE_TIME));
953         }
954 
955         // Let each directory add this non-local component to the cost
956         // based on the place locality. This should be enough to
957         // dwarf any cost variants among truly local places
958         if (!isLocal) {
959             for (final DirectoryEntry d : entries) {
960                 d.addCost(REMOTE_COST_OVERHEAD);
961             }
962         }
963 
964         logger.debug("Doing addEntries for {} new entries", entries.size());
965         addEntries(entries);
966 
967         // Notify peers if entries are being added locally
968         if (isLocal && !this.peerDirectories.isEmpty()) {
969             // This may fail if the peer is not up yet. That is normal.
970             for (final DirectoryEntry peer : this.peerDirectories) {
971                 if (this.heartbeat.isAlive(peer.getKey())) {
972                     registerWith(peer, entries, false);
973                 } else if (logger.isDebugEnabled()) {
974                     logger.debug("Not registering {} with peer {}, not alive right now", entries.size(), peer.getKey());
975                 }
976             }
977         }

1021     /**
1022      * Get tne next logical entry based on current dataId and last place visited
1023      *
1024      * @param dataId key to entryMap, dataType::serviceType, e.g. UNKNOWN::ID
1025      * @param payload the payload being routed
1026      * @param lastPlace place agent visited last, this is not stateless
1027      * @param entries map of DirectoryEntry stored in this directory
1028      * @return List of DirectoryEntry with next place to go or empty list if none
1029      */
1030     protected List<DirectoryEntry> nextKeys(final String dataId, final IBaseDataObject payload, @Nullable final DirectoryEntry lastPlace,
1031             final DirectoryEntryMap entries) {
1032         // Find the entry list for the type being requested
1033         final DirectoryEntryList currentList = getWildcardedEntryList(dataId, entries);
1034 
1035         // Nothing for the dataId or any wildcarded versions, we are done
1036         if ((currentList == null) || currentList.isEmpty()) {
1037             logger.debug("nextKey - nothing found here for {}", dataId);
1038             return Collections.emptyList();
1039         }
1040 
1041         // remove denied entries
1042         currentList.removeIf(de -> de.getLocalPlace() != null && de.getLocalPlace().isDenied(payload.currentForm()));
1043 
1044         if (currentList.isEmpty()) {
1045             logger.debug("nextKeys - no non-DENIED entries found here for {}", dataId);
1046             return Collections.emptyList();
1047         }
1048         // The list we are building for return to the caller
1049         final List<DirectoryEntry> keyList = new ArrayList<>();
1050 
1051         // The dataId this time is different from the last place
1052         // visited, so we can just choose from the list of the lowest
1053         // expense places and get on with it
1054         DirectoryEntry trialEntry = currentList.getEntry(0);
1055         if (lastPlace == null || (!lastPlace.getDataId().equals(dataId) && !trialEntry.getServiceLocation().equals(lastPlace.getServiceLocation()))) {
1056             logger.debug("doing first in list for {}", trialEntry);
1057             keyList.add(currentList.pickOneOf(trialEntry.getExpense()));
1058         } else {
1059             // Trying a particular "dataType::serviceType" pair again
1060             for (int i = 0; i < currentList.size(); i++) {
1061                 trialEntry = currentList.getEntry(i);
1062 
1063                 // Skip entry if less/same expensive. Includes the obvious
1064                 // test, plus evaluation of whether we would choose a
1065                 // particular non-local place if it was here. If we wouldn't
1066                 // choose it if it was here, we certainly aren't willing
1067                 // to move to get it.
1068                 final int te = trialEntry.getExpense() % REMOTE_EXPENSE_OVERHEAD;
1069                 final int le = lastPlace.getExpense() % REMOTE_EXPENSE_OVERHEAD;
1070 
1071                 // Always skip service cheaper than what we already did
1072                 if (te < le) {
1073                     logger.debug("nextKey skip lower cost {}", trialEntry.getFullKey());
1074                     continue;
1075                 }
1076 
1077                 // If relaying, we want to be hopping closer to the target
1078                 if ((te == le) && (trialEntry.getExpense() >= lastPlace.getExpense())
1079                         && !trialEntry.getServiceHostUrl().equals(lastPlace.getServiceHostUrl())) {
1080                     logger.debug("nextKey skip equal cost {}", trialEntry.getFullKey());
1081                     continue;
1082                 }
1083 
1084                 // If equal or lower cost, no point in using the entry
1085                 if ((trialEntry.getExpense() <= lastPlace.getExpense()) && trialEntry.getServiceHostUrl().equals(lastPlace.getServiceHostUrl())) {
1086                     logger.debug("nextKey skip lower cost not relaying {}", trialEntry.getFullKey());
1087                     continue;
1088                 }
1089 
1090                 // Entry is more expense and different service
1091                 logger.debug("nextKey - doing next in list");
1092                 keyList.add(currentList.pickOneOf(trialEntry.getExpense()));
1093                 break;
1094             }
1095 
1096         }
1097 
1098         return keyList;
1099     }

1136     /**
1137      * Handle the routing for a payload
1138      *
1139      * @param d the visiting payload
1140      */
1141     protected void handleRouting(final IBaseDataObject d) {
1142         // The source entry we are interested in is the one that got us
1143         // here. The "lastPlaceVisited" should be my own key, so we want
1144         // the one before that.
1145         DirectoryEntry sourceEntry = d.getPenultimatePlaceVisited();
1146 
1147         // Source entry still null?
1148         if (sourceEntry == null) {
1149             logger.debug("Payload had no source entry and no places visited. " + "Using my own directory key, which is probably wrong.");
1150             sourceEntry = this.getDirectoryEntry();
1151         }
1152 
1153         // Last place visited shows the key that cause the payload
1154         // to arrive at this place since it is logged into the history
1155         // just before calling this method. The dataId on this entry
1156         // is the entry that finally was selected for use, so we reuse
1157         // it here in the proper entry map.
1158         final DirectoryEntry thisEntry = d.getLastPlaceVisited();
1159         final String dataId = thisEntry.getDataId();
1160 
1161         if (logger.isDebugEnabled()) {
1162             logger.debug("Relay payload '{}' arrived with form {} coming from {} arrival entry {} arrival dataId={}", d.shortName(), d.currentForm(),
1163                     sourceEntry.getKey(), thisEntry.getKey(), dataId);
1164         }
1165 
1166         // Where we want to go from here
1167         List<DirectoryEntry> destination = nextKeys(dataId, d, sourceEntry);
1168 
1169         if (logger.isDebugEnabled()) {
1170             logger.debug("Selected {} entries {} from incoming {} and data id {} current form={}", destination.size(), destination,
1171                     sourceEntry.getKey(), dataId, d.currentForm());
1172         }
1173 
1174         // Replace the current form with the full key version of same
1175         d.popCurrentForm();
1176         for (final DirectoryEntry destEntry : destination) {
1177             d.pushCurrentForm(destEntry.getKey());
1178         }
1179 
1180         if (logger.isDebugEnabled()) {
1181             logger.debug("Leaving relay gateway with current form {}", d.getAllCurrentForms());
1182         }
1183     }

1247      * @return count of how many keys removed

1252             logger.debug("Cannot remove remote places in standalone nodes");
1253             return 0;
1254         }
1255 
1256         if ((keys == null) || keys.isEmpty()) {
1257             logger.warn("Ignoring null or empty key list for irdRemovePlaces");
1258             return 0;
1259         }
1260 
1261         // Validate remote input
1262         for (final String key : keys) {
1263             if (!KeyManipulator.isValid(key)) {
1264                 logger.warn("Ignoring irdRemovePlaces called with {} keys due to invalid key {}", keys.size(), key);
1265                 return 0;
1266             }
1267         }
1268 
1269         // Note we don't just pull the dataId from the map
1270         // because we anticipate the incoming keys will be
1271         // wildcarded as places go away rather than individual
1272         // service proxy keys
1273         final List<DirectoryEntry> matches = new ArrayList<>();
1274         for (final String key : keys) {
1275             final List<DirectoryEntry> m = this.entryMap.removeAllMatching(key);
1276             matches.addAll(m);
1277         }
1278 
1279         final int count = matches.size();
1280         if (logger.isDebugEnabled()) {
1281             logger.debug("Found {} entries for removal matching {} keys={}", count, keys.size(), keys);
1282         }
1283 
1284         // Nothing do to, nothing to propagate
1285         if (count == 0) {
1286             return 0;
1287         }
1288 
1289         int localCount = 0;
1290 
1291         // Count the locals and hit the observers
1292         for (final DirectoryEntry e : matches) {
1293             if (isLocal(e.getKey())) {
1294                 localCount++;
1295             }
1296             // Notify observers of entry removal
1297             this.observerManager.placeRemove(e.getFullKey());
1298         }
1299 
1300         // Notify peers if local entries are being removed
1301         if (!this.peerDirectories.isEmpty() && (localCount > 0)) {
1302             // This may fail if the peer is not up. That is normal.
1303             for (final DirectoryEntry peer : this.peerDirectories) {
1304                 if (this.heartbeat.isAlive(peer.getKey())) {
1305                     logger.debug("Deregistering {} keys from peer {}", keys.size(), peer);
1306                     deregisterFrom(peer, keys, false);
1307                 }
1308             }
1309         }
1310 
1311         final List<String> localKeys = new ArrayList<>();
1312         for (final DirectoryEntry match : matches) {
1313             final String key = match.getKey();
1314             if (isLocal(key)) {
1315                 logger.debug("Removed {} putting in local bucket", key);
1316                 localKeys.add(key);
1317             }
1318         }
1319 
1320         return localKeys.size();
1321     }

1323     /**
1324      * Private helper to deregister entry from directories
1325      *
1326      * @param dir the remote directory to deregister from
1327      * @param keys the list of keys the place can handle (SERVICE_PROXY)
1328      * @param propagating true if propagating across levels
1329      */
1330     protected void deregisterFrom(final DirectoryEntry dir, final List<String> keys, final boolean propagating) {
1331         try {
1332             // Follow the logic to irdRemovePlaces on the remote side
1333             new DirectoryAdapter().outboundRemovePlaces(dir.getKey(), keys, propagating);
1334         } catch (Exception ex) {
1335             logger.error("DirectoryPlace.deregisterFrom: " + "Problem talking to directory " + dir.getKey() + " to deregister keys", ex);
1336         }

1339     /**
1340      * Shutdown this place and deregister and notify any peers and observers that this directory is closing
1341      */
1342     @Override
1343     public void shutDown() {
1344         if (this.shutdownInitiated) {
1345             return;
1346         }
1347         this.shutdownInitiated = true;
1348         this.running = false;
1349 
1350         logger.debug("Initiating directory shutdown");
1351 
1352         if (this.heartbeat != null) {
1353             this.heartbeat.shutDown();
1354         }
1355 
1356         if (!this.emissaryNode.isStandalone()) {
1357 
1358             // Notify peers of my demise
1359             for (final DirectoryEntry peer : this.peerDirectories) {
1360                 logger.debug("Sending fail msg to peer {}", peer);
1361                 sendFailMessage(peer, myKey, true);
1362             }
1363         }
1364 
1365         // Remove all entries and notify all observers
1366         final List<DirectoryEntry> matches = this.entryMap.collectAllMatching("*.*.*.*");
1367         for (final DirectoryEntry e : matches) {
1368             this.observerManager.placeRemove(e.getFullKey());
1369         }
1370 
1371         // Nuke em
1372         this.entryMap.clear();
1373 
1374         // Remove peers and Notify all observers that we are leaving the group
1375         this.peerDirectories.clear();
1376         this.observerManager.peerUpdate(this.peerDirectories);
1377 
1378         unbindFromNamespace();
1379         logger.info("Done shutting down DirectoryPlace");
1380 
1381     }

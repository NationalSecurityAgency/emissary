9   /**
10   * A class of utility methods for manipulating dictionary keys. Keys are stored in the dictionary with the following
11   * format:
12   *
13   * <code>dataType.serviceName.serviceType.location$expense</code>
14   * 
15   * Example: UNKNOWN.servicename.ID.tcp://host.dom.com:8001/FilePlace$5050
16   * 
17   * <table>
18   * <caption>For the example key above the return values are as follows:</caption>
19   * <tr>
20   * <td>routine</td>
21   * <td>return value</td>
22   * </tr>
23   * <tr>
24   * <td>dataType</td>
25   * <td>UNKOWN</td>
26   * </tr>
27   * <tr>
28   * <td>serviceName</td>
29   * <td>servicename</td>
30   * </tr>
31   * <tr>
32   * <td>serviceType</td>
33   * <td>ID</td>
34   * </tr>
35   * <tr>
36   * <td>serviceHost</td>
37   * <td>host.dom.com:8001</td>
38   * </tr>
39   * <tr>
40   * <td>serviceHostURL</td>
41   * <td>http://host.dom.com:8001/</td>
42   * </tr>
43   * <tr>
44   * <td>serviceLocation</td>
45   * <td>http://host.dom.com:8001/FilePlace</td>
46   * </tr>
47   * <tr>
48   * <td>expense</td>
49   * <td>5050</td>
50   * </tr>
51   * <tr>
52   * <td>serviceClassname</td>
53   * <td>FilePlace</td>
54   * </tr>
55   * <tr>
56   * <td>dataID</td>
57   * <td>UNKOWN::ID</td>
58   * </tr>
59   * <tr>
60   * <td>defaultDirectoryKey</td>
61   * <td>*.*.*.http://host.dom.com:8001/DirectoryPlace</td>
62   * </tr>
63   * </table>
64   *
65   */
66  public class KeyManipulator implements Serializable {

71      /**
72       * How we separate portions of the key
73       */
74      public static final char SEPARATOR = '.';
75      public static final char HOSTSEPARATOR = ':';
76      public static final char CLASSSEPARATOR = '/';
77      public static final String DOLLAR = "$";
78      public static final String DATAIDSEPARATOR = "::";
79      public static final String WILDCARD_THREE = "*.*.*.";

81      /**
82       * How many portions (separated by SEPARATOR) are in a complete key
83       */
84      public static final int NUMTUPLES = 4;

86      private static final String doubleSlash = "//";

88      /**
89       * Make a key from parts
90       *
91       * @param dataType the first part of the key
92       * @param serviceName the second part of the key
93       * @param serviceType the third part of the key
94       * @param serviceLocation the fourth part of the key
95       */
96      public static String makeKey(final String dataType, final String serviceName, final String serviceType, final String serviceLocation) {
97          return dataType + SEPARATOR + serviceName + SEPARATOR + serviceType + SEPARATOR + serviceLocation;
98      }

103     public static String getDataType(final String key) {
104         final int firstSeparator = key.indexOf(SEPARATOR);

124     /**
125      * Performs wildcard (? | *) string matching for dictionary key searches.
126      */
127     public static boolean gmatch(final String s, final String p) {
128         return gmatch2(s.toCharArray(), p.toCharArray(), 0, 0);
129     }
130 
131     /**
132      * Performs wildcard (? | *) character array matching for dictionary key searches.
133      */
134     public static boolean gmatch(final char[] s, final char[] p) {
135         return gmatch2(s, p, 0, 0);
136     }

138     private static boolean gmatch2(final char[] s, final char[] p, final int spos, final int ppos) {
139         // fastest interpreted version
140         if (p.length == ppos) {
141             return s.length == spos;
142         }
143         if (s.length == spos) {
144             return false;
145         }
146         final char scc = s[spos];
147         final char c = p[ppos];
148         if (c == '?') {
149             if (scc > 0) {
150                 return gmatch2(s, p, spos + 1, ppos + 1);
151             }
152             return false;
153         } else if (c == '*') {
154             final int ppos2 = ppos + 1;
155             if (p.length == ppos2) {
156                 return true;
157             }
158             int spos2 = spos;
159             while (s.length > ++spos2) {
160                 if (p[ppos2] == s[spos2] && gmatch2(s, p, spos2, ppos2)) {
161                     return true;
162                 }
163             }
164             return false;
165         } else if (c == scc) {
166             return gmatch2(s, p, spos + 1, ppos + 1);
167         } else {
168             return false;
169         }
170     }

172     /**
173      * Return whether the key passed in is complete or not This is useful in determining if the key must be wildcarded for a
174      * query.
175      *
176      * @param key the key to check
177      */
178     public static boolean isKeyComplete(final String key) {
179         if (numTuplesInKey(key) < KeyManipulator.NUMTUPLES) {
180             return false;
181         }
182         if (getServiceHost(key).length() == 0) {
183             return false;
184         }
185         return true;
186     }

188     /**
189      * Return the number of tuples in the Key passed in
190      */
191     public static int numTuplesInKey(final String key) {
192         int count = 0;
193         final byte[] keyBytes = key.getBytes();
194 
195         for (int i = 0; i < keyBytes.length; i++) {
196             if (keyBytes[i] == KeyManipulator.SEPARATOR) {
197                 count++;
198             }
199             // Dont count separators in the hostname portion
200             if ((i > 0) && (keyBytes[i] == CLASSSEPARATOR) && (keyBytes[i - 1] == CLASSSEPARATOR)) {
201                 break;
202             }
203         }
204 
205         // There is one more tuple than separators x.x.x.x
206         return count + 1;
207     }

232     /**
233      * Returns the hostname:port from a dictionary formatted key
234      */
235     public static String getServiceHost(final String key) {
236         final String location = getServiceLocation(key);
237         final int ds = location.indexOf(doubleSlash);
238 
239         if (ds > -1) {
240             final int cs = location.indexOf(CLASSSEPARATOR, ds + 2);
241 
242             if (cs > -1) {
243                 return location.substring(ds + 2, cs);
244             }
245             return "";
246         }
247         return "";
248     }

274     /**
275      * Returns the service location (host:port/className) field from a dictionary formatted key.
276      */
277     public static String getServiceLocation(final String key) {
278         final int firstSeparator = key.indexOf(SEPARATOR);
279         final int secondSeparator = key.indexOf(SEPARATOR, firstSeparator + 1);
280         final int thirdSeparator = key.indexOf(SEPARATOR, secondSeparator + 1);
281         final int fourthSeparator = key.indexOf(DOLLAR, thirdSeparator + 1);
282 
283         if (thirdSeparator >= 0) {
284             if (fourthSeparator > 0) {
285                 return key.substring(thirdSeparator + 1, fourthSeparator);
286             }
287             return key.substring(thirdSeparator + 1);
288         }
289         return "";
290     }

299     public static int getExpense(final String key, final int dflt) {
300         final int pos = key.lastIndexOf(DOLLAR);
301         int expense = dflt;
302         try {
303             expense = Integer.parseInt(key.substring(pos + 1));
304         } catch (NumberFormatException ignored) {
305             // It's optional...
306         }
307         return expense;
308     }

310     /**
311      * Add expense record onto a key
312      */
313     public static String addExpense(final String key, final int expense) {
314         if (getExpense(key, -99) == expense) {
315             return key;
316         }
317 
318         if (key.indexOf(DOLLAR) > -1) {
319             return key.substring(0, key.lastIndexOf(DOLLAR) + 1) + expense;
320         }
321 
322         return key + DOLLAR + expense;
323     }

326      * Strip the expense from a key if present
327      *
328      * @param key any key
329      * @return the modified key

331     public static String removeExpense(final String key) {
332         final int pos = key.indexOf(DOLLAR);
333         if (pos != -1) {
334             return key.substring(0, pos);
335         }
336         return key;

366     /**
367      * Add in the remote overhead to the "to" key if it was a move
368      *
369      * @param from the key we came from
370      * @param to the key we went to
371      * @return the &quot;to&quot; key with remote overhead or as it was
372      */
373     public static String addRemoteCostIfMoved(final String from, final String to) {
374 
375         // If no move, just return the "to" key
376         if (getServiceHost(from).equals(getServiceHost(to))) {
377             return to;
378         }
379 
380         // Add remote overhead to the "to" key
381         return addExpense(to, getExpense(to) + IDirectoryPlace.REMOTE_EXPENSE_OVERHEAD);
382     }

384     /**
385      * Mangle origKey such that the host on proxyKey becomes a proxy for it
386      *
387      * @param origKey the original key
388      * @param proxyKey the key representing the place that wants to proxy
389      * @param dfltExp default expense to use if origKey has none or -1 for no dflt
390      * @return a key with the original data type, service type and service name and expense but the new place location from
391      *         proxyKey
392      */
393     public static String makeProxyKey(final String origKey, final String proxyKey, final int dfltExp) {
394         if (isLocalTo(origKey, proxyKey) && dfltExp > -1) {
395             return addExpense(origKey, getExpense(origKey, dfltExp));
396         }
397         final int newExp = getExpense(origKey, dfltExp);
398         return getDataType(origKey) + SEPARATOR + getServiceName(origKey) + SEPARATOR + getServiceType(origKey) + SEPARATOR
399                 + getServiceLocation(proxyKey) + (newExp > -1 ? (DOLLAR + newExp) : "");
400     }

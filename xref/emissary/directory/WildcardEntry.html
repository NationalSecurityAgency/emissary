13  /**
14   * Handle the details of a wildcard directory entry including iterating through the possible directory match strings A
15   * dataId will be wildcarded like this, based on dash and paren markings in the dataType.
16   * <ul>
17   * <li>FOO-BAR(ASCII-TRANSLIT)-BAZ(SHAZAM)-BAM_MESSAGE</li>
18   * <li>FOO-BAR(ASCII-TRANSLIT)-BAZ(*)-BAM_MESSAGE</li>
19   * <li>FOO-BAR(*)-BAZ(*)-BAM_MESSAGE</li>
20   * <li>FOO-BAR(*)-BAZ(*)-*</li>
21   * <li>FOO-BAR(*)-*</li>
22   * <li>FOO-*</li>
23   * <li></li>
24   * </ul>
25   *
26   * <ol>
27   * <li>The complete unchanged entry is returned first, so an exact match will work like it always has worked
28   * before.</li>
29   * <li>Next the parenthetical expressions are wildcarded out starting from the right and moving left until all have been
30   * changed to wildcards. Each step wildcards another parenthetical expression without reverting the previous one. So you
31   * currently cannot get a wildcard match on the non-rightmost expression without every one to the right of it also being
32   * wildcarded.</li>
33   * <li>Finally, each dash delimited expression is wildcarded, then eliminated starting from the right until there is
34   * nothing left but the pure wildcard entry &quot;*&quot;. The wildcarded parenthetical expressions are left wildcarded
35   * for this stage, so as things are dropped off, you can only match wildcarded parentheticals that are still remaining
36   * to the left.</li>
37   * </ol>
38   */
39  public class WildcardEntry {
40  
41      private static final Logger logger = LoggerFactory.getLogger(WildcardEntry.class);
42  
43      String dataType;
44      @Nullable
45      String serviceType;
46      @Nullable
47      List<String> wc = null;
48  
49      private static final char DASH = '-';
50      private static final char OPEN = '(';
51      private static final char CLOS = ')';
52      private static final String KSEP = KeyManipulator.DATAIDSEPARATOR;
53      private static final int KSPL = KSEP.length();
54  
55      private static final String DASH_WC = "-*"; // DASH Wildcard
56      private static final String PAREN_WC = "(*)"; // PAREN Wildcard
57      private static final String PURE_WC = "*"; // PURE Wildcard
58  
59      public WildcardEntry(final String s) {
60          parseEntry(s);
61      }
62  
63      /**
64       * Parse the entry into the ordered list of wildcard entries
65       */
66      private synchronized void parseEntry(final String entry) {
67          final int pos = entry.indexOf(KSEP);
68          if (pos > 0) {
69              this.dataType = entry.substring(0, pos);
70              this.serviceType = entry.substring(pos + KSPL);
71          } else {
72              this.dataType = entry;
73              this.serviceType = null;
74          }
75      }
76  
77      /**
78       * Iterator over the wildcard entries in order
79       */
80      public Iterator<String> iterator() {
81          load();
82          return this.wc.iterator();
83      }
84  
85      /**
86       * Get the results of parsing as a set
87       */
88      public Set<String> asSet() {
89          load();
90          return new HashSet<>(this.wc);
91      }
92  
93      /**
94       * Lazy load of the list containing wildcarded parts
95       */
96      private synchronized void load() {
97          if (this.wc == null) {
98              this.wc = new ArrayList<>();
99  
100             // Put the original on the list first
101             if (this.serviceType != null) {
102                 this.wc.add(this.dataType + KSEP + this.serviceType);
103             } else {
104                 this.wc.add(this.dataType);
105             }
106 
107             // Replace paren parts one at a time from the right
108             String dt = this.dataType;
109             int lastOpen = this.dataType.length();
110             boolean done = false;
111             while (!done) {
112                 for (int i = lastOpen - 1; i >= 0; i--) {
113                     if (dt.charAt(i) == OPEN) {
114                         final int clos = dt.indexOf(CLOS, i);
115                         if (clos > i) {
116                             dt = dt.substring(0, i) + PAREN_WC + dt.substring(clos + 1);
117                             if (this.serviceType != null) {
118                                 this.wc.add(dt + KSEP + this.serviceType);
119                             } else {
120                                 this.wc.add(dt);
121                             }
122                             lastOpen = i;
123                             break;
124                         }
125                         done = true;
126                         break;
127                     }
128                     if (i == 0) {
129                         done = true;
130                     }
131                 }
132             }
133 
134             // Now working with the totally replaced paren string
135             // handle the dashed parts
136 
137             for (int i = dt.length() - 1; i >= 0; i--) {
138                 if (dt.charAt(i) == DASH) {
139                     if (this.serviceType != null) {
140                         this.wc.add(dt.substring(0, i) + DASH_WC + KSEP + this.serviceType);
141                     } else {
142                         this.wc.add(dt.substring(0, i) + DASH_WC);
143                     }
144                 }
145             }
146             if (this.serviceType != null) {
147                 this.wc.add(PURE_WC + KSEP + this.serviceType);
148             } else {
149                 this.wc.add(PURE_WC);
150             }
151         }
152     }
153 
154     /**
155      * Select an entry from the map
156      *
157      * @param dataId the string to wildcard
158      * @param map the map to choose from
159      * @return the found entry
160      */
161     public static DirectoryEntryList getWildcardedEntry(final String dataId, final DirectoryEntryMap map) {
162 
163         final DirectoryEntryList matches = new DirectoryEntryList();
164 
165         final WildcardEntry we = new WildcardEntry(dataId);
166         logger.debug("Got a set of size {} from {}", we.size(), dataId);
167 
168         for (final String w : we.asSet()) {
169             final DirectoryEntryList found = map.get(w);
170             if (found != null) {
171                 logger.debug("Found a wildcard match on {} size={}", w, found.size());
172                 matches.addAll(found);
173             } else {
174                 logger.debug("SKipping {} nothing in map", w);
175             }
176         }
177         return matches;
178     }
179 
180     /**
181      * Size of this wildcard entry set
182      */
183     public int size() {
184         load();
185         return this.wc.size();
186     }
187 }

35  /**
36   * DirectoryProviderPlace can be extended to enable registration for service discovery and access. This incorporates key
37   * attributes such as cost, representing the data processing speed of the place, and quality, indicating how effectively
38   * the place performs its function. Together, these attributes determine the total expense of the service and can be
39   * leveraged for processing optimization.
40   */
41  public abstract class DirectoryProviderPlace implements IServiceProviderPlace {

94      /**
95       * Help the constructor get the place running
96       *
97       * @param theDir name of our directory
98       */
99      protected void setupPlace(@Nullable String theDir, String placeLocation, boolean register) throws IOException {
100 
101         // Customize the logger to the runtime class
102         logger = LoggerFactory.getLogger(this.getClass());
103 
104         // The order of the following initialization calls
105         // is touchy. NPE all over if you mess up here.
106 
107         if (register) {
108             // Set ServicePlace config items
109             configureServicePlace(placeLocation);
110         }
111 
112         // Backwards compatibility setup items
113         DirectoryEntry firstentry = new DirectoryEntry(keys.get(0));
114         myKey = firstentry.getKey();
115         serviceCost = firstentry.getCost();
116         serviceQuality = firstentry.getQuality();
117         placeName = firstentry.getServiceLocation();
118 
119 
120         // configure directory references
121         if (!(this instanceof DirectoryPlace)) {
122             localizeDirectory(theDir);
123             logger.debug("Our localizedDirectory is {}", dirPlace);
124         } else {
125             logger.debug("Not localizing directory since we are a directory");
126         }
127     }

129     protected void setupPlacePostHook(boolean register) {
130 
131         // Bind to the namespace before registering
132         // our keys. This allows incoming traffic to find
133         // us as soon as they see the keys
134         for (String key : keys) {
135             String bindKey = KeyManipulator.getServiceLocation(key);
136             logger.debug("Binding myself into the namespace as {}", bindKey);
137             Namespace.bind(bindKey, this);
138         }
139 
140         if (register) {
141             // Register with the directory
142             // This pushes all our keys out to the directory which
143             // sends them on in turn to peers, &c. in the p2p network
144             register();
145         }
146     }

148     /**
149      * Get a local reference to the directory.
150      *
151      * @param theDir key for the directory to use, if null will look up default name
152      * @return true if it worked
153      */
154     private boolean localizeDirectory(@Nullable String theDir) {
155         // Get a local (non proxy) copy of the directory if possible!
156         // Looking up both if nothing is provided
157         if (theDir == null) {
158             try {
159                 localDirPlace = DirectoryPlace.lookup();
160                 dirPlace = localDirPlace.toString();
161             } catch (EmissaryException ex) {
162                 if (EmissaryServer.exists() && !(this instanceof DirectoryPlace)) {
163                     logger.warn("Unable to find DirectoryPlace in local namespace", ex);
164                     return false;
165                 }
166             }
167         } else {
168             dirPlace = theDir;
169             localDirPlace = null;
170             try {
171                 String myUrl = KeyManipulator.getServiceHostUrl(keys.get(0));
172                 String dirUrl = KeyManipulator.getServiceHostUrl(dirPlace);
173                 if (Strings.CS.equals(dirUrl, myUrl)) {
174                     localDirPlace = (IDirectoryPlace) Namespace.lookup(KeyManipulator.getServiceLocation(theDir));
175                 } else {
176                     logger.debug("Not localizing directory since dirPlace {} is not equal to myUrl {}", dirPlace, myUrl);
177                 }
178             } catch (EmissaryException ex) {
179                 logger.error("Exception attempting to get local reference to directory", ex);
180                 return false;
181             }
182         }
183         return true;
184     }

205     /**
206      * Configuration items read here are:
207      *
208      * <ul>
209      * <li>PLACE_NAME: place name portion of key, required</li>
210      * <li>SERVICE_NAME: service name portion of key, required</li>
211      * <li>SERVICE_TYPE: service type portion of key, required</li>
212      * <li>SERVICE_DESCRIPTION: description of place, required</li>
213      * <li>SERVICE_COST: cost of service provided, required</li>
214      * <li>SERVICE_QUALITY: quality of service provided, required</li>
215      * <li>SERVICE_PROXY: list of service proxy types for key</li>
216      * <li>SERVICE_KEY: full 4 part keys with expense</li>
217      * </ul>
218      *
219      * @param placeLocation the specified placeLocation or a full four part key to register with
220      */
221     protected void configureServicePlace(@Nullable String placeLocation) throws IOException {
222         Configurator configG = getConfigurator();
223         serviceDescription = configG.findStringEntry(SERVICE_DESCRIPTION);
224         if (serviceDescription == null || serviceDescription.length() == 0) {
225             serviceDescription = "Description not available";
226         }
227 
228         if (placeLocation == null) {
229             placeLocation = this.getClass().getSimpleName();
230         }
231 
232         String locationPart = placeLocation;
233         if (KeyManipulator.isKeyComplete(placeLocation)) {
234             keys.add(placeLocation); // save as first in list
235             locationPart = KeyManipulator.getServiceLocation(placeLocation);
236         } else if (!placeLocation.contains("://")) {
237             EmissaryNode node = new EmissaryNode();
238             locationPart = "http://" + node.getNodeName() + ":" + node.getNodePort() + "/" + placeLocation;
239         }
240 
241 
242         // Build keys the old fashioned way from parts specified in the config
243         String placeName = configG.findStringEntry(PLACE_NAME);
244         String serviceName = configG.findStringEntry(SERVICE_NAME);
245         String serviceType = configG.findStringEntry(SERVICE_TYPE);
246         int serviceCost = configG.findIntEntry(SERVICE_COST, -1);
247         int serviceQuality = configG.findIntEntry(SERVICE_QUALITY, -1);
248 
249         // Bah.
250         if (placeName != null && placeName.length() == 0) {
251             placeName = null;
252         }
253         if (serviceName != null && serviceName.length() == 0) {
254             serviceName = null;
255         }
256         if (serviceType != null && serviceType.length() == 0) {
257             serviceType = null;
258         }
259 
260         if (placeName != null && serviceName != null && serviceType != null && serviceCost > -1 && serviceQuality > -1) {
261             // pick up the proxies(save full 4-tuple keys!)
262             for (String sp : configG.findEntries(SERVICE_PROXY)) {
263                 DirectoryEntry de = new DirectoryEntry(sp, serviceName, serviceType, locationPart, serviceDescription, serviceCost, serviceQuality);
264                 keys.add(de.getFullKey());
265             }
266             // pick up the denied proxies(save full 4-tuple keys!)
267             for (String sp : configG.findEntries(SERVICE_PROXY_DENY)) {
268                 DirectoryEntry de = new DirectoryEntry(sp, serviceName, serviceType, locationPart, serviceDescription, serviceCost, serviceQuality);
269                 denyList.add(de.getDataType());
270             }
271         } else {
272             // May be configured the new way, but warn if there is a mixture of
273             // null and non-null items using the old-fashioned way. Perhaps the
274             // user just missed one of them
275             int nullCount = 0;
276             if (placeName == null) {
277                 nullCount++;
278             }
279             if (serviceName == null) {
280                 nullCount++;
281             }
282             if (serviceType == null) {
283                 nullCount++;
284             }
285             if (serviceCost == -1) {
286                 nullCount++;
287             }
288             if (serviceQuality == -1) {
289                 nullCount++;
290             }
291 
292             if (nullCount > 0 && nullCount < 5) {
293                 throw new IOException(
294                         "Missing configuration items. Please check SERVICE_NAME, SERVICE_TYPE, PLACE_NAME, SERVICE_COST, SERVICE_QUALITY");
295 
296             }
297         }
298 
299         // Now build any keys the new way
300         for (String k : configG.findEntries(SERVICE_KEY)) {
301             if (KeyManipulator.isKeyComplete(k)) {
302                 keys.add(k);
303             } else {
304                 logger.warn("SERVICE_KEY '{}' is missing parts and cannot be used", k);
305             }
306         }
307 
308         // Make sure some keys were defined one way or the other
309         if (keys.isEmpty()) {
310             throw new IOException("NO keys were defined. Please configure at least one "
311                     + "SERVICE_KEY or SERVICE_NAME/SERVICE_TYPE/SERVICE_PROXY group");
312         }
313     }

332     /**
333      * Convenience method a lot of places use. Removes all items from the current form stack that this place has proxies for
334      *
335      * @param d a data object whose current form will be expunged of my proxies
336      * @return count of how many items removed
337      */
338     protected int nukeMyProxies(IBaseDataObject d) {
339         List<String> nukem = new ArrayList<>();
340         int sz = d.currentFormSize();
341         Set<String> serviceProxies = getProxies();
342 
343         if (serviceProxies.contains("*")) {
344             d.replaceCurrentForm(null); // clear it out
345             return sz;
346         }
347 
348         // listem
349         for (int i = 0; i < sz; i++) {
350             String form = d.currentFormAt(i);
351             if (serviceProxies.contains(form)) {
352                 nukem.add(form);
353             } else {
354                 // cardem
355                 WildcardEntry wc = new WildcardEntry(form);
356                 if (!Collections.disjoint(wc.asSet(), serviceProxies)) {
357                     nukem.add(form);
358                 }
359             }
360         }
361 
362         // nukem
363         for (String f : nukem) {
364             int pos = d.searchCurrentForm(f);
365             if (pos != -1) {
366                 d.deleteCurrentFormAt(pos);
367             }
368         }
369 
370         return nukem.size();
371     }

433     /**
434      * Add a service proxy to a running place. Duplicates are ignored.
435      *
436      * @param serviceProxy the new proxy string to add
437      */
438     @Override
439     public void addServiceProxy(String serviceProxy) {
440         // Add new one to the top key in the list
441         DirectoryEntry de = new DirectoryEntry(keys.get(0));
442         boolean keyAdded = false;
443         if (!de.getDataType().equals(serviceProxy)) {
444             // Clear out the placeholder
445             if (de.getDataType().equals(UNUSED_PROXY)) {
446                 keys.remove(0);
447             }
448 
449             de.setDataType(serviceProxy);
450 
451             if (!keys.contains(de.getFullKey())) {
452                 keys.add(de.getFullKey());
453 
454                 // Register the new proxy in the directory
455                 logger.debug("Registering new key {}", de.getKey());
456                 register(de.getFullKey());
457                 keyAdded = true;
458             }
459         }
460 
461         if (!keyAdded) {
462             logger.debug("Duplicate service proxy {} ignored", serviceProxy);
463         }
464     }

487     /**
488      * Register a single service proxy key
489      *
490      * @param key the new key to register
491      */
492     protected void register(String key) {
493         logger.debug("Registering key {}", key);
494         // Cannot register if we have no directory
495         // If we are the directory, its no problem though
496         if (dirPlace == null) {
497             if (!(this instanceof IDirectoryPlace)) {
498                 logger.debug("Directory is null: cannot register anything. Illegal configuration.");
499             }
500             return;
501         }
502 
503         // Register place and all proxies by building up a list
504         // of our interests to send to the directory
505         List<String> keylist = new ArrayList<>();
506         keylist.add(key);
507         registerWithDirectory(keylist);
508     }

510     /**
511      * Register our interest in all of our serviceProxies Sends only one message to the directory to cover all service
512      * proxies (a scalability issue for large systems)
513      */
514     protected void register() {
515         logger.debug("Registering: {}", this);
516 
517         // Cannot register if we have no directory
518         // If we are the directory, its no problem though
519         if (dirPlace == null) {
520             if (!(this instanceof IDirectoryPlace)) {
521                 logger.debug("Directory is null: cannot register anything. Illegal configuration.");
522             }
523             return;
524         }
525 
526         // Register place and all proxies by building up a list
527         // of our current interests to send to the directory
528         registerWithDirectory(new ArrayList<>(keys));
529     }

531     /**
532      * Register keys with the directory
533      *
534      * @param keylist the keys to register
535      */
536     protected void registerWithDirectory(List<String> keylist) {
537         try {
538             if (localDirPlace == null) {
539                 // This should never happen, we require a local
540                 // directory on every JVM...
541                 logger.error("Key registration requires a DirectoryPlace in every Emissary Node");
542             } else {
543                 logger.debug("Registering my {} keys {}", keylist.size(), keylist);
544                 localDirPlace.addPlaces(keylist);
545             }
546         } catch (RuntimeException e) {
547             logger.warn("Register ERROR for keys {}", keylist, e);
548         }
549     }

552     /**
553      * Deregister keys from the directory
554      *
555      * @param keys the keys to register
556      */
557     protected void deregisterFromDirectory(List<String> keys) {
558         try {
559             if (localDirPlace == null && dirPlace != null) {
560                 // This should never happen, we require a local
561                 // directory on every JVM...
562                 logger.debug("Deregistering my {} proxies {} from remote dir {}", keys.size(), keys, dirPlace);
563                 DirectoryAdapter da = new DirectoryAdapter();
564                 da.outboundRemovePlaces(dirPlace, keys, false);
565             } else if (localDirPlace != null) {
566                 logger.debug("Deregistering my {} proxies {}", keys.size(), keys);
567                 localDirPlace.removePlaces(keys);
568             }
569         } catch (RuntimeException e) {
570             logger.warn("Deregister ERROR keys={}", keys, e);
571         }
572     }

574     /**
575      * Remove a service proxy from the running place. Proxy strings not found registered will be ignored Will remove all
576      * keys that match the supplied proxy
577      *
578      * @param serviceProxy the proxy string to remove
579      */
580     @Override
581     public void removeServiceProxy(String serviceProxy) {
582         // List of keys to deregister
583         List<String> keylist = new ArrayList<>();
584 
585         // nb. no enhanced for loop due to remove
586         for (Iterator<String> i = keys.iterator(); i.hasNext();) {
587             String k = i.next();
588             String kproxy = KeyManipulator.getDataType(k);
589             if (kproxy.equals(serviceProxy)) {
590                 keylist.add(KeyManipulator.removeExpense(k));
591                 i.remove();
592             }
593         }
594         if (!keylist.isEmpty()) {
595             deregisterFromDirectory(keylist);
596         } else {
597             logger.debug("Unknown service proxy {} ignored", serviceProxy);
598         }
599 
600         // Make sure we leave something on the keys list
601         if (keys.isEmpty() && !keylist.isEmpty()) {
602             DirectoryEntry de = new DirectoryEntry(keylist.get(0));
603             de.setCost(serviceCost);
604             de.setQuality(serviceQuality);
605             de.setDataType(UNUSED_PROXY);
606             keys.add(de.getFullKey());
607         }
608     }

610     /**
611      * Remove and deregister a key
612      *
613      * @param key the full key (with expense) to deregister
614      */
615     @Override
616     public void removeKey(String key) {
617         String keyWithOutExpense = KeyManipulator.removeExpense(key);
618 
619         if (keys.remove(key)) {
620             List<String> keylist = new ArrayList<>();
621             keylist.add(keyWithOutExpense);
622             deregisterFromDirectory(keylist);
623 
624             if (keys.isEmpty()) {
625                 DirectoryEntry de = new DirectoryEntry(key);
626                 de.setDataType(UNUSED_PROXY);
627                 keys.add(de.getFullKey());
628             }
629         } else {
630             logger.debug("Key specified for removal not found: {}", key);
631         }
632     }

634     /**
635      * Stop all threads if any. Can be overridden by, deregister from directory, and remove from the namespace. derived
636      * classes if needed
637      */
638     @Override
639     public void shutDown() {
640         logger.debug("Called shutDown()");
641 
642         // Remove from directory a list of keys without expense
643         deregisterFromDirectory(keys.stream().map(KeyManipulator::removeExpense).collect(Collectors.toList()));
644 
645         // Unbind from namespace
646         unbindFromNamespace();
647     }

649     protected void unbindFromNamespace() {
650         keys.stream().map(KeyManipulator::getServiceLocation).forEach(Namespace::unbind);

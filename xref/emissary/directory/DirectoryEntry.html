227     /**
228      * Set the expense, cost and quality from the expense
229      * 
230      * @param expense the expense to use
231      */
232     protected void setCqeFromExp(final int expense) {
233         if (expense >= 100) {
234             final int invQual = expense % 100;
235             this.theQuality = 100 - invQual;
236             this.theCost = (expense - invQual) / 100;
237         } else {
238             // Should give expense of 0
239             this.theCost = 0;
240             this.theQuality = 100;
241         }
242 
243         this.theExpense = expense;
244 
245     }

310     /**
311      * Set key and precompute stuff we need
312      * 
313      * @see #buildKey
314      * @param key the key
315      */
316     protected void setKey(final String key) {
317         this.theKey = KeyManipulator.removeExpense(key);
318         this.serviceType = KeyManipulator.getServiceType(this.theKey);
319         this.serviceName = KeyManipulator.getServiceName(this.theKey);
320         this.dataType = KeyManipulator.getDataType(key);
321         this.dataId = this.dataType + KeyManipulator.DATAIDSEPARATOR + this.serviceType;
322         this.serviceLocation = KeyManipulator.getServiceLocation(key);
323         this.serviceHostUrl = KeyManipulator.getServiceHostUrl(key);
324         final int exp = KeyManipulator.getExpense(key, -1);
325         if (exp > -1) {
326             setCqeFromExp(exp);
327         }
328     }

494      * 
495      * @param cost the cost

498     public static int calculateExpense(final int cost, final int quality) {
499         return (cost * 100) + (100 - quality);

512     /**
513      * Determine if local by looking up in namespace
514      * 
515      * @return true if local
516      */
517     public boolean isLocal() {
518         if (!this.lookupAttempted) {
519             try {
520                 setLocalPlace((IServiceProviderPlace) Namespace.lookup(this.serviceLocation));
521             } catch (NamespaceException e) {
522                 // empty catch block
523             }
524             logger.debug("NS Lookup for locality on {}{}", this.serviceLocation, this.localPlace == null ? " failed" : " passed");
525 
526             this.lookupAttempted = true;
527         }
528         return this.localPlace != null;
529     }

558     /**
559      * Build an entry from the supplied xml fragment
560      * 
561      * @param e a JDOM Element
562      */
563     public static DirectoryEntry fromXml(final Element e) {
564         final String key = e.getChildTextTrim(KEY);
565         final String desc = e.getChildTextTrim(DESC);
566         final int cost = JDOMUtil.getChildIntValue(e, COST);
567         final int quality = JDOMUtil.getChildIntValue(e, QUALITY);
568         return new DirectoryEntry(key, desc, cost, quality);
569     }

571     /**
572      * Turn this entry into an xml fragment
573      */
574     public Element getXml() {
575         final Element root = new Element(ENTRY);
576         root.addContent(JDOMUtil.simpleElement(KEY, this.theKey));
577         root.addContent(JDOMUtil.simpleElement(DESC, this.description));
578         root.addContent(JDOMUtil.simpleElement(COST, this.theCost));
579         root.addContent(JDOMUtil.simpleElement(QUALITY, this.theQuality));
580         root.addContent(JDOMUtil.simpleElement(EXPENSE, this.theExpense));
581         return root;

87      /**
88       * Setup to manage heartbeats to remote directories
89       *
90       * @param directoryKey Key for directory I act on behalf of
91       * @param dirList list of directory keys for remote directories
92       * @param initialDelaySeconds seconds to wait before initial timer task
93       * @param intervalSeconds how often the timeer task kicks off
94       */
95      public HeartbeatManager(final String directoryKey, @Nullable final List<String> dirList, final int initialDelaySeconds,
96              final int intervalSeconds) {
97          this.initialDelaySeconds = initialDelaySeconds;
98          this.intervalSeconds = intervalSeconds;
99  
100         logger.debug("Starting with initialDelay={}, interval={}", initialDelaySeconds, intervalSeconds);
101 
102         // new daemon timer
103         this.timer = new Timer("HeartbeatManager", true);
104 
105         // Save directory key
106         this.thisDirectory = directoryKey;
107 
108         // Save each directory in dirList along with good
109         // starting health value
110         if (dirList != null) {
111             for (final String key : dirList) {
112                 addRemoteDirectory(key);
113             }
114         }
115 
116         // "smooth" execution every 30 seconds starting in 2 minutes
117         this.timer.schedule(new HeartbeatTask(), (this.initialDelaySeconds * 1000L), (this.intervalSeconds * 1000L));
118     }

226     /**
227      * Set health status and call the trigger on transition
228      *
229      * @param key key for directory
230      * @param status new status
231      * @param reason the exception message on bad status
232      */
233     protected void healthReport(final String key, final boolean status, final String reason) {
234         final String dirKey = KeyManipulator.getDefaultDirectoryKey(key);
235         final Health v = this.directories.get(dirKey);
236         if (v == null) {
237             logger.error("Not monitoring directory {}", dirKey);
238             return;
239         }
240 
241         final boolean wasAlive = v.isAlive();
242         final boolean wasHealthy = v.isHealthy();
243 
244         v.addReport(status, reason);
245 
246         final boolean isAlive = v.isAlive();
247         final boolean isHealthy = v.isHealthy();
248 
249         if (logger.isDebugEnabled()) {
250             logger.debug("Reporting on {} status={}, wasAlive/Healthy={}/{}, isAlive/Healthy={}/{}", key, status, wasAlive, wasHealthy, isAlive,
251                     isHealthy);
252         }
253 
254         if (wasAlive && !isAlive) {
255             // We had a permanent failure.
256             takeFailureAction(key, true);
257         } else if (wasHealthy && !isHealthy) {
258             // We had a negative transition.
259             takeFailureAction(key, false);
260         } else if (!wasHealthy && isHealthy) {
261             // We had a positive transition.
262             takeSuccessAction(key);
263         }
264     }

303     /**
304      * The Task thread, monitors all remote directories
305      */
306     class HeartbeatTask extends TimerTask {
307         @Override
308         public void run() {
309             try {
310                 logger.debug("Running timer task on {} directories", HeartbeatManager.this.directories.size());
311                 for (final String dir : HeartbeatManager.this.directories.keySet()) {
312                     heartbeat(dir);
313                 }
314                 logger.debug("Ending the HeartbeatTask run method");
315             } catch (Exception e) {
316                 logger.error("Unexpected problem in heartbeat timer", e);
317             }
318         }
319     }

321     /**
322      * Send a heartbeat message to the directory represented by key and take follow-on actions as appropriate Called from
323      * the timer task normally, but can be called externally by the impatient
324      *
325      * @see emissary.directory.DirectoryPlace#heartbeatRemoteDirectory(String)
326      * @param key key representing the directory to heartbeat
327      * @return true if the directory referenced by key is up
328      */
329     public final boolean heartbeat(final String key) {
330         boolean isup = false;
331         try {
332             logger.debug("Sending heartbeat msg to {}", key);
333             EmissaryResponse response = getHeartbeat(this.thisDirectory, key);
334             if (response.getStatus() == 200) {
335                 healthReport(key, true, response.getContentString());
336                 isup = true;
337             } else {
338                 healthReport(key, false, response.getContentString());
339                 isup = false;
340             }
341         } catch (Exception e) {
342             logger.error("Cannot perform heartbeat", e);
343             healthReport(key, false, e.getMessage());
344             isup = false;
345         }
346         return isup;
347     }

353     public static EmissaryResponse getHeartbeat(String fromPlace, String toPlace, EmissaryClient client) {
354         final String directoryUrl = KeyManipulator.getServiceHostUrl(toPlace);
355         final HttpPost method = client.createHttpPost(directoryUrl, EmissaryClient.context, "/Heartbeat.action");
356         final String loc = KeyManipulator.getServiceLocation(toPlace);
357 
358         final List<NameValuePair> nvps = new ArrayList<>();
359         nvps.add(new BasicNameValuePair(HeartbeatAdapter.FROM_PLACE_NAME, fromPlace));
360         nvps.add(new BasicNameValuePair(HeartbeatAdapter.TO_PLACE_NAME, loc));
361         method.setEntity(new UrlEncodedFormEntity(nvps, StandardCharsets.UTF_8));
362 
363         return client.send(method);
364     }

367     /**
368      * Holder class for the health information for a single remote directory
369      */
370     class Health {
371         // Count consecutive failures
372         private int failCounter = 0;
373 
374         private String lastMessage;
375 
376         /**
377          * Create a new Health object with the specified status and msg
378          *
379          * @param isAlive true if the initial status is no failures
380          * @param msg the initial msg value
381          */
382         public Health(final boolean isAlive, final String msg) {
383             setStatus(isAlive, msg);
384         }
385 
386         /**
387          * Accumulate another heartbeat result
388          *
389          * @param v the most recent status
390          * @param msg the most recent message
391          */
392         public void addReport(final boolean v, final String msg) {
393             this.lastMessage = msg;
394             if (v) {
395                 this.failCounter = 0;
396             } else {
397                 this.failCounter++;
398             }
399         }
400 
401         /**
402          * Set the status now
403          *
404          * @param isAlive false means permanent failure indicated
405          * @param message message to asocciate with this statsu
406          */
407         void setStatus(final boolean isAlive, final String message) {
408             if (!isAlive) {
409                 this.failCounter = HeartbeatManager.this.permanentFailThreshold;
410                 this.lastMessage = message;
411             } else {
412                 this.failCounter = 0;
413                 this.lastMessage = message;
414             }
415         }
416 
417         /**
418          * Report our health status
419          *
420          * @return true if failed less than threshold times
421          */
422         public boolean isHealthy() {
423             return this.failCounter < HeartbeatManager.this.failThreshold;
424         }
425 
426         /**
427          * Report our aliveness status
428          *
429          * @return true if failed less than permanent threshold times
430          */
431         public boolean isAlive() {
432             return this.failCounter < HeartbeatManager.this.permanentFailThreshold;
433         }
434 
435         /**
436          * Access to the last saved message
437          */
438         public String getLastMessage() {
439             return this.lastMessage;
440         }
441     }

94      /**
95       * Setup to manage heartbeats to remote directories
96       *
97       * @param directoryKey Key for directory I act on behalf of
98       * @param initialDelaySeconds seconds to wait before initial timer task
99       * @param intervalSeconds how often the timeer task kicks off
100      * @param dirList list of directory keys for remote directories
101      */
102     public HeartbeatManager(final String directoryKey, final int initialDelaySeconds, final int intervalSeconds,
103             @Nullable final List<String> dirList) {
104         this.initialDelaySeconds = initialDelaySeconds;
105         this.intervalSeconds = intervalSeconds;
106 
107         logger.debug("Starting with initialDelay={}, interval={}", initialDelaySeconds, intervalSeconds);
108 
109         // new daemon timer
110         this.timer = new Timer("HeartbeatManager", true);
111 
112         // Save directory key
113         this.thisDirectory = directoryKey;
114 
115         // Save each directory in dirList along with good
116         // starting health value
117         if (dirList != null) {
118             for (final String key : dirList) {
119                 addRemoteDirectory(key);
120             }
121         }
122 
123         // "smooth" execution every 30 seconds starting in 2 minutes
124         this.timer.schedule(new HeartbeatTask(), this.initialDelaySeconds * 1000L, this.intervalSeconds * 1000L);
125     }

233     /**
234      * Set health status and call the trigger on transition
235      *
236      * @param key key for directory
237      * @param status new status
238      * @param reason the exception message on bad status
239      */
240     protected void healthReport(final String key, final boolean status, final String reason) {
241         final String dirKey = KeyManipulator.getDefaultDirectoryKey(key);
242         final Health v = this.directories.get(dirKey);
243         if (v == null) {
244             logger.error("Not monitoring directory {}", dirKey);
245             return;
246         }
247 
248         final boolean wasAlive = v.isAlive();
249         final boolean wasHealthy = v.isHealthy();
250 
251         v.addReport(status, reason);
252 
253         final boolean isAlive = v.isAlive();
254         final boolean isHealthy = v.isHealthy();
255 
256         if (logger.isDebugEnabled()) {
257             logger.debug("Reporting on {} status={}, wasAlive/Healthy={}/{}, isAlive/Healthy={}/{}", key, status, wasAlive, wasHealthy, isAlive,
258                     isHealthy);
259         }
260 
261         if (wasAlive && !isAlive) {
262             // We had a permanent failure.
263             takeFailureAction(key, true);
264         } else if (wasHealthy && !isHealthy) {
265             // We had a negative transition.
266             takeFailureAction(key, false);
267         } else if (!wasHealthy && isHealthy) {
268             // We had a positive transition.
269             takeSuccessAction(key);
270         }
271     }

273     /**
274      * Notify our directory that there was a falure
275      *
276      * @param key string key of the directory that failed
277      */
278     public void takeFailureAction(final String key, final boolean permanent) {
279         final String myKey = KeyManipulator.getServiceLocation(this.thisDirectory);
280         try {
281             final IRemoteDirectory d = (IRemoteDirectory) Namespace.lookup(myKey);
282             final int count = d.irdFailDirectory(key, permanent);
283             logger.info("Notified {} of failed directory {}{}, {} keys removed", myKey, key, permanent ? " permanently!" : "", count);
284         } catch (NamespaceException ne) {
285             logger.error("Tried to fail a remote directory {} but cannot look up my own directory using {}", key, myKey, ne);
286         }
287     }

289     /**
290      * Notify our directory that a directory has been contacted. This could initiate a zone transfer or other action
291      *
292      * @param key string key of the directory that was contacted
293      */
294     void takeSuccessAction(final String key) {
295         final String myKey = KeyManipulator.getServiceLocation(this.thisDirectory);
296         try {
297             final DirectoryPlace d = (DirectoryPlace) Namespace.lookup(myKey);
298             if (d.isStaticPeer(key)) {
299                 logger.info("Notifying {} of re-established contact with {}", myKey, key);
300                 d.contactedRemoteDirectory(key);
301             } else {
302                 logger.info("Ignoring contact with non-configured peer {}", key);
303             }
304         } catch (NamespaceException ne) {
305             logger.error("Tried to reestablish a remote directory " + key + " but cannot look up my own directory using " + myKey, ne);
306         }
307     }

310     /**
311      * The Task thread, monitors all remote directories
312      */
313     class HeartbeatTask extends TimerTask {
314         @Override
315         public void run() {
316             try {
317                 logger.debug("Running timer task on {} directories", HeartbeatManager.this.directories.size());
318                 for (final String dir : HeartbeatManager.this.directories.keySet()) {
319                     heartbeat(dir);
320                 }
321                 logger.debug("Ending the HeartbeatTask run method");
322             } catch (RuntimeException e) {
323                 logger.error("Unexpected problem in heartbeat timer", e);
324             }
325         }
326     }

328     /**
329      * Send a heartbeat message to the directory represented by key and take follow-on actions as appropriate Called from
330      * the timer task normally, but can be called externally by the impatient
331      *
332      * @see emissary.directory.DirectoryPlace#heartbeatRemoteDirectory(String)
333      * @param key key representing the directory to heartbeat
334      * @return true if the directory referenced by key is up
335      */
336     public final boolean heartbeat(final String key) {
337         boolean isup = false;
338         try {
339             logger.debug("Sending heartbeat msg to {}", key);
340             EmissaryResponse response = getHeartbeat(this.thisDirectory, key);
341             if (response.getStatus() == 200) {
342                 healthReport(key, true, response.getContentString());
343                 isup = true;
344             } else {
345                 healthReport(key, false, response.getContentString());
346                 isup = false;
347             }
348         } catch (RuntimeException e) {
349             logger.error("Cannot perform heartbeat", e);
350             healthReport(key, false, e.getMessage());
351             isup = false;
352         }
353         return isup;
354     }

356     public static EmissaryResponse getHeartbeat(String fromPlace, String toPlace) {
357         return getHeartbeat(fromPlace, toPlace, new EmissaryClient());
358     }

360     public static EmissaryResponse getHeartbeat(String fromPlace, String toPlace, EmissaryClient client) {
361         final String directoryUrl = KeyManipulator.getServiceHostUrl(toPlace);
362         final HttpPost method = client.createHttpPost(directoryUrl, EmissaryClient.context, "/Heartbeat.action");
363         final String loc = KeyManipulator.getServiceLocation(toPlace);
364 
365         final List<NameValuePair> nvps = new ArrayList<>();
366         nvps.add(new BasicNameValuePair(HeartbeatAdapter.FROM_PLACE_NAME, fromPlace));
367         nvps.add(new BasicNameValuePair(HeartbeatAdapter.TO_PLACE_NAME, loc));
368         method.setEntity(new UrlEncodedFormEntity(nvps, StandardCharsets.UTF_8));
369 
370         return client.send(method);
371     }

374     /**
375      * Holder class for the health information for a single remote directory
376      */
377     class Health {
378         // Count consecutive failures
379         private int failCounter = 0;
380 
381         private String lastMessage;
382 
383         /**
384          * Create a new Health object with the specified status and msg
385          *
386          * @param isAlive true if the initial status is no failures
387          * @param msg the initial msg value
388          */
389         public Health(final boolean isAlive, final String msg) {
390             setStatus(isAlive, msg);
391         }
392 
393         /**
394          * Accumulate another heartbeat result
395          *
396          * @param v the most recent status
397          * @param msg the most recent message
398          */
399         public void addReport(final boolean v, final String msg) {
400             this.lastMessage = msg;
401             if (v) {
402                 this.failCounter = 0;
403             } else {
404                 this.failCounter++;
405             }
406         }
407 
408         /**
409          * Set the status now
410          *
411          * @param isAlive false means permanent failure indicated
412          * @param message message to asocciate with this statsu
413          */
414         void setStatus(final boolean isAlive, final String message) {
415             if (!isAlive) {
416                 this.failCounter = HeartbeatManager.this.permanentFailThreshold;
417                 this.lastMessage = message;
418             } else {
419                 this.failCounter = 0;
420                 this.lastMessage = message;
421             }
422         }
423 
424         /**
425          * Report our health status
426          *
427          * @return true if failed less than threshold times
428          */
429         public boolean isHealthy() {
430             return this.failCounter < HeartbeatManager.this.failThreshold;
431         }
432 
433         /**
434          * Report our aliveness status
435          *
436          * @return true if failed less than permanent threshold times
437          */
438         public boolean isAlive() {
439             return this.failCounter < HeartbeatManager.this.permanentFailThreshold;
440         }
441 
442         /**
443          * Access to the last saved message
444          */
445         public String getLastMessage() {
446             return this.lastMessage;
447         }
448     }

96      /**
97       * Initialization phase hook for the filter with provided filter configuration
98       * 
99       * @param theConfigG passed in configuration object, usually DropOff's config
100      * @param filterName the configured name of this filter or null for the default
101      * @param theFilterConfig the configuration for the specific filter
102      */
103     @Override
104     public void initialize(final emissary.config.Configurator theConfigG, @Nullable final String filterName,
105             final emissary.config.Configurator theFilterConfig) {
106         this.configG = theConfigG;
107         if (filterName != null) {
108             setFilterName(filterName);
109         }
110         loadFilterConfiguration(theFilterConfig);
111         loadFilterCondition(theConfigG);
112         loadOutputSpec(theConfigG);
113         this.dropOffUtil = new DropOffUtil(theConfigG);
114         initializeOutputTypes(this.filterConfig);
115     }

117     private final void loadFilterCondition(final Configurator parentConfig) {
118         this.filterConditionSpec = parentConfig.findStringEntry("FILTER_CONDITION_" + getFilterName(), null);
119 
120         // format FILTER_CONDITION_<filtername> = profilename:clazz just like dropoff filter config
121         if (!StringUtils.isEmpty(filterConditionSpec)) {
122             final String name;
123             final String clazz;
124             Configurator filterConfig = null;
125             final int colpos = filterConditionSpec.indexOf(':');
126             if (colpos > -1) {
127                 name = filterConditionSpec.substring(0, colpos);
128                 clazz = filterConditionSpec.substring(colpos + 1);
129                 final String filterConfigName = parentConfig.findStringEntry(name);
130                 if (filterConfigName != null) {
131                     try {
132                         filterConfig = ConfigUtil.getConfigInfo(filterConfigName);
133                     } catch (IOException configError) {
134                         logger.warn("Specified filter configuration {} cannot be loaded", filterConfigName);
135                         return;
136                     }
137                 }
138             } else {
139                 clazz = filterConditionSpec;
140             }
141 
142             try {
143                 final Object filterConditionObj = emissary.core.Factory.create(clazz);
144 
145                 if (filterConditionObj != null && filterConditionObj instanceof IFilterCondition) {
146                     this.filterCondition = (IFilterCondition) filterConditionObj;
147                     // initialize using the config
148                     filterCondition.initialize(filterConfig);
149                 } else {
150                     logger.warn("Failed to initialize filter condition {}. Filter does not implement IFilterCondition", getFilterName());
151                 }
152             } catch (Throwable t) {
153                 // failed to initialize
154                 logger.error("Failed in initialize filter condition {} with argument {} and message {}", getFilterName(), filterConditionSpec,
155                         t.getMessage(), t);
156             }
157 
158         }
159     }

167         if (config != null) {
168             this.outputTypes = config.findEntriesAsSet("OUTPUT_TYPE");
169             this.logger.debug("Loaded {} output types for filter {}", this.outputTypes.size(), this.outputTypes);
170             this.blacklist = config.findEntriesAsSet("BLACKLIST");
171             this.logger.debug("Loaded {} blacklist types for filter {}", this.blacklist.size(), this.blacklist);
172         } else {
173             this.logger.debug("InitializeCustom has null filter config");
174         }
175     }

197     /**
198      * Load the filter configuration with precendence of provided, named, default Preference order for loading
199      * configurations
200      * <ol>
201      * <li>[filter-package].FILTER_NAME.cfg</li>
202      * <li>[filter-package].[filter-class]-FILTER_NAME.cfg</li>
203      * <li>[filter-package].[filter-class].cfg</li>
204      * </ol>
205      * 
206      * @param suppliedFilterConfig configuration to use when not null
207      */
208     protected void loadFilterConfiguration(@Nullable final Configurator suppliedFilterConfig) {
209         if (suppliedFilterConfig != null) {
210             this.filterConfig = suppliedFilterConfig;
211             return;
212         }
213 
214         final List<String> configPreferences = new ArrayList<String>();
215 
216         if (getFilterName() != null) {
217             configPreferences.add(this.getClass().getPackage().getName() + "." + getFilterName() + ConfigUtil.CONFIG_FILE_ENDING);
218             configPreferences.add(this.getClass().getName() + "-" + getFilterName() + ConfigUtil.CONFIG_FILE_ENDING);
219         }
220         configPreferences.add(this.getClass().getName() + ConfigUtil.CONFIG_FILE_ENDING);
221 
222         this.logger.debug("Looking for filter configuration preferences {}", configPreferences);
223         try {
224             this.filterConfig = emissary.config.ConfigUtil.getConfigInfo(configPreferences);
225         } catch (IOException iox) {
226             this.logger.debug("Could not find filter configuration for {}", getFilterName(), iox);
227         }
228     }

231      * Run the filter for a set of documents
232      * 
233      * @param list collection of IBaseDataObject to run the filter on
234      * @param params map of params
235      * @return status value
236      */
237     @Override
238     public int filter(final List<IBaseDataObject> list, final Map<String, Object> params) {
239         // Important to process them in order, if not already sorted
240 
241         int status = 0;
242         for (final IBaseDataObject d : list) {
243             status = filter(d, params);
244         }
245         return status;
246     }

335     /**
336      * Get bytes as UTF-8 converted from specified charset
337      * 
338      * @param value the contents
339      * @param start position to start subarray
340      * @param len length of subarray
341      * @param charset the charset of the bytes in value
342      */
343     protected String normalizeBytes(final byte[] value, final int start, final int len, @Nullable final String charset) {
344         String s = null;
345 
346         if (charset != null) {
347             try {
348                 s = new String(value, start, len, charset);
349             } catch (UnsupportedEncodingException ex) {
350                 this.logger.debug("Error encoding string", ex);
351             }
352         }
353 
354         if (s == null) {
355             // from exception or no charset
356             s = new String(value, start, len);
357         }
358 
359         return s;
360     }

362     /**
363      * Extract the charset from the payload or defaultCharset
364      * 
365      * @param d the payload
366      * @param defaultCharset the default
367      * @return the charset or defualtCharset if none
368      */
369     protected String getCharset(final IBaseDataObject d, final String defaultCharset) {
370         String lang = d.getFontEncoding();
371         if (lang == null || lang.toUpperCase().indexOf("ASCII") != -1 || lang.toUpperCase().indexOf("8859-1") != -1) {
372             final String s = d.getStringParameter("HTML_CHARSET");
373             if (s != null) {
374                 lang = s;
375             }
376         }
377         if (lang == null || lang.toUpperCase().indexOf("ASCII") != -1 || lang.toUpperCase().indexOf("8859-1") != -1) {
378             final String s = d.getStringParameter("MIME_CHARSET");
379             if (s != null) {
380                 lang = s;
381             }
382         }
383         if (lang == null) {
384             return defaultCharset;
385         } else {
386             return JavaCharSet.get(lang);
387         }
388     }

400     /**
401      * Determine is this payload should be output by this filter Usually by the primary view or one of the alternate views
402      * being on the outputTypes list from the run-time type configuration stream for the filter in question.
403      *
404      * @param types types to check
405      * @return true if any one of the types is outputtable
406      */
407     protected boolean isOutputtable(final Collection<String> types) {
408         if (this.outputTypes.contains("*")) {
409             this.logger.debug("Outputtable due to wildcard in output types");
410             return true;
411         }
412 
413         final boolean canOutput = !Collections.disjoint(this.outputTypes, types);
414         if (canOutput && this.logger.isDebugEnabled()) {
415             final Set<String> outputFor = new HashSet<String>();
416             for (final String s : this.outputTypes) {
417                 if (types.contains(s)) {
418                     outputFor.add(s);
419                 }
420             }
421             this.logger.debug("Outputtable due to non-disjoint sets: {}", outputFor);
422         }
423         return canOutput;
424     }

426     /**
427      * Makes a set of the file type, current form with and without the .PrimaryView qualifier and all the alternate view
428      * names Result set can be passed to {@link #isOutputtable(Collection)} for checking
429      *
430      * @param d the payload
431      */
432     protected Set<String> getTypesToCheck(final IBaseDataObject d) {
433         final Set<String> checkTypes = getPrimaryTypesToCheck(d);
434         for (final String viewName : d.getAlternateViewNames()) {
435             checkTypes.addAll(getTypesToCheckForNamedView(d, viewName));
436         }
437         checkTypes.addAll(getTypesToCheckForNamedView(d, METADATA_VIEW_NAME));
438         checkTypes.add(ALL_ALT_VIEWS);
439         return checkTypes;
440     }
441 
442     protected Set<String> getTypesToCheckForNamedView(final IBaseDataObject d, final String viewName) {
443         final Set<String> checkTypes = new HashSet<String>();
444         final String lang = this.dropOffUtil.getLanguage(d);
445         final String fileType = DropOffUtil.getFileType(d);
446         final String currentForm = d.currentForm();
447 
448         // skip over blacklisted alt views
449         if (this.blacklist.contains(viewName) || this.blacklist.contains(fileType + "." + viewName)) {
450             return checkTypes;
451         }
452 
453         checkTypes.add(fileType);
454         checkTypes.add(fileType + "." + viewName);
455         checkTypes.add("*." + viewName);
456 
457         if (!"NONE".equals(lang)) {
458             checkTypes.add(lang);
459             checkTypes.add(lang + "." + viewName);
460             checkTypes.add(lang + "." + fileType);
461             checkTypes.add(lang + "." + fileType + "." + viewName);
462         }
463 
464         if (currentForm != null && !fileType.equals(currentForm)) {
465             checkTypes.add(currentForm);
466             checkTypes.add(currentForm + "." + viewName);
467             if (!"NONE".equals(lang)) {
468                 checkTypes.add(lang + "." + currentForm);
469                 checkTypes.add(lang + "." + currentForm + "." + viewName);
470             }
471         }
472         this.logger.debug("Types to be checked for named view {}: {}", viewName, checkTypes);
473         return checkTypes;
474     }

476     /**
477      * Makes a set of the file type, current form with and without the .PrimaryView qualifier. Result set can be passed to
478      * {@link #isOutputtable(Collection)} for checking whether the primary view should be output
479      *
480      * @param d the payload
481      */
482     protected Set<String> getPrimaryTypesToCheck(final IBaseDataObject d) {
483         final Set<String> checkTypes = getTypesToCheckForNamedView(d, PRIMARY_VIEW_NAME);
484         final String lang = this.dropOffUtil.getLanguage(d);
485         checkTypes.add(lang + LANGUAGE_VIEW);
486         checkTypes.add(ALL_LANGUAGE_VIEWS);
487         checkTypes.add(ALL_PRIMARY_VIEWS);
488         return checkTypes;
489     }

50       * Method to convert payload(s) to an output type

59      /**
60       * Initialization phase hook for the filter with provided filter configuration
61       *
62       * @param theConfigG passed in configuration object, usually DropOff's config
63       * @param filterName the configured name of this filter or null for the default
64       * @param theFilterConfig the configuration for the specific filter
65       */
66      @Override
67      public void initialize(final Configurator theConfigG, Configurator>final String filterName, final Configurator theFilterConfig) {
68          super.initialize(theConfigG, filterName, theFilterConfig);
69          initOutputConfig();
70          initRollConfig();
71          initFilenameGenerator();
72          setupLocalOutputDir();
73          setupRoller();
74      }

76      /**
77       * Initialize the output config vars
78       */
79      protected void initOutputConfig() {
80          this.defaultOutputPath = this.filterConfig.findStringEntry(OUTPUT_PATH, defaultOutputPath);
81          this.outputPath = Paths.get(this.defaultOutputPath);
82      }
83  
84      /**
85       * Initialize a file name generator
86       */
87      protected void initFilenameGenerator() {
88          this.fileNameGenerator = new DateStampFilenameGenerator(StringUtils.isNotBlank(filterName) ? "." + filterName.toLowerCase() : "");
89      }
90  
91      /**
92       * Initialize the roll specific vars
93       */
94      protected void initRollConfig() {
95          this.maxRollFileSize = (int) this.filterConfig.findSizeEntry(MAX_ROLL_FILE_SIZE, maxRollFileSize);
96          this.maxOutputAppenders = this.filterConfig.findIntEntry(MAX_OUTPUT_APPENDERS, AgentPool.computePoolSize());
97          this.rollInterval = this.filterConfig.findLongEntry(CFG_ROLL_INTERVAL, rollInterval);
98          this.rollIntervalUnits = TimeUnit.valueOf(this.filterConfig.findStringEntry(ROLL_INTERVAL_UNIT, rollIntervalUnits.toString()));
99      }
100 
101     /**
102      * Create the local output directories
103      */
104     @SuppressWarnings("SystemExitOutsideMain")
105     protected void setupLocalOutputDir() {
106         if (!Files.exists(this.outputPath)) {
107             logger.info("Attempting to create {} output directory, {}", getFilterName(), this.outputPath);
108             try {
109                 Files.createDirectories(this.outputPath);
110             } catch (IOException e) {
111                 logger.error("Unable to create directory for {} output, exiting immediately.", getFilterName(), e);
112                 System.exit(1);
113             }
114         }
115     }
116 
117     /**
118      * Create the {@link JournaledCoalescer} and {@link Roller}
119      */
120     @SuppressWarnings("SystemExitOutsideMain")
121     protected void setupRoller() {
122         try {
123             this.rollable = createRollable();
124             this.roller = createRoller();
125             manageRoller();
126             logger.info("Added Roller for {} running every {} {}(s) or on size {} (bytes).", getFilterName(), this.rollInterval,
127                     this.rollIntervalUnits, this.maxRollFileSize);
128         } catch (Exception ex) {
129             if (ex instanceof InterruptedException) {
130                 Thread.currentThread().interrupt();
131             }
132             logger.error("Unable to instantiate Roller for handling {} file output", getFilterName(), ex);
133             System.exit(1);
134         }
135     }
136 
137     /**
138      * Create the rollable resource
139      *
140      * @return the specific journaled coalescer for the filter
141      * @throws IOException if there is an issue with the output path
142      * @throws InterruptedException if the journal is interrupted
143      */
144     protected IJournaler createRollable() throws IOException, InterruptedException {
145         return new JournaledCoalescer(this.outputPath, this.fileNameGenerator, this.maxOutputAppenders);
146     }
147 
148     /**
149      * Create the object to manage the state of the roll
150      *
151      * @return the roller object
152      */
153     protected Roller createRoller() {
154         return new Roller(this.maxRollFileSize, this.rollIntervalUnits, this.rollInterval, this.rollable);
155     }
156 
157     /**
158      * Add the roller to the roll manager
159      */
160     protected void manageRoller() {
161         RollManager.getManager().addRoller(this.roller);
162     }

174     @Override
175     public int filter(final List<IBaseDataObject> payloadList, final Map<String, Object> params) {
176         int code;
177         try (KeyedOutput ko = this.rollable.getOutput()) {
178             params.put("CONTENT_URI_" + getFilterName(), "file://" + ko.getFinalDestination().toString());
179             params.put("CONTENT_FORMAT_" + getFilterName(), getFilterName());
180             code = filter(payloadList, params, ko);
181             if (code == STATUS_SUCCESS) {
182                 ko.commit();
183             }
184         } catch (IOException e) {
185             logger.error("IOException during dropoff.", e);
186             code = STATUS_FAILURE;
187         }
188         return code;
189     }
190 
191     @Override
192     public int filter(final List<IBaseDataObject> list, final Map<String, Object> params, final OutputStream output) {
193 
194         // We subtract 1 from the list because the first element is currently assumed to be the TLD
195         list.get(0).putParameter("DESCENDANT_COUNT", list.size() - 1);
196 
197         try {
198             output.write(convert(list, params));
199             if (appendNewLine) {
200                 output.write("\n".getBytes());
201             }
202         } catch (IOException iox) {
203             logger.warn("Could not write to log filter", iox);
204             return STATUS_FAILURE;
205         }
206         return STATUS_SUCCESS;
207     }

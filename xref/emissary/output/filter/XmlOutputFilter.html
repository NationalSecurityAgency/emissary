15  /**
16   * Filter that writes unadorned data as utf-8.
17   */
18  public class XmlOutputFilter extends AbstractRollableFilter {
19  
20      /**
21       * Initialize reads the configuration items for this filter
22       * 
23       * @param configG the configurator to read from
24       * @param filterName the configured name of this filter or null for the default
25       * @param filterConfig the configuration for the specific filter
26       */
27      @Override
28      public void initialize(Configurator configG, @Nullable String filterName, Configurator filterConfig) {
29          if (filterName == null) {
30              setFilterName("XML");
31          }
32          super.initialize(configG, filterName, filterConfig);
33          this.appendNewLine = false;
34      }
35  
36      @Override
37      public byte[] convert(final List<IBaseDataObject> list, final Map<String, Object> params) throws IOException {
38          return PayloadUtil.toXmlString(list).getBytes();
39      }
40  
41      /**
42       * Main to test output types
43       */
44      @SuppressWarnings("SystemOut")
45      public static void main(String[] args) {
46          String name = args.length > 0 ? args[0] : null;
47  
48          XmlOutputFilter filter = new XmlOutputFilter();
49          try {
50              Configurator config = ConfigUtil.getConfigInfo(DropOffPlace.class);
51              filter.initialize(config, name);
52              System.out.println("Output types " + filter.outputTypes);
53          } catch (Exception ex) {
54              System.err.println("Cannot configure filter: " + ex.getMessage());
55          }
56      }

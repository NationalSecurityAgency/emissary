46  public class JsonOutputFilter extends AbstractRollableFilter {
47  
48      protected Set<String> denylistFields = new TreeSet<>();
49      protected Set<String> denylistPrefixes = new TreeSet<>();
50      protected Set<String> allowlistFields = new TreeSet<>();
51      protected Set<String> allowlistPrefixes = new TreeSet<>();
52      protected Map<String, Set<String>> denylistValues;
53      protected Set<String> stripPrefixes;

55      protected boolean emitPayload = true;

59      @Override
60      public void initialize(final Configurator theConfigG, @Nullable Configurator>final String filterName, final Configurator theFilterConfig) {
61          if (filterName == null) {
62              setFilterName("JSON");
63          }
64          super.initialize(theConfigG, filterName, theFilterConfig);
65          this.allowlistFields.addAll(this.filterConfig.findEntries("EXTRA_PARAM"));
66          this.allowlistPrefixes.addAll(this.filterConfig.findEntries("EXTRA_PREFIX"));
67          this.denylistFields.addAll(this.filterConfig.findEntries("DENYLIST_FIELD"));
68          this.denylistPrefixes.addAll(this.filterConfig.findEntries("DENYLIST_PREFIX"));
69          this.denylistValues = this.filterConfig.findStringMatchMultiMap("DENYLIST_VALUE_");
70          this.stripPrefixes = this.filterConfig.findEntriesAsSet("STRIP_PARAM_PREFIX");
71          this.emitPayload = this.filterConfig.findBooleanEntry("EMIT_PAYLOAD", true);
72          initFilenameGenerator();
73          initJsonMapper();
74      }
75  
76      @Override
77      protected void initFilenameGenerator() {
78          this.fileNameGenerator = new DateFilterFilenameGenerator("json");
79      }
80  
81      /**
82       * Initialize the Jackson json object mapper
83       */
84      protected void initJsonMapper() {
85          jsonMapper = new ObjectMapper();
86          jsonMapper.registerModule(new IbdoModule());
87          jsonMapper.registerModule(new JavaTimeModule());
88          jsonMapper.addMixIn(IBaseDataObject.class, emitPayload ? IbdoPayloadMixin.class : IbdoParameterMixin.class);
89          // the id in addFilter must match the annotation for JsonFilter
90          jsonMapper.setFilterProvider(new SimpleFilterProvider().addFilter("param_filter", new IbdoParameterFilter()));
91      }
92  
93      @Override
94      public byte[] convert(final List<IBaseDataObject> list, final Map<String, Object> params) throws IOException {
95          return jsonMapper.writeValueAsBytes(list);
96      }

98      class IbdoParameterFilter extends SimpleBeanPropertyFilter {
99  
100         protected final boolean outputAll;
101         protected final boolean emptyDenylist;
102         protected final boolean denylistStar;
103         protected final boolean emptyAllowlist;
104         protected final boolean allowlistStar;
105         private char keyReplacement = '_';
106 
107         public IbdoParameterFilter() {
108             // if all collections are empty, then output everything
109             this.allowlistStar = (allowlistFields.contains("*") || allowlistFields.contains("ALL"));
110             this.denylistStar = (denylistFields.contains("*") || denylistFields.contains("ALL"));
111             this.emptyDenylist = CollectionUtils.isEmpty(denylistFields) && CollectionUtils.isEmpty(denylistPrefixes);
112             this.emptyAllowlist = CollectionUtils.isEmpty(allowlistFields) && CollectionUtils.isEmpty(allowlistPrefixes);
113             this.outputAll = emptyDenylist && (allowlistStar || emptyAllowlist);
114         }
115 
116         @Override
117         public void serializeAsField(Object pojo, JsonGenerator jgen, SerializerProvider provider, PropertyWriter writer) throws Exception {
118 
119             String key = writer.getName();
120             @SuppressWarnings("unchecked")
121             Collection<Object> values = (Collection<Object>) ((Map<?, ?>) pojo).get(key);
122 
123             if (includeParameter(key)) {
124                 Collection<Object> write = filter(key, values);
125                 if (CollectionUtils.isNotEmpty(write)) {
126                     // customize the key
127                     jgen.writeFieldName(transform(key));
128 
129                     // only write the element
130                     ((MapProperty) writer).setValue(write);
131                     writer.serializeAsElement(write, jgen, provider);
132                 }
133             }
134         }
135 
136         protected boolean includeParameter(String key) {
137             if (outputAll) {
138                 return true;
139             }
140 
141             // check the allow/deny list first
142             if (denylistFields.contains(key)) {
143                 return false;
144             } else if (allowlistFields.contains(key)) {
145                 return true;
146             }
147 
148             // see if there is a hit on the denylist prefix
149             for (final String prefix : denylistPrefixes) {
150                 if (key.startsWith(prefix)) {
151                     return false;
152                 }
153             }
154 
155             // omit/emit all parameters if '*' or 'ALL'
156             if (denylistStar) {
157                 return false;
158             } else if (allowlistStar) {
159                 return true;
160             }
161 
162             // there is a hit on the allow-list prefix, but it is on the deny-list
163             for (final String prefix : allowlistPrefixes) {
164                 if (key.startsWith(prefix)) {
165                     return true;
166                 }
167             }
168 
169             // if we were only given an deny-list, output all keys
170             return emptyAllowlist;
171         }
172 
173         protected Collection<Object> filter(String key, Collection<Object> values) {
174             Set<Object> keep = new TreeSet<>();
175             for (final Object value : values) {
176                 if (!(denylistValues.containsKey(key) && denylistValues.get(key).contains(value.toString()))) {
177                     keep.add(value);
178                 }
179             }
180             return keep;
181         }
182 
183         protected String transform(String name) {
184             return normalize(strip(name.toUpperCase()));
185         }
186 
187         protected String strip(String name) {
188             for (final String prefix : stripPrefixes) {
189                 if (name.startsWith(prefix)) {
190                     return name.substring(prefix.length());
191                 }
192             }
193             return name;
194         }
195 
196         protected String normalize(String name) {
197             boolean changed = false;
198             char[] ch = name.toCharArray();
199             for (int i = 0; i < ch.length; i++) {
200                 if (!Character.isLetterOrDigit(ch[i]) && Character.compare(ch[i], '_') != 0 && Character.compare(ch[i], '.') != 0) {
201                     ch[i] = keyReplacement;
202                     changed = true;
203                 }
204             }
205             if (changed) {
206                 return new String(ch);
207             }
208 
209             return name;
210         }
211     }

224     /**
225      * Add some fields to the ibdo before output This is only needed if custom fields need to be written for the ibdo
226      */
227     class IbdoSerializer extends JsonSerializer<IBaseDataObject> {
228 
229         @Override
230         public void serialize(IBaseDataObject ibdo, JsonGenerator jgen, SerializerProvider provider) throws IOException {
231             jgen.writeStartObject();
232             JavaType javaType = provider.constructType(IBaseDataObject.class);
233             BeanDescription beanDesc = provider.getConfig().introspect(javaType);
234             JsonSerializer<Object> serializer = BeanSerializerFactory.instance.findBeanOrAddOnSerializer(provider, javaType, beanDesc,
235                     provider.isEnabled(MapperFeature.USE_STATIC_TYPING));
236 
237             // add some custom fields here
238             jgen.writeObjectField("id", dropOffUtil.getBestIdFrom(ibdo));
239             jgen.writeObjectField("processedTimestamp", TimeUtil.getCurrentDateFullISO8601());
240 
241             serializer.unwrappingSerializer(null).serialize(ibdo, jgen, provider);
242             jgen.writeEndObject();
243         }
244     }

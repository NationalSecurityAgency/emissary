83       * Setup configuration items we need and build the output filter
84       */
85      protected void configurePlace() {
86          // Set configuration info on file paths
87          this.dropOffUtil = new DropOffUtil(configG);
88          this.doSynchronized = configG.findBooleanEntry("SYNCHRONIZED_PROCESS", false);
89          this.failurePolicyTerminate = configG.findBooleanEntry("FAILURE_TERMINATES_CHAIN", true);
90          this.outputCompletionPayloadSize = configG.findBooleanEntry("OUTPUT_COMPLETION_PAYLOAD_SIZE", false);
91          // Build and store all the filter that are desired IN THE ORDER SPECIFIED
92          final List<String> filterClasses = configG.findEntries("OUTPUT_FILTER");
93          initializeFilters(filterClasses);
94  
95      }

97      /**
98       * Start up the requested filter
99       * 
100      * @param filterClasses the name:class values of the configured filter for this drop off
101      */
102     protected void initializeFilters(final List<String> filterClasses) {
103         for (final String entry : filterClasses) {
104             final String name;
105             final String clazz;
106             Configurator filterConfig = null;
107             final int colpos = entry.indexOf(':');
108             if (colpos > -1) {
109                 name = entry.substring(0, colpos);
110                 clazz = entry.substring(colpos + 1);
111                 final String filterConfigName = configG.findStringEntry(name);
112                 if (filterConfigName != null) {
113                     try {
114                         filterConfig = ConfigUtil.getConfigInfo(filterConfigName);
115                     } catch (IOException configError) {
116                         logger.warn("Specified filter configuration {} cannot be loaded", filterConfigName);
117                         continue;
118                     }
119                 }
120             } else {
121                 name = null;
122                 clazz = entry;
123             }
124 
125             try {
126                 final Object filter = emissary.core.Factory.create(clazz);
127                 if (filter != null && filter instanceof IDropOffFilter) {
128                     final IDropOffFilter f = (IDropOffFilter) filter;
129                     f.initialize(configG, name, filterConfig);
130                     addFilter(f);
131                 } else {
132                     logger.error("Misconfigured filter {} is not an IDropOffFilter instance, ignoring it", clazz);
133                 }
134             } catch (RuntimeException ex) {
135                 logger.error("Unable to create or initialize {}", clazz, ex);
136             }
137         }
138 
139         // Collect the set of content types to elide
140         this.elideContentForms = configG.findEntriesAsSet("ELIDE_CONTENT");
141 
142         // collect the set of no-nuke forms
143         this.noNukeForms = configG.findEntriesAsSet("NO_NUKE_FORM");
144 
145         if (logger.isInfoEnabled()) {
146             logger.debug("Setting ELIDE_CONTENT forms to " + this.elideContentForms);
147             final StringBuilder sb = new StringBuilder("Output Filters:");
148             if (this.outputFilters.size() > 0) {
149                 for (final IDropOffFilter f : this.outputFilters) {
150                     sb.append(" ").append(f.getFilterName()).append("(").append(f.getClass().getName()).append(")");
151                 }
152             } else {
153                 sb.append(" NONE!");
154             }
155             logger.info(sb.toString());
156         }
157 
158         if (logger.isDebugEnabled()) {
159             final IBaseDataObject fakePayload = DataObjectFactory.getInstance(new byte[0], "fakename", Form.UNKNOWN);
160             for (final IDropOffFilter filter : getFilters()) {
161                 final String name = filter.getFilterName();
162                 final String spec = filter.getOutputSpec();
163 
164                 logger.debug("Adding filter={}, spec={}, sample={}, class={}", name, spec, this.dropOffUtil.getPathFromSpec(spec, fakePayload),
165                         filter.getClass().getSimpleName());
166             }
167         }
168     }

180     /**
181      * "HD" agent calls this method when visiting the place. If you use {@link emissary.core.MobileAgent} this method is
182      * never called. This method overrides {@link ServiceProviderPlace} and allows this processing place to have access to
183      * all payloads wanting to be dropped off in a single list.
184      * 
185      * @param payloadList list of IBaseDataObject from an {@link emissary.core.HDMobileAgent}
186      */
187     @Override
188     public List<IBaseDataObject> agentProcessHeavyDuty(final List<IBaseDataObject> payloadList) throws Exception {
189 
190         logger.debug("Entering DropOffPlace.agentProcessHeavyDuty with {} payload items", payloadList.size());
191 
192         // Prepare each incoming payload object
193         for (final IBaseDataObject d : payloadList) {
194             try {
195                 // checking to see if any object in the tree is marked as not outputable
196                 if (!d.isOutputable()) {
197                     logger.info("Skipping object since it is not able to be output ID:{}", this.dropOffUtil.getBestId(d, payloadList.get(0)));
198                     return Collections.emptyList();
199                 }
200 
201                 // Process the payload item with HDcontext=true
202                 processData(d, true);
203             } catch (RuntimeException e) {
204                 logger.error("Place.process threw:", e);
205                 d.addProcessingError("agentProcessHD(" + myKey + "): " + e);
206 
207                 if (!d.currentForm().equals(Form.ERROR)) {
208                     d.pushCurrentForm(Form.ERROR);
209                 }
210             }
211         }
212 
213         // Prepare the data and metadata for filter output
214         final Map<String, Object> filterParams = new HashMap<>();
215         preFilterHook(payloadList, filterParams);
216 
217         // Run the filter on the output, indicating that
218         // the records are pre-sorted, if the filter cares
219         runOutputFilters(payloadList, filterParams);
220 
221         // Any cleanup operations needed
222         postFilterHook(payloadList, filterParams);
223 
224         if (!payloadList.isEmpty()) {
225             // Should have been sorted by the prefilter hook
226 
227             // Just report the TLD object ID
228             final IBaseDataObject tld = payloadList.get(0);
229 
230 
231             if (outputCompletionPayloadSize && tld.hasContent()) {
232                 logger.info(
233                         "Finished DropOff for object {}, with external id: {}, with total processing time: {}ms, with filetype: {}, payload size: {} bytes",
234                         tld.getInternalId(), this.dropOffUtil.getBestId(tld, tld),
235                         Duration.between(tld.getCreationTimestamp(), Instant.now()).toMillis(),
236                         tld.getFileType(), tld.getChannelSize());
237             } else {
238                 logger.info("Finished DropOff for object {}, with external id: {}, with total processing time: {}ms, with filetype: {}",
239                         tld.getInternalId(), this.dropOffUtil.getBestId(tld, tld),
240                         Duration.between(tld.getCreationTimestamp(), Instant.now()).toMillis(),
241                         tld.getFileType());
242             }
243 
244         }
245 
246         // Execute 'Dispose Runnables' to tidy up resources used with SeekableByteChannelFactory implementations
247         DisposeHelper.execute(payloadList);
248 
249         // This place does not sprout, return an empty list
250         return Collections.emptyList();
251     }

253     /**
254      * Called by MobileAgent through ServiceProviderPlace to handle a single payload
255      * 
256      * @param tData the payload to work on
257      */
258     @Override
259     public void process(final IBaseDataObject tData) {
260         if (DataUtil.isEmpty(tData)) {
261             logger.warn("null/empty data object");
262             return;
263         }
264 
265         // checking to see if the object is marked as not outputable
266         if (!tData.isOutputable()) {
267             logger.warn("Skipping object since it is not able to be output ID:{}", this.dropOffUtil.getBestId(tData, tData));
268             return;
269         }
270 
271         // synchronization can be set by config file entry
272         if (this.doSynchronized) {
273             synchronized (this) {
274                 processData(tData, false);
275             }
276         } else {
277             processData(tData, false);
278         }
279 
280         // Execute 'Dispose Runnables' to tidy up resources used with SeekableByteChannelFactory implementations
281         DisposeHelper.execute(tData);
282     }

284     /**
285      * Prepare a list of payload object to be filtered
286      * 
287      * @param payloadList the list of items that were eligible for output
288      * @param filterParams metadata needed for the output filter
289      */
290     public void preFilterHook(final List<IBaseDataObject> payloadList, final Map<String, Object> filterParams) {
291         // Sort the list of records
292         Collections.sort(payloadList, new ShortNameComparator());
293         filterParams.put(IDropOffFilter.PRE_SORTED, Boolean.TRUE);
294         filterParams.put(IDropOffFilter.TLD_PARAM, payloadList.get(0));
295 
296         // Prepare the metadata
297         this.dropOffUtil.processMetadata(payloadList);
298     }

300     /**
301      * Clean up after all filter are done
302      * 
303      * @param payloadList the list of items that were eligible for output
304      * @param filterParams metadata needed for the output filter
305      */
306     public void postFilterHook(final List<IBaseDataObject> payloadList, final Map<String, Object> filterParams) {
307         // remove the current forms we used or could have used
308         for (final IBaseDataObject dataObject : payloadList) {
309             // Save off no-nuke forms
310             final List<String> saveForms = new ArrayList<>();
311             for (final String nnf : this.noNukeForms) {
312                 if (dataObject.searchCurrentForm(nnf) > -1) {
313                     saveForms.add(nnf);
314                 }
315             }
316             // nuke 'em
317             this.nukeMyProxies(dataObject);
318 
319             // Restore the no-nukes
320             for (final String sf : saveForms) {
321                 dataObject.pushCurrentForm(sf);
322             }
323         }
324 
325     }

327     /**
328      * Internal method to process a single data object
329      * 
330      * @param tData the payload to work on or prepare
331      * @param haveList true if in HD context
332      */
333     protected void processData(final IBaseDataObject tData, final boolean haveList) {
334 
335         logger.debug("DropOff is working on {}, current form is {}", tData.shortName(), tData.getAllCurrentForms());
336 
337         final StringBuilder poppedForms = new StringBuilder();
338 
339         String prevBin = "";
340 
341         // skip the I/O for some types for all filter
342         for (int i = 0; i < tData.currentFormSize(); i++) {
343             final String cf = tData.currentFormAt(i);
344             if (this.elideContentForms.contains(cf)) {
345                 tData.setData(("[[ " + tData.getAllCurrentForms() + " content elided in DropOffPlace. ]]").getBytes());
346             }
347         }
348 
349         // Write out data for all the destinations we area proxy for, popping
350         // them off the stack as they are handled.
351 
352         final Set<String> serviceProxies = getProxies();
353         final Set<String> cfSet = new HashSet<>();
354         for (int i = 0; i < tData.currentFormSize(); i++) {
355             final String cf = tData.currentFormAt(i);
356 
357             if (serviceProxies.contains(cf) || serviceProxies.contains("*")) {
358                 // Record extra drop offs in the transform history since
359                 // we are taking precedence over the normal agent/directory mechanism
360                 // Do this before popping the extra destination so that it
361                 // is not just lost forever. Since the agent appends
362                 // one entry to the history when it sends the agent here,
363                 // that's the top current form and
364                 // we don't want to duplicate that one
365 
366                 if (!prevBin.equals(cf) && (i > 0) && !cfSet.contains(cf) && !("UNKNOWN".equals(cf) || cf.endsWith("-PROCESSED"))) {
367                     // e.g.: [dataType].DROP_OFF.IO.host.dom:port/DropOffPlace
368                     final DirectoryEntry de = getDirectoryEntry();
369                     de.setDataType("[" + cf + "]");
370                     tData.appendTransformHistory(de.getKey());
371                 }
372 
373                 // Accumulate forms we have handled in poppedForms
374                 if (poppedForms.length() > 0) {
375                     poppedForms.append(" ");
376                 }
377                 poppedForms.append(cf);
378                 cfSet.add(cf);
379 
380                 prevBin = cf;
381             }
382         }
383 
384         // Record the list of forms
385         tData.setParameter("POPPED_FORMS", poppedForms.toString());
386 
387         // Do the output filtering now if we aren't in HD mode
388         if (!haveList) {
389 
390             // There currently are no extra params needed
391             final Map<String, Object> filterParams = new HashMap<>();
392 
393             runOutputFilters(tData, filterParams);
394 
395             // Actually remove the current forms we used or could have used
396             this.nukeMyProxies(tData);
397 
398             logger.debug("DropOff finished with {}", tData.shortName());
399         }
400     }

402     /**
403      * Run all the output filter
404      * 
405      * @param target either IBaseDataObject or List thereof
406      * @param filterParams other parameters that filter need
407      */
408     @SuppressWarnings("unchecked")
409     protected void runOutputFilters(final Object target, final Map<String, Object> filterParams) {
410 
411         IBaseDataObject doTarget = null;
412         List<IBaseDataObject> listTarget = null;
413         if (target instanceof IBaseDataObject) {
414             doTarget = (IBaseDataObject) target;
415         } else if (target instanceof List) {
416             listTarget = (List<IBaseDataObject>) target;
417         } else {
418             logger.error("Cannot run filter on {}", target.getClass().getName());
419             return;
420         }
421 
422         // Write output onto each of the filter that have been
423         // configured, as long as they work
424         for (final IDropOffFilter filter : this.outputFilters) {
425             final long start = System.currentTimeMillis();
426 
427             // call the filter to output its data
428             int filterStatus = IDropOffFilter.STATUS_FAILURE;
429             try {
430                 if (listTarget != null && filter.isOutputtable(listTarget)) {
431                     filterStatus = filter.filter(listTarget, filterParams);
432                 } else if (doTarget != null && filter.isOutputtable(doTarget)) {
433                     filterStatus = filter.filter(doTarget, filterParams);
434                 } else {
435                     logger.debug("Filter {} not Outputtable for {}", filter.getFilterName(), listTarget != null ? "list" : "single payload");
436                     filterStatus = IDropOffFilter.STATUS_SUCCESS;
437                 }
438                 logger.debug("Filter {} took {}s - {}", filter.getFilterName(), (System.currentTimeMillis() - start) / 1000.0, filterStatus);
439             } catch (RuntimeException e) {
440                 logger.error("Filter {} failed", filter.getFilterName(), e);
441             }
442 
443             if ((filterStatus != IDropOffFilter.STATUS_SUCCESS) && this.failurePolicyTerminate) {
444                 logger.error("DropOff Filter chain terminated at {} due to error return status", filter.getFilterName());
445                 break;
446             }
447         }
448     }

94      protected void configurePlace() {
95          // Set configuration info on file paths
96          this.dropOffUtil = new DropOffUtil(configG);
97          this.doSynchronized = configG.findBooleanEntry("SYNCHRONIZED_PROCESS", false);
98          this.failurePolicyTerminate = configG.findBooleanEntry("FAILURE_TERMINATES_CHAIN", true);
99          this.outputObjectMetrics = configG.findBooleanEntry("OUTPUT_OBJECT_METRICS", false);
100         this.objectMetricsFields = configG.findEntries("OBJECT_METRICS_FIELDS");
101         this.outputObjectMetricsLatency = configG.findStringEntry("OBJECT_METRICS_LATENCY", "");
102         String outputTldMetricsLatencyFormat = configG.findStringEntry("OBJECT_METRICS_LATENCY_FORMAT", "yyyy-MM-dd'T'HH:mm:ss.SSSXXX");
103         this.latencyDateTimeFormatter = DateTimeFormatter.ofPattern(outputTldMetricsLatencyFormat);
104         this.outputCompletionPayloadSize = configG.findBooleanEntry("OUTPUT_COMPLETION_PAYLOAD_SIZE", false);
105         // Build and store all the filter that are desired IN THE ORDER SPECIFIED
106         final List<String> filterClasses = configG.findEntries("OUTPUT_FILTER");
107         initializeFilters(filterClasses);
108 
109         objectMetricsLog = LoggerFactory.getLogger("objectMetrics");
110     }

112     /**
113      * Start up the requested filter
114      * 
115      * @param filterClasses the name:class values of the configured filter for this drop off
116      */
117     protected void initializeFilters(final List<String> filterClasses) {
118         for (final String entry : filterClasses) {
119             final String name;
120             final String clazz;
121             Configurator filterConfig = null;
122             final int colpos = entry.indexOf(':');
123             if (colpos > -1) {
124                 name = entry.substring(0, colpos);
125                 clazz = entry.substring(colpos + 1);
126                 final String filterConfigName = configG.findStringEntry(name);
127                 if (filterConfigName != null) {
128                     try {
129                         filterConfig = ConfigUtil.getConfigInfo(filterConfigName);
130                     } catch (IOException configError) {
131                         logger.warn("Specified filter configuration {} cannot be loaded", filterConfigName);
132                         continue;
133                     }
134                 }
135             } else {
136                 name = null;
137                 clazz = entry;
138             }
139 
140             try {
141                 final Object filter = emissary.core.Factory.create(clazz);
142                 if (filter != null && filter instanceof IDropOffFilter) {
143                     final IDropOffFilter../emissary/output/filter/IDropOffFilter.html#IDropOffFilter">IDropOffFilter f = (IDropOffFilter) filter;
144                     f.initialize(configG, name, filterConfig);
145                     addFilter(f);
146                 } else {
147                     logger.error("Misconfigured filter {} is not an IDropOffFilter instance, ignoring it", clazz);
148                 }
149             } catch (Exception ex) {
150                 logger.error("Unable to create or initialize {}", clazz, ex);
151             }
152         }
153 
154         // Collect the set of content types to elide
155         this.elideContentForms = configG.findEntriesAsSet("ELIDE_CONTENT");
156 
157         // collect the set of no-nuke forms
158         this.noNukeForms = configG.findEntriesAsSet("NO_NUKE_FORM");
159 
160         if (logger.isInfoEnabled()) {
161             logger.debug("Setting ELIDE_CONTENT forms to " + this.elideContentForms);
162             final StringBuilder sb = new StringBuilder("Output Filters:");
163             if (this.outputFilters.size() > 0) {
164                 for (final IDropOffFilter f : this.outputFilters) {
165                     sb.append(" ").append(f.getFilterName()).append("(").append(f.getClass().getName()).append(")");
166                 }
167             } else {
168                 sb.append(" NONE!");
169             }
170             logger.info(sb.toString());
171         }
172 
173         if (logger.isDebugEnabled()) {
174             final IBaseDataObject fakePayload = DataObjectFactory.getInstance(new byte[0], "fakename", Form.UNKNOWN);
175             for (final IDropOffFilter filter : getFilters()) {
176                 final String name = filter.getFilterName();
177                 final String spec = filter.getOutputSpec();
178 
179                 logger.debug("Adding filter={}, spec={}, sample={}, class={}", name, spec, this.dropOffUtil.getPathFromSpec(spec, fakePayload),
180                         filter.getClass().getSimpleName());
181             }
182         }
183     }

195     /**
196      * "HD" agent calls this method when visiting the place. If you use {@link emissary.core.MobileAgent} this method is
197      * never called. This method overrides {@link ServiceProviderPlace} and allows this processing place to have access to
198      * all payloads wanting to be dropped off in a single list.
199      * 
200      * @param payloadList list of IBaseDataObject from an {@link emissary.core.HDMobileAgent}
201      */
202     @Override
203     public List<IBaseDataObject> agentProcessHeavyDuty(final List<IBaseDataObject> payloadList) throws Exception {
204 
205         logger.debug("Entering DropOffPlace.agentProcessHeavyDuty with {} payload items", payloadList.size());
206 
207         // Prepare each incoming payload object
208         for (final IBaseDataObject d : payloadList) {
209             try {
210                 // checking to see if any object in the tree is marked as not outputable
211                 if (!d.isOutputable()) {
212                     logger.info("Skipping object since it is not able to be output ID:{}", this.dropOffUtil.getBestId(d, payloadList.get(0)));
213                     return Collections.emptyList();
214                 }
215 
216                 // Process the payload item with HDcontext=true
217                 processData(d, true);
218             } catch (Exception e) {
219                 logger.error("Place.process threw:", e);
220                 d.addProcessingError("agentProcessHD(" + myKey + "): " + e);
221 
222                 if (!d.currentForm().equals(Form.ERROR)) {
223                     d.pushCurrentForm(Form.ERROR);
224                 }
225             }
226         }
227 
228         // Prepare the data and metadata for filter output
229         final Map<String, Object> filterParams = new HashMap<>();
230         preFilterHook(payloadList, filterParams);
231 
232         // Run the filter on the output, indicating that
233         // the records are pre-sorted, if the filter cares
234         runOutputFilters(payloadList, filterParams);
235 
236         // Any cleanup operations needed
237         postFilterHook(payloadList, filterParams);
238 
239         if (!payloadList.isEmpty()) {
240             // Should have been sorted by the prefilter hook
241 
242             // Just report the TLD object ID
243             final IBaseDataObject tld = payloadList.get(0);
244 
245             if (outputObjectMetrics) {
246                 objectMetricsLog.info(appendEntries(outputObjectMetrics(tld, objectMetricsFields)), "Finished DropOff");
247             }
248 
249             if (outputCompletionPayloadSize && tld.data() != null) {
250                 logger.info(
251                         "Finished DropOff for object {}, with external id: {}, with total processing time: {}ms, with filetype: {}, payload size: {} bytes",
252                         tld.getInternalId(), this.dropOffUtil.getBestId(tld, tld), (new Date().getTime() - tld.getCreationTimestamp().getTime()),
253                         tld.getFileType(), tld.dataLength());
254             } else {
255                 logger.info("Finished DropOff for object {}, with external id: {}, with total processing time: {}ms, with filetype: {}",
256                         tld.getInternalId(), this.dropOffUtil.getBestId(tld, tld), (new Date().getTime() - tld.getCreationTimestamp().getTime()),
257                         tld.getFileType());
258             }
259 
260         }
261 
262         // Execute 'Dispose Runnables' to tidy up resources used with SeekableByteChannelFactory implementations
263         DisposeHelper.execute(payloadList);
264 
265         // This place does not sprout, return an empty list
266         return Collections.emptyList();
267     }

269     /**
270      * Called by MobileAgent through ServiceProviderPlace to handle a single payload
271      * 
272      * @param tData the payload to work on
273      */
274     @Override
275     public void process(final IBaseDataObject tData) {
276         if (DataUtil.isEmpty(tData)) {
277             logger.warn("null/empty data object");
278             return;
279         }
280 
281         // checking to see if the object is marked as not outputable
282         if (!tData.isOutputable()) {
283             logger.warn("Skipping object since it is not able to be output ID:{}", this.dropOffUtil.getBestId(tData, tData));
284             return;
285         }
286 
287         // synchronization can be set by config file entry
288         if (this.doSynchronized) {
289             synchronized (this) {
290                 processData(tData, false);
291             }
292         } else {
293             processData(tData, false);
294         }
295 
296         // Execute 'Dispose Runnables' to tidy up resources used with SeekableByteChannelFactory implementations
297         DisposeHelper.execute(tData);
298     }

300     /**
301      * Prepare a list of payload object to be filtered
302      * 
303      * @param payloadList the list of items that were eligible for output
304      * @param filterParams metadata needed for the output filter
305      */
306     public void preFilterHook(final List<IBaseDataObject> payloadList, final Map<String, Object> filterParams) {
307         // Sort the list of records
308         Collections.sort(payloadList, new ShortNameComparator());
309         filterParams.put(IDropOffFilter.PRE_SORTED, Boolean.TRUE);
310         filterParams.put(IDropOffFilter.TLD_PARAM, payloadList.get(0));
311 
312         // Prepare the metadata
313         this.dropOffUtil.processMetadata(payloadList);
314     }

316     /**
317      * Clean up after all filter are done
318      * 
319      * @param payloadList the list of items that were eligible for output
320      * @param filterParams metadata needed for the output filter
321      */
322     public void postFilterHook(final List<IBaseDataObject> payloadList, final Map<String, Object> filterParams) {
323         // remove the current forms we used or could have used
324         for (final IBaseDataObject dataObject : payloadList) {
325             // Save off no-nuke forms
326             final List<String> saveForms = new ArrayList<>();
327             for (final String nnf : this.noNukeForms) {
328                 if (dataObject.searchCurrentForm(nnf) > -1) {
329                     saveForms.add(nnf);
330                 }
331             }
332             // nuke 'em
333             this.nukeMyProxies(dataObject);
334 
335             // Restore the no-nukes
336             for (final String sf : saveForms) {
337                 dataObject.pushCurrentForm(sf);
338             }
339         }
340 
341     }

343     /**
344      * Internal method to process a single data object
345      * 
346      * @param tData the payload to work on or prepare
347      * @param haveList true if in HD context
348      */
349     protected void processData(final IBaseDataObject tData, final boolean haveList) {
350 
351         logger.debug("DropOff is working on {}, current form is {}", tData.shortName(), tData.getAllCurrentForms());
352 
353         final StringBuilder poppedForms = new StringBuilder();
354 
355         String prevBin = "";
356 
357         // skip the I/O for some types for all filter
358         for (int i = 0; i < tData.currentFormSize(); i++) {
359             final String cf = tData.currentFormAt(i);
360             if (this.elideContentForms.contains(cf)) {
361                 tData.setData(("[[ " + tData.getAllCurrentForms() + " content elided in DropOffPlace. ]]").getBytes());
362             }
363         }
364 
365         // Write out data for all the destinations we area proxy for, popping
366         // them off the stack as they are handled.
367 
368         final Set<String> serviceProxies = getProxies();
369         final Set<String> cfSet = new HashSet<>();
370         for (int i = 0; i < tData.currentFormSize(); i++) {
371             final String cf = tData.currentFormAt(i);
372 
373             if (serviceProxies.contains(cf) || serviceProxies.contains("*")) {
374                 // Record extra drop offs in the transform history since
375                 // we are taking precedence over the normal agent/directory mechanism
376                 // Do this before popping the extra destination so that it
377                 // is not just lost forever. Since the agent appends
378                 // one entry to the history when it sends the agent here,
379                 // that's the top current form and
380                 // we don't want to duplicate that one
381 
382                 if (!prevBin.equals(cf) && (i > 0) && !cfSet.contains(cf) && !("UNKNOWN".equals(cf) || cf.endsWith("-PROCESSED"))) {
383                     // e.g.: [dataType].DROP_OFF.IO.host.dom:port/DropOffPlace
384                     final DirectoryEntry de = getDirectoryEntry();
385                     de.setDataType("[" + cf + "]");
386                     tData.appendTransformHistory(de.getKey());
387                 }
388 
389                 // Accumulate forms we have handled in poppedForms
390                 if (poppedForms.length() > 0) {
391                     poppedForms.append(" ");
392                 }
393                 poppedForms.append(cf);
394                 cfSet.add(cf);
395 
396                 prevBin = cf;
397             }
398         }
399 
400         // Record the list of forms
401         tData.setParameter("POPPED_FORMS", poppedForms.toString());
402 
403         // Do the output filtering now if we aren't in HD mode
404         if (!haveList) {
405 
406             // There currently are no extra params needed
407             final Map<String, Object> filterParams = new HashMap<>();
408 
409             runOutputFilters(tData, filterParams);
410 
411             // Actually remove the current forms we used or could have used
412             this.nukeMyProxies(tData);
413 
414             logger.debug("DropOff finished with {}", tData.shortName());
415         }
416     }

418     /**
419      * Run all the output filter
420      * 
421      * @param target either IBaseDataObject or List thereof
422      * @param filterParams other parameters that filter need
423      */
424     @SuppressWarnings("unchecked")
425     protected void runOutputFilters(final Object target, final Map<String, Object> filterParams) {
426 
427         IBaseDataObject doTarget = null;
428         List<IBaseDataObject> listTarget = null;
429         if (target instanceof IBaseDataObject) {
430             doTarget = (IBaseDataObject) target;
431         } else if (target instanceof List) {
432             listTarget = (List<IBaseDataObject>) target;
433         } else {
434             logger.error("Cannot run filter on {}", target.getClass().getName());
435             return;
436         }
437 
438         // Write output onto each of the filter that have been
439         // configured, as long as they work
440         for (final IDropOffFilter filter : this.outputFilters) {
441             final long start = System.currentTimeMillis();
442 
443             // call the filter to output its data
444             int filterStatus = IDropOffFilter.STATUS_FAILURE;
445             try {
446                 if (listTarget != null && filter.isOutputtable(listTarget)) {
447                     filterStatus = filter.filter(listTarget, filterParams);
448                 } else if (doTarget != null && filter.isOutputtable(doTarget)) {
449                     filterStatus = filter.filter(doTarget, filterParams);
450                 } else {
451                     logger.debug("Filter {} not Outputtable for {}", filter.getFilterName(), listTarget != null ? "list" : "single payload");
452                     filterStatus = IDropOffFilter.STATUS_SUCCESS;
453                 }
454                 logger.debug("Filter {} took {}s - {}", filter.getFilterName(), ((System.currentTimeMillis() - start) / 1000.0), filterStatus);
455             } catch (Exception e) {
456                 logger.error("Filter {} failed", filter.getFilterName(), e);
457             }
458 
459             if ((filterStatus != IDropOffFilter.STATUS_SUCCESS) && this.failurePolicyTerminate) {
460                 logger.error("DropOff Filter chain terminated at {} due to error return status", filter.getFilterName());
461                 break;
462             }
463         }
464     }

502     /**
503      * Create a one line json output of specified parameters
504      *
505      * @param ibdo IBaseDataObject for the Top Level Document
506      * @param outputParams List of parameters to output
507      */
508     protected Map<String, String> outputObjectMetrics(IBaseDataObject ibdo, List<String> outputParams) {
509         // Create output map
510         Map<String, String> jsonMap = new HashMap<>();
511         jsonMap.put("InternalId", ibdo.getInternalId().toString());
512 
513         long processingLatency;
514         if (outputObjectMetricsLatency.isEmpty() || !ibdo.hasParameter(outputObjectMetricsLatency)) {
515             processingLatency = Instant.now().toEpochMilli() - ibdo.getCreationTimestamp().getTime();
516         } else {
517             processingLatency = Instant.now().toEpochMilli()
518                     - Instant.from(latencyDateTimeFormatter.parse(ibdo.getStringParameter(outputObjectMetricsLatency))).toEpochMilli();
519         }
520         jsonMap.put("ProcessingLatency", String.valueOf(processingLatency));
521 
522         // Add other specified Parameters
523         for (String param : outputParams) {
524             if (ibdo.hasParameter(param)) {
525                 jsonMap.put(param, ibdo.getStringParameter(param));
526             }
527         }
528         return jsonMap;
529     }

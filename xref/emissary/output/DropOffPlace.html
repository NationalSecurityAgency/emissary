84      protected void configurePlace() {
85          // Set configuration info on file paths
86          this.dropOffUtil = new DropOffUtil(configG);
87          this.doSynchronized = configG.findBooleanEntry("SYNCHRONIZED_PROCESS", false);
88          this.failurePolicyTerminate = configG.findBooleanEntry("FAILURE_TERMINATES_CHAIN", true);
89          this.outputCompletionPayloadSize = configG.findBooleanEntry("OUTPUT_COMPLETION_PAYLOAD_SIZE", false);
90          // Build and store all the filter that are desired IN THE ORDER SPECIFIED
91          final List<String> filterClasses = configG.findEntries("OUTPUT_FILTER");
92          initializeFilters(filterClasses);
93  
94      }

96      /**
97       * Start up the requested filter
98       * 
99       * @param filterClasses the name:class values of the configured filter for this drop off
100      */
101     protected void initializeFilters(final List<String> filterClasses) {
102         for (final String entry : filterClasses) {
103             final String name;
104             final String clazz;
105             Configurator filterConfig = null;
106             final int colpos = entry.indexOf(':');
107             if (colpos > -1) {
108                 name = entry.substring(0, colpos);
109                 clazz = entry.substring(colpos + 1);
110                 final String filterConfigName = configG.findStringEntry(name);
111                 if (filterConfigName != null) {
112                     try {
113                         filterConfig = ConfigUtil.getConfigInfo(filterConfigName);
114                     } catch (IOException configError) {
115                         logger.warn("Specified filter configuration {} cannot be loaded", filterConfigName);
116                         continue;
117                     }
118                 }
119             } else {
120                 name = null;
121                 clazz = entry;
122             }
123 
124             try {
125                 final Object filter = emissary.core.Factory.create(clazz);
126                 if (filter != null && filter instanceof IDropOffFilter) {
127                     final IDropOffFilter../emissary/output/filter/IDropOffFilter.html#IDropOffFilter">IDropOffFilter f = (IDropOffFilter) filter;
128                     f.initialize(configG, name, filterConfig);
129                     addFilter(f);
130                 } else {
131                     logger.error("Misconfigured filter {} is not an IDropOffFilter instance, ignoring it", clazz);
132                 }
133             } catch (Exception ex) {
134                 logger.error("Unable to create or initialize {}", clazz, ex);
135             }
136         }
137 
138         // Collect the set of content types to elide
139         this.elideContentForms = configG.findEntriesAsSet("ELIDE_CONTENT");
140 
141         // collect the set of no-nuke forms
142         this.noNukeForms = configG.findEntriesAsSet("NO_NUKE_FORM");
143 
144         if (logger.isInfoEnabled()) {
145             logger.debug("Setting ELIDE_CONTENT forms to " + this.elideContentForms);
146             final StringBuilder sb = new StringBuilder("Output Filters:");
147             if (this.outputFilters.size() > 0) {
148                 for (final IDropOffFilter f : this.outputFilters) {
149                     sb.append(" ").append(f.getFilterName()).append("(").append(f.getClass().getName()).append(")");
150                 }
151             } else {
152                 sb.append(" NONE!");
153             }
154             logger.info(sb.toString());
155         }
156 
157         if (logger.isDebugEnabled()) {
158             final IBaseDataObject fakePayload = DataObjectFactory.getInstance(new byte[0], "fakename", Form.UNKNOWN);
159             for (final IDropOffFilter filter : getFilters()) {
160                 final String name = filter.getFilterName();
161                 final String spec = filter.getOutputSpec();
162 
163                 logger.debug("Adding filter={}, spec={}, sample={}, class={}", name, spec, this.dropOffUtil.getPathFromSpec(spec, fakePayload),
164                         filter.getClass().getSimpleName());
165             }
166         }
167     }

179     /**
180      * "HD" agent calls this method when visiting the place. If you use {@link emissary.core.MobileAgent} this method is
181      * never called. This method overrides {@link ServiceProviderPlace} and allows this processing place to have access to
182      * all payloads wanting to be dropped off in a single list.
183      * 
184      * @param payloadList list of IBaseDataObject from an {@link emissary.core.HDMobileAgent}
185      */
186     @Override
187     public List<IBaseDataObject> agentProcessHeavyDuty(final List<IBaseDataObject> payloadList) throws Exception {
188 
189         logger.debug("Entering DropOffPlace.agentProcessHeavyDuty with {} payload items", payloadList.size());
190 
191         // Prepare each incoming payload object
192         for (final IBaseDataObject d : payloadList) {
193             try {
194                 // checking to see if any object in the tree is marked as not outputable
195                 if (!d.isOutputable()) {
196                     logger.info("Skipping object since it is not able to be output ID:{}", this.dropOffUtil.getBestId(d, payloadList.get(0)));
197                     return Collections.emptyList();
198                 }
199 
200                 // Process the payload item with HDcontext=true
201                 processData(d, true);
202             } catch (Exception e) {
203                 logger.error("Place.process threw:", e);
204                 d.addProcessingError("agentProcessHD(" + myKey + "): " + e);
205 
206                 if (!d.currentForm().equals(Form.ERROR)) {
207                     d.pushCurrentForm(Form.ERROR);
208                 }
209             }
210         }
211 
212         // Prepare the data and metadata for filter output
213         final Map<String, Object> filterParams = new HashMap<>();
214         preFilterHook(payloadList, filterParams);
215 
216         // Run the filter on the output, indicating that
217         // the records are pre-sorted, if the filter cares
218         runOutputFilters(payloadList, filterParams);
219 
220         // Any cleanup operations needed
221         postFilterHook(payloadList, filterParams);
222 
223         if (!payloadList.isEmpty()) {
224             // Should have been sorted by the prefilter hook
225 
226             // Just report the TLD object ID
227             final IBaseDataObject tld = payloadList.get(0);
228 
229 
230             if (outputCompletionPayloadSize && tld.hasContent()) {
231                 logger.info(
232                         "Finished DropOff for object {}, with external id: {}, with total processing time: {}ms, with filetype: {}, payload size: {} bytes",
233                         tld.getInternalId(), this.dropOffUtil.getBestId(tld, tld),
234                         Duration.between(Instant.now(), tld.getCreationTimestamp()).toMillis(),
235                         tld.getFileType(), tld.getChannelSize());
236             } else {
237                 logger.info("Finished DropOff for object {}, with external id: {}, with total processing time: {}ms, with filetype: {}",
238                         tld.getInternalId(), this.dropOffUtil.getBestId(tld, tld),
239                         Duration.between(Instant.now(), tld.getCreationTimestamp()).toMillis(),
240                         tld.getFileType());
241             }
242 
243         }
244 
245         // Execute 'Dispose Runnables' to tidy up resources used with SeekableByteChannelFactory implementations
246         DisposeHelper.execute(payloadList);
247 
248         // This place does not sprout, return an empty list
249         return Collections.emptyList();
250     }

252     /**
253      * Called by MobileAgent through ServiceProviderPlace to handle a single payload
254      * 
255      * @param tData the payload to work on
256      */
257     @Override
258     public void process(final IBaseDataObject tData) {
259         if (DataUtil.isEmpty(tData)) {
260             logger.warn("null/empty data object");
261             return;
262         }
263 
264         // checking to see if the object is marked as not outputable
265         if (!tData.isOutputable()) {
266             logger.warn("Skipping object since it is not able to be output ID:{}", this.dropOffUtil.getBestId(tData, tData));
267             return;
268         }
269 
270         // synchronization can be set by config file entry
271         if (this.doSynchronized) {
272             synchronized (this) {
273                 processData(tData, false);
274             }
275         } else {
276             processData(tData, false);
277         }
278 
279         // Execute 'Dispose Runnables' to tidy up resources used with SeekableByteChannelFactory implementations
280         DisposeHelper.execute(tData);
281     }

283     /**
284      * Prepare a list of payload object to be filtered
285      * 
286      * @param payloadList the list of items that were eligible for output
287      * @param filterParams metadata needed for the output filter
288      */
289     public void preFilterHook(final List<IBaseDataObject> payloadList, final Map<String, Object> filterParams) {
290         // Sort the list of records
291         Collections.sort(payloadList, new ShortNameComparator());
292         filterParams.put(IDropOffFilter.PRE_SORTED, Boolean.TRUE);
293         filterParams.put(IDropOffFilter.TLD_PARAM, payloadList.get(0));
294 
295         // Prepare the metadata
296         this.dropOffUtil.processMetadata(payloadList);
297     }

299     /**
300      * Clean up after all filter are done
301      * 
302      * @param payloadList the list of items that were eligible for output
303      * @param filterParams metadata needed for the output filter
304      */
305     public void postFilterHook(final List<IBaseDataObject> payloadList, final Map<String, Object> filterParams) {
306         // remove the current forms we used or could have used
307         for (final IBaseDataObject dataObject : payloadList) {
308             // Save off no-nuke forms
309             final List<String> saveForms = new ArrayList<>();
310             for (final String nnf : this.noNukeForms) {
311                 if (dataObject.searchCurrentForm(nnf) > -1) {
312                     saveForms.add(nnf);
313                 }
314             }
315             // nuke 'em
316             this.nukeMyProxies(dataObject);
317 
318             // Restore the no-nukes
319             for (final String sf : saveForms) {
320                 dataObject.pushCurrentForm(sf);
321             }
322         }
323 
324     }

326     /**
327      * Internal method to process a single data object
328      * 
329      * @param tData the payload to work on or prepare
330      * @param haveList true if in HD context
331      */
332     protected void processData(final IBaseDataObject tData, final boolean haveList) {
333 
334         logger.debug("DropOff is working on {}, current form is {}", tData.shortName(), tData.getAllCurrentForms());
335 
336         final StringBuilder poppedForms = new StringBuilder();
337 
338         String prevBin = "";
339 
340         // skip the I/O for some types for all filter
341         for (int i = 0; i < tData.currentFormSize(); i++) {
342             final String cf = tData.currentFormAt(i);
343             if (this.elideContentForms.contains(cf)) {
344                 tData.setData(("[[ " + tData.getAllCurrentForms() + " content elided in DropOffPlace. ]]").getBytes());
345             }
346         }
347 
348         // Write out data for all the destinations we area proxy for, popping
349         // them off the stack as they are handled.
350 
351         final Set<String> serviceProxies = getProxies();
352         final Set<String> cfSet = new HashSet<>();
353         for (int i = 0; i < tData.currentFormSize(); i++) {
354             final String cf = tData.currentFormAt(i);
355 
356             if (serviceProxies.contains(cf) || serviceProxies.contains("*")) {
357                 // Record extra drop offs in the transform history since
358                 // we are taking precedence over the normal agent/directory mechanism
359                 // Do this before popping the extra destination so that it
360                 // is not just lost forever. Since the agent appends
361                 // one entry to the history when it sends the agent here,
362                 // that's the top current form and
363                 // we don't want to duplicate that one
364 
365                 if (!prevBin.equals(cf) && (i > 0) && !cfSet.contains(cf) && !("UNKNOWN".equals(cf) || cf.endsWith("-PROCESSED"))) {
366                     // e.g.: [dataType].DROP_OFF.IO.host.dom:port/DropOffPlace
367                     final DirectoryEntry de = getDirectoryEntry();
368                     de.setDataType("[" + cf + "]");
369                     tData.appendTransformHistory(de.getKey());
370                 }
371 
372                 // Accumulate forms we have handled in poppedForms
373                 if (poppedForms.length() > 0) {
374                     poppedForms.append(" ");
375                 }
376                 poppedForms.append(cf);
377                 cfSet.add(cf);
378 
379                 prevBin = cf;
380             }
381         }
382 
383         // Record the list of forms
384         tData.setParameter("POPPED_FORMS", poppedForms.toString());
385 
386         // Do the output filtering now if we aren't in HD mode
387         if (!haveList) {
388 
389             // There currently are no extra params needed
390             final Map<String, Object> filterParams = new HashMap<>();
391 
392             runOutputFilters(tData, filterParams);
393 
394             // Actually remove the current forms we used or could have used
395             this.nukeMyProxies(tData);
396 
397             logger.debug("DropOff finished with {}", tData.shortName());
398         }
399     }

401     /**
402      * Run all the output filter
403      * 
404      * @param target either IBaseDataObject or List thereof
405      * @param filterParams other parameters that filter need
406      */
407     @SuppressWarnings("unchecked")
408     protected void runOutputFilters(final Object target, final Map<String, Object> filterParams) {
409 
410         IBaseDataObject doTarget = null;
411         List<IBaseDataObject> listTarget = null;
412         if (target instanceof IBaseDataObject) {
413             doTarget = (IBaseDataObject) target;
414         } else if (target instanceof List) {
415             listTarget = (List<IBaseDataObject>) target;
416         } else {
417             logger.error("Cannot run filter on {}", target.getClass().getName());
418             return;
419         }
420 
421         // Write output onto each of the filter that have been
422         // configured, as long as they work
423         for (final IDropOffFilter filter : this.outputFilters) {
424             final long start = System.currentTimeMillis();
425 
426             // call the filter to output its data
427             int filterStatus = IDropOffFilter.STATUS_FAILURE;
428             try {
429                 if (listTarget != null && filter.isOutputtable(listTarget)) {
430                     filterStatus = filter.filter(listTarget, filterParams);
431                 } else if (doTarget != null && filter.isOutputtable(doTarget)) {
432                     filterStatus = filter.filter(doTarget, filterParams);
433                 } else {
434                     logger.debug("Filter {} not Outputtable for {}", filter.getFilterName(), listTarget != null ? "list" : "single payload");
435                     filterStatus = IDropOffFilter.STATUS_SUCCESS;
436                 }
437                 logger.debug("Filter {} took {}s - {}", filter.getFilterName(), ((System.currentTimeMillis() - start) / 1000.0), filterStatus);
438             } catch (Exception e) {
439                 logger.error("Filter {} failed", filter.getFilterName(), e);
440             }
441 
442             if ((filterStatus != IDropOffFilter.STATUS_SUCCESS) && this.failurePolicyTerminate) {
443                 logger.error("DropOff Filter chain terminated at {} due to error return status", filter.getFilterName());
444                 break;
445             }
446         }
447     }

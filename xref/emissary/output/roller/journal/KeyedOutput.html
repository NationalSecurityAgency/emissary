3   import jakarta.annotation.Nullable;
4   import org.slf4j.Logger;
5   import org.slf4j.LoggerFactory;

7   import java.io.IOException;
8   import java.io.OutputStream;
9   import java.nio.ByteBuffer;
10  import java.nio.channels.SeekableByteChannel;
11  import java.nio.file.Path;
12  
13  /**
14   * Lightweight wrapper holding reference to a pooled object and the pool, which returns the channel to the pool on
15   * close.
16   */
17  public class KeyedOutput extends OutputStream implements SeekableByteChannel {
18  
19      static final Logger LOG = LoggerFactory.getLogger(KeyedOutput.class);
20      @Nullable
21      JournaledChannelPool pool;
22      @Nullable
23      JournaledChannel jc;
24  
25      KeyedOutput(final JournaledChannelPool pool, final JournaledChannel jc) {
26          this.pool = pool;
27          this.jc = jc;
28      }
29  
30      public Path getFinalDestination() {
31          return this.pool.getDirectory().resolve(this.pool.getKey());
32      }
33  
34      @Override
35      public void write(final int b) throws IOException {
36          this.jc.write(b);
37      }
38  
39      @Override
40      public void write(final byte[] bs, final int off, final int len) throws IOException {
41          this.jc.write(bs, off, len);
42      }
43  
44      @Override
45      public int write(final ByteBuffer src) throws IOException {
46          return this.jc.write(src);
47      }
48  
49      @Override
50      public long position() throws IOException {
51          return this.jc.position();
52      }
53  
54      /* Unsupported operations */
55      @Override
56      public SeekableByteChannel position(final long newPosition) throws IOException {
57          throw new UnsupportedOperationException("This operation is not permitted");
58      }
59  
60      @Override
61      public long size() throws IOException {
62          return this.jc.size();
63      }
64  
65      @Override
66      public boolean isOpen() {
67          return (this.jc != null) && this.jc.isOpen();
68      }
69  
70      /**
71       * Commits writes to underlying storage. This method should only be called after a successful write.
72       * 
73       * @throws IOException If there is some I/O problem.
74       */
75      public final void commit() throws IOException {
76          this.jc.commit();
77      }
78  
79      /**
80       * Closes this Channel/Output Stream by releasing resources to underlying this.pool.
81       * 
82       * @throws IOException If there is some I/O problem.
83       */
84      @Override
85      public void close() throws IOException {
86          if (this.pool == null) {
87              return;
88          }
89          LOG.debug("Returning object to pool {}", this.jc.path);
90          this.pool.free(this.jc);
91          this.pool = null;
92          this.jc = null;
93      }
94  
95      @Override
96      public int read(final ByteBuffer dst) throws IOException {
97          throw new UnsupportedOperationException("This operation is not permitted");
98      }
99  
100     @Override
101     public SeekableByteChannel truncate(final long size) throws IOException {
102         throw new UnsupportedOperationException("This operation is not permitted");
103     }
104 }

5   import java.nio.ByteBuffer;

8   /**
9    * Journal Entry containing a value, generally a file name, and an offset.
10   */
11  public final class JournalEntry {
12      static final byte SEP = 0x00;
13      final String val;
14      long offset;
15  
16      public JournalEntry(final String val, final long offset) {
17          if (val == null || val.length() > 512) {
18              throw new IllegalArgumentException("Val must be present and cannot exceed 512 bytes");
19          }
20          this.val = val;
21          this.offset = offset;
22      }
23  
24      void serialize(final ByteBuffer b) {
25          b.putInt(this.val.length());
26          b.put(SEP);
27          b.put(this.val.getBytes());
28          b.put(SEP);
29          b.putLong(this.offset);
30      }
31  
32      @Override
33      public String toString() {
34          return "JournalEntry{" + "value=" + val + ", offset=" + offset + '}';
35      }
36  
37      static JournalEntry deserialize(final ByteBuffer b) {
38          final int keyLen = b.getInt();
39          validateSep(b.get());
40          final byte[] keyBytes = new byte[keyLen];
41          b.get(keyBytes, 0, keyLen);
42          validateSep(b.get());
43          final long offset = b.getLong();
44          return new JournalEntry(new String(keyBytes), offset);
45      }
46  
47      static void validateSep(final byte b) {
48          if (SEP != b) {
49              throw new IllegalArgumentException("Null byte separator expected " + b);
50          }
51      }
52  
53      @Override
54      public int hashCode() {
55          int hash = 7;
56          hash = 83 * hash + Objects.hashCode(this.val);
57          return hash;
58      }
59  
60      @Override
61      public boolean equals(@Nullable final Object obj) {
62          if (this == obj) {
63              return true;
64          }
65          if (obj == null) {
66              return false;
67          }
68          if (getClass() != obj.getClass()) {
69              return false;
70          }
71          final JournalEntry other = (JournalEntry) obj;
72          if (!Objects.equals(this.val, other.val)) {
73              return false;
74          }
75          return true;
76      }
77  
78      public String getVal() {
79          return this.val;
80      }
81  
82      public long getOffset() {
83          return this.offset;
84      }
85  }

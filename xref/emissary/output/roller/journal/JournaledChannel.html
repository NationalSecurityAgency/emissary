15  /**
16   * Wrapper class to allow for use of underlying channel in either OutputStream code or WritableChannel.
17   */
18  public class JournaledChannel extends OutputStream implements SeekableByteChannel {
19  
20      static final Logger LOG = LoggerFactory.getLogger(JournaledChannel.class);
21      // 128k
22      static final int BUFF_SIZE = 128 * 1024;
23      @Nullable
24      FileChannel fc;
25      Path path;
26      @Nullable
27      JournalEntry e;
28      final int index;
29      final JournalWriter journal;
30      @Nullable
31      ByteBuffer directBuff;
32  
33      JournaledChannel(final Path path, final String key, final int index) throws IOException {
34          this.fc = FileChannel.open(path, StandardOpenOption.CREATE_NEW, StandardOpenOption.READ, StandardOpenOption.WRITE);
35          this.path = path;
36          this.index = index;
37          this.journal = new JournalWriter(path.getParent(), path.getFileName().toString(), key);
38          this.directBuff = ByteBuffer.allocateDirect(BUFF_SIZE);
39          writeEntry();
40      }
41  
42      @Nullable
43      private byte[] b1 = null;
44  
45      @Override
46      public void write(final int b) throws IOException {
47          if (this.b1 == null) {
48              this.b1 = new byte[1];
49          }
50          this.b1[0] = (byte) b;
51          this.write(this.b1);
52      }
53  
54      @Override
55      public void write(final byte[] bs, final int off, final int len) throws IOException {
56          if ((off < 0) || (off > bs.length) || (len < 0) || ((off + len) > bs.length) || ((off + len) < 0)) {
57              throw new IndexOutOfBoundsException();
58          } else if (len == 0) {
59              return;
60          }
61  
62          int remaining = len;
63          int offset = off;
64          while (remaining > 0) {
65              final int limit = Math.min(remaining, this.directBuff.capacity());
66              this.directBuff.clear();
67              this.directBuff.put(bs, offset, limit);
68              this.directBuff.flip();
69              while (this.directBuff.hasRemaining()) {
70                  if (this.fc.write(this.directBuff) <= 0) {
71                      throw new IllegalStateException("no bytes written");
72                  }
73              }
74              offset += limit;
75              remaining -= limit;
76          }
77      }
78  
79      @Override
80      public int write(final ByteBuffer src) throws IOException {
81          // doing this to avoid java bug that caused direct memory leaks.
82          // could be removed in java 9 when they provide a JVM argument to
83          // limit caching
84          if (!src.isDirect()) {
85              int written = 0;
86              while (src.hasRemaining()) {
87                  this.directBuff.clear();
88                  while (this.directBuff.hasRemaining() && src.hasRemaining()) {
89                      this.directBuff.put(src.get());
90                  }
91                  this.directBuff.flip();
92                  written += this.fc.write(this.directBuff);
93              }
94              return written;
95          } else {
96              return this.fc.write(src);
97          }
98      }

121     /**
122      * Sets the position of the channel according to the current entry. Should only be called by the pool.
123      * 
124      * @throws IOException If there is some I/O problem.
125      */
126     void setPosition() throws IOException {
127         if (this.e.getOffset() != this.fc.position()) {
128             this.fc.position(this.e.getOffset());
129         }
130     }
131 
132     /**
133      * Commits writes to underlying storage. This method should only be called after a successful write.
134      * 
135      * @throws IOException If there is some I/O problem.
136      */
137     public final void commit() throws IOException {
138         writeEntry();
139     }
140 
141     private void writeEntry() throws IOException {
142         final JournalEntry entry = new JournalEntry(this.path.toString(), this.fc.position());
143         this.journal.write(entry);
144         this.e = entry;
145     }
146 
147     /**
148      * Closes this Channel/Output Stream by releasing resources to underlying pool. Further calls result in unspecified
149      * behavior
150      * 
151      * @throws IOException If there is some I/O problem.
152      */
153     @Override
154     public void close() throws IOException {
155         if (this.fc != null) {
156             this.fc.close();
157             this.fc = null;
158         }
159         this.journal.close();
160         this.e = null;
161         this.directBuff = null;
162     }
163 
164     /**
165      * Opens this object for writing and sets position according to current Journal Entry.
166      * 
167      * @throws IOException If there is some I/O problem.
168      */
169     void open() throws IOException {
170         LOG.debug("Opening channel for writing {}", this.path);
171         if (this.fc.position() != this.e.getOffset()) {
172             this.fc.position(this.e.getOffset());
173         }
174     }

14  import static emissary.output.roller.journal.Journal.SEP;
15  
16  /**
17   * BG Write ahead log to track progress, often files, that have been successfully flushed to disk. The file itself
18   * contains metadata at the start, followed by a collection of log entries of fixed length, currently 1024. Leading
19   * metadata is in the format:
20   *
21   * <code>
22   * [Journal Magic][Journal Version][Journal key.size()][Journal key.getBytes()][Journal Sequence Number][null byte][1-n Journal Entries]
23   * </code>
24   *
25   * Record format:
26   *
27   * <code>
28   * [Journal sequence number][null byte][Entry value.size()][null byte][Entry val.getBytes()][null byte][position][null padded to fixed len]
29   * </code>
30   */
31  public class JournalWriter implements Closeable {
32  
33      private final ReentrantLock lock = new ReentrantLock();
34      @Nullable
35      private ByteBuffer b = ByteBuffer.allocateDirect(Journal.ENTRY_LENGTH);
36      // full path to journal file
37      final Path journalPath;
38      // current sequence value
39      private long sequence;
40      byte version;
41      String key;
42      // persisted journal
43      @Nullable
44      FileChannel journal;
45      JournalEntry prev;
46  
47      public JournalWriter(final Path dir, final String key) throws IOException {
48          this(dir, key, key);
49      }
50  
51      public JournalWriter(final Path dir, final String journalFileName, final String key) throws IOException {
52          this.journalPath = dir.resolve(journalFileName + Journal.EXT);
53          this.key = key;
54          checkJournal();
55      }
56  
57      private void checkJournal() throws IOException {
58          if (Files.exists(journalPath) && Files.size(journalPath) > 0L) {
59              throw new IllegalStateException("Journals Are Immutable");
60          } else {
61              Files.deleteIfExists(journalPath);
62          }
63      }
64  
65      /**
66       * Write to journal
67       *
68       * @return position difference between last entry and current
69       */
70      public long write(JournalEntry e) throws IOException {
71          lock.lock();
72          try {
73              if (journal == null) {
74                  writeHeader();
75              }
76              b.clear();
77              b.putLong(++sequence);
78              b.put(SEP);
79              e.serialize(b);
80              // fixed record length format so zero out everything
81              nullpad();
82              write();
83              return prev == null ? e.offset : e.offset - prev.offset;
84          } finally {
85              prev = e;
86              lock.unlock();
87          }
88      }
89  
90      private void write() throws IOException {
91          b.flip();
92          journal.write(b);
93          b.clear();
94      }
95  
96      private void writeHeader() throws IOException {
97          this.journal = FileChannel.open(journalPath, StandardOpenOption.CREATE, StandardOpenOption.READ, StandardOpenOption.WRITE);
98          b.clear();
99          b.put(Journal.MAGIC);
100         b.put(Journal.CURRENT_VERSION);
101         byte[] keyBytes = key.getBytes();
102         b.putInt(keyBytes.length);
103         for (int i = 0; i < keyBytes.length; i++) {
104             if (b.remaining() == 0) {
105                 write();
106             }
107             b.put(keyBytes[i]);
108         }
109         if (b.remaining() < Journal.NINE) {
110             write();
111         }
112         sequence = System.currentTimeMillis();
113         b.putLong(sequence);
114         b.put(SEP);
115         write();
116     }
117 
118     // fill buffer with zeros from current position to limit
119     private void nullpad() {
120         while (b.hasRemaining()) {
121             b.put(SEP);
122         }
123     }
124 
125     /**
126      * Closes underlying journal channel.
127      */
128     @Override
129     public void close() throws IOException {
130         lock.lock();
131         try {
132             if (journal != null) {
133                 journal.close();
134             }
135             journal = null;
136             b = null;
137         } finally {
138             lock.unlock();
139         }
140     }
141 }

99      /**
100      * Set up config for this class
101      * <ul>
102      * <li>ID_PARAMETER : multiple parameter values, ordered list of how to build a EMISSARY_ID</li>
103      * <li>ID : backwards compatibility for ID_PARAMETER only used if ID_PARAMETER does not exist</li>
104      * <li>DATE_PARAMETER : multiple parameter values, ordered list of how to build a date path</li>
105      * <li>DATE_FORMAT: string to use for SimpleDateFormat, ordered list</li>
106      * <li>OUTPUT_FILE_PREFIX: string to use when generating random filenames, dflt: TXT</li>
107      * <li>UUID_IN_OUTPUT_FILENAMES: boolean [true]</li>
108      * <li>AUTO_GENERATED_ID_PREFIX: prefix to use for an auto-generated id</li>
109      * </ul>
110      */
111     protected void configure(final Configurator configG) {
112         Configurator actualConfigG = configG;
113         if (actualConfigG == null) {
114             try {
115                 actualConfigG = ConfigUtil.getConfigInfo(DropOffUtil.class);
116             } catch (IOException e) {
117                 logger.error("Cannot open default config file", e);
118             }
119         }
120 
121         if (actualConfigG != null) {
122             this.placeOutputData = actualConfigG.findStringEntry("OUTPUT_DATA", "outputData");
123             this.unixRoot = actualConfigG.findStringEntry("UNIX_ROOT", null);
124             this.executrix = new Executrix(actualConfigG);
125             this.idTokens = actualConfigG.findEntries("ID_PARAMETER");
126             this.autoGeneratedIdPrefix = actualConfigG.findStringEntry("AUTO_GENERATED_ID_PREFIX");
127 
128             // truncate the prefix if necessary
129             if (!StringUtils.isBlank(this.autoGeneratedIdPrefix) && (this.autoGeneratedIdPrefix.length() > 4)) {
130                 this.autoGeneratedIdPrefix = this.autoGeneratedIdPrefix.substring(0, 4);
131             }
132             if (this.idTokens.size() == 0) {
133                 this.idTokens = actualConfigG.findEntries("ID");
134             }
135             this.dateTokens = actualConfigG.findEntries("DATE_PARAMETER");
136             this.parentParams = actualConfigG.findEntries("PARENT_PARAM");
137 
138 
139             this.defaultEventDateToNow = actualConfigG.findBooleanEntry(DEFAULT_EVENT_DATE_TO_NOW, this.defaultEventDateToNow);
140 
141             prefix = actualConfigG.findStringEntry("OUTPUT_FILE_PREFIX", prefix);
142             this.uuidInOutputFilenames = actualConfigG.findBooleanEntry("UUID_IN_OUTPUT_FILENAMES", this.uuidInOutputFilenames);
143             // limit the length of the FILEXT param
144             this.maxFilextLen = actualConfigG.findIntEntry("MAX_FILEXT_LEN", this.maxFilextLen);
145             if (this.maxFilextLen < 0) {
146                 this.maxFilextLen = Integer.MAX_VALUE;
147             }
148         } else {
149             logger.debug("Configuration is null for DropOffUtil, using defaults");
150             this.executrix = new Executrix();
151         }

191     public String makeWayForIncomingFile(final IBaseDataObject DO, final String suffix, final String spec) {
192         // Construct output file from DropOff output path and IBaseDataObject
193         // filename
194         final String outputFile = getShortOutputFileName(DO, spec);
195 
196         // Return value when finally set
197         final String fileName = outputFile + suffix;
198 
199         // Set up path and make it writable
200         if (!setupPath(fileName)) {
201             return null;
202         }
203 
204         if (removeExistingFile(fileName)) {
205             return fileName;
206         }
207 
208         return null;
209     }

215         // If a file already exists under this name, we're going
216         // to have to move it aside. This is going to break the link
217         // if it is an attachment to another parent document.
218         final Path theFile = Paths.get(fileName);
219         try {
220             Files.deleteIfExists(theFile);

222             logger.error("Trouble removing file: {}", theFile, e);
223         }
224         return !Files.exists(theFile);

227     /**
228      * mkdir -p to a path and make sure it is writable
229      *
230      * @param fileName the file name, including directory and filename parts
231      * @return true iff it works
232      */
233     public boolean setupPath(final String fileName) {
234         final String pathName = fileName.substring(0, fileName.lastIndexOf(SEPARATOR));
235         final Path thePath = Paths.get(pathName);
236 
237         // If the specified output directory doesn't exist try creating it
238         if (!Files.exists(thePath)) {
239             int tryCount = 1;
240             do {
241                 try {
242                     Files.createDirectories(thePath);
243                 } catch (IOException e) {
244                     logger.warn("Trouble setting up directories:{}", thePath, e);
245                 }
246                 if (!Files.exists(thePath)) {
247                     try {
248                         Thread.sleep(50L * tryCount);
249                     } catch (InterruptedException e) {
250                         Thread.currentThread().interrupt();
251                     }
252                 }
253                 tryCount++;
254             } while (!Files.exists(thePath) && tryCount <= 10);
255 
256             if (!Files.exists(thePath)) {
257                 logger.warn("Cannot create directory for output: {} in {} attempts", thePath, tryCount);
258                 return false;
259             }
260 
261             if (tryCount > 2 && logger.isDebugEnabled()) {
262                 logger.debug("Output path created for {} but it took {} attempts", thePath, tryCount);
263             }
264         }
265 
266         // If the specified output directory doesn't have write
267         // permission try to fix it
268         if (!Files.isWritable(thePath)) {
269             logger.warn("Dont have write permission for {}, setting it now", pathName);
270             if (!thePath.toFile().setWritable(true) || !Files.isWritable(thePath)) {
271                 logger.warn("Cannot write to directory for output: {}", thePath);
272                 return false;
273             }
274         }
275 
276         return true;
277     }

295     /**
296      * Create a path name from the spec for the specified spec Specs understand the following 'language' of replacement
297      * stuff. Anything not understood is a literal
298      *
299      * <pre>
300      * <code>@TLD{'KEY'}</code> is to pull the named KEY from the top level document Metadata
301      * <code>@META{'KEY'}</code> is to pull the named KEY from the MetaData
302      * %U% = USER
303      * %I% = INPUT_FILE_NAME (whole thing)
304      * %S% = INPUT_FILE SHORT NAME
305      * %P% = INPUT_FILE PATH (all but short name)
306      * %i% = INPUT_FILE_NAME with slashes converted to underscores
307      * %p% = INPUT_FILE PATH with slashes converted to underscores
308      * %F% = FILETYPE
309      * %L% = LANGUAGE
310      * %G% = DTG multi directory layout yyyy-mm-dd/hh/mi(div)10
311      * %R% = ROOT (Unix or Win depending on OS)
312      * %B% = ID for the payload depending on type (no -att-)
313      * %b% = ID for the payload depending on type (with -att-)
314      * %Y% = Four digit year
315      * %M% = Two digit month
316      * %D% = Two digit day of month
317      * %J% = Three digit ordinal day of the year
318      * </pre>
319      *
320      * @param specArg the incoming specification
321      * @param d the payload we are making a path for
322      * @param tld the top level document in the d family, possibly null
323      * @return string path name with correct separators for this OS
324      */
325     public String getPathFromSpec(final String specArg, @Nullable final IBaseDataObjectt">IBaseDataObject d, @Nullable final IBaseDataObject tld) {
326         final StringBuffer sb = new StringBuffer(128);
327 
328         // Provide a default spec, just like the old days...
329         String spec = specArg;
330         if (spec == null) {
331             spec = "%R%/@TLD{'TARGETBIN'}/%S%";
332         }
333 
334         for (int i = 0; i < spec.length(); i++) {
335             final char c = spec.charAt(i);
336 
337             if (c == '%' && i < spec.length() - 2) {
338                 final char t = spec.charAt(i + 1);
339                 final char x = spec.charAt(i + 2);
340 
341                 if (x == c) {
342                     switch (t) {
343                         case 'U':
344                             if (tld != null) {
345                                 sb.append(nvl(tld.getParameter("UserName"), "no-userid"));
346                             } else if (d != null) {
347                                 sb.append(nvl(d.getParameter("UserName"), "no-userid"));
348                             }
349                             break;
350                         case 'S':
351                             if (d != null) {
352                                 sb.append(d.shortName());
353                             }
354                             break;
355                         case 'I':
356                             if (d != null) {
357                                 sb.append(d.getFilename());
358                             }
359                             break;
360                         case 'i':
361                             if (d != null) {
362                                 sb.append(d.getFilename().replaceAll("[/\\\\]", "_"));
363                             }
364                             break;
365                         case 'P':
366                             if (d != null) {
367                                 sb.append(d.getFilename().substring(0, d.getFilename().length() - d.shortName().length()));
368                             }
369                             break;
370                         case 'p':
371                             if (d != null) {
372                                 sb.append(d.getFilename().substring(0, d.getFilename().length() - d.shortName().length()).replaceAll("[/\\\\]", "_"));
373                             }
374                             break;
375                         case 'F':
376                             if (d != null) {
377                                 sb.append(nvl(cleanSpecPath(d.getFileType()), "NONE"));
378                             }
379                             break;
380                         case 'L':
381                             if (d != null) {
382                                 sb.append(nvl(d.getParameter("LANGUAGE"), "NONE"));
383                             }
384                             break;
385                         case 'G':
386                             if (tld != null) {
387                                 sb.append(datePath(cleanSpecPath(tld.getStringParameter("DTG"))));
388                             } else if (d != null) {
389                                 sb.append(datePath(cleanSpecPath(d.getStringParameter("DTG"))));
390                             }
391                             break;
392                         case 'R':
393                             sb.append(getRootPath());
394                             break;
395                         case 'B':
396                             if (tld != null) {
397                                 sb.append(cleanSpecPath(getBestIdFrom(tld)));
398                             } else if (d != null) {
399                                 sb.append(cleanSpecPath(getBestIdFrom(d)));
400                             }
401                             break;
402                         case 'b':
403                             sb.append(cleanSpecPath((tld != null) ? getBestIdFrom(tld) : getBestIdFrom(d)));
404                             final String sn = d.shortName();
405                             final int pos = sn.indexOf(emissary.core.Family.SEP);
406                             if (pos > 0) {
407                                 sb.append(sn.substring(pos));
408                             }
409                             break;
410                         case 'Y':
411                             sb.append(TimeUtil.getDate("yyyy", "GMT"));
412                             break;
413                         case 'M':
414                             sb.append(TimeUtil.getDate("MM", "GMT"));
415                             break;
416                         case 'D':
417                             sb.append(TimeUtil.getDate("dd", "GMT"));
418                             break;
419                         case 'J':
420                             sb.append(TimeUtil.getDate("DDD", "GMT"));
421                             break;
422                         default:
423                             sb.append(c).append(t).append(x);
424                     }
425                     i += 2; // SUPPRESS CHECKSTYLE ModifiedControlVariable
426                 } else {
427                     // No trailing % after character
428                     sb.append(c);
429                 }
430 
431             } else if (c == '@' && i < spec.length() - 8 && spec.charAt(i + 1) == 'M' && spec.charAt(i + 2) == 'E' && spec.charAt(i + 3) == 'T'
432                     && spec.charAt(i + 4) == 'A' && spec.charAt(i + 5) == '{' && spec.charAt(i + 6) == '\'') {
433 
434                 final int endpos = spec.indexOf("'", i + 7);
435                 if (endpos > i + 7) {
436                     final String token = spec.substring(i + 7, endpos);
437                     final String value = cleanSpecPath(d.getStringParameter(token));
438                     sb.append(nvl(value, "NO-" + token));
439                     i += 8 + token.length(); // META{'token'} SUPPRESS CHECKSTYLE ModifiedControlVariable
440                 } else {
441                     sb.append(c);
442                 }
443             } else if (c == '@' && i < spec.length() - 7 && spec.charAt(i + 1) == 'T' && spec.charAt(i + 2) == 'L' && spec.charAt(i + 3) == 'D'
444                     && spec.charAt(i + 4) == '{' && spec.charAt(i + 5) == '\'' && tld != null) {
445                 final int endpos = spec.indexOf("'", i + 6);
446                 if (endpos > i + 6) {
447                     final String token = spec.substring(i + 6, endpos);
448                     final String value = cleanSpecPath(tld.getStringParameter(token));
449                     sb.append(nvl(value, "NO-" + token));
450                     i += 7 + token.length(); // TLD{'token'} SUPPRESS CHECKSTYLE ModifiedControlVariable
451                 } else {
452                     sb.append(c);
453                 }
454             } else {
455                 sb.append(c);
456             }
457 
458         }
459 
460         String answer = sb.toString();
461 
462         // Set the proper path separator
463         answer = answer.replace('\\', '/');
464         answer = answer.replaceAll("\\.([/\\\\])", "_$1");
465 
466         return answer;
467     }

473     /**
474      * Extract the ID from the payload. The ID from the payload is specified in the cfg file. An ID = SHORTNAME will use the
475      * shortname. An ID = AUTO_GENERATED_ID will use an auto gen uuid. If no id value is found, defaults to using an auto
476      * generated id.
477      *
478      * @return id
479      */
480     public String getBestIdFrom(final IBaseDataObject d) {
481         for (final String s : this.idTokens) {
482             if (!StringUtils.isBlank(d.getStringParameter(s))) {
483                 return d.getStringParameter(s);
484             }
485             if (SHORTNAME.equals(s)) {
486                 final String shortName = d.shortName();
487                 // if shortname is not blank use it, if blank move on...
488                 if (!StringUtils.isBlank(shortName)) {
489                     return shortName;
490                 }
491             }
492             if (AUTO_GENERATED_ID.equals(s)) {
493                 return getRandomUUID(d);
494             }
495         }
496         // if nothing else worked, use an auto gen id
497         return getRandomUUID(d);
498     }

592     /**
593      * Replace any file separators that are not for this platform with the correct one
594      */
595     public String fixFileNameSeparators(final String s) {
596         // other platform;
597         String badfs = null;
598 
599         if ("/".equals(SEPARATOR)) {
600             badfs = "\\";
601         } else {
602             badfs = "/";
603         }
604 
605         final StringBuilder ret = new StringBuilder(s.length());
606 
607         for (int i = 0; i < s.length(); i++) {
608             if (s.charAt(i) == badfs.charAt(0)) {
609                 ret.append(SEPARATOR);
610             } else {
611                 ret.append(s.charAt(i));
612             }
613         }
614 
615         return ret.toString();
616     }

625     /**
626      * @param dtg expected format yyyymmddhhmmss
627      * @return yyyy-mm-dd/hh/(mm%10)
628      */
629     protected String datePath(@Nullable final String dtg) {
630         if (dtg == null) {
631             return TimeUtil.getDateAsPath(Instant.now());
632         } else {
633             return dtg.substring(0, 4) + "-" + // yyyy
634                     dtg.substring(4, 6) + "-" + // mm
635                     dtg.substring(6, 8) + "/" + // dd
636                     dtg.substring(8, 10) + "/" + // HH
637                     dtg.substring(10, 11) + "0"; // M0

641     /**
642      * Make a dot file name from the supplied path Ex: /path/to/file.txt --&gt; /path/to/.file.txt
643      *
644      * @param fullName the full path and filename of the file
645      * @return the path as is but with a leading dot in the filename
646      */
647     public String makeDotFile(final String fullName) {
648         final int fpos = fullName.lastIndexOf("/");
649         final int rpos = fullName.lastIndexOf("\\");
650         if (fpos == -1 && rpos == -1) {
651             return "." + fullName;
652         }
653         final int pos = Math.max(fpos, rpos);
654         return fullName.substring(0, pos + 1) + "." + fullName.substring(pos + 1);
655     }

808     /**
809      * Extract the ID from the payload. The ID from the payload is specified in the cfg file. An ID = SHORTNAME will use the
810      * shortname. An ID = AUTO_GENERATED_ID will use an auto gen uuid. If no id value is found, defaults to using an auto
811      * generated id.
812      *
813      * @param d the payload
814      * @param tld if a param is specified and cannot be found in the d method parameter, tries to get param from tld.
815      * @return the id based the best ID available, shortname or auto gen id. If no id is found, defaults to auto gen.
816      */
817     public String getBestId(final IBaseDataObjectt">IBaseDataObject d, @Nullable final IBaseDataObject tld) {
818 
819         for (final String s : this.idTokens) {
820             if (AUTO_GENERATED_ID.equals(s)) {
821                 String parentAutoGeneratedId = null;
822                 if (tld != null) {
823                     parentAutoGeneratedId = tld.getStringParameter(PARENT_AUTO_GENERATED_ID);
824                 }
825                 if (StringUtils.isBlank(parentAutoGeneratedId)) {
826                     String uuid = getRandomUUID(d);
827                     if (tld != null) {
828                         tld.setParameter(PARENT_AUTO_GENERATED_ID, uuid);
829                     }
830                     if (!StringUtils.isBlank(uuid)) {
831                         final String component = d.shortName();
832                         final int pos = component.indexOf(Family.SEP);
833                         if (pos > -1) {
834                             uuid += component.substring(pos);
835                         }
836                         return uuid;
837                     }
838                 }
839                 String uuid = null;
840                 // try getting param from the tld
841                 if (!StringUtils.isBlank(parentAutoGeneratedId) && (tld != null)) {
842                     uuid = tld.getStringParameter(PARENT_AUTO_GENERATED_ID);
843                 }
844                 if (!StringUtils.isBlank(uuid)) {
845                     final String component = d.shortName();
846                     final int pos = component.indexOf(Family.SEP);
847                     if (pos > -1) {
848                         uuid += component.substring(pos);
849                     }
850                     d.setParameter(AUTO_GENERATED_ID, "yes");
851                     return uuid;
852                 }
853 
854             }
855 
856             if (SHORTNAME.equals(s)) {
857                 final String shortName = d.shortName();
858 
859                 // if shortname is not blank use it, if blank move on...
860                 if (!StringUtils.isBlank(shortName)) {
861                     return shortName;
862                 }
863 
864                 String path = d.getStringParameter(s);
865                 // try getting shortname from the tld
866                 if (StringUtils.isBlank(path) && (tld != null)) {
867                     path = tld.getStringParameter(s);
868                 }
869                 if (!StringUtils.isBlank(path)) {
870                     final String component = d.shortName();
871                     final int pos = component.indexOf(Family.SEP);
872                     if (pos > -1) {
873                         path += component.substring(pos);
874                     }
875                     return path;
876                 }
877 
878             }
879             // the param from the tld has priority over any child
880             if ((tld != null) && !StringUtils.isBlank(tld.getStringParameter(s))) {
881                 String path = tld.getStringParameter(s);
882                 if (!StringUtils.isBlank(path)) {
883                     final String component = d.shortName();
884                     final int pos = component.indexOf(Family.SEP);
885                     if (pos > -1) {
886                         path += component.substring(pos);
887                     }
888                     return path;
889                 }
890             }
891             // if the param is not in the tld
892             if (!StringUtils.isBlank(d.getStringParameter(s))) {
893                 return d.getStringParameter(s);
894             }
895 
896         }
897 
898         // if nothing works, auto gen an id
899         final String uuid = getRandomUUID(d);
900         if (tld != null) {
901             tld.setParameter(PARENT_AUTO_GENERATED_ID, uuid);
902         }
903         return uuid;
904     }

971     /**
972      * Extracts from the provided {@link IBaseDataObject} the last file extension from each value in the "Original-Filename"
973      * parameter, if that value contains "." before its last character. If one or more file extensions are extracted, the
974      * IBaseDataObject's "FILEXT" parameter is set as the unique set of extracted file extensions, converted to lowercase.
975      *
976      * @param p IBaseDataObject to process
977      *
978      */
979     void extractUniqueFileExtensions(IBaseDataObject p) {
980         if (p.hasParameter(ORIGINAL_FILENAME)) {
981             final Set<String> extensions = new HashSet<>();
982             for (Object filename : p.getParameter(ORIGINAL_FILENAME)) {
983                 final String fn = (String) filename;
984                 if (StringUtils.isNotEmpty(fn) && fn.lastIndexOf('.') > -1) {
985                     final int pos = fn.lastIndexOf('.') + 1;
986                     if (pos < fn.length()) {
987                         final String fext = fn.substring(pos).toLowerCase();
988                         if (fext.length() > 0 && fext.length() <= this.maxFilextLen) {
989                             extensions.add(fext);
990                         }
991                     }
992                 }
993             }
994             if (!extensions.isEmpty()) {
995                 p.setParameter("FILEXT", extensions);
996             }
997         }
998     }

1000     /**
1001      * Process metadata before doing any output
1002      *
1003      * @param payloadList list of items being dropped off that may need initial metadata computations
1004      */
1005     public void processMetadata(final List<IBaseDataObject> payloadList) {
1006 
1007         // Keep track of parent's filetype to output
1008         // relies on the attachments being sorted
1009         final Map<String, String> parentTypes = new HashMap<String, String>();
1010         final IBaseDataObject tld = payloadList.get(0);
1011         final List<String> extended_filetypes = new ArrayList<>();
1012         parentTypes.put("1", tld.getFileType());
1013         for (int i = 0; i < parentParams.size(); i++) {
1014             final String param = parentParams.get(i);
1015             if (tld.hasParameter(param)) {
1016                 parentTypes.put("1" + param, tld.getStringParameter(param));
1017             }
1018         }
1019 
1020         for (final IBaseDataObject p : payloadList) {
1021             final int level = StringUtils.countMatches(p.shortName(), emissary.core.Family.SEP) + 1;
1022             // save specified metadata items for children to grab
1023             parentTypes.put("" + level, p.getFileType());
1024 
1025             extractUniqueFileExtensions(p);
1026 
1027             if (p.getStringParameter("EXTENDED_FILETYPE") == null) {
1028                 extended_filetypes.clear();
1029                 for (final Map.Entry<String, Collection<Object>> entry : p.getParameters().entrySet()) {
1030                     final String key = entry.getKey();
1031                     if (key != null && key.endsWith("_FILETYPE")) {
1032                         for (final Object value : entry.getValue()) {
1033                             final String vs = value.toString();
1034                             if (!extended_filetypes.contains(vs)) {
1035                                 extended_filetypes.add(vs);
1036                             }
1037                         }
1038                     }
1039                 }
1040                 if (!extended_filetypes.isEmpty()) {
1041                     final StringBuilder extft = new StringBuilder(getFileType(p));
1042                     for (int j = 0; j < extended_filetypes.size(); j++) {
1043                         final String s = extended_filetypes.get(j);
1044                         extft.append("//").append(s);
1045                     }
1046                     p.setParameter("EXTENDED_FILETYPE", extft.toString());
1047                 }
1048             }
1049 
1050             for (int j = 0; j < parentParams.size(); j++) {
1051                 final String param = parentParams.get(j);
1052                 if (p.hasParameter(param)) {
1053                     parentTypes.put("" + level + param, p.getStringParameter(param));
1054                 } else {
1055                     // Must clear to keep my children from getting their uncle's value
1056                     parentTypes.remove("" + level + param);
1057                 }
1058 
1059             }
1060 
1061             if (level > 1) {
1062                 // then it might want some PARENT info
1063 
1064                 final int parentLevel = level - 1;
1065                 final String pType = parentTypes.get("" + parentLevel);
1066                 if (pType != null && pType.length() > 0) {
1067                     p.setParameter("PARENT_FILETYPE", pType);
1068                 } else {
1069                     p.setParameter("PARENT_FILETYPE", parentTypes.get("1"));
1070                 }
1071                 for (int j = 0; j < parentParams.size(); j++) {
1072                     final String param = parentParams.get(j);
1073                     int plvl = parentLevel;
1074                     while (plvl > 1 && !parentTypes.containsKey("" + plvl + param)) {
1075                         plvl--;
1076                     }
1077                     if (StringUtils.isNotBlank(parentTypes.get(plvl + param))) {
1078                         p.setParameter("PARENT_" + param, parentTypes.get(plvl + param));
1079                     }
1080                 }
1081             }
1082 
1083             if (p.hasExtractedRecords()) {
1084                 final List<IBaseDataObject> childObjList = p.getExtractedRecords();
1085                 Collections.sort(childObjList, new ShortNameComparator());
1086                 for (final IBaseDataObject child : childObjList) {
1087                     final int parentLevel = StringUtils.countMatches(child.shortName(), emissary.core.Family.SEP);
1088                     final String parentFileType = parentTypes.get("" + parentLevel);
1089                     if (parentFileType != null) {
1090                         child.setParameter("PARENT_FILETYPE", parentFileType);
1091                     }
1092                     for (int k = 0; k < parentParams.size(); k++) {
1093                         final String param = parentParams.get(k);
1094                         int plvl = parentLevel;
1095                         while (plvl > 1 && !parentTypes.containsKey("" + plvl + param)) {
1096                             plvl--;
1097                         }
1098                         if (StringUtils.isNotBlank(parentTypes.get(plvl + param))) {
1099                             child.setParameter("PARENT_" + param, parentTypes.get(plvl + param));
1100                         }
1101                     }
1102                 }
1103             }
1104         }
1105     }

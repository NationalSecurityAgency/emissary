1   package emissary.output.io;

3   import emissary.directory.EmissaryNode;
4   import emissary.util.io.FileNameGenerator;
5   
6   import org.apache.commons.lang3.StringUtils;
7   
8   import java.time.LocalDateTime;
9   import java.time.ZoneId;
10  import java.time.format.DateTimeFormatter;
11  import java.util.UUID;
12  
13  /**
14   * Creates a Filename Generator that contains a timestamp (julian day), uuid, node name/host, and filter name. Example
15   * filename with a filter param of json is: 20232231650_7959b045-d895-4b34-bed2-8800b5071dcd_localhost_json
16   */
17  public class DateFilterFilenameGenerator implements FileNameGenerator {
18  
19      protected static final DateTimeFormatter DATE_PATTERN = DateTimeFormatter.ofPattern("yyyyDDDHHmm");
20      public static final char DELIMITER = '_';
21      public static final char DASH = '-';
22      private final String filterNamePart;
23      private static final String NODE_NAME = System.getProperty(EmissaryNode.NODE_NAME_PROPERTY);
24  
25      /**
26       * Create a file name generator that contains date, uuid, node name, filter extension.
27       * 
28       * @param filterName filter name used to create file extension
29       */
30      public DateFilterFilenameGenerator(String filterName) {
31          this.filterNamePart = (StringUtils.isNotBlank(filterName) ? DELIMITER + filterName.replace(DELIMITER, DASH) : StringUtils.EMPTY);
32      }
33  
34      /**
35       *
36       * @return the next unique filename from this generator
37       */
38      @Override
39      public String nextFileName() {
40          return createFileName(filterNamePart);
41      }
42  
43      public static String createFileName(String extension) {
44          return String.format("%s%s%s%s%s%s", now(), DELIMITER, UUID.randomUUID(), DELIMITER, NODE_NAME, extension);
45      }
46  
47      private static String now() {
48          return DATE_PATTERN.format(LocalDateTime.now(ZoneId.systemDefault()));
49      }
50  
51  }

132     @SuppressWarnings("SystemOut")
133     public static void main(final String[] args) throws IOException, EmissaryException {
134 
135 
136         //
137         // Evaluate arguments to the static main
138         //
139         // Need config path and startup config file on command line
140         if (args.length < 1 || args.length > 3) {
141             System.err.println("Usage: java emissary.admin.Startup " + "[-start|-add|-delete] [config_path] config_file");
142             return;
143         }
144 
145         final String startupConfigFile;
146         if (args.length == 1) {
147             directoryAction = setAction(ACTIONSTART);
148             if (args[0].startsWith("/") || args[0].toUpperCase(Locale.getDefault()).startsWith("HTTP")) {
149                 startupConfigFile = args[0];
150             } else {
151                 startupConfigFile = ConfigUtil.getConfigFile(args[0]);
152             }
153         } else if (args.length == 2) {
154             directoryAction = setAction(ACTIONSTART);
155             startupConfigFile = makeConfig(args[0], args[1]);
156         } else {
157             directoryAction = setAction(args[0]);
158             startupConfigFile = makeConfig(args[1], args[2]);
159         }
160 
161         final Startup start = new Startup(startupConfigFile, new EmissaryNode());
162         start.start();
163 
164         logger.info("The system is up and running fine. All ahead Warp-7.");
165     }

168     /**
169      * Start the system
170      */
171     public void start() throws EmissaryException {
172         final boolean bootStatus = bootstrap();
173 
174         if (!bootStatus) {
175             throw new EmissaryException("Unable to bootstrap the system");
176         }
177 
178         // bootstrap now only starts the processing places (this allows
179         // derived classes to hold off starting the pickup places until
180         // they've completed their own set-up). So we have to startup
181         // the pickup places here.
182         startPickUpPlaces();
183 
184         if (!verifyNoInvisiblePlacesStarted() && node.isStrictStartupMode()) {
185             invisPlacesStartedInStrictMode = true;
186         }
187     }

209     public boolean bootstrap() {
210 
211         //
212         // Setup the Local Directories in a hashtable
213         //
214         final boolean status = localDirectorySetup(this.localDirectories);
215 
216         if (!status) {
217             logger.warn("Startup: local directory setup failed.");
218             return false;
219         }
220 
221         //
222         // Set up the rest of the Places except no pickups
223         //
224         sortPlaces(this.hostsConfig.findEntries("PLACE"));
225 
226         logger.info("Ready to start {} place(s) and {} PickUp place(s).", hashListSize(this.placeLists), hashListSize(this.pickupLists));
227 
228         logger.info("Processing non-pickup places...");
229         startMapOfPlaces(this.placeLists);
230 
231         //
232         // Wait for all places to get started and registered
233         //
234         this.stopAndWaitForPlaceCreation();
235 
236         logger.debug("Done with bootstrap phase");
237         return true;
238     }

256     void startMapOfPlaces(final Map<String, Set<String>> m) {
257 
258         if (hashListSize(m) > 0) {
259             for (final Set<String> placeList : m.values()) {
260                 final boolean status = placeSetup(directoryAction, this.localDirectories, places, placeList);
261 
262                 if (!status) {
263                     logger.warn("Startup: places setup failed!");
264                     return;
265                 }
266             }
267 
268         }
269 
270         logger.debug("done with map of {} places", hashListSize(m));
271     }

273     /**
274      * Count all entries in lists of a map
275      */
276     private static int hashListSize(@Nullable final Map<String, Set<String>> m) {
277         int total = 0;
278         if (m != null) {
279             for (final Set<String> l : m.values()) {
280                 if (l != null) {
281                     total += l.size();
282                 }
283             }
284         }
285         return total;

288     protected boolean localDirectorySetup(final Map<String, String> localDirectoriesArg) {
289 
290         final List<String> hostParameters = this.hostsConfig.findEntries("LOCAL_DIRECTORY");
291 
292         final long start = System.currentTimeMillis();
293         final Map<String, String> dirStarts = new HashMap<>();
294         EmissaryNode emissaryNode = EmissaryServer.getInstance().getNode();
295         for (final String thePlaceLocation : hostParameters) {
296 
297             final String host = placeHost(thePlaceLocation);
298 
299             if (KeyManipulator.isLocalTo(thePlaceLocation, "http://" + this.node.getNodeName() + ":" + this.node.getNodePort() + "/StartupEngine")) {
300                 final String thePlaceClassStr = PlaceStarter.getClassString(thePlaceLocation);
301                 if (logger.isInfoEnabled()) {
302                     logger.info("Doing local startup for directory {}({}) ", getLocationName(thePlaceLocation), thePlaceClassStr);
303                 }
304                 final IServiceProviderPlace p = PlaceStarter.createPlace(thePlaceLocation, null, thePlaceClassStr, null, emissaryNode);
305                 if (p != null) {
306                     dirStarts.put(host, thePlaceLocation);
307                     localDirectoriesArg.put(host, p.toString());
308                 } else {
309                     localDirectoriesArg.remove(thePlaceLocation);
310                     logger.warn("Giving up on directory {}", thePlaceLocation);
311                 }
312             } else {
313                 logger.warn("Directory location is not local: {}", thePlaceLocation);
314             }
315         }
316 
317         // All local directories must be up before proceeding
318         logger.debug("Waiting for all local directories to start, expecting {}", dirStarts.size());
319         int prevCount = 0;
320         while (localDirectoriesArg.size() + this.failedLocalDirectories.size() < dirStarts.size()) {
321             final int newCount = localDirectoriesArg.size() + this.failedLocalDirectories.size();
322             if (newCount > prevCount && newCount < dirStarts.size()) {
323                 logger.info("Completed {} of {} local directories", localDirectoriesArg.size(), dirStarts.size());
324                 prevCount = newCount;
325             }
326 
327             try {
328                 Thread.sleep(10);
329             } catch (InterruptedException e) {
330                 Thread.currentThread().interrupt();
331             }
332         }
333 
334         if (logger.isDebugEnabled()) {
335             logger.debug("Directories all up in {}s", (System.currentTimeMillis() - start) / 1000.0);
336         }
337 
338         return true;
339     }

341     /**
342      * Start all places on the list on a thread, return control immediately. All places in hostParameters list must be for
343      * the same host:port!
344      */
345     protected boolean placeSetup(final int directoryActionArg, final Map<String, String> localDirectoriesArg, final Map<String, String> placesArg,
346             final Set<String> hostParameters) {
347 
348         // Track how many places we are trying to start
349         this.placesToStart.addAll(hostParameters);
350 
351         final Thread t = new Thread(() -> {
352 
353             final String thePlaceHost = placeHost(hostParameters.stream().findFirst().get());
354 
355             final String localDirectory = localDirectoriesArg.get(thePlaceHost);
356 
357             if (localDirectory == null) {
358                 hostParameters.forEach(placesToStart::remove);
359                 if (failedLocalDirectories.get(thePlaceHost) != null) {
360                     logger.warn("Skipping {} due to previously failed directory", thePlaceHost);
361                 } else {
362                     logger.warn("Skipping {} due to local Directory not found", thePlaceHost);
363                 }
364                 return;
365             }
366 
367             if (directoryActionArg != DIRECTORYSTART && directoryActionArg != DIRECTORYADD) {
368                 hostParameters.forEach(placesToStart::remove);
369                 return;
370             }
371 
372             logger.debug("Using localDir={} to create {} places on {}", localDirectory, hostParameters.size(), thePlaceHost);
373 
374             // Create a stream of places that can be configured to start in parallel
375             boolean parallelPlaceStartup = hostsConfig.findBooleanEntry(PARALLEL_PLACE_STARTUP_CONFIG, false);
376             Stream<String> hostParametersStream = StreamSupport.stream(hostParameters.spliterator(), parallelPlaceStartup);
377             logger.info("Using parallel place startup: {}", hostParametersStream.isParallel());
378 
379             // Start everything in hostParameters
380             // (PLACE lines from cfg file for a given host
381             hostParametersStream.forEach(thePlaceLocation -> {
382                 placeName(thePlaceLocation);
383 
384                 // Get the class name and Class object for what we want to make
385                 final String thePlaceLocName = getLocationName(thePlaceLocation);
386                 final String thePlaceClassString = PlaceStarter.getClassString(thePlaceLocation);
387                 StringBuilder startupBuilder =
388                         new StringBuilder("Doing local startup on ")
389                                 .append(thePlaceLocName)
390                                 .append("(")
391                                 .append(thePlaceClassString).append(")...");
392                 if (thePlaceClassString == null) {
393                     startupBuilder.append("skipping, no class string!!");
394                     placesToStart.remove(thePlaceLocation);
395                     logger.warn(startupBuilder.toString());
396                     return;
397                 }
398                 logger.debug("Starting place {}", thePlaceLocation);
399                 if (KeyManipulator.isLocalTo(thePlaceLocation, String.format("http://%s:%s/StartupEngine", node.getNodeName(), node.getNodePort()))) {
400                     if (directoryActionArg == DIRECTORYADD && Namespace.exists(thePlaceLocation)) {
401                         // logger.info("Local place already exists: {}", thePlaceLocation);
402                         startupBuilder.append("local place already exists");
403                         placesToStart.remove(thePlaceLocation);
404                         // add place to placeAlreadyStarted list, so can be verified in verifyNoInvisibleStartPlaces
405                         placeAlreadyStarted.add(thePlaceLocation.substring(thePlaceLocation.lastIndexOf("/") + 1));
406                         logger.info(startupBuilder.toString());
407                         return;
408                     }
409 
410                     final IServiceProviderPlace p = PlaceStarter.createPlace(thePlaceLocation, null, thePlaceClassString, localDirectory);
411                     if (p != null) {
412                         placesArg.put(thePlaceLocation, thePlaceLocation);
413                         startupBuilder.append("done!");
414                         logger.info(startupBuilder.toString());
415                     } else {
416                         // logger.error("{} failed to start!", thePlaceLocation);
417                         failedPlaces.add(thePlaceLocation);
418                         placesToStart.remove(thePlaceLocation);
419                         startupBuilder.append("FAILED!!");
420                         logger.error(startupBuilder.toString());
421                     }
422                 }
423             });
424         });
425         t.start();
426         return true;
427     }

429     /**
430      * Check to see if all the places have started and been registered in the directory. This doesn't account for
431      * directories, just things started with a "PLACE" tag
432      */
433     protected void stopAndWaitForPlaceCreation() {
434         int numPlacesExpected = this.placesToStart.size();
435         int numPlacesFound;
436         int numPlacesFoundPreviously = 0;
437 
438         logger.info("Waiting for {} places to start {}", placesToStart.size(),
439                 placesToStart.stream().map(s -> StringUtils.substringAfterLast(s, "/")).sorted().collect(Collectors.toList()));
440         do {
441             if (this.placesToStart.size() != numPlacesExpected) {
442                 logger.info("Now waiting for {} places to start. (originally {} places)", this.placesToStart.size(), numPlacesExpected);
443                 numPlacesExpected = this.placesToStart.size();
444             }
445 
446             numPlacesFound = places.size();
447 
448             if (numPlacesFound >= numPlacesExpected) {
449                 boolean failedPlaceStartups = false;
450 
451                 if (!this.failedPlaces.isEmpty()) {
452                     failedPlaceStartups = true;
453                     String failedPlaceList = String.join("; ", this.failedPlaces);
454                     logger.warn("The following places have failed to start: {}", failedPlaceList);
455                 }
456                 if (!CoordinationPlace.getFailedCoordinationPlaces().isEmpty()) {
457                     failedPlaceStartups = true;
458                     String failedCoordPlaceList = String.join("; ", CoordinationPlace.getFailedCoordinationPlaces());
459                     logger.warn("The following coordination places have failed to start: {}", failedCoordPlaceList);
460                 }
461 
462                 // check if strict startup & places/coordination places failed, if yes, shut down server
463                 if (this.node.isStrictStartupMode() && failedPlaceStartups) {
464                     logger.error("Server failed to start due to Strict mode being enabled.  To disable strict mode, " +
465                             "run server start command without the --strict flag");
466                     logger.error("Server shutting down");
467                     System.exit(1);
468                 }
469 
470                 // normal termination of the loop
471                 logger.debug("Woohoo! {} of {} places are up and running.", numPlacesFound, numPlacesExpected);
472                 break;
473             }
474 
475             try {
476                 Thread.sleep(500);
477             } catch (InterruptedException e) {
478                 Thread.currentThread().interrupt();
479             }
480 
481             if (numPlacesFound != numPlacesFoundPreviously) {
482                 numPlacesFoundPreviously = numPlacesFound;
483 
484                 final float percentageUp = (float) numPlacesFound / (float) numPlacesExpected;
485                 final String leadString;
486                 if (percentageUp < 0.20) {
487                     leadString = "Hmmm... only ";
488                 } else if (percentageUp < 0.40) {
489                     leadString = "Ok, now ";
490                 } else if (percentageUp < 0.60) {
491                     leadString = "Making progress, ";
492                 } else if (percentageUp < 0.80) {
493                     leadString = "Over half way there! ";
494                 } else if (percentageUp < 0.95) {
495                     leadString = "Almost ready! ";
496                 } else if (numPlacesFound + 1 == numPlacesExpected) {
497                     leadString = "One more to go... ";
498                 } else {
499                     leadString = "Yeah! ";
500                 }
501 
502                 logger.debug("{}{} of {} places are up and running.", leadString, numPlacesFound, numPlacesExpected);
503             }
504 
505         } while (true); // break terminated loop
506     }

508     /**
509      * sort all the PLACE entries into either a processing place or a pickup place
510      */
511     protected void sortPlaces(final List<String> placeList) {
512 
513         for (final String location : placeList) {
514             final String className = PlaceStarter.getClassString(location);
515             if (className == null) {
516                 continue;
517             }
518 
519             try {
520                 sortPickupOrPlace(location, PickUpPlace.implementsPickUpPlace(Class.forName(className)) ? this.pickupLists : this.placeLists);
521             } catch (ClassNotFoundException e) {
522                 logger.error("Could not create place {}", className, e);
523             }
524         }
525     }

527     private static void sortPickupOrPlace(String theLocation, Map<String, Set<String>> placeList) {
528         final String host = placeHost(theLocation);
529         Set<String> l = placeList.computeIfAbsent(host, k -> new LinkedHashSet<>());
530         if (l.contains(theLocation)) {
531             logger.warn("Sorting places found duplicate {}({}), skipping!", getLocationName(theLocation), PlaceStarter.getClassString(theLocation));
532         } else {
533             l.add(theLocation);
534         }
535     }

541     /**
542      * Verifies the active directory places vs places started up. Log if any places are started without being announced in
543      * start-up.
544      *
545      * @return true if no invisible places started, false if yes
546      */
547     public static boolean verifyNoInvisiblePlacesStarted() {
548         try {
549             IDirectoryPlace dirPlace = DirectoryPlace.lookup();
550             List<DirectoryEntry> dirEntries = dirPlace.getEntries();
551             for (DirectoryEntry entry : dirEntries) {
552                 // add place names of active places. getLocalPlace() returns null for any place that failed to start
553                 if (entry.getLocalPlace() != null) {
554                     activeDirPlaces.add(entry.getLocalPlace().getPlaceName());
555                 }
556             }
557 
558             // remove DirectoryPlace from activeDirPlaces. DirectoryPlace is started up automatically in order to
559             // start all other places, so it isn't per se "announced", but it is known and logged
560             activeDirPlaces.removeIf(dir -> dir.equalsIgnoreCase("DirectoryPlace"));
561         } catch (EmissaryException e) {
562             throw new EmissaryRuntimeException(e);
563         }
564 
565         // compares place names in active dirs and active places, removes them from set if found
566         for (String thePlaceLocation : places.values()) {
567             activeDirPlaces.removeIf(dir -> dir.equalsIgnoreCase(placeName(thePlaceLocation)));
568         }
569 
570         // places that are attempted to startup but are already up are added to separate list
571         // this will only check if places are added to that list
572         if (!placeAlreadyStarted.isEmpty()) {
573             for (String thePlaceLocation : placeAlreadyStarted) {
574                 activeDirPlaces.removeIf(dir -> dir.equalsIgnoreCase(thePlaceLocation));
575             }
576         }
577 
578         // if any places are left in active dir keys, they are places not announced on startup
579         if (!activeDirPlaces.isEmpty()) {
580             logger.warn("{} place(s) started up without being announced! {}", activeDirPlaces.size(), activeDirPlaces);
581             return false;
582         }
583 
584         return true;
585     }

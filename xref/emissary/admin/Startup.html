160     /**
161      * Start the system
162      */
163     public void start() throws EmissaryException {
164         final boolean bootStatus = bootstrap();
165 
166         if (!bootStatus) {
167             throw new EmissaryException("Unable to bootstrap the system");
168         }
169 
170         // bootstrap now only starts the processing places (this allows
171         // derived classes to hold off starting the pickup places until
172         // they've completed their own set-up). So we have to startup
173         // the pickup places here.
174         startPickUpPlaces();
175 
176         if (!verifyNoInvisiblePlacesStarted() && node.isStrictStartupMode()) {
177             invisPlacesStartedInStrictMode = true;
178         }
179     }

201     public boolean bootstrap() {
202 
203         //
204         // Setup the Local Directories in a hashtable
205         //
206         final boolean status = localDirectorySetup(this.localDirectories);
207 
208         if (!status) {
209             logger.warn("Startup: local directory setup failed.");
210             return false;
211         }
212 
213         //
214         // Setup the rest of the Places except no pickups
215         //
216         sortPlaces(this.hostsConfig.findEntries("PLACE"));
217 
218         logger.info("Ready to start {}  places and {} PickUp places.", hashListSize(this.placeLists), hashListSize(this.pickupLists));
219 
220         logger.info("Processing non-pickup places...");
221         startMapOfPlaces(this.placeLists);
222 
223         //
224         // Wait for all places to get started and registered
225         //
226         this.stopAndWaitForPlaceCreation();
227 
228         logger.info("Done with bootstrap phase");
229         return true;
230     }

248     void startMapOfPlaces(final Map<String, Set<String>> m) {
249 
250         if (hashListSize(m) > 0) {
251             for (final Set<String> placeList : m.values()) {
252                 final boolean status = placeSetup(directoryAction, this.localDirectories, places, placeList);
253 
254                 if (!status) {
255                     logger.warn("Startup: places setup failed!");
256                     return;
257                 }
258             }
259 
260         }
261 
262         logger.info("done with map of {} places", hashListSize(m));
263     }

265     /**
266      * Count all entries in lists of a map
267      */
268     private int hashListSize(@Nullable final Map<String, Set<String>> m) {
269         int total = 0;
270         if (m != null) {
271             for (final Set<String> l : m.values()) {
272                 if (l != null) {
273                     total += l.size();
274                 }
275             }
276         }
277         return total;

280     protected boolean localDirectorySetup(final Map<String, String> localDirectoriesArg) {
281 
282         final List<String> hostParameters = this.hostsConfig.findEntries("LOCAL_DIRECTORY");
283 
284         final long start = System.currentTimeMillis();
285         final Map<String, String> dirStarts = new HashMap<>();
286         EmissaryNodeode">EmissaryNode emissaryNode = new EmissaryNode();
287         for (final String thePlaceLocation : hostParameters) {
288 
289             final String host = placeHost(thePlaceLocation);
290 
291             if (KeyManipulator.isLocalTo(thePlaceLocation, "http://" + this.node.getNodeName() + ":" + this.node.getNodePort() + "/StartupEngine")) {
292                 logger.info("Doing local startup for directory {} ", thePlaceLocation);
293                 final String thePlaceClassStr = PlaceStarter.getClassString(thePlaceLocation);
294                 final IServiceProviderPlace p = PlaceStarter.createPlace(thePlaceLocation, null, thePlaceClassStr, null, emissaryNode);
295                 if (p != null) {
296                     dirStarts.put(host, thePlaceLocation);
297                     localDirectoriesArg.put(host, p.toString());
298                 } else {
299                     localDirectoriesArg.remove(thePlaceLocation);
300                     logger.warn("Giving up on directory {}", thePlaceLocation);
301                 }
302             } else {
303                 logger.warn("Directory location is not local: {}", thePlaceLocation);
304             }
305         }
306 
307         // All local directories must be up before proceeding
308         logger.debug("Waiting for all local directories to start, expecting {}", dirStarts.size());
309         int prevCount = 0;
310         while (localDirectoriesArg.size() + this.failedLocalDirectories.size() < dirStarts.size()) {
311             final int newCount = localDirectoriesArg.size() + this.failedLocalDirectories.size();
312             if (newCount > prevCount && newCount < dirStarts.size()) {
313                 logger.info("Completed {} of {} local directories", localDirectoriesArg.size(), dirStarts.size());
314                 prevCount = newCount;
315             }
316 
317             try {
318                 Thread.sleep(10);
319             } catch (InterruptedException e) {
320                 Thread.currentThread().interrupt();
321             }
322         }
323 
324         if (logger.isInfoEnabled()) {
325             logger.info("Directories all up in {}s", (System.currentTimeMillis() - start) / 1000.0);
326         }
327 
328         return true;
329     }

331     /**
332      * Start all places on the list on a thread, return control immediately. All places in hostParameters list must be for
333      * the same host:port!
334      */
335     protected boolean placeSetup(final int directoryActionArg, final Map<String, String> localDirectoriesArg, final Map<String, String> placesArg,
336             final Set<String> hostParameters) {
337 
338         // Track how many places we are trying to start
339         this.placesToStart.addAll(hostParameters);
340 
341         final Thread t = new Thread(() -> {
342 
343             final String thePlaceHost = placeHost(hostParameters.stream().findFirst().get());
344 
345             final String localDirectory = localDirectoriesArg.get(thePlaceHost);
346 
347             if (localDirectory == null) {
348                 hostParameters.forEach(Startup.this.placesToStart::remove);
349                 if (Startup.this.failedLocalDirectories.get(thePlaceHost) != null) {
350                     logger.warn("Skipping {} due to previously failed directory", thePlaceHost);
351                 } else {
352                     logger.warn("Skipping {} : local Directory not found", thePlaceHost);
353                 }
354                 return;
355             }
356 
357             if (directoryActionArg != DIRECTORYSTART && directoryActionArg != DIRECTORYADD) {
358                 hostParameters.forEach(Startup.this.placesToStart::remove);
359                 return;
360             }
361 
362             logger.info("Using localDir={} to create {} places on {}", localDirectory, hostParameters.size(), thePlaceHost);
363 
364             // Create a stream of places that can be configured to start in parallel
365             boolean parallelPlaceStartup = hostsConfig.findBooleanEntry(PARALLEL_PLACE_STARTUP_CONFIG, false);
366             Stream<String> hostParametersStream = StreamSupport.stream(hostParameters.spliterator(), parallelPlaceStartup);
367             logger.info("Parallel place startup: {}", hostParametersStream.isParallel());
368 
369             // Start everything in hostParameters
370             // (PLACE lines from cfg file for a given host
371             hostParametersStream.forEach(thePlaceLocation -> {
372                 placeName(thePlaceLocation);
373 
374                 // Get the class name and Class object for what we want to make
375                 final String thePlaceClassString = PlaceStarter.getClassString(thePlaceLocation);
376                 if (thePlaceClassString == null) {
377                     logger.warn("Skipping {}, no class string", thePlaceLocation);
378                     Startup.this.placesToStart.remove(thePlaceLocation);
379                     return;
380                 }
381                 logger.debug("Starting place {}", thePlaceLocation);
382                 if (KeyManipulator.isLocalTo(thePlaceLocation,
383                         String.format("http://%s:%s/StartupEngine", Startup.this.node.getNodeName(), Startup.this.node.getNodePort()))) {
384                     if (directoryActionArg == DIRECTORYADD && Namespace.exists(thePlaceLocation)) {
385                         logger.info("Local place already exists: {}", thePlaceLocation);
386                         Startup.this.placesToStart.remove(thePlaceLocation);
387                         // add place to placeAlreadyStarted list, so can be verified in verifyNoInvisibleStartPlaces
388                         placeAlreadyStarted.add(thePlaceLocation.substring(thePlaceLocation.lastIndexOf("/") + 1));
389                         return;
390                     }
391 
392                     logger.info("Doing local startup on place {}", thePlaceLocation);
393                     final String thePlaceClassStr = PlaceStarter.getClassString(thePlaceLocation);
394                     final IServiceProviderPlace p =
395                             PlaceStarter.createPlace(thePlaceLocation, null, thePlaceClassStr, localDirectory);
396                     if (p != null) {
397                         placesArg.put(thePlaceLocation, thePlaceLocation);
398                     } else {
399                         logger.error("{} failed to start!", thePlaceLocation);
400                         Startup.this.failedPlaces.add(thePlaceLocation);
401                         Startup.this.placesToStart.remove(thePlaceLocation);
402                     }
403                 }
404             });
405         });
406         t.start();
407         return true;
408     }

410     /**
411      * Check to see if all the places have started and been registered in the directory. This doesn't account for
412      * directories, just things started with a "PLACE" tag
413      */
414     protected void stopAndWaitForPlaceCreation() {
415         int numPlacesExpected = this.placesToStart.size();
416         int numPlacesFound;
417         int numPlacesFoundPreviously = 0;
418 
419         logger.info("Waiting for {} places to start.", numPlacesExpected);
420         do {
421             if (this.placesToStart.size() != numPlacesExpected) {
422                 logger.info("NOW Waiting for {} places to start. (originally {} places)", this.placesToStart.size(), numPlacesExpected);
423                 numPlacesExpected = this.placesToStart.size();
424             }
425 
426             numPlacesFound = places.size();
427 
428             if (numPlacesFound >= numPlacesExpected) {
429                 boolean failedPlaceStartups = false;
430 
431                 if (!this.failedPlaces.isEmpty()) {
432                     failedPlaceStartups = true;
433                     String failedPlaceList = String.join("; ", this.failedPlaces);
434                     logger.warn("The following places have failed to start: {}", failedPlaceList);
435                 }
436                 if (!CoordinationPlace.getFailedCoordinationPlaces().isEmpty()) {
437                     failedPlaceStartups = true;
438                     String failedCoordPlaceList = String.join("; ", CoordinationPlace.getFailedCoordinationPlaces());
439                     logger.warn("The following coordination places have failed to start: {}", failedCoordPlaceList);
440                 }
441 
442                 // check if strict startup & places/coordination places failed, if yes, shut down server
443                 if (this.node.isStrictStartupMode() && failedPlaceStartups) {
444                     logger.error(
445                             "Server failed to start due to Strict mode being enabled.  To disable strict mode, " +
446                                     "run server start command without the --strict flag");
447                     logger.error("Server shutting down");
448                     System.exit(1);
449                 }
450 
451                 // normal termination of the loop
452                 logger.info("Woohoo! {} of {} places are up and running.", numPlacesFound, numPlacesExpected);
453                 break;
454             }
455 
456             try {
457                 Thread.sleep(500);
458             } catch (InterruptedException e) {
459                 Thread.currentThread().interrupt();
460             }
461 
462             if (numPlacesFound != numPlacesFoundPreviously) {
463                 numPlacesFoundPreviously = numPlacesFound;
464 
465                 final float percentageUp = (float) numPlacesFound / (float) numPlacesExpected;
466                 final String leadString;
467                 if (percentageUp < 0.20) {
468                     leadString = "Hmmm... only ";
469                 } else if (percentageUp < 0.40) {
470                     leadString = "Ok, now ";
471                 } else if (percentageUp < 0.60) {
472                     leadString = "Making progress, ";
473                 } else if (percentageUp < 0.80) {
474                     leadString = "Over half way there! ";
475                 } else if (percentageUp < 0.95) {
476                     leadString = "Almost ready! ";
477                 } else if (numPlacesFound + 1 == numPlacesExpected) {
478                     leadString = "One more to go... ";
479                 } else {
480                     leadString = "Yeah! ";
481                 }
482 
483                 logger.info("{}{} of {} places are up and running.", leadString, numPlacesFound, numPlacesExpected);
484             }
485 
486         } while (true); // break terminated loop
487     }

489     /**
490      * sort all the PLACE entries into either a processing place or a pickup place
491      */
492     protected void sortPlaces(final List<String> allPlaces) {
493 
494         for (final String theLocation : allPlaces) {
495             Startup.placeName(theLocation);
496             final String className = PlaceStarter.getClassString(theLocation);
497             if (className == null) {
498                 continue;
499             }
500 
501 
502             try {
503                 if (PickUpPlace.implementsPickUpPlace(Class.forName(className))) {
504                     sortPickupOrPlace(theLocation, this.pickupLists);
505                 } else {
506                     sortPickupOrPlace(theLocation, this.placeLists);
507                 }
508             } catch (ClassNotFoundException e) {
509                 logger.error("Could not create place {}", className, e);
510             }
511         }
512     }

514     private void sortPickupOrPlace(String theLocation, Map<String, Set<String>> placeList) {
515         final String host = placeHost(theLocation);
516         Set<String> l = placeList.computeIfAbsent(host, k -> new LinkedHashSet<>());
517         l.add(theLocation);
518     }

520     /**
521      * Verifies the active directory places vs places started up. Log if any places are started without being announced in
522      * start-up.
523      * 
524      * @return true if no invisible places started, false if yes
525      */
526     public static boolean verifyNoInvisiblePlacesStarted() {
527         try {
528             IDirectoryPlace dirPlace = DirectoryPlace.lookup();
529             List<DirectoryEntry> dirEntries = dirPlace.getEntries();
530             for (DirectoryEntry entry : dirEntries) {
531                 // add place names of active places. getLocalPlace() returns null for any place that failed to start
532                 if (entry.getLocalPlace() != null) {
533                     activeDirPlaces.add(entry.getLocalPlace().getPlaceName());
534                 }
535             }
536 
537             // remove DirectoryPlace from activeDirPlaces. DirectoryPlace is started up automatically in order to
538             // start all other places, so it isn't per se "announced", but it is known and logged
539             activeDirPlaces.removeIf(dir -> dir.equalsIgnoreCase("DirectoryPlace"));
540         } catch (EmissaryException e) {
541             throw new RuntimeException(e);
542         }
543 
544         // compares place names in active dirs and active places, removes them from set if found
545         for (String thePlaceLocation : places.values()) {
546             activeDirPlaces.removeIf(dir -> dir.equalsIgnoreCase(placeName(thePlaceLocation)));
547         }
548 
549         // places that are attempted to startup but are already up are added to separate list
550         // this will only check if places are added to that list
551         if (!placeAlreadyStarted.isEmpty()) {
552             for (String thePlaceLocation : placeAlreadyStarted) {
553                 activeDirPlaces.removeIf(dir -> dir.equalsIgnoreCase(thePlaceLocation));
554             }
555         }
556 
557         // if any places are left in active dir keys, they are places not announced on startup
558         if (!activeDirPlaces.isEmpty()) {
559             logger.warn("{} place(s) started up without being announced! {}", activeDirPlaces.size(), activeDirPlaces);
560             return false;
561         }
562 
563         return true;
564     }

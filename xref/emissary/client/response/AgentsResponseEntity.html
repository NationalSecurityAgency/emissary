15  @XmlRootElement(name = "agents")

35          }
36      }

40      }

48      }
49  
50      @Override
51      public void append(BaseEntity e) {
52          addErrors(e.getErrors());
53          if (e instanceof AgentsResponseEntity) {
54              AgentsResponseEntity pre = this.getClass().cast(e);
55              addClusterAgents(pre.getLocal());
56          }
57      }
58  
59      @Override
60      public void dumpToConsole() {
61          Stream.concat(cluster.stream(), Stream.of(local)).filter(agentList -> agentList.getHost() != null).collect(Collectors.toSet()).stream()
62                  .sorted(Comparator.comparing(AgentList::getHost)).forEach(agentList -> {
63                      if (!agentList.getAgents().isEmpty()) {
64                          agentList.dumpToConsole();
65                      }
66                  });
67      }

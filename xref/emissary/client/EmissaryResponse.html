23  @SuppressWarnings("AvoidObjectArrays")
24  public class EmissaryResponse {
25  
26      private static final Logger logger = LoggerFactory.getLogger(EmissaryResponse.class);
27  
28      final int status;
29      @Nullable
30      final Object content;
31      final String contentType;
32      final Header[] headers;
33      final ObjectMapper objectMapper = new ObjectMapper();
34  
35      public EmissaryResponse(ClassicHttpResponse response) {
36          int tempStatus = response.getCode();
37          String tempContent;
38          headers = response.getHeaders();
39          Header[] contentHeaders = response.getHeaders(HttpHeaders.CONTENT_TYPE);
40          if (contentHeaders.length > 0) {
41              contentType = contentHeaders[0].getValue();
42              if (contentHeaders.length > 1) {
43                  logger.warn("Too many content headers: {}", contentHeaders.length);
44                  if (logger.isDebugEnabled()) {
45                      Arrays.stream(contentHeaders).sequential().forEach(ch -> logger.debug("Header -> {}", ch));
46                  }
47              }
48          } else {
49              logger.debug("No content type header, setting to plain text");
50              contentType = MediaType.TEXT_PLAIN;
51          }
52          try {
53              HttpEntity entity = response.getEntity();
54              if (entity == null) {
55                  logger.debug("No entity");
56                  tempContent = "";
57              } else {
58                  tempContent = IOUtils.toString(entity.getContent(), StandardCharsets.UTF_8);
59              }
60          } catch (UnsupportedOperationException | IOException e) {
61              tempContent = e.getMessage();
62              tempStatus = 500;
63              logger.error("There was an issue generating the response", e);
64          }
65          logger.debug("response was: {} with content: {}", tempStatus, tempContent);
66          status = tempStatus;
67          content = tempContent;
68      }
69  
70      public int getStatus() {
71          return status;
72      }
73  
74      public Header[] getHeaders() {
75          return headers;
76      }
77  
78      @Nullable
79      public String getContentString() {
80          if (content == null) {
81              return null;
82          }
83  
84          if (status == HttpStatus.SC_OK) {
85              return content.toString();
86          } else {
87              return "Bad request -> status: " + status + " message: " + content;
88          }
89      }
90  
91      public <T extends BaseEntity> T getContent(Class<T> mapper) {
92          if (content == null) {
93              return null;
94          }
95          try {
96              if (status == HttpStatus.SC_OK) {
97                  return objectMapper.readValue(content.toString(), mapper);
98              } else {
99                  return makeErrorEntity(content.toString(), mapper);
100             }
101         } catch (IOException e) {
102             logger.error("Error mapping object to {}", mapper.getCanonicalName(), e);
103             return makeErrorEntity(content.toString(), mapper);
104         }
105     }
106 
107     private static <T extends BaseEntity> T makeErrorEntity(String msg, Class<T> mapper) {
108         T r = null;
109         try {
110             Object c = Class.forName(mapper.getName()).getDeclaredConstructor().newInstance();
111             r = mapper.cast(c);
112         } catch (ReflectiveOperationException e) {
113             logger.error("Problem creating new {}", mapper.getName(), e);
114         }
115         if (r != null) {
116             r.addError(msg);
117         }
118         return r;
119     }

123         @Override
124         public EmissaryResponse handleResponse(ClassicHttpResponse response) throws IOException {
125             HttpEntity entity = response.getEntity();
126             EmissaryResponse er = new EmissaryResponse(response);
127             EntityUtils.consume(entity);
128             return er;
129         }

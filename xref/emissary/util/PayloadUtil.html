42      protected static void configure() {
43          Configurator configG = null;
44          try {
45              configG = ConfigUtil.getConfigInfo(PayloadUtil.class);
46          } catch (IOException e) {
47              logger.error("Cannot open default config file", e);
48          }
49  
50          // fill lists with data from PayloadUtil.cfg for transform itinerary/history
51          if (configG != null) {
52              for (String fileType : configG.findEntries(REDUCED_HISTORY)) {
53                  setFileTypeHistoryPreference(fileType, REDUCED_HISTORY);
54              }
55              for (String fileType : configG.findEntries(NO_URL)) {
56                  setFileTypeHistoryPreference(fileType, NO_URL);
57              }
58          }
59      }

61      /**
62       * Check if fileType is already mapped to history preference. If not, set fileType -$gt; preference in map
63       *
64       * @param fileType current fileType pulled from cfg
65       * @param preference preference associated with current cfg fileType
66       */
67      protected static void setFileTypeHistoryPreference(String fileType, String preference) {
68          if (historyPreference.containsKey(fileType)) {
69              if (historyPreference.get(fileType).equals(preference)) {
70                  // log if filetype is assigned in the same preference more than once
71                  logger.warn("FileType {} is assigned to {} in cfg more than once.", fileType, preference);
72              } else {
73                  // log if filetype already has previous preference assignment
74                  logger.warn("FileType {} already has history preference {} assigned. {} will be ignored.", fileType,
75                          historyPreference.get(fileType), preference);
76              }
77          } else {
78              historyPreference.put(fileType, preference);
79          }
80      }

82      /**
83       * Try really hard to get a meaningful name for a payload object
84       */
85      public static String getName(final Object o) {
86          String payloadName = o.getClass().getName();
87          if (o instanceof IBaseDataObject) {
88              payloadName = ((IBaseDataObject) o).shortName();
89          } else if (o instanceof Collection) {
90              final Iterator<?> pi = ((Collection<?>) o).iterator();
91              if (pi.hasNext()) {
92                  payloadName = ((IBaseDataObject) pi.next()).shortName() + "(" + ((Collection<?>) o).size() + ")";
93              }
94          }
95          return payloadName;
96      }

108     /**
109      * Generate a string about the payload object
110      * 
111      * @param payload the payload to describe
112      */
113     public static String getPayloadDisplayString(final IBaseDataObject payload) {
114         final StringBuilder sb = new StringBuilder();
115         final List<TransformHistory.History> th = payload.getTransformHistory().getHistory();
116         final String fileName = payload.getFilename();
117         final String fileType = payload.getFileType();
118         final List<String> currentForms = payload.getAllCurrentForms();
119         final Date creationTimestamp = payload.getCreationTimestamp();
120 
121         sb.append("\n").append("filename: ").append(fileName).append("\n").append("   creationTimestamp: ").append(creationTimestamp).append("\n")
122                 .append("   currentForms: ").append(currentForms).append("\n").append("   filetype: ").append(fileType).append("\n")
123                 .append("   transform history (").append(th.size()).append(") :").append("\n");
124 
125         // transform history output
126         String historyCase = configureHistoryCase(fileType);
127         if (historyCase.equals(REDUCED_HISTORY)) {
128             // found reduced history match, output only dropoff
129             sb.append("   ** reduced transform history **").append("\n").append("     dropOff -> ")
130                     .append(payload.getLastPlaceVisited())
131                     .append("\n");
132         } else {
133             for (final TransformHistory.History h : th) {
134                 sb.append(" ");
135                 if (h.wasCoordinated()) {
136                     sb.append(" ");
137                 }
138                 // check is NO_URL or not
139                 sb.append("    ").append(historyCase.equals(NO_URL) ? h.getKeyNoUrl() : h.getKey()).append("\n");
140             }
141         }
142         return sb.toString();
143     }

145     /**
146      * Check if fileType in PayloadUtil.cfg matches current payload fileType
147      *
148      * @param fileType current payload fileType
149      * @return string for output case
150      */
151     public static String configureHistoryCase(String fileType) {
152         if (historyPreference.containsKey(fileType)) {
153             return historyPreference.get(fileType);
154         }
155         // no match for current fileType, return empty string
156         return "";
157     }

159     /**
160      * Generate a one-line string about the payload object
161      * 
162      * @param payload the payload to describe
163      */
164     public static String getPayloadOneLineString(final IBaseDataObject payload) {
165         final StringBuilder sb = new StringBuilder();
166         final String fn = payload.getFilename();
167         final int attPos = fn.indexOf(Family.SEP);
168         if (attPos != -1) {
169             sb.append(fn.substring(attPos + 1)).append(" ");
170         }
171         final List<String> th = payload.transformHistory(true);
172         String prev = "";
173         for (final String h : th) {
174             final int pos = h.indexOf(".");
175             if (pos > 0) {
176                 final String prefix = h.substring(0, pos);
177                 if (!prev.equals(prefix)) {
178                     if (prev.length() != 0) {
179                         sb.append(",");
180                     }
181                     sb.append(prefix);
182                     prev = prefix;
183                 }
184             }
185         }
186         sb.append(">>").append(payload.getAllCurrentForms());
187         sb.append("//").append(payload.getFileType());
188         sb.append("//").append(payload.getCreationTimestamp());
189         return sb.toString();
190     }

192     /**
193      * Turn the payload into an xml jdom document
194      * 
195      * @param d the payload
196      */
197     public static Document toXml(final IBaseDataObject d) {
198         final Element root = new Element("payload");
199         root.addContent(JDOMUtil.protectedElement("name", d.getFilename()));
200         final Element cf = new Element("current-forms");
201         for (final String c : d.getAllCurrentForms()) {
202             cf.addContent(JDOMUtil.simpleElement("current-form", c));
203         }
204         root.addContent(cf);
205         root.addContent(JDOMUtil.simpleElement("encoding", d.getFontEncoding()));
206         root.addContent(JDOMUtil.simpleElement("filetype", d.getFileType()));
207         root.addContent(JDOMUtil.simpleElement("classification", d.getClassification()));
208         final Element th = new Element("transform-history");
209         for (final String s : d.transformHistory(true)) {
210             th.addContent(JDOMUtil.simpleElement("itinerary-step", s));
211         }
212         root.addContent(th);
213         if (d.getProcessingError() != null) {
214             root.addContent(JDOMUtil.simpleElement("processing-error", d.getProcessingError()));
215         }
216         final Element meta = new Element("metadata");
217         for (final String key : d.getParameters().keySet()) {
218             final Element m = JDOMUtil.protectedElement("param", d.getStringParameter(key));
219             m.setAttribute("name", key);
220             meta.addContent(m);
221         }
222         root.addContent(meta);
223 
224         if (d.header() != null) {
225             root.addContent(JDOMUtil.protectedElement("header", d.header()));
226         }
227         if (d.dataLength() > 0) {
228             root.addContent(JDOMUtil.protectedElement("data", d.data()));
229         }
230         if (d.footer() != null) {
231             root.addContent(JDOMUtil.protectedElement("footer", d.footer()));
232         }
233 
234         // Alt views
235         if (d.getNumAlternateViews() > 0) {
236             final Element views = new Element("views");
237             for (final String av : d.getAlternateViewNames()) {
238                 final Element v = JDOMUtil.protectedElement("view", d.getAlternateView(av));
239                 v.setAttribute("name", av);
240                 views.addContent(v);
241             }
242             root.addContent(views);
243         }
244 
245         logger.debug("Produced xml document for {}", d.shortName());
246         return new Document(root);
247     }

258     /**
259      * Turn a list of payload into an xml jdom ocument
260      * 
261      * @param list the payload list
262      */
263     public static Document toXml(final List<IBaseDataObject> list) {
264         final Element root = new Element("payload-list");
265         for (final IBaseDataObject d : list) {
266             final Document doc = toXml(d);
267             root.addContent(doc.detachRootElement());
268             logger.debug("Adding xml content for {} to document", d.shortName());
269         }
270         return new Document(root);
271     }

282     /**
283      * Print formatted metadata key:value pairs
284      */
285     private static final String SEP = ": ";
286 
287     public static String printFormattedMetadata(final IBaseDataObject payload) {
288         final StringBuilder out = new StringBuilder();
289         out.append(LS);
290         for (final Map.Entry<String, Collection<Object>> entry : payload.getParameters().entrySet()) {
291             out.append(entry.getKey()).append(SEP).append(entry.getValue()).append(LS);
292         }
293         return out.toString();
294     }

296     /**
297      * Checks whether the form complies with form rules established by a regex
298      *
299      * Approved forms can contain alpha-numerics, '-', '_', '()', '/', '+'
300      *
301      * @param form The form to be tested
302      * @return Whether the form is Emissary compliant
303      */
304     public static boolean isValidForm(String form) {
305         return validFormRegex.matcher(form).find();

7   import org.slf4j.Logger;
8   import org.slf4j.LoggerFactory;

10  import java.io.IOException;
11  
12  public class CustomRolloverLogbackAppender<E> extends RollingFileAppender<E> {
13      private long start = System.currentTimeMillis();
14      private int rolloverTime = 15;
15      private boolean isConfigured = false;
16  
17      protected static final Logger logger = LoggerFactory.getLogger(CustomRolloverLogbackAppender.class);
18  
19      public void configure() {
20          try {
21              Configurator configG = ConfigUtil.getConfigInfo(getClass());
22              rolloverTime = configG.findIntEntry("ROLLOVER_TIME_MINUTES", rolloverTime);
23          } catch (IOException e) {
24              logger.error("Could not read from the config for the CustomRolloverLogbackAppender ", e);
25          }
26      }
27  
28      @Override
29      public void rollover() {
30  
31          if (!isConfigured) {
32              configure();
33              isConfigured = true;
34          }
35  
36          long currentTime = System.currentTimeMillis();
37          int maxIntervalSinceLastLoggingInMillis = rolloverTime * 60 * 1000;
38  
39          if ((currentTime - start) >= maxIntervalSinceLastLoggingInMillis) {
40              super.rollover();
41              start = System.currentTimeMillis();
42          }
43      }
44  }

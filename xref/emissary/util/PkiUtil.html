31      private static final String FILE_PRE = "file://";
32      private static final Pattern ENV_VARIABLE_PATTERN = Pattern.compile("\\$\\{(\\w+)}");
33  
34      private static final String BASE_64_ENCODED_DATA = "base64EncodedData";
35      // get between 64 and 4096 characters of Base64-encoded cert data
36      private static final Pattern CERT_PATTERN = Pattern.compile(
37              "^-----BEGIN CERTIFICATE-----$(?<" + BASE_64_ENCODED_DATA + ">(?i)[a-z0-9+/=\\r\\n]{64,4096})^-----END CERTIFICATE-----$",
38              Pattern.MULTILINE);
39  
40      private static final Logger log = LoggerFactory.getLogger(PkiUtil.class);
41  
42      /* build the key/trust store from props */
43      @Nullable
44      public static KeyStore buildStore(@Nullable final String path, final char[] pazz, final String type)
45              throws IOException, GeneralSecurityException {
46          if ((path == null) || path.isEmpty()) {
47              return null;
48          }
49          final KeyStore keyStore = KeyStore.getInstance(type);
50          String pemData = FileUtils.readFileToString(new File(path), US_ASCII);
51          if (isPemCertificate(pemData)) {
52              loadKeyStore(keyStore, pemData);
53          } else {
54              try (InputStream is = Files.newInputStream(Paths.get(path))) {
55                  keyStore.load(is, pazz);
56              }
57          }
58          return keyStore;
59      }
60  
61      protected static boolean isPemCertificate(String data) {
62          return CERT_PATTERN.matcher(data).find();
63      }
64  
65      private static void loadKeyStore(KeyStore keyStore, String pemData)
66              throws CertificateException, IOException, NoSuchAlgorithmException, KeyStoreException {
67          // initialize the keystore
68          keyStore.load(null, null);
69  
70          int certCount = 0;
71          int matcherPosition = 0;
72          Matcher matcher = CERT_PATTERN.matcher(pemData);
73          while (matcher.find(matcherPosition)) {
74              byte[] derBytes = DatatypeConverter.parseBase64Binary(matcher.group(BASE_64_ENCODED_DATA).trim());
75              CertificateFactory certificateFactory = CertificateFactory.getInstance("X.509");
76              X509Certificate x509Certificate = (X509Certificate) certificateFactory.generateCertificate(new ByteArrayInputStream(derBytes));
77              keyStore.setCertificateEntry("cert_" + certCount++, x509Certificate);
78              matcherPosition = matcher.end();
79          }
80      }
81  
82      /*
83       * Build char array from password, load from file or read from environment variable.
84       */
85      @Nullable
86      public static char[] loadPassword(@Nullable final String password) throws IOException {
87          if (password == null) {
88              return null;
89          }
90          String realPw;
91          if (password.startsWith(FILE_PRE)) {
92              final String pth = password.substring(FILE_PRE.length());
93              log.debug("Loading key password from file " + pth);
94              try (BufferedReader r = new BufferedReader(new FileReader(pth))) {
95                  realPw = r.readLine();
96              }
97              if (realPw == null) {
98                  throw new IOException("Unable to load store password from " + password);
99              }
100         } else {
101             Matcher matcher = ENV_VARIABLE_PATTERN.matcher(password);
102             if (matcher.matches()) {
103                 realPw = System.getenv(matcher.group(1));
104             } else {
105                 realPw = password;
106             }
107         }
108         return realPw.toCharArray();
109     }

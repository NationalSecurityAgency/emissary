2   package emissary.util;
3   
4   import emissary.util.io.ReadOutput;
5   
6   import jakarta.annotation.Nullable;
7   
8   import java.io.IOException;
9   import java.io.PrintStream;
10  import java.time.Instant;
11  
12  public class WatcherThread extends Thread {
13      @Nullable
14      private Process proc = null;
15      private int delay = 1000;
16      private boolean flag = true;
17  
18      public WatcherThread() {}
19  
20      public WatcherThread(Process pp) {
21          proc = pp;
22      }
23  
24      public void setDelay(int nn) {
25          delay = nn;
26      }
27  
28      public void setProcess(Process pp) {
29          proc = pp;
30      }
31  
32      public void killMe() {
33          flag = false;
34      }
35  
36      /**
37       * I've chosen to use an internal flag (called flag) to represent the state of the running thread. Since this thread
38       * spends most of its time sleeping, this will work fine. Otherwise, we'd have to use the interrupt method of the
39       * thread.
40       */
41      @Override
42      public void run() {
43          int nTries = 10;
44          int partialDelay = delay / nTries;
45          int ii = 0;
46  
47          while (flag && (ii++ < nTries)) {
48              if (proc != null) {
49                  try {
50                      sleep(partialDelay);
51  
52                      if (ii == nTries) {
53                          try {
54                              proc.exitValue();
55                          } catch (IllegalThreadStateException itsx) {
56                              proc.destroy();
57                              proc = null;
58                              flag = false;
59                          }
60                      }
61                  } catch (InterruptedException ix) {
62                      Thread.currentThread().interrupt();
63                  }
64              }
65          }
66      }
67  
68      @SuppressWarnings("SystemOut")
69      public static void main(String[] args) throws IOException {
70          WatcherThread wt = new WatcherThread();
71          String name = wt.getClass().getName();
72  
73          if (args.length != 2) {
74              System.out.println("usage: " + name + " time cmd");
75  
76              return;
77          }
78  
79          wt.setDelay(Integer.parseInt(args[0]));
80  
81          System.out.println(name + " begins at " + Instant.now());
82  
83          Process proc = Runtime.getRuntime().exec(args[1]);
84  
85          wt.setProcess(proc);
86  
87          PrintStream out = System.out;
88          PrintStream err = System.err;
89  
90          Thread t1 = new Thread(new ReadOutput(proc.getInputStream(), name, out));
91          Thread t2 = new Thread(new ReadOutput(proc.getErrorStream(), name, err));
92  
93          wt.start();
94          t1.start();
95          t2.start();
96  
97          try {
98              proc.waitFor();
99              wt.killMe();
100             t1.join();
101             t2.join();
102         } catch (InterruptedException ix) {
103             Thread.currentThread().interrupt();
104         }
105 
106         System.out.println(name + " ends at " + Instant.now());
107 
108 
109     }
110 }

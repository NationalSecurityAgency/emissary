13  /**
14   * This implements a simple caching mechanism for {@link Constructor} lookups. For example if the same class constructor
15   * is looked up repeatedly, using this cache may be able to avoid a lot of JVM reflection overhead.
16   *
17   * <p>
18   * Note that the cache implementation may have a small capacity and/or be thread-specific, so storing something in the
19   * cache does not <i>guarantee</i> that it will be indefinitely cached or that the cached value will be visible to other
20   * threads.
21   */
22  @SuppressWarnings("AvoidObjectArrays")
23  public final class ConstructorLookupCache {
24  
25      private static final Logger logger = LoggerFactory.getLogger(ConstructorLookupCache.class);
26  
27      /**
28       * Represents a known result for looking up a class constructor that can handle a specific set of argument types.
29       */
30      private static final class KnownConstructor {
31  
32          /** The class being constructed. */
33          private final Class<?> clazz;
34  
35          /** The argument types that will be passed to the constructor. */
36          private final Class<?>[] argTypes;
37  
38          /**
39           * A constructor for {@link #clazz} that accepts the argument types from {@link #argTypes}.
40           */
41          private final Constructor<?> constructor;
42  
43          /**
44           * Create a binding between a class, some constructor argument types, and a constructor.
45           *
46           * @param clazz The class being constructed.
47           * @param argTypes The argument types that will be passed to the constructor.
48           * @param constructor A constructor for {@code clazz} that can accept the argument types specified in {@code argTypes}.
49           */
50          public KnownConstructor(final Class<?> clazz, final Class<?>[] argTypes, final Constructor<?> constructor) {
51              this.clazz = clazz;
52              this.argTypes = argTypes;
53              this.constructor = constructor;
54          }
55  
56          /**
57           * Get the stored constructor if it matches the given class and argument types.
58           *
59           * @param desiredClazz The type of class being constructed.
60           * @param desiredArgTypes The argument types that will be passed to the constructor.
61           * @return If the class and argument types match the ones that are stored, this returns the stored constructor that is
62           *         assumed to be compatible with those argument types and create an object of the given type; otherwise
63           *         {@code null}.
64           */
65          @Nullable
66          public Constructor<?> getConstructor(final Class<?> desiredClazz, final Class<?>[] desiredArgTypes) {
67              // The stored class has to match the desired class.
68              if (!this.clazz.equals(desiredClazz)) {
69                  return null; // Non-matching class.
70              }
71  
72              // The stored argument count has to match the desired
73              // argument count.
74              final Class<?>[] myArgTypes = this.argTypes;
75              if (myArgTypes.length != desiredArgTypes.length) {
76                  return null; // The number of arguments doesn't match.
77              }
78  
79              // The stored argument types have to match the desired
80              // argument types.
81              for (int i = 0; i < myArgTypes.length; i++) {
82                  if (myArgTypes[i] == null) {
83                      if (desiredArgTypes[i] != null) {
84                          return null; // Non-matching null argument type.
85                      }
86                  } else if (!myArgTypes[i].equals(desiredArgTypes[i])) {
87                      return null; // Non-matching non-null argument type.
88                  }
89              }
90  
91              // At this point we know that we have the desired class
92              // and arguments.
93              return this.constructor;
94          }
95      }
96  
97      /**
98       * A cached constructor lookup result. If a thread is asked to invoke the same constructor repeatedly, we can cache the
99       * result of the lookup to avoid some costly reflection calls.
100      */
101     private static final ThreadLocal<SoftReference<KnownConstructor>> cachedConstructorLookup = new ThreadLocal<>();
102 
103     /** A table mapping boxed classes to their primitive types. */
104     private static final Map<Class<?>, Class<?>> primClass = new HashMap<>();
105 
106     // Initialize the mappings.
107     static {
108         primClass.put(Integer.class, Integer.TYPE);
109         primClass.put(Boolean.class, Boolean.TYPE);
110         primClass.put(Float.class, Float.TYPE);
111         primClass.put(Character.class, Character.TYPE);
112         primClass.put(Long.class, Long.TYPE);
113         primClass.put(Double.class, Double.TYPE);
114         primClass.put(Byte.class, Byte.TYPE);
115     }
116 
117     /**
118      * Convert a boxed type into its primitive type.
119      * 
120      * @param clazz The type of interest.
121      * @return If {@code clazz} is a boxed primitive, return the primitive type; otherwise just return {@code clazz}.
122      */
123     private static Class<?> getPrim(final Class<?> clazz) {
124         final Class<?> prim = primClass.get(clazz);
125         return (prim != null) ? prim : clazz;
126     }
127 
128     /**
129      * Look for a constructor for the given class type which can accept the given argument types, without using the lookup
130      * cache.
131      *
132      * @param clazz The type of object that will be constructed.
133      * @param argTypes The types of the arguments that will be passed to the class constructor.
134      * @return If {@code clazz} has a constructor that can accept the argument types in {@code argTypes}, this returns the
135      *         matching constructor; otherwise {@code null}.
136      */
137     @Nullable
138     private static Constructor<?> directConstructorLookup(final Class<?> clazz, final Class<?>[] argTypes) {
139         // Look for an exact match.
140         try {
141             return clazz.getConstructor(argTypes);
142         } catch (NoSuchMethodException e) {
143             logger.debug("No constructor for [{}] in Factory.create())", clazz.getName());
144         }
145 
146         // There was no exact match, so look through the existing
147         // constructors for an assignable match.
148         NEXT_CANDIDATE_CONSTRUCTOR: for (final Constructor<?> candidate : clazz.getConstructors()) {
149             final Class<?>[] ctypes = candidate.getParameterTypes();
150 
151             if (logger.isDebugEnabled()) {
152                 logger.debug("Checking:{}, {}", clazz.getName(), clazz);
153                 logger.debug("   types   :{}", Arrays.toString(ctypes));
154                 logger.debug("   numParms:{} =? {}", ctypes.length, argTypes.length);
155             }
156 
157             // If the candidate constructor doesn't have the same
158             // number of arguments, it definitely isn't compatible
159             // with the desired argument types.
160             if (ctypes.length != argTypes.length) {
161                 continue NEXT_CANDIDATE_CONSTRUCTOR;
162             }
163 
164             // The candidate takes the right number of arguments, so
165             // compare its expected types to the types that will be
166             // passed. If the given type is not assignable to the
167             // expected type, then this constructor is not compatible.
168             for (int j = 0; j < argTypes.length; j++) {
169                 final Class<?> a = ctypes[j];
170                 final Class<?> b = argTypes[j];
171 
172                 if ((a != null) && (b != null)) {
173                     if (a.isAssignableFrom(b)) {
174                         logger.debug("   param={}  assignable {}", a, b);
175                     } else {
176                         final Class<?> bPrim = getPrim(b);
177                         if (a.isAssignableFrom(bPrim)) {
178                             logger.debug("   param={}  assignable {}", a, b);
179                         } else {
180                             logger.debug("   param={}  !assignable {}", a, b);
181                             continue NEXT_CANDIDATE_CONSTRUCTOR;
182                         }
183                     }
184                 }
185             }
186 
187             // If we reach here, the current candidate constructor is
188             // a compatible match.
189             return candidate;
190         }
191 
192         // None of the class's available constructors are compatible
193         // with the given argument types.
194         return null;
195     }
196 
197     /**
198      * Look for a constructor in the cache.
199      *
200      * @param clazz The class to be constructed.
201      * @param argTypes The argument types that the caller intends to pass to the constructor.
202      * @return If a matching constructor is in the cache, return it. Otherwise {@code null}.
203      */
204     @Nullable
205     public static Constructor<?> get(final Class<?> clazz, final Class<?>[] argTypes) {
206         // Currently there is at most one cached lookup per thread, so
207         // we can just check if the current thread knows about the
208         // given class and constructor arguments.
209         final SoftReference<KnownConstructor> softResult = cachedConstructorLookup.get();
210         if (softResult == null) {
211             return null; // Nothing is currently cached in this thread.
212         }
213 
214         final KnownConstructor knownResult = softResult.get();
215         if (knownResult == null) {
216             return null; // There was something cached but it's been lost.
217         }
218 
219         // We do have a cached lookup. It can be used iff it matches the
220         // given class and constructor arguments.
221         return knownResult.getConstructor(clazz, argTypes);
222     }
223 
224     /**
225      * Store a constructor lookup in the cache.
226      *
227      * @param clazz The class to be constructed.
228      * @param argTypes The argument types that the caller intends to pass to the constructor.
229      * @param constructor A constructor for {@code clazz} that can accept the argument types specified in {@code argTypes}.
230      */
231     public static void put(final Class<?> clazz, final Class<?>[] argTypes, final Constructor<?> constructor) {
232         cachedConstructorLookup.set(new SoftReference<>(new KnownConstructor(clazz, argTypes, constructor)));
233     }
234 
235     /**
236      * Look for a constructor for the given class type which can accept the given argument types.
237      *
238      * @param clazz The class to be constructed.
239      * @param argTypes The argument types that the caller intends to pass to the constructor.
240      * @return A matching constructor for the specified class, or {@code null} if no such constructor was found.
241      */
242     public static Constructor<?> lookup(final Class<?> clazz, final Class<?>[] argTypes) {
243         final Constructor<?> cachedConstructor = get(clazz, argTypes);
244         if (cachedConstructor != null) {
245             // We found the constructor in the cache.
246             return cachedConstructor;
247         } else {
248             // The desired constructor is not currently cached, so
249             // look it up directly.
250             final Constructor<?> uncachedConstructor = directConstructorLookup(clazz, argTypes);
251 
252             // If we got a result, cache it before returning.
253             if (uncachedConstructor != null) {
254                 put(clazz, argTypes, uncachedConstructor);
255             }
256             return uncachedConstructor;
257         }
258     }
259 
260     /**
261      * Destroy the ThreadLocal cache object
262      */
263     public static void unload() {
264         cachedConstructorLookup.remove();
265     }
266 
267     /** This is a static utility class, so prevent instantiation. */
268     private ConstructorLookupCache() {}
269 }

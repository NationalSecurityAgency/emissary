11  import java.io.PrintStream;

13  /**
14   * This class implements a tree state machine scanner that searches text backwards starting from the end. A list of
15   * strings is provided as the keywords to be searched. This class is useful for a relatively small set of keywords.
16   * Larger keyword lists can be used if you have memory!
17   *
18   * @author ce
19   * @version 1.0
20   */
21  @SuppressWarnings("AvoidObjectArrays")
22  public class BackwardsTreeScanner {
23  
24      // The internal structure of the offset-keyword id array
25      // contained in the hit vectors
26  
27      /** Root node of tree state diagram. Always start a search from here! */
28      private State root = new State((byte) 0);
29  
30      /**
31       * Optional empty constructor
32       */
33      public BackwardsTreeScanner() {
34  
35      }
36  
37      /**
38       * Constructor that delegates learning keywords to resetKeywords method.
39       *
40       * @param keywordStrings - array of keywords to learn
41       * @throws Exception - thrown if problem encountered
42       */
43      public BackwardsTreeScanner(String[] keywordStrings) throws Exception {
44          resetKeywords(keywordStrings);
45      }
46  
47      /**
48       * Resets keywords and internal State learns them. This method destroys previous state.
49       *
50       * @param keywordStrings - String of keywords to learn
51       * @throws Exception - if problem encountered while learning
52       */
53      public synchronized void resetKeywords(String[] keywordStrings) throws Exception {
54          // make byte arrays
55          // Original list of keywords stored in byte array form.
56          byte[][] keywords = new byte[keywordStrings.length][];
57          root = new State((byte) 0); // reset state
58          // and learn them
59          for (int i = 0; i < keywords.length; i++) {
60              keywords[i] = keywordStrings[i].getBytes();
61              root.learn(keywordStrings[i].getBytes(), i);
62          }
63          // root.print(System.out);
64      }
65  
66      /**
67       * This scans the byte array backwards from the offset. Each hit is added to the result vector. We stop when all
68       * possibilities are found
69       */
70      public synchronized int scan(byte[] data, int offset, HitList result) throws Exception {
71          if (result == null) {
72              throw new Exception("Null result vector in 3rd parameter of scan()");
73          }
74          // reset the state machine
75          State state = root;
76          int curPos = offset;
77          while (state != null && curPos >= 0) {
78              // Save any matches for this state.
79              // get the next character
80              byte ch = data[curPos];
81              // move to the next state. Really complicated, right?
82              if (ch < 0) {
83                  state = state.nextStates[256 + (int) ch];
84              } else {
85                  state = state.nextStates[ch];
86              }
87              // move the the previous character
88              if (state != null && state.matches != null) {
89                  for (int i = 0; i < state.matches.length; i++) {
90                      int id = state.matches[i];
91                      Hit tmp = new Hit(curPos, id);
92                      result.add(tmp);
93                  }
94              }
95              curPos--;
96          }
97          return curPos;
98      }
99  
100     /*
101      * This class implements a state machine that can learn character sequences.
102      */
103     public class State {
104         // Each state has 256 transitions leaving it to new states based
105         // on a single ascii character. If there is no next state for a
106         // character, then the next state will be null.
107         public State[] nextStates = new State[256];
108         // Each state can be visited by a single character. This is it!
109         public byte gotHereBy;
110         // A list of keyword ids that are matched at this state.
111         @Nullable
112         public int[] matches = null;
113 
114         // constructor
115         public State(byte gotHereBy) {
116             this.gotHereBy = gotHereBy;
117         }
118 
119         public void learn(byte[] word, int id) throws Exception {
120             learn(word, word.length - 1, id);
121         }
122 
123         /**
124          * Walk through the keyword backwards. Adding states to the root (or current state) when they don't exists. At the end,
125          * record the keyword id in the ending state.
126          * 
127          * Warning this is recursive, but that is OK for small keywords.
128          */
129         public void learn(byte[] word, int wordLoc, int id) throws Exception {
130             if (word == null) {
131                 throw new Exception("null keyword in BackwardsTreeScanner.learn()");
132             }
133             if (wordLoc >= word.length) {
134                 throw new Exception("char pos > word length:" + wordLoc + ">" + word.length);
135             }
136             if (wordLoc < 0) {
137                 // we are finished because this is the first character,
138                 // so save the id in this state. We want the matches to be
139                 // in an array so this is a little harder than a vector thing.
140                 if (matches == null) {
141                     matches = new int[0];
142                 }
143                 int[] newMatches = new int[matches.length + 1];
144                 System.arraycopy(matches, 0, newMatches, 0, matches.length);
145                 matches = newMatches;
146                 matches[matches.length - 1] = id;
147             } else {
148                 // Get the next character in the word
149                 byte nextChar = word[wordLoc];
150                 // See if the state already exists
151                 State nextState = nextStates[nextChar];
152                 if (nextState == null) {
153                     // Make a new state because it isn't there yet.
154                     nextState = nextStates[nextChar] = new State(nextChar);
155                 }
156                 // Learn the rest of the keyword in the new state.
157                 nextState.learn(word, wordLoc - 1, id);
158             }
159         }
160 
161         public void print(PrintStream out) {
162             print(out, "root:");
163         }
164 
165         // Make a pretty picture.
166         public void print(PrintStream out, String prefix) {
167             if (gotHereBy < ' ' || gotHereBy > '~') {
168                 out.println(prefix + "-> " + "(byte)" + gotHereBy);
169             } else {
170                 out.println(prefix + "-> " + (char) gotHereBy);
171             }
172             if (matches != null) {
173                 out.print(prefix + "ids [");
174                 for (int match : matches) {
175                     out.print(" " + match);
176                 }
177                 out.println(" ]");
178             }
179             for (State nextState : nextStates) {
180                 if (nextState != null) {
181                     nextState.print(out, prefix + "  ");
182                 }
183             }
184         }
185     }

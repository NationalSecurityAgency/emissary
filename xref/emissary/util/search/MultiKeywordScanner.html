23      public void loadKeywords(final String[] keywordsArg) {
24          this.keywords = keywordsArg;
25  
26          try {
27              if (null == this.treeScanner) {
28                  this.treeScanner = new BackwardsTreeScanner(keywordsArg);
29              } else {
30                  this.treeScanner.resetKeywords(keywordsArg);
31              }
32          } catch (Exception e) {
33              logger.error("Could not create BackwardsTreeScanner", e);
34          }
35  
36          buildSkipTable();
37      }

39      private void buildSkipTable() {
40          final int numKeywords = this.keywords.length;
41  
42          for (int i = 0; i < numKeywords; i++) {
43              final byte[] keyword = this.keywords[i].getBytes();
44              final int keywordLength = keyword.length;
45              if (i == 0) {
46                  this.standardSkip = keywordLength;
47              } else if (this.standardSkip > keywordLength) {
48                  this.standardSkip = keywordLength;
49              }
50          }
51  
52          for (int i = 0; i < 256; i++) {
53              this.skipTable[i] = this.standardSkip;
54          }
55  
56          for (int i = 0; i < numKeywords; i++) {
57              final byte[] keyword = this.keywords[i].getBytes();
58              final int keywordLength = keyword.length;
59              for (int j = 0; j < (keywordLength - 1); j++) {
60                  final int byteValue = get256Value(keyword[j]);
61                  final int skip = keywordLength - (j + 1);
62                  if (skip < this.skipTable[byteValue]) {
63                      this.skipTable[byteValue] = skip;
64                  }
65              }
66          }
67      }
68  
69      private static int get256Value(final byte b) {
70          return ((int) b) & 0xff;
71      }

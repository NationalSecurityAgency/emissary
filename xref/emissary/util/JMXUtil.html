7   import java.lang.management.ManagementFactory;
8   import javax.management.MBeanServer;
9   import javax.management.ObjectName;
10  
11  /**
12   * Utility class for JMX operations.
13   */
14  public class JMXUtil {
15  
16      private static final Logger logger = LoggerFactory.getLogger(JMXUtil.class);
17  
18      /**
19       * Simple method (hopefully) to register an object with the JMXServer.
20       *
21       * @param obj object to register
22       */
23      public static void registerMBean(final Object obj) {
24          registerMBean(obj, obj.getClass().getName());
25      }
26  
27      public static void registerMBean(@Nullable final Object obj, final String name) {
28          if (Boolean.parseBoolean(System.getProperty("emissary.jmx.disabled"))) {
29              logger.info("Emissary JMX is disabled via emissary.jmx.disabled property");
30              return;
31          }
32  
33          if (obj != null) {
34              logger.debug("Regisering MBean for Class --> {}", obj.getClass().getName());
35  
36              final MBeanServer mbs = ManagementFactory.getPlatformMBeanServer();
37              try {
38                  final ObjectName objName = new ObjectName("emissary:name=" + name);
39                  if (!mbs.isRegistered(objName)) {
40                      mbs.registerMBean(obj, objName);
41                  }
42              } catch (Exception e) {
43                  logger.error("Caught exception trying to register MBean for class {}", obj.getClass().getName(), e);
44              }
45          } else {
46              logger.warn("Received request to register a null object");
47          }
48      }
49  
50      /** This class is not meant to be instantiated. */
51      private JMXUtil() {}
52  }

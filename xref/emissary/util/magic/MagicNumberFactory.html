55      /**
56       * Public method to parse a byte array representing the magic file into a list containing MagicNumber objects which are
57       * also nested with continuations as child MagicNumber instances.
58       *
59       * @param configData the byte[] representing the magic file
60       * @param zeroDepthErrorList logs errors with zero depth entries
61       * @param continuationErrorMap logs errors with continuations - these are entries with depths &gt; 0
62       * @param swallowParseException boolean whether to swallow or propogate ParseExceptions that are IGNORABLE_DATATYPE_MSGS
63       * @return a {@link List}.
64       */
65      public static List<MagicNumber> buildMagicNumberList(byte[] configData, @Nullable List<String> zeroDepthErrorList,
66              @Nullable Map<String, List<String>> continuationErrorMap, boolean swallowParseException) {
67  
68          List<MagicNumber> magicNumberList = new ArrayList<>();
69          MagicNumber finger = null;
70          int currentDepth = -1;
71          List<MagicNumber> extensions = new ArrayList<>();
72          try (BufferedReader reader = new BufferedReader(new InputStreamReader(new ByteArrayInputStream(configData), MagicNumber.DEFAULT_CHARSET))) {
73              String s;
74              int counter = 0;
75              while ((s = reader.readLine()) != null) {
76                  counter++;
77                  if (s == null || s.length() == 0 || s.charAt(0) == '#') {
78                      continue;
79                  }
80                  int depth = getEntryDepth(s);
81                  if (depth < 0) {
82                      continue;
83                  }
84                  try {
85                      if (depth == 0 && !extensions.isEmpty()) {
86                          if (finger == null) {
87                              extensions = null;
88                              extensions = new ArrayList<>();
89                          } else {
90                              addExtensionsLayer(extensions, finger);
91                              extensions = null;
92                              extensions = new ArrayList<>();
93                              finger = null;
94                          }
95                      }
96                      if (depth == 0) {
97                          MagicNumber item = parseAndStore(magicNumberList, s, swallowParseException);
98                          if (item != null) {
99                              finger = item;
100                             currentDepth = -1;
101                         } else {
102                             finger = null;
103                         }
104                     } else if (depth > 0) {
105 
106                         if (finger == null) {
107                             continue;
108                         }
109                         if (currentDepth < 0) {
110                             currentDepth = depth;
111                         }
112                         if (currentDepth == depth) {
113                             parseAndStore(extensions, s, swallowParseException);
114                         } else if (currentDepth < depth) {
115                             if (extensions.isEmpty()) {
116                                 finger = null;
117                                 currentDepth = -1;
118                                 continue;
119                             }
120                             currentDepth = depth;
121                             addExtensionsLayer(extensions, finger);
122                             extensions = null;
123                             extensions = new ArrayList<>();
124                             parseAndStore(extensions, s, swallowParseException);
125                         }
126                     }
127                 } catch (Exception e) {
128 
129                     if (continuationErrorMap == null || zeroDepthErrorList == null) {
130                         continue;
131                     }
132                     if (swallowParseException &&
133                             (e.getClass() == ParseException.class) &&
134                             IGNORABLE_DATATYPE_MSGS.contains(e.getMessage())) {
135                         // swallow this cause said we don't care
136                         continue;
137                     }
138 
139 
140                     if (depth > 0) {
141                         MagicNumber mItem = magicNumberList.get(magicNumberList.size() - 1);
142                         String signature = mItem.toString();
143                         List<String> failedExtensions = continuationErrorMap.get(signature);
144                         if (failedExtensions == null) {
145                             failedExtensions = new ArrayList<>();
146                             continuationErrorMap.put(mItem.toString(), failedExtensions);
147                         }
148                         failedExtensions.add("[MAGIC LINE# " + counter + "] " + s);
149                     } else if (depth == 0) {
150                         zeroDepthErrorList.add("[MAGIC LINE# " + counter + "] " + s);
151                     }
152                 }
153             }
154             if (finger != null && !extensions.isEmpty()) {
155                 addExtensionsLayer(extensions, finger);
156             }
157         } catch (IOException ioe) {
158             log.error("Caught IOException on buildMagicNumberList (throwing a runtime exception): {}", ioe.getMessage(), ioe);
159             // Doing all of this in memory - yes, one could erroneously use one of the IO objects but ...
160             throw new EmissaryRuntimeException(ioe);
161         }
162         return magicNumberList;
163     }

165     /**
166      * Private method for parsing entries and storing them into the target storage data structure which is a list.
167      *
168      *
169      * @param entry the magic number String entry
170      * @param storage a {@link List} where the MagicNumber instances will be placed
171      * @param swallowParseException should we swallow Ignorable ParseException or bubble them up
172      * @return the MagicNumber instance created
173      * @throws Exception if one occurs while parsing the entry
174      * @see #buildMagicNumber(java.lang.String, boolean)
175      */
176     private static MagicNumber parseAndStore(List<MagicNumber> storage, String entry, boolean swallowParseException) throws Exception {
177         MagicNumber item = buildMagicNumber(entry, swallowParseException);
178         if (item != null) {
179             storage.add(item);
180         }
181         return item;
182     }

184     /**
185      * Private method to store the list of MagicNumber instances and adds them a new layer of continuations in the target
186      * MagicNumber
187      *
188      * @param target the MagicNumber instance acting as the parent for the continuations
189      * @param extensions a {@link List} of continuations which are MagicNumber instances
190      */
191     private static void addExtensionsLayer(List<MagicNumber> extensions, MagicNumber target) {
192         MagicNumber[] extensionArray = new MagicNumber[extensions.size()];
193         int index = 0;
194         for (MagicNumber m : extensions) {
195             extensionArray[index++] = m;
196         }
197         target.addDependencyLayer(extensionArray);
198     }

209     /**
210      * Parses a magic number entry and prepares a magic number item
211      * 
212      * @param entry line to parse
213      * @param swallowParseException should we swallow Ignorable ParseException or bubble them up
214      */
215     public static MagicNumber buildMagicNumber(String entry, boolean swallowParseException) throws ParseException {
216 
217         String[] columns = prepareEntry(entry);
218         MagicNumber item = new MagicNumber();
219 
220         try {
221             // column A parsing
222             item.depth = getEntryDepth(columns[0]);
223             item.offsetUnary = resolveOffsetUnary(columns);
224             item.offset = resolveOffset(columns, item);
225         } catch (Exception e) {
226             throw new ParseException("Error on column 0:" + columns[0] + ". " + e.getMessage());
227         }
228         try {
229             // columb B parsing
230             item.dataType = resolveDataType(columns);
231             item.dataTypeLength = getDataTypeByteLength(item);
232             item.mask = resolveMask(columns, item);
233         } catch (Exception e) {
234             if (swallowParseException) {
235                 // This means you put TRUE in SWALLOW_IGNORABLE_EXCEPTIONS in a UnixFilePlace.cfg file
236                 // so let's log at debug level so you can hide these message easily
237                 log.debug("Warning unable to read column 1\t: {} - {}", columns[1], e.getMessage());
238             } else {
239                 log.error("original entry   \t: {}", entry);
240                 log.error("Error on column 1\t: {}", columns[1], e);
241             }
242             throw new ParseException("Parse Error on column 1:" + columns[1] + ". " + e.getMessage());
243         }
244         try {
245             // column C parsing
246             item.unaryOperator = resolveUnary(columns, item);
247             item.value = resolveValue(columns, item);
248             item.dataTypeLength = item.value.length;
249         } catch (Exception e) {
250             throw new ParseException("Error on column 2:" + columns[2] + ". " + e.getMessage());
251         }
252         // column D parsing
253         item.description = columns[3];
254         return item;
255     }

257     /**
258      * Tokenizes an entry into four columns by tab or space while recognizing escape sequences.
259      */
260     private static String[] tokenizeEntry(String entry) {
261         int index = 0;
262         String[] columns = new String[4];
263         columns[0] = EMPTYSTRING;
264         columns[1] = EMPTYSTRING;
265         columns[2] = EMPTYSTRING;
266         columns[3] = EMPTYSTRING;
267         for (int i = 0; i < entry.length(); i++) {
268             char c = entry.charAt(i);
269             if (c == '\\' && i != (entry.length() - 1) && entry.charAt(i + 1) == ' ') {
270                 columns[index] += ' ';
271                 i++;
272             } else if (c == ' ' || c == '\t') {
273                 while (entry.length() > (i + 1)) {
274                     if (entry.charAt(i + 1) == ' ' || entry.charAt(i + 1) == '\t') {
275                         i++;
276                     } else {
277                         break;
278                     }
279                 }
280                 index++;
281             } else {
282                 columns[index] += c;
283             }
284             if (index == 3) {
285                 if (entry.length() > (i + 1)) {
286                     columns[index] = entry.substring(i + 1);
287                 }
288                 break;
289             }
290         }
291         return columns;
292     }

296      */
297     private static String[] prepareEntry(String entry) throws ParseException {
298         if (entry == null) {
299             throw new ParseException(ENTRY_NOT_NULL_RULE);
300         }
301         String subject = entry;
302         int invalidOperatorIndex = subject.indexOf(" = ");
303         if (invalidOperatorIndex > 0) {
304             String tail = subject.length() > (invalidOperatorIndex + 3) ? ' ' + subject.substring(invalidOperatorIndex + 3) : "";
305             subject = subject.substring(0, invalidOperatorIndex) + tail;
306         }
307 
308         String[] columns = tokenizeEntry(subject);
309         for (int count = 0; count < columns.length; count++) {
310             if (count == 3 && columns[count].length() == 0 && !(columns[0].charAt(0) == '>')) {
311                 // columns[count] = NULL_DESCRIPTION;
312 
313             } else if (columns[count].length() == 0 && count < 3) {
314                 throw new ParseException(ENTRY_4COLUMN_RULE);
315             }
316         }
317         return columns;
318     }

323     private static int resolveOffset(String[] columns, MagicNumber item) throws ParseException {
324         String entry = columns[0];
325         if (item.depth > 0) {
326             entry = entry.substring(item.depth);
327         }
328         if (entry.charAt(0) == '&') {
329             entry = entry.substring(1);
330         } else if (entry.charAt(0) == '(' && entry.charAt(entry.length() - 1) == ')') {
331             entry = entry.substring(1, entry.length() - 1);
332         }
333         try {
334             return MagicMath.stringToInt(entry);
335         } catch (NumberFormatException e) {
336             throw new ParseException(e + ": Malformed offset value");
337         }
338     }

340     private static char resolveOffsetUnary(String[] columns) {
341         if (columns[0].charAt(0) == '&') {
342             return '&';
343         }
344         return (char) 0;

347     public static int getEntryDepth(String entry) {
348         if (entry.length() == 0 || (entry.charAt(0) != '>' && !Character.isDigit(entry.charAt(0)))) {
349             return -1;
350         }
351         int depth = 0;
352         for (; depth < entry.length(); depth++) {
353             if (entry.charAt(depth) != '>') {
354                 break;
355             }
356         }
357         return depth;
358     }

361     // COLUMN B: BYTE&maskValue
362     // -----------------------------------------------------------------------
363     private static int resolveDataType(String[] columns) throws ParseException {
364         initTypeMap();
365 
366         String subject = columns[1];
367         if (subject.startsWith("search")) {
368             throw new ParseException(UNSUPPORTED_DATATYPE_MSG_SEARCH);
369         }
370         if (subject.equals("regex")) {
371             throw new ParseException(UNSUPPORTED_DATATYPE_MSG_REGEX);
372         }
373         if (subject.charAt(0) == 'u' || subject.charAt(0) == 'U') {
374             throw new ParseException(UNSUPORTED_DATATYPE_MSG_UNSIGNED);
375         }
376 
377         // parse out any masking
378         int ix = subject.indexOf("&") > 0 ? subject.indexOf("&") : subject.indexOf("/");
379         if (ix > 0) {
380             subject = columns[1].substring(0, ix);
381         }
382         int dataTypeId = lookupDataType(subject);
383         if (dataTypeId < 0) {
384             throw new ParseException("Unsupported Data Type: " + subject);
385         }
386         return dataTypeId;
387     }

389     private static int lookupDataType(String arg) {
390         int dataTypeIdInt = typeMap.get(arg.toUpperCase(Locale.getDefault()));
391         switch (dataTypeIdInt) {
392             case MagicNumber.TYPE_DATE:
393                 return -1;
394             case MagicNumber.TYPE_BEDATE:
395                 return -1;
396             case MagicNumber.TYPE_LEDATE:
397                 return -1;
398             default:
399                 return dataTypeIdInt;
400         }
401     }

403     @Nullable
404     private static byte[] resolveMask(String[] columns, MagicNumber item) {
405         int ix = columns[1].indexOf("&");
406         if (ix > 0) {
407             byte[] maskValues = MagicMath.stringToByteArray(columns[1].substring(ix + 1));
408             MagicMath.setLength(maskValues, item.dataTypeLength);
409         }
410         return null;
411     }

413     // -----------------------------------------------------------------------
414     // COLUMN C: [UNARY_OPERATOR][Some value like 0x00]
415     // -----------------------------------------------------------------------
416     private static byte[] resolveValue(String[] columns, MagicNumber item) {
417         String subject = columns[2];
418 
419         if (item.dataType == MagicNumber.TYPE_STRING && !(subject.length() == 1 && subject.charAt(0) == 'x')) {
420             byte[] strVal = MagicMath.parseEscapedString(subject);
421             item.dataTypeLength = strVal.length;
422             return strVal;
423         } else if (subject.length() == 1 && subject.charAt(0) == 'x') {
424             item.substitute = true;
425             return new byte[0];
426         }
427 
428         int unaryLen = unaryPrefixLength(subject);
429         if (unaryLen > 0) {
430             subject = subject.substring(unaryLen);
431         }
432         if (subject.toUpperCase(Locale.getDefault()).endsWith("L")) {
433             subject = subject.substring(0, subject.length() - 1);
434         }
435         byte[] valueArray = MagicMath.stringToByteArray(subject);
436         valueArray = MagicMath.setLength(valueArray, item.dataTypeLength);
437 
438         if (item.mask != null) {
439             valueArray = MagicMath.mask(valueArray, item.mask);
440         }
441         if (item.dataType == MagicNumber.TYPE_LELONG) {
442             MagicMath.longEndianSwap(valueArray, 0);
443         } else if (item.dataType == MagicNumber.TYPE_LESHORT) {
444             MagicMath.shortEndianSwap(valueArray, 0);
445         }
446         return valueArray;
447     }

449     private static int unaryPrefixLength(@Nullable String s) {
450 
451         if (s == null || s.length() == 0) {
452             return 0;
453         }
454 
455         char op = s.charAt(0);
456         int len = s.length();
457         if (!Character.isDigit(op)) {
458             switch (op) {
459                 case MagicNumber.MAGICOPERATOR_AND:
460                     return 1;
461                 case MagicNumber.MAGICOPERATOR_GTHAN:
462                     return len > 1 && s.charAt(1) == MagicNumber.MAGICOPERATOR_AND ? 2 : 1;
463                 case MagicNumber.MAGICOPERATOR_LTHAN:
464                     return len > 1 && s.charAt(1) == MagicNumber.MAGICOPERATOR_AND ? 2 : 1;
465                 case MagicNumber.MAGICOPERATOR_OR:
466                     return 1;
467                 case MagicNumber.MAGICOPERATOR_BWAND:
468                     return 1;
469                 case MagicNumber.MAGICOPERATOR_BWNOT:
470                     return 1;
471                 case MagicNumber.MAGICOPERATOR_NOT:
472                     return 1;
473                 default:
474                     return 0;
475             }
476         }
477         return 0;
478     }

480     private static int getDataTypeByteLength(MagicNumber item) {
481         int dataTypeId = item.dataType;
482         switch (dataTypeId) {
483             case MagicNumber.TYPE_STRING:
484                 if (item.value == null) {
485                     return -1;
486                 } else {
487                     return item.value.length;
488                 }
489             case MagicNumber.TYPE_BYTE:
490                 return 1;
491             case MagicNumber.TYPE_SHORT:
492                 return 2;
493             case MagicNumber.TYPE_BESHORT:
494                 return 2;
495             case MagicNumber.TYPE_LESHORT:
496                 return 2;
497             case MagicNumber.TYPE_LONG:
498                 return 4;
499             case MagicNumber.TYPE_BELONG:
500                 return 4;
501             case MagicNumber.TYPE_LELONG:
502                 return 4;
503             case MagicNumber.TYPE_BEDATE:
504                 return 4;
505             case MagicNumber.TYPE_LEDATE:
506                 return 4;
507             default:
508                 return -1;
509         }
510     }

512     private static char resolveUnary(String[] columns, MagicNumber item) throws ParseException {
513         int unaryLen = unaryPrefixLength(columns[2]);
514         if (item.dataType == MagicNumber.TYPE_STRING || unaryLen == 0) {
515             return MagicNumber.MAGICOPERATOR_DEFAULT;
516         } else if (unaryLen == 1) {
517             return columns[2].charAt(0);
518         } else if (unaryLen == 2 && columns[2].charAt(0) == MagicNumber.MAGICOPERATOR_LTHAN) {
519             return MagicNumber.MAGICOPERATOR_EQUAL_LTHAN;
520         } else if (unaryLen == 2 && columns[2].charAt(0) == MagicNumber.MAGICOPERATOR_GTHAN) {
521             return MagicNumber.MAGICOPERATOR_EQUAL_GTHAN;
522         } else {
523             throw new ParseException("Unrecognized unary prefix");
524         }
525     }

527     @Nullable
528     public static String resolveReverseDataType(int dataTypeId) {
529         initTypeMap();
530         for (Map.Entry<String, Integer> entry : typeMap.entrySet()) {
531             if (entry.getValue().intValue() == dataTypeId) {
532                 return entry.getKey();
533             }
534         }
535         return null;
536     }

538     private static void initTypeMap() {
539         if (typeMap != null) {
540             return;
541         }
542         typeMap = new TreeMap<>();
543         typeMap.put(MagicNumber.TYPE_KEY_BYTE, MagicNumber.TYPE_BYTE);
544         typeMap.put(MagicNumber.TYPE_KEY_SHORT, MagicNumber.TYPE_SHORT);
545         typeMap.put(MagicNumber.TYPE_KEY_LONG, MagicNumber.TYPE_LONG);
546         typeMap.put(MagicNumber.TYPE_KEY_STRING, MagicNumber.TYPE_STRING);
547         typeMap.put(MagicNumber.TYPE_KEY_DATE, MagicNumber.TYPE_DATE);
548         typeMap.put(MagicNumber.TYPE_KEY_BESHORT, MagicNumber.TYPE_BESHORT);
549         typeMap.put(MagicNumber.TYPE_KEY_BELONG, MagicNumber.TYPE_BELONG);
550         typeMap.put(MagicNumber.TYPE_KEY_BEDATE, MagicNumber.TYPE_BEDATE);
551         typeMap.put(MagicNumber.TYPE_KEY_LESHORT, MagicNumber.TYPE_LESHORT);
552         typeMap.put(MagicNumber.TYPE_KEY_LELONG, MagicNumber.TYPE_LELONG);
553         typeMap.put(MagicNumber.TYPE_KEY_LEDATE, MagicNumber.TYPE_LEDATE);

51      /**
52       * Public method to parse a byte array representing the magic file into a list containing MagicNumber objects which are
53       * also nested with continuations as child MagicNumber instances.
54       *
55       * @param configData the byte[] representing the magic file
56       * @param zeroDepthErrorList logs errors with zero depth entries
57       * @param continuationErrorMap logs errors with continuations - these are entries with depths &gt; 0
58       * @param swallowParseException boolean whether to swallow or propogate ParseExceptions that are IGNORABLE_DATATYPE_MSGS
59       * @return a {@link List}.
60       */
61      public static List<MagicNumber> buildMagicNumberList(byte[] configData, @Nullable List<String> zeroDepthErrorList,
62              @Nullable Map<String, List<String>> continuationErrorMap, boolean swallowParseException) {
63  
64          List<MagicNumber> magicNumberList = new ArrayList<>();
65          MagicNumber finger = null;
66          int currentDepth = -1;
67          List<MagicNumber> extensions = new ArrayList<>();
68          try (BufferedReader reader = new BufferedReader(new InputStreamReader(new ByteArrayInputStream(configData), MagicNumber.DEFAULT_CHARSET))) {
69              String s;
70              int counter = 0;
71              while ((s = reader.readLine()) != null) {
72                  counter++;
73                  if (s == null || s.length() == 0 || s.charAt(0) == '#')
74                      continue;
75                  int depth = getEntryDepth(s);
76                  if (depth < 0)
77                      continue;
78                  try {
79                      if (depth == 0 && extensions.size() > 0) {
80                          if (finger == null) {
81                              extensions = null;
82                              extensions = new ArrayList<>();
83                          } else {
84                              addExtensionsLayer(extensions, finger);
85                              extensions = null;
86                              extensions = new ArrayList<>();
87                              finger = null;
88                          }
89                      }
90                      if (depth == 0) {
91                          MagicNumber item = parseAndStore(magicNumberList, s, swallowParseException);
92                          if (item != null) {
93                              finger = item;
94                              currentDepth = -1;
95                          } else {
96                              finger = null;
97                          }
98                      } else if (depth > 0) {
99  
100                         if (finger == null)
101                             continue;
102                         if (currentDepth < 0)
103                             currentDepth = depth;
104                         if (currentDepth == depth) {
105                             parseAndStore(extensions, s, swallowParseException);
106                         } else if (currentDepth < depth) {
107                             if (extensions.size() == 0) {
108                                 finger = null;
109                                 currentDepth = -1;
110                                 continue;
111                             }
112                             currentDepth = depth;
113                             addExtensionsLayer(extensions, finger);
114                             extensions = null;
115                             extensions = new ArrayList<>();
116                             parseAndStore(extensions, s, swallowParseException);
117                         }
118                     }
119                 } catch (Exception e) {
120 
121                     if (continuationErrorMap == null || zeroDepthErrorList == null)
122                         continue;
123 
124                     if (swallowParseException &&
125                             (e.getClass() == ParseException.class) &&
126                             IGNORABLE_DATATYPE_MSGS.contains(e.getMessage())) {
127                         // swallow this cause said we don't care
128                         continue;
129                     }
130 
131 
132                     if (depth > 0) {
133                         MagicNumber mItem = magicNumberList.get(magicNumberList.size() - 1);
134                         String signature = mItem.toString();
135                         List<String> failedExtensions = continuationErrorMap.get(signature);
136                         if (failedExtensions == null) {
137                             failedExtensions = new ArrayList<>();
138                             continuationErrorMap.put(mItem.toString(), failedExtensions);
139                         }
140                         failedExtensions.add("[MAGIC LINE# " + counter + "] " + s);
141                     } else if (depth == 0) {
142                         zeroDepthErrorList.add("[MAGIC LINE# " + counter + "] " + s);
143                     }
144                 }
145             }
146             if (finger != null && extensions.size() > 0)
147                 addExtensionsLayer(extensions, finger);
148         } catch (IOException ioe) {
149             log.error("Caught IOException on buildMagicNumberList (throwing a runtime exception): {}", ioe.getMessage(), ioe);
150             /** Doing all of this in memory - yes, one could erroneously use one of the IO objects but ... */
151             throw new RuntimeException(ioe);
152         }
153         return magicNumberList;
154     }

167     private static MagicNumber parseAndStore(List<MagicNumber> storage, String entry, boolean swallowParseException) throws Exception {
168         MagicNumber item = buildMagicNumber(entry, swallowParseException);
169         if (item != null)
170             storage.add(item);
171         return item;

199     /**
200      * Parses a magic number entry and prepares a magic number item
201      * 
202      * @param entry line to parse
203      * @param swallowParseException should we swallow Ignorable ParseException or bubble them up
204      */
205     public static MagicNumber buildMagicNumber(String entry, boolean swallowParseException) throws ParseException {
206 
207         String[] columns = prepareEntry(entry);
208         MagicNumberml#MagicNumber">MagicNumber item = new MagicNumber();
209 
210         try {
211             // column A parsing
212             item.depth = getEntryDepth(columns[0]);
213             item.offsetUnary = resolveOffsetUnary(columns, item);
214             item.offset = resolveOffset(columns, item);
215         } catch (Exception e) {
216             throw new ParseException("Error on column 0:" + columns[0] + ". " + e.getMessage());
217         }
218         try {
219             // columb B parsing
220             item.dataType = resolveDataType(columns, item);
221             item.dataTypeLength = getDataTypeByteLength(columns, item);
222             item.mask = resolveMask(columns, item);
223         } catch (Exception e) {
224             if (swallowParseException) {
225                 // This means you put TRUE in SWALLOW_IGNORABLE_EXCEPTIONS in a UnixFilePlace.cfg file
226                 // so let's log at debug level so you can hide these message easily
227                 log.debug("Warning unable to read column 1\t: {} - {}", columns[1], e.getMessage());
228             } else {
229                 log.error("original entry   \t: {}", entry);
230                 log.error("Error on column 1\t: {}", columns[1], e);
231             }
232             throw new ParseException("Parse Error on column 1:" + columns[1] + ". " + e.getMessage());
233         }
234         try {
235             // column C parsing
236             item.unaryOperator = resolveUnary(columns, item);
237             item.value = resolveValue(columns, item);
238             item.dataTypeLength = item.value.length;
239         } catch (Exception e) {
240             throw new ParseException("Error on column 2:" + columns[2] + ". " + e.getMessage());
241         }
242         // column D parsing
243         item.description = columns[3];
244         return item;
245     }

247     /**
248      * Tokenizes an entry into four columns by tab or space while recognizing escape sequences.
249      */
250     private static String[] tokenizeEntry(String entry) {
251         int index = 0;
252         String[] columns = new String[4];
253         columns[0] = EMPTYSTRING;
254         columns[1] = EMPTYSTRING;
255         columns[2] = EMPTYSTRING;
256         columns[3] = EMPTYSTRING;
257         for (int i = 0; i < entry.length(); i++) {
258             char c = entry.charAt(i);
259             if (c == '\\' && i != (entry.length() - 1) && entry.charAt(i + 1) == ' ') {
260                 columns[index] += ' ';
261                 i++;
262             } else if (c == ' ' || c == '\t') {
263                 while (entry.length() > (i + 1)) {
264                     if (entry.charAt(i + 1) == ' ' || entry.charAt(i + 1) == '\t')
265                         i++;
266                     else
267                         break;
268                 }
269                 index++;
270             } else {
271                 columns[index] += c;
272             }
273             if (index == 3) {
274                 if (entry.length() > (i + 1))
275                     columns[index] = entry.substring(i + 1);
276                 break;
277             }
278         }
279         return columns;
280     }

285     private static String[] prepareEntry(String entry) throws ParseException {
286         if (entry == null)
287             throw new ParseException(ENTRY_NOT_NULL_RULE);
288         String subject = entry;
289         int invalidOperatorIndex = subject.indexOf(" = ");
290         if (invalidOperatorIndex > 0) {
291             String tail = subject.length() > (invalidOperatorIndex + 3) ? ' ' + subject.substring(invalidOperatorIndex + 3) : "";
292             subject = subject.substring(0, invalidOperatorIndex) + tail;
293         }
294 
295         String[] columns = tokenizeEntry(subject);
296         for (int count = 0; count < columns.length; count++) {
297             if (count == 3 && columns[count].length() == 0 && !(columns[0].charAt(0) == '>')) {
298                 // columns[count] = NULL_DESCRIPTION;
299 
300             } else if (columns[count].length() == 0 && count < 3) {
301                 throw new ParseException(ENTRY_4COLUMN_RULE);
302             }
303         }
304         return columns;
305     }

310     private static int resolveOffset(String[] columns, MagicNumber item) throws ParseException {
311         String entry = columns[0];
312         if (item.depth > 0)
313             entry = entry.substring(item.depth);
314         if (entry.charAt(0) == '&')
315             entry = entry.substring(1);
316         else if (entry.charAt(0) == '(' && entry.charAt(entry.length() - 1) == ')')
317             entry = entry.substring(1, entry.length() - 1);
318         try {
319             return MagicMath.stringToInt(entry);
320         } catch (NumberFormatException e) {
321             throw new ParseException("Malformatted offset value: " + entry);
322         }
323     }

331     public static int getEntryDepth(String entry) {
332         if (entry.length() == 0 || (entry.charAt(0) != '>' && !Character.isDigit(entry.charAt(0)))) {
333             return -1;
334         }
335         int depth = 0;
336         for (; depth < entry.length(); depth++) {
337             if (entry.charAt(depth) != '>')
338                 break;
339         }
340         return depth;
341     }

346     private static int resolveDataType(String[] columns, MagicNumber item) throws ParseException {
347         initTypeMap();
348 
349         String subject = columns[1];
350         if (subject.startsWith("search")) {
351             throw new ParseException(UNSUPPORTED_DATATYPE_MSG_SEARCH);
352         }
353         if (subject.equals("regex")) {
354             throw new ParseException(UNSUPPORTED_DATATYPE_MSG_REGEX);
355         }
356         if (subject.charAt(0) == 'u' || subject.charAt(0) == 'U') {
357             throw new ParseException(UNSUPORTED_DATATYPE_MSG_UNSIGNED);
358         }
359 
360         // parse out any masking
361         int ix = subject.indexOf("&") > 0 ? subject.indexOf("&") : subject.indexOf("/");
362         if (ix > 0)
363             subject = columns[1].substring(0, ix);
364 
365         int dataTypeId = lookupDataType(subject);
366         if (dataTypeId < 0)
367             throw new ParseException("Unsupported Data Type: " + subject);
368         return dataTypeId;
369     }

371     private static int lookupDataType(String arg) {
372         int dataTypeIdInt = typeMap.get(arg.toUpperCase());
373         switch (dataTypeIdInt) {
374             case MagicNumber.TYPE_DATE:
375                 return -1;
376             case MagicNumber.TYPE_BEDATE:
377                 return -1;
378             case MagicNumber.TYPE_LEDATE:
379                 return -1;
380             default:
381                 return dataTypeIdInt;
382         }
383     }

385     @Nullable
386     private static byte[] resolveMask(String[] columns, MagicNumber item) {
387         int ix = columns[1].indexOf("&");
388         if (ix > 0) {
389             byte[] maskValues = MagicMath.stringToByteArray(columns[1].substring(ix + 1));
390             maskValues = MagicMath.setLength(maskValues, item.dataTypeLength);
391         }
392         return null;
393     }

395     // -----------------------------------------------------------------------
396     // COLUMN C: [UNARY_OPERATOR][Some value like 0x00]
397     // -----------------------------------------------------------------------
398     private static byte[] resolveValue(String[] columns, MagicNumber item) throws ParseException {
399         String subject = columns[2];
400 
401         if (item.dataType == MagicNumber.TYPE_STRING && !(subject.length() == 1 && subject.charAt(0) == 'x')) {
402             byte[] strVal = MagicMath.parseEscapedString(subject);
403             item.dataTypeLength = strVal.length;
404             return strVal;
405         } else if (subject.length() == 1 && subject.charAt(0) == 'x') {
406             item.substitute = true;
407             return new byte[0];
408         }
409 
410         int unaryLen = unaryPrefixLength(subject);
411         if (unaryLen > 0)
412             subject = subject.substring(unaryLen);
413         if (subject.toUpperCase().endsWith("L"))
414             subject = subject.substring(0, subject.length() - 1);
415 
416         byte[] valueArray = MagicMath.stringToByteArray(subject);
417         valueArray = MagicMath.setLength(valueArray, item.dataTypeLength);
418 
419         if (item.mask != null)
420             valueArray = MagicMath.mask(valueArray, item.mask);
421         if (item.dataType == MagicNumber.TYPE_LELONG)
422             MagicMath.longEndianSwap(valueArray, 0);
423         else if (item.dataType == MagicNumber.TYPE_LESHORT)
424             MagicMath.shortEndianSwap(valueArray, 0);
425         return valueArray;
426     }

428     private static int unaryPrefixLength(@Nullable String s) {
429 
430         if (s == null || s.length() == 0)
431             return 0;
432 
433         char op = s.charAt(0);
434         int len = s.length();
435         if (!Character.isDigit(op)) {
436             switch (op) {
437                 case MagicNumber.MAGICOPERATOR_AND:
438                     return 1;
439                 case MagicNumber.MAGICOPERATOR_GTHAN:
440                     return len > 1 && s.charAt(1) == MagicNumber.MAGICOPERATOR_AND ? 2 : 1;
441                 case MagicNumber.MAGICOPERATOR_LTHAN:
442                     return len > 1 && s.charAt(1) == MagicNumber.MAGICOPERATOR_AND ? 2 : 1;
443                 case MagicNumber.MAGICOPERATOR_OR:
444                     return 1;
445                 case MagicNumber.MAGICOPERATOR_BWAND:
446                     return 1;
447                 case MagicNumber.MAGICOPERATOR_BWNOT:
448                     return 1;
449                 case MagicNumber.MAGICOPERATOR_NOT:
450                     return 1;
451                 default:
452                     return 0;
453             }
454         }
455         return 0;
456     }

458     private static int getDataTypeByteLength(String[] columns, MagicNumber item) {
459         int dataTypeId = item.dataType;
460         switch (dataTypeId) {
461             case MagicNumber.TYPE_STRING:
462                 if (item.value == null)
463                     return -1;
464                 else
465                     return item.value.length;
466             case MagicNumber.TYPE_BYTE:
467                 return 1;
468             case MagicNumber.TYPE_SHORT:
469                 return 2;
470             case MagicNumber.TYPE_BESHORT:
471                 return 2;
472             case MagicNumber.TYPE_LESHORT:
473                 return 2;
474             case MagicNumber.TYPE_LONG:
475                 return 4;
476             case MagicNumber.TYPE_BELONG:
477                 return 4;
478             case MagicNumber.TYPE_LELONG:
479                 return 4;
480             case MagicNumber.TYPE_BEDATE:
481                 return 4;
482             case MagicNumber.TYPE_LEDATE:
483                 return 4;
484             default:
485                 return -1;
486         }

489     private static char resolveUnary(String[] columns, MagicNumber item) throws ParseException {
490         int unaryLen = unaryPrefixLength(columns[2]);
491         if (item.dataType == MagicNumber.TYPE_STRING || unaryLen == 0)
492             return MagicNumber.MAGICOPERATOR_DEFAULT;
493         else if (unaryLen == 1)
494             return columns[2].charAt(0);
495         else if (unaryLen == 2 && columns[2].charAt(0) == MagicNumber.MAGICOPERATOR_LTHAN) {
496             return MagicNumber.MAGICOPERATOR_EQUAL_LTHAN;
497         } else if (unaryLen == 2 && columns[2].charAt(0) == MagicNumber.MAGICOPERATOR_GTHAN) {
498             return MagicNumber.MAGICOPERATOR_EQUAL_GTHAN;
499         } else
500             throw new ParseException("Unrecognized unary prefix");
501     }

505         initTypeMap();

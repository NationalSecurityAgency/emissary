127     /**
128      * Tests the sample and if successful provides the description
129      */
130     @Nullable
131     public String describe(byte[] data) {
132         String desc = describeSelf(data);
133         if (desc == null) {
134             return null;
135         }
136         StringBuilder sb = new StringBuilder(desc);
137         return escapeBackspace(describeDependents(data, sb, 0));
138     }

140     /**
141      * Private formatting method for escaping backspaces
142      */
143     private static String escapeBackspace(String desc) {
144         StringBuilder s = new StringBuilder();
145         for (int i = 0; i < desc.length(); i++) {
146             if (desc.charAt(i) == '\\' && (i + 1) < desc.length() && desc.charAt(i + 1) == 'b') {
147                 s = new StringBuilder(s.substring(0, s.length() - 1));
148                 i++;
149                 continue;
150             }
151             s.append(desc.charAt(i));
152         }
153         return s.toString();
154     }

167     /**
168      * Private method to format output - mainly for description substitutions
169      */
170     private String format(String desc, byte[] data) {
171 
172         if (!substitute) {
173             return desc;
174         }
175         Deque<Character> chars = new ArrayDeque<>();
176         for (int i = desc.length() - 1; i >= 0; --i) {
177             chars.push(desc.charAt(i));
178         }
179         StringBuilder sb = new StringBuilder();
180 
181         while (!chars.isEmpty()) {
182             Character next = chars.pop();
183             if (!chars.isEmpty() && next == '%') {
184                 char subType = chars.pop();
185                 if (dataType == TYPE_STRING) {
186                     if (offset < (data.length - 2)) {
187                         String sub = new String(Objects.requireNonNull(getElement(data, offset, 1)), DEFAULT_CHARSET);
188                         sb.append(sub);
189                     }
190                 } else if (subType == 'c' || subType == 's') {
191 
192                     byte[] subData = getElement(data, offset, dataTypeLength);
193                     if (subData != null) {
194                         String sub = new String(subData, DEFAULT_CHARSET);
195                         sb.append(sub);
196                     }
197 
198                 } else {
199 
200                     byte[] subData = getElement(data, offset, dataTypeLength);
201                     if (subData != null) {
202                         String sub = MagicMath.byteArrayToString(subData, 10);
203                         sb.append(sub);
204                     }
205                 }
206 
207                 if (subType == 'l' && !chars.isEmpty() && chars.peek() == 'd') {
208                     chars.pop();
209                 }
210                 continue;
211             }
212             sb.append(next.charValue());
213         }
214         return sb.toString();
215     }

217     /**
218      * Tests dependent children
219      */
220     private String describeDependents(byte[] data, StringBuilder sb, int layer) {
221         log.debug("DESCRIBING DEPENDENTS at layer {}", layer);
222         if (dependencies == null || layer >= dependencies.size()) {
223             log.debug("Not enough dependents for layer {}", layer);
224             return sb.toString();
225         }
226 
227         boolean shouldContinue = false;
228         MagicNumber[] dependentItems = dependencies.get(layer);
229         log.debug("Found {} items at layer {}", dependentItems.length, layer);
230         for (MagicNumber dependentItem : dependentItems) {
231             String s = dependentItem.describeSelf(data);
232 
233             if (s != null) {
234                 if (sb.length() > 0) {
235                     sb.append(' ');
236                 }
237                 sb.append(s);
238                 shouldContinue = true;
239             }
240         }
241 
242         if (!shouldContinue) {
243             return sb.toString();
244         }
245         return describeDependents(data, sb, layer + 1);
246     }

256         return testNumeric(subject);

259     /**
260      * Tests numeric byte data only
261      */
262     private boolean testNumeric(byte[] data) {
263         if (substitute) {
264             return true;
265         }
266         byte[] mValues = value;
267 
268         log.debug("Unary Operator: {}", unaryOperator);
269 
270         int end = mValues.length;
271         switch (unaryOperator) {
272             case MAGICOPERATOR_AND:
273             case MAGICOPERATOR_BWAND:
274                 for (int i = 0; i < end; i++) {
275                     if (data[i] != mValues[i]) {
276                         return false;
277                     }
278                 }
279                 return true;
280             case MAGICOPERATOR_GTHAN:
281                 for (int i = 0; i < end; i++) {
282                     if ((data[i] & 0xFF) < (mValues[i] & 0xFF)) {
283                         return false;
284                     }
285                     if (i == end - 1 && data[i] == mValues[i]) {
286                         return false;
287                     }
288                 }
289                 return true;
290             case MAGICOPERATOR_LTHAN:
291                 for (int i = 0; i < end; i++) {
292                     if ((data[i] & 0xFF) > (mValues[i] & 0xFF)) {
293                         return false;
294                     }
295                     if (i == end - 1 && data[i] == mValues[i]) {
296                         return false;
297                     }
298                 }
299                 return true;
300             case MAGICOPERATOR_OR:
301                 for (int i = 0; i < end; i++) {
302                     if (data[i] == mValues[i]) {
303                         return true;
304                     }
305                 }
306                 return false;
307             case MAGICOPERATOR_BWNOT:
308             case MAGICOPERATOR_NOT:
309                 for (int i = 0; i < end; i++) {
310                     if (data[i] != mValues[i]) {
311                         return true;
312                     }
313                 }
314                 return false;
315             case MAGICOPERATOR_EQUAL_GTHAN:
316                 for (int i = 0; i < end; i++) {
317                     if ((data[i] & 0xFF) < (mValues[i] & 0xFF)) {
318                         return false;
319                     }
320                 }
321                 return true;
322             case MAGICOPERATOR_EQUAL_LTHAN:
323                 for (int i = 0; i < end; i++) {
324                     if ((data[i] & 0xFF) > (mValues[i] & 0xFF)) {
325                         return false;
326                     }
327                 }
328                 return true;
329             default:
330                 throw new IllegalStateException(
331                         "This MagicNumber instance is configured incorrectly. The unary operator is set to an unknown or unconfigured value.");
332 
333         }
334     }

336     /**
337      * Retrieves the data sample
338      */
339     @Nullable
340     private static byte[] getElement(@Nullable byte[] data, int offset, int length) {
341         if (data == null) {
342             return null;
343         }
344         if (data.length < (offset + length)) {
345             return null;
346         }
347         byte[] subject = new byte[length];
348         System.arraycopy(data, offset, subject, 0, subject.length);
349         return subject;
350     }

363     /**
364      * Re-creates the string magic number entry for this number only
365      *
366      * @return a String represention of the entry
367      */
368     @Override
369     public String toString() {
370 
371         StringBuilder sb = new StringBuilder();
372         sb.append(">".repeat(Math.max(0, depth)));
373         if (offsetUnary > 0) {
374             sb.append(offsetUnary);
375         }
376 
377         if (offset == 0) {
378             sb.append("0");
379         } else {
380             sb.append(MagicMath.HEX_PREFIX);
381             sb.append(Integer.toHexString(offset));
382         }
383 
384         sb.append('\t');
385         sb.append(MagicNumberFactory.resolveReverseDataType(dataType));
386         if (mask != null && mask.length > 0) {
387             sb.append('&');
388             sb.append(MagicMath.byteArrayToHexString(mask));
389         }
390 
391         sb.append('\t');
392         if (unaryOperator == MAGICOPERATOR_EQUAL_LTHAN) {
393             sb.append("<=");
394         } else if (unaryOperator == MAGICOPERATOR_EQUAL_GTHAN) {
395             sb.append(">=");
396         } else {
397             sb.append(unaryOperator);
398         }
399         sb.append(MagicMath.byteArrayToHexString(value));
400 
401         sb.append('\t');
402         sb.append(description);
403         return sb.toString();
404     }

406     /**
407      * Private method to create the string plus continuations
408      */
409     private String toString(@Nullable StringBuilder sbuf, int depth) {
410         StringBuilder sb = sbuf;
411         int d = depth;
412         if (sb == null) {
413             sb = new StringBuilder(description);
414         }
415         if (dependencies == null || d >= dependencies.size()) {
416             return sb.toString();
417         }
418         MagicNumber[] dependentItems = dependencies.get(d);
419         for (MagicNumber dependentItem : dependentItems) {
420             sb.append('\n');
421             sb.append(dependentItem.toString());
422         }
423         return toString(sb, d + 1);
424     }

46      public static String byteArrayToHexString(byte[] b) {
47          StringBuilder sb = new StringBuilder();
48          for (int i = 0; i < b.length; i++) {
49              if (i == 0) {
50                  sb.append(HEX_PREFIX);
51              }
52              sb.append(Integer.toString((int) b[i], 16));
53          }
54          return sb.toString();
55      }

57      public static byte[] parseEscapedString(String s) {
58          List<Number> array = new ArrayList<>();
59          Deque<Character> chars = new ArrayDeque<>();
60          for (int i = (s.length() - 1); i >= 0; i--) {
61              chars.push(s.charAt(i));
62          }
63          while (!chars.isEmpty()) {
64              Character c = chars.pop();
65              String val = EMPTYSTRING;
66              if (c == '\\') {
67                  if (chars.isEmpty()) {
68                      array.add(32);
69                      break;
70                  }
71                  Character next = chars.peek();
72                  if (literals[next] > 0) {
73                      array.add(literals[next]);
74                      chars.pop();
75                  } else if (Character.isDigit(next)) {
76                      int max = 3;
77                      while (!chars.isEmpty() && Character.isDigit(next) && max-- > 0) {
78                          val += chars.pop();
79                          if (!chars.isEmpty()) {
80                              next = chars.peek();
81                          }
82                      }
83                      array.add(new BigInteger(val, 8));
84                      val = EMPTYSTRING;
85                  } else if (next == 'x') {
86                      chars.pop(); // pop the hex symbol
87                      val += chars.pop();
88                      val += chars.pop();
89                      array.add(new BigInteger(val, 16));
90                      val = EMPTYSTRING;
91                  }
92                  continue;
93              }
94              array.add((int) c);
95          }
96          byte[] bytes = new byte[array.size()];
97          Iterator<Number> iter = array.iterator();
98          for (int i = 0; i < bytes.length; i++) {
99              Number num = iter.next();
100             bytes[i] = num.byteValue();
101         }
102         return bytes;
103     }

105     public static byte[] stringToByteArray(String s) {
106         if (s.startsWith(HEX_PREFIX)) {
107             return hexStringToByteArray(s);
108         } else if (!s.equals(ZERO) && s.startsWith(PRE_OCT)) {
109             return octalStringToByteArray(s.substring(1));
110         } else {
111             return decimalStringToByteArray(s);
112         }
113     }

115     @Nullable
116     public static byte[] stringToByteArray(int arraySize, @Nullable String stringValue) {
117         if (stringValue == null || stringValue.length() == 0) {
118             return null;
119         }
120         if (stringValue.length() > 2 && HEX_PREFIX.equals(stringValue.substring(0, 2))) {
121             return hexStringToByteArray(stringValue);
122         } else {
123             return integerToByteArray(arraySize, stringToLong(stringValue));
124         }
125     }

136     public static byte[] setLength(byte[] data, int length) {
137 
138         int actualSize = data.length;
139         for (int i = 0; i < data.length; i++) {
140             if (data[i] == 0) {
141                 actualSize--;
142             } else {
143                 break;
144             }
145         }
146         if (data.length == length) {
147             return data;
148         } else if (actualSize > length) {
149             throw new ByteArrayPrecisionException(BYTEARRAY_PRECISION_ERROR_RULE);
150         }
151 
152         if (length == 0) {
153             return new byte[0];
154         }
155 
156         byte[] newValues = new byte[length];
157         int ix = data.length - 1;
158         for (int i = (length - 1); i >= 0; i--) {
159             if (ix < 0) {
160                 newValues[i] = (byte) 0;
161             } else {
162                 newValues[i] = data[ix--];
163             }
164         }
165         return newValues;
166     }

168     public static byte[] mask(byte[] data, byte[] maskValues) {
169         byte[] target = new byte[data.length];
170         for (int i = 0; i < data.length; i++) {
171             target[i] = (byte) (data[i] & maskValues[i]);
172         }
173         return target;
174     }

176     public static byte[] hexStringToByteArray(String s) {
177         String subject = s;
178         if (subject.startsWith(HEX_PREFIX)) {
179             subject = subject.substring(2);
180         }
181         if (subject.length() % 2 != 0) {
182             subject = ZERO + subject;
183         }
184         byte[] array = new byte[subject.length() / 2];
185         for (int i = 0; i < array.length; i++) {
186             int b = Integer.parseInt(subject.substring(i * 2, i * 2 + 2), 16);
187             array[i] = (byte) (0xff & b);
188         }
189         return array;
190     }

196     public static int stringToInt(String s) {
197         if (s.startsWith(HEX_PREFIX)) {
198             return new BigInteger(s.substring(2), 16).intValue();
199         } else if (!s.equals("0") && s.startsWith(PRE_OCT)) {
200             return new BigInteger(s.substring(1), 8).intValue();
201         } else {
202             return new BigInteger(s, 10).intValue();
203         }
204     }

207     public static byte[] integerToByteArray(int arraySize, long integerValue) {
208         byte[] valueBytes = new byte[arraySize];
209         for (int i = 0; i < arraySize; i++) {
210             valueBytes[arraySize - i - 1] = (byte) (integerValue >>> (i * 8) & 0xff);
211         }
212         return valueBytes;
213     }

215     public static long stringToLong(String stringValue) {
216         if (stringValue.length() > 2 && HEX_PREFIX.equals(stringValue.substring(0, 2))) {
217             return Long.parseLong(stringValue.substring(2), 16);
218         } else if (stringValue.length() > 1 && stringValue.charAt(0) == '0') {
219             return Long.parseLong(stringValue.substring(1), 8);
220         } else {
221             return Long.parseLong(stringValue, 10);
222         }
223     }

237     public static String byteArrayToString(byte[] data, int radix) {
238         int actualSize = data.length;
239         for (int i = 0; i < data.length; i++) {
240             if (data[i] == 0) {
241                 actualSize--;
242             } else {
243                 break;
244             }
245         }
246         if (actualSize == 0) {
247             return ZERO;
248         }
249         byte[] adjustedData = setLength(data, actualSize);
250         BigInteger value = new BigInteger(adjustedData);
251         return value.toString(radix);
252     }

269     public static void longEndianSwap(byte[] array, int offset) {
270 
271         if (array.length < (offset + 4)) {
272             throw new ArrayIndexOutOfBoundsException(array.length + 1);
273         }
274         byte t = array[offset];
275         array[offset] = array[offset + 3];
276         array[offset + 3] = t;
277         t = array[offset + 1];
278         array[offset + 1] = array[offset + 2];
279         array[offset + 2] = t;
280 
281     }

284         if (array.length < (offset + 2)) {
285             throw new ArrayIndexOutOfBoundsException(array.length + 1);
286         }
287 
288         byte t = array[offset];
289         array[offset] = array[offset + 1];

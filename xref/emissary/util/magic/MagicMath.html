48      public static String byteArrayToHexString(byte[] b) {
49          StringBuilder sb = new StringBuilder();
50          for (int i = 0; i < b.length; i++) {
51              if (i == 0) {
52                  sb.append(HEX_PREFIX);
53              }
54              sb.append(Integer.toString((int) b[i], 16));
55          }
56          return sb.toString();
57      }

59      public static byte[] parseEscapedString(String s) {
60          List<Number> array = new ArrayList<>();
61          Deque<Character> chars = new ArrayDeque<>();
62          for (int i = s.length() - 1; i >= 0; i--) {
63              chars.push(s.charAt(i));
64          }
65          while (!chars.isEmpty()) {
66              Character c = chars.pop();
67              String val = EMPTYSTRING;
68              if (c == '\\') {
69                  if (chars.isEmpty()) {
70                      array.add(32);
71                      break;
72                  }
73                  Character next = chars.peek();
74                  if (literals[next] > 0) {
75                      array.add(literals[next]);
76                      chars.pop();
77                  } else if (Character.isDigit(next)) {
78                      int max = 3;
79                      while (!chars.isEmpty() && Character.isDigit(next) && max-- > 0) {
80                          val += chars.pop();
81                          if (!chars.isEmpty()) {
82                              next = chars.peek();
83                          }
84                      }
85                      array.add(new BigInteger(val, 8));
86                      val = EMPTYSTRING;
87                  } else if (next == 'x') {
88                      chars.pop(); // pop the hex symbol
89                      val += chars.pop();
90                      val += chars.pop();
91                      array.add(new BigInteger(val, 16));
92                      val = EMPTYSTRING;
93                  }
94                  continue;
95              }
96              array.add((int) c);
97          }
98          byte[] bytes = new byte[array.size()];
99          Iterator<Number> iter = array.iterator();
100         for (int i = 0; i < bytes.length; i++) {
101             Number num = iter.next();
102             bytes[i] = num.byteValue();
103         }
104         return bytes;
105     }

107     public static byte[] stringToByteArray(String s) {
108         if (s.startsWith(HEX_PREFIX)) {
109             return hexStringToByteArray(s);
110         } else if (!s.equals(ZERO) && s.startsWith(PRE_OCT)) {
111             return octalStringToByteArray(s.substring(1));
112         } else {
113             return decimalStringToByteArray(s);
114         }
115     }

118     public static byte[] stringToByteArray(int arraySize, @Nullable String stringValue) {
119         if (stringValue == null || stringValue.length() == 0) {
120             return null;
121         }
122         if (stringValue.length() > 2 && HEX_PREFIX.equals(stringValue.substring(0, 2))) {
123             return hexStringToByteArray(stringValue);
124         } else {
125             return integerToByteArray(arraySize, stringToLong(stringValue));
126         }

138     public static byte[] setLength(byte[] data, int length) {
139 
140         int actualSize = data.length;
141         for (int i = 0; i < data.length; i++) {
142             if (data[i] == 0) {
143                 actualSize--;
144             } else {
145                 break;
146             }
147         }
148         if (data.length == length) {
149             return data;
150         } else if (actualSize > length) {
151             throw new ByteArrayPrecisionException(BYTEARRAY_PRECISION_ERROR_RULE);
152         }
153 
154         if (length == 0) {
155             return new byte[0];
156         }
157 
158         byte[] newValues = new byte[length];
159         int ix = data.length - 1;
160         for (int i = length - 1; i >= 0; i--) {
161             if (ix < 0) {
162                 newValues[i] = (byte) 0;
163             } else {
164                 newValues[i] = data[ix--];
165             }
166         }
167         return newValues;
168     }

170     public static byte[] mask(byte[] data, byte[] maskValues) {
171         byte[] target = new byte[data.length];
172         for (int i = 0; i < data.length; i++) {
173             target[i] = (byte) (data[i] & maskValues[i]);
174         }
175         return target;
176     }

178     public static byte[] hexStringToByteArray(String s) {
179         String subject = s;
180         if (subject.startsWith(HEX_PREFIX)) {
181             subject = subject.substring(2);
182         }
183         if (subject.length() % 2 != 0) {
184             subject = ZERO + subject;
185         }
186         byte[] array = new byte[subject.length() / 2];
187         for (int i = 0; i < array.length; i++) {
188             int b = Integer.parseInt(subject.substring(i * 2, i * 2 + 2), 16);
189             array[i] = (byte) (0xff & b);
190         }
191         return array;
192     }

198     public static int stringToInt(String s) {
199         if (s.startsWith(HEX_PREFIX)) {
200             return new BigInteger(s.substring(2), 16).intValue();
201         } else if (!s.equals("0") && s.startsWith(PRE_OCT)) {
202             return new BigInteger(s.substring(1), 8).intValue();
203         } else {
204             return new BigInteger(s, 10).intValue();
205         }
206     }

209     public static byte[] integerToByteArray(int arraySize, long integerValue) {
210         byte[] valueBytes = new byte[arraySize];
211         for (int i = 0; i < arraySize; i++) {
212             valueBytes[arraySize - i - 1] = (byte) (integerValue >>> (i * 8) & 0xff);
213         }
214         return valueBytes;
215     }

217     public static long stringToLong(String stringValue) {
218         if (stringValue.length() > 2 && HEX_PREFIX.equals(stringValue.substring(0, 2))) {
219             return Long.parseLong(stringValue.substring(2), 16);
220         } else if (stringValue.length() > 1 && stringValue.charAt(0) == '0') {
221             return Long.parseLong(stringValue.substring(1), 8);
222         } else {
223             return Long.parseLong(stringValue, 10);
224         }
225     }

239     public static String byteArrayToString(byte[] data, int radix) {
240         int actualSize = data.length;
241         for (int i = 0; i < data.length; i++) {
242             if (data[i] == 0) {
243                 actualSize--;
244             } else {
245                 break;
246             }
247         }
248         if (actualSize == 0) {
249             return ZERO;
250         }
251         byte[] adjustedData = setLength(data, actualSize);
252         BigInteger value = new BigInteger(adjustedData);
253         return value.toString(radix);

271     public static void longEndianSwap(byte[] array, int offset) {
272 
273         if (array.length < (offset + 4)) {
274             throw new ArrayIndexOutOfBoundsException(array.length + 1);
275         }
276         byte t = array[offset];
277         array[offset] = array[offset + 3];
278         array[offset + 3] = t;
279         t = array[offset + 1];
280         array[offset + 1] = array[offset + 2];
281         array[offset + 2] = t;
282 
283     }

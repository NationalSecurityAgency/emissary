5   /**
6    * Listener interface for disk space monitoring events. Implementations receive callbacks when disk space thresholds are
7    * crossed.
8    */
9   public interface DiskSpaceListener {
10  
11      /**
12       * Called when disk space usage exceeds the configured pause threshold.
13       *
14       * @param path the filesystem path being monitored
15       * @param usedPercent the current disk usage as a percentage (0-100)
16       * @param freeBytes the current number of free bytes available
17       */
18      void onDiskSpaceExceeded(Path path, double usedPercent, long freeBytes);
19  
20      /**
21       * Called when disk space usage falls below the configured resume threshold.
22       *
23       * @param path the filesystem path being monitored
24       * @param usedPercent the current disk usage as a percentage (0-100)
25       * @param freeBytes the current number of free bytes available
26       */
27      void onDiskSpaceRecovered(Path path, double usedPercent, long freeBytes);
28  }

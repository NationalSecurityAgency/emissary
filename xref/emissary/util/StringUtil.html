5   public class StringUtil {
6   
7       /**
8        * This function corrects the string before placing it in an XML doc (not all characters are allowed in XML)
9        *
10       * @param in String the raw string
11       * @return String the corrected string where invalid characters become "/0x"+ascii value of character
12       */
13      public static String escapeInvalidXml(String in) {
14          // this is stolen from the implementation of Verifier.checkCharacterData(in);
15          for (int i = 0, len = in.length(); i < len; i++) {
16              char c = in.charAt(i);
17              if (!(c > 0x1F && c < 0xD800) && !Verifier.isXMLCharacter(c)) {
18                  return in.substring(0, i) + "/0x" + Integer.toHexString(c) + escapeInvalidXml(in.substring(i + 1));
19              }
20          }
21          return in;
22      }
23  
24      /** This class is not meant to be instantiated. */
25      private StringUtil() {}

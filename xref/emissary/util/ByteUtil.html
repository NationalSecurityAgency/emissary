125     /**
126      * Check if byte at position in array is a control or blank space byte
127      *
128      * @param b a byte array
129      * @param pos a position in the byte array
130      * @return true if byte at pos in array b is a control or blank space byte
131      */
132     public static boolean isControlOrBlankSpace(byte[] b, int pos) {
133         if (b[pos] == Ascii_DEL || b[pos] <= Ascii_SP) {
134             return true;
135         }
136         if (b[pos] == Ascii_b && pos > 0 && b[pos - 1] == Ascii_ESC) {
137             return true;
138         }
139 
140         // Check if the current pos is the first byte in a UTF-8 C1
141         // control character (U+0080..U+009f).
142         final int curr = b[pos] & 0xff;
143         final int next = (pos < (b.length - 1)) ? (b[pos + 1] & 0xff) : -1;
144         if ((curr == 0xc2) && (next >= 0x80) && (next <= 0x9f)) {
145             return true;
146         }
147 
148         // Check if the current pos is the second byte in a UTF-8 C1
149         // control character (U+0080..U+009f).
150         final int prev = (pos > 0) ? (b[pos - 1] & 0xff) : -1;
151         return (prev == 0xc2) && (curr >= 0x80) && (curr <= 0x9f);
152     }

154     /**
155      * Glue two byte arrays together into one
156      * 
157      * @param a the first byte array
158      * @param b the second byte array
159      * @return the whole
160      */
161     public static byte[] glue(@Nullable byte[] a, @Nullable byte[] b) {
162         if (a == null) {
163             return b;
164         }
165         if (b == null) {
166             return a;
167         }
168         return glue(a, 0, a.length - 1, b, 0, b.length - 1);
169     }

171     /**
172      * Glue three byte arrays together into one
173      * 
174      * @param a the first byte array
175      * @param b the second byte array
176      * @param c the third byte array
177      * @return the whole
178      */
179     public static byte[] glue(@Nullable byte[] a, @Nullable byte[] b, @Nullable byte[] c) {
180         if (a == null) {
181             return glue(b, c);
182         }
183         if (b == null) {
184             return glue(a, c);
185         }
186         if (c == null) {
187             return glue(a, b);
188         }
189         return glue(a, 0, a.length - 1, b, 0, b.length - 1, c, 0, c.length - 1);
190     }

192     /**
193      * Glue two byte arrays together into one
194      * 
195      * @param a the first byte array
196      * @param astart starting position in a
197      * @param aend ending position in a
198      * @param b the second byte array
199      * @param bstart starting position in b
200      * @param bend ending position in b
201      * @return the whole
202      */
203     public static byte[] glue(byte[] a, int astart, int aend, byte[] b, int bstart, int bend) {
204         int alen = aend - astart + 1;
205         int blen = bend - bstart + 1;
206 
207         byte[] rslt = new byte[alen + blen];
208         System.arraycopy(a, astart, rslt, 0, alen);
209         System.arraycopy(b, bstart, rslt, alen, blen);
210         return rslt;
211     }

213     /**
214      * Glue three byte arrays together into one
215      * 
216      * @param a the first byte array
217      * @param astart starting position in a
218      * @param aend ending position in a
219      * @param b the second byte array
220      * @param bstart starting position in b
221      * @param bend ending position in b
222      * @param c the third byte array
223      * @param cstart starting position in c
224      * @param cend ending position in c
225      * @return the whole
226      */
227     public static byte[] glue(byte[] a, int astart, int aend, byte[] b, int bstart, int bend, byte[] c, int cstart, int cend) {
228         int alen = aend - astart + 1;
229         int blen = bend - bstart + 1;
230         int clen = cend - cstart + 1;
231 
232         byte[] rslt = new byte[alen + blen + clen];
233         System.arraycopy(a, astart, rslt, 0, alen);
234         System.arraycopy(b, bstart, rslt, alen, blen);
235         System.arraycopy(c, cstart, rslt, alen + blen, clen);
236         return rslt;
237     }

239     /**
240      * Split a byte array at the specified position
241      * 
242      * @param a the byte array
243      * @param pos the split position (a[pos] goes to the second part)
244      */
245     public static List<byte[]> split(@Nullable byte[] a, int pos) {
246         List<byte[]> list = new ArrayList<>();
247         if (a != null && pos > 0 && pos <= a.length) {
248             byte[] part1 = new byte[pos];
249             byte[] part2 = new byte[a.length - pos];
250             System.arraycopy(a, 0, part1, 0, pos);
251             System.arraycopy(a, pos, part2, 0, part2.length);
252             list.add(part1);
253             list.add(part2);
254         } else {
255             // Just give back the original
256             list.add(a);
257         }
258         return list;
259     }

261     /**
262      * Given a byte-array and a start offset, return a string of the bytes between the start position and a carriage return
263      * byte. In essence, this is grabbing a line of input where the byte array is composed of several lines of input.
264      * 
265      * @param data The byte array of input data.
266      * @param pos The initial start offset.
267      * @return A string created from the bytes found from the start offset to the carriage return byte.
268      */
269     public static String grabLine(byte[] data, int pos) {
270         String ret = null;
271         int eolnPos = -1;
272         for (int i = pos; i < data.length; i++) {
273             if (data[i] == '\n') {
274                 eolnPos = i;
275                 break;
276             }
277         }
278         if (eolnPos != -1) {
279             // String up to the found \n pos
280             ret = new String(data, pos, eolnPos - pos + 1);
281         } else {
282             // String to end of buffer
283             ret = new String(data, pos, data.length - pos);
284         }
285         return ret;
286     }

289      * Scans a byte array looking for non-printable values.
290      * 
291      * @param bytes the bytes to be scanned.
292      * @return whether or not there were non-printable values.
293      */
294     public static boolean hasNonPrintableValues(final byte[] bytes) {
295         boolean badCharacters = false;
296 
297         for (byte aByte : bytes) {
298             if (aByte < 9 || (aByte > 13 && aByte < 32)) {
299                 badCharacters = true;
300                 break;
301             }
302         }
303 
304         return badCharacters;
305     }

307     /**
308      * Creates a hex string of a sha256 hash for a byte[].
309      * 
310      * @param bytes to be hashed
311      * @return the hex string of a sha256 hash of the bytes.
312      */
313     @Nullable
314     public static String sha256Bytes(final byte[] bytes) {
315         try {
316             final MessageDigest md = MessageDigest.getInstance("SHA-256");
317             final byte[] hash = md.digest(bytes);
318 
319             final StringBuilder hexString = new StringBuilder(2 * hash.length);
320             for (byte b : hash) {
321                 final String hex = Integer.toHexString(0xff & b);
322 
323                 if (hex.length() == 1) {
324                     hexString.append('0');
325                 }
326                 hexString.append(hex);
327             }
328             return hexString.toString();
329         } catch (NoSuchAlgorithmException e) {
330             return null;
331         }
332     }

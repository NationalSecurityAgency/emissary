17  /**
18   * Help determine type of data from various data file mappings The name of the file gives some context to the mappings
19   * it provides.
20   */
21  public class TypeEngine {
22      // Private logger
23      private static final Logger logger = LoggerFactory.getLogger(TypeEngine.class);
24  
25      // Map of name to Configurator
26      protected Map<String, Configurator> contextMapping = new HashMap<>();
27  
28      // Map of name to Map for extra mappings
29      protected Map<String, Map<String, String>> extraMapping = new HashMap<>();
30  
31      /**
32       * Configure from a Configurator of the delegating class
33       */
34      public TypeEngine(Configurator configG) {
35          List<String> l = configG.findEntries("TYPE_ENGINE_FILE");
36          if (CollectionUtils.isNotEmpty(l)) {
37              configure(l);
38          }
39      }
40  
41      /**
42       * Configure from a list of config files
43       */
44      public TypeEngine(List<String> configFiles) {
45          configure(configFiles);
46      }
47  
48      /**
49       * Just build an empty one
50       */
51      public TypeEngine() {}
52  
53      /**
54       * Configure it
55       */
56      public void configure(@Nullable List<String> configFiles) {
57  
58          if (configFiles == null) {
59              logger.info("No files specified for type engine");
60              return;
61          }
62  
63          for (String name : configFiles) {
64              try {
65                  Configurator c = ConfigUtil.getConfigInfo(name);
66                  String engineName = c.findStringEntry("ENGINE_TYPE", name);
67                  logger.debug("TypeEngine loaded {} as {}", name, engineName);
68                  contextMapping.put(engineName, c);
69              } catch (IOException e) {
70                  logger.error("TypeEngine unable to read {}", name, e);
71              }
72          }
73      }
74  
75      /**
76       * Look up label in specified engine
77       * 
78       * @param engine name of the engine to use
79       * @param label LHS part of equation to lookup
80       * @return RHS part of mapping or null if none found
81       */
82      @Nullable
83      public String getForm(@Nullable String engine, @Nullable String label) {
84  
85          // check params
86          if (engine == null || label == null) {
87              logger.debug("Cannot process null arg engine={}, label={}", engine, label);
88              return null;
89          }
90  
91          String ret = null;
92  
93          // Look up an override mapping
94          Map<String, String> extra = extraMapping.get(engine);
95          if (extra != null) {
96              ret = extra.get(label.toUpperCase(Locale.getDefault()));
97          }
98  
99          // Grab the specified engine and do the default lookup
100         if (ret == null) {
101             Configurator c = contextMapping.get(engine);
102             if (c != null) {
103                 ret = c.findStringEntry(label.toUpperCase(Locale.getDefault()), null);
104                 if (logger.isDebugEnabled() && ret != null) {
105                     logger.debug("Found {} while looking up type for {}", ret, label.toUpperCase(Locale.getDefault()));
106                 }
107             }
108         }
109 
110         return ret;
111     }
112 
113     /**
114      * Add an extra mapping into the specified engine
115      */
116     public void addType(String engine, String label, String value) {
117         Map<String, String> extra = extraMapping.computeIfAbsent(engine, k -> new HashMap<>());
118         extra.put(label, value);
119     }
120 
121 
122     /**
123      * Chop down to file extension and look it up in specified engine
124      */
125     @Nullable
126     public String getFormByExtension(String engine, @Nullable String fn) {
127         if (fn == null) {
128             return null;
129         }
130 
131         int idx = fn.lastIndexOf('.');
132         if (idx != -1) {
133             return getForm(engine, fn.substring(idx + 1));
134         }
135         return getForm(engine, fn);
136     }
137 }

7   import org.slf4j.Logger;
8   import org.slf4j.LoggerFactory;

10  import java.io.File;
11  import java.io.IOException;
12  import java.util.Set;
13  
14  /**
15   * Class that performs simple checks to determine if an executable, directory, and/or file exists on the host machine.
16   */
17  public class DependencyCheck {
18  
19      private static final Logger logger = LoggerFactory.getLogger(DependencyCheck.class);
20  
21      Configurator config;
22  
23      public static final String REQUIRED_EXECUTABLE = "REQUIRED_EXECUTABLE";
24      public static final String REQUIRED_DIRECTORY = "REQUIRED_DIRECTORY";
25      public static final String REQUIRED_FILE = "REQUIRED_FILE";
26      public static final String OPTIONAL_EXECUTABLE = "OPTIONAL_EXECUTABLE";
27      public static final String OPTIONAL_DIRECTORY = "OPTIONAL_DIRECTORY";
28      public static final String OPTIONAL_FILE = "OPTIONAL_FILE";
29  
30      public DependencyCheck() throws IOException {
31          config = ConfigUtil.getConfigInfo(DependencyCheck.class);
32      }
33  
34      public DependencyCheck(Configurator config) {
35          this.config = config;
36      }
37  
38      /**
39       * Static method to determine if an executable is found.
40       * 
41       * @param executable name (can be fully qualified path).
42       * @return return true if any of the executable is found, false otherwise.
43       */
44      public static boolean executableExists(String executable) {
45          boolean exists = false;
46          String[] command = new String[] {"which", executable};
47          Executrix exec = new Executrix();
48          int returnVal = exec.execute(command);
49          if (returnVal == 0) {
50              exists = true;
51          }
52          return exists;
53      }
54  
55      /**
56       * Static method to determine if a fully qualified directory exists
57       *
58       * @param pathToDir path to the directory to check
59       * @return true if exist, false otherwise.
60       */
61      public static boolean directoryExists(String pathToDir) {
62          boolean dirExists = false;
63          {
64              try {
65                  dirExists = new File(pathToDir).getCanonicalFile().isDirectory();
66              } catch (IOException e) {
67                  dirExists = false;
68              }
69          }
70          return dirExists;
71      }
72  
73      /**
74       * Static method to determine if a fully qualified file exists
75       *
76       * @param pathToFile path to the file to check
77       * @return true if exists, false otherwise.
78       */
79      public static boolean fileExists(String pathToFile) {
80          boolean dirExists = false;
81          {
82              try {
83                  dirExists = new File(pathToFile).getCanonicalFile().isFile();
84              } catch (IOException e) {
85                  dirExists = false;
86              }
87          }
88          return dirExists;
89      }
90  
91      public Set<String> getDependencies(String key) {
92          return config.findEntriesAsSet(key);
93      }
94  
95      public void printDependencyReport() {
96          Set<String> reqExeSet = getDependencies(REQUIRED_EXECUTABLE);
97          Set<String> reqDirSet = getDependencies(REQUIRED_DIRECTORY);
98          Set<String> reqFileSet = getDependencies(REQUIRED_FILE);
99          Set<String> optExeSet = getDependencies(OPTIONAL_EXECUTABLE);
100         Set<String> optDirSet = getDependencies(OPTIONAL_DIRECTORY);
101         Set<String> optFileSet = getDependencies(OPTIONAL_FILE);
102 
103         for (String reqExe : reqExeSet) {
104             boolean exists = DependencyCheck.executableExists(reqExe);
105             logger.info("RequiredExecutable: {} exists: {}", reqExe, exists);
106         }
107         for (String reqDir : reqDirSet) {
108             boolean exists = DependencyCheck.directoryExists(reqDir);
109             logger.info("RequiredDirectory {} exists: {}", reqDir, exists);
110         }
111         for (String reqFile : reqFileSet) {
112             boolean exists = DependencyCheck.fileExists(reqFile);
113             logger.info("RequiredFile: {} exists: {}", reqFile, exists);
114         }
115         for (String optExe : optExeSet) {
116             boolean exists = DependencyCheck.executableExists(optExe);
117             logger.info("OptionalExecutable: {} exists: {}", optExe, exists);
118         }
119         for (String optDir : optDirSet) {
120             boolean exists = DependencyCheck.directoryExists(optDir);
121             logger.info("OptionalDirectory {} exists: {}", optDir, exists);
122         }
123         for (String optFile : optFileSet) {
124             boolean exists = DependencyCheck.fileExists(optFile);
125             logger.info("OptionalFile: {} exists: {}", optFile, exists);
126         }
127     }
128 
129     public static void main(String[] args) throws IOException {
130         DependencyCheck dependencyCheck = new DependencyCheck();
131         dependencyCheck.printDependencyReport();
132     }
133 
134 }

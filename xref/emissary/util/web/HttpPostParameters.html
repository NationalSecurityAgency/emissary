1   package emissary.util.web;
2   
3   import jakarta.annotation.Nullable;
4   import org.slf4j.Logger;
5   import org.slf4j.LoggerFactory;
6   
7   import java.io.UnsupportedEncodingException;
8   import java.net.URLEncoder;
9   import java.nio.charset.StandardCharsets;
10  
11  /**
12   * Hold onto fielded data that should be posted to a URL in x-www-urlencoded format.
13   *
14   * For convenience also keep it around in unencoded format
15   */
16  public class HttpPostParameters {
17  
18      private static final Logger logger = LoggerFactory.getLogger(HttpPostParameters.class);
19  
20      private final StringBuilder thePostData = new StringBuilder();
21      private final StringBuilder thePostDataEncoded = new StringBuilder();
22  
23      /**
24       * Create an empty parameter list
25       */
26      public HttpPostParameters() {}
27  
28      /**
29       * Create and add one parameter
30       * 
31       * @param s the field name
32       * @param v the value
33       */
34      public HttpPostParameters(final String s, final String v) {
35          this.add(s, v);
36      }
37  
38      /**
39       * Add a field and value
40       * 
41       * @param s the field name
42       * @param v the value (can be null)
43       */
44      public void add(final String s, @Nullable final String v) {
45          if (this.thePostDataEncoded.length() != 0) {
46              this.thePostData.append("&");
47              this.thePostDataEncoded.append("&");
48          }
49  
50          this.thePostData.append(s).append("=");
51          try {
52              this.thePostDataEncoded.append(URLEncoder.encode(s, StandardCharsets.UTF_8.name())).append("=");
53          } catch (NullPointerException e) {
54              logger.error("Null first parameter to add method.", e);
55          } catch (UnsupportedEncodingException e) {
56              logger.error("Bad encoding UTF-8", e);
57          }
58          if (v != null) {
59              this.thePostData.append(v);
60              try {
61                  this.thePostDataEncoded.append(URLEncoder.encode(v, "UTF-8"));
62              } catch (UnsupportedEncodingException e1) {
63                  logger.error("Bad encoding UTF-8", e1);
64              }
65          }
66      }
67  
68      /**
69       * Get the length of the encoded stuff for the Content-length header
70       */
71      public int length() {
72          return this.thePostDataEncoded.length();
73      }
74  
75      /**
76       * Get the x-www-urlencoded string
77       */
78      public String toPostString() {
79          return this.thePostDataEncoded.toString();
80      }
81  
82      /**
83       * Get the x-www-urlencoded string with '?' preceeding
84       */
85      public String toGetString() {
86          return "?" + this.thePostDataEncoded.toString();
87      }
88  
89      /**
90       * Get the plain old data
91       */
92      @Override
93      public String toString() {
94          return this.thePostData.toString();
95      }
96  }

13  /**
14   * This class provides a seekable channel for a portion, or window, within the provided ReadableByteChannel. The
15   * underlying window size is configured on instantiation. If you intend to move the positions for stateful processing,
16   * you should provide a buffer size, <code>buffsize</code>, greater than the maximum amount you wish to operate against
17   * at any one time.
18   *
19   * <p>
20   * This implementation should be able to hold a maximum window of ~4GB. This implementation uses on heap buffers so be
21   * wary of using Integer.MAX_VALUE as that can cause and OOME.
22   */
23  public class WindowedSeekableByteChannel implements SeekableByteChannel {
24      private static final Logger logger = LoggerFactory.getLogger(WindowedSeekableByteChannel.class);
25  
26      /**
27       * The input source
28       */
29      private final ReadableByteChannel in;
30  
31      /**
32       * estimated length. We have to estimate because we are buffering into a window and may not be at the end. We read ahead
33       * to keep buffers full, but there can be additional data.
34       */
35      long estimatedLength;
36  
37      /**
38       * The earliest position we can move to. Essentially, position of the underlying Channel that is at position 0 of buff1
39       */
40      long minposition;
41  
42      /* Maximum amount of data allowed in memory */
43      // long maxwindow;
44  
45      /** flag if we've reached the end of the underlying channel */
46      private boolean endofchannel;
47  
48      /**
49       * Internal buffers for windowed content
50       */
51      private ByteBuffer buff1;
52      private ByteBuffer buff2;
53  
54      /**
55       * Creates a new instance and populates buffers with data.
56       */
57      public WindowedSeekableByteChannel(final ReadableByteChannel in, final int buffsize) throws IOException {
58  
59          logger.debug("WindowSeekableByteChannel created with buffer size = {}", buffsize);
60  
61          if ((in == null) || !in.isOpen()) {
62              throw new IllegalArgumentException("Channel must be open and not null:");
63          }
64  
65          this.in = in;
66          int capacity = buffsize / 2;
67          if ((buffsize % 2) == 1) {
68              capacity++;
69          }
70  
71          this.buff1 = ByteBuffer.allocate(capacity);
72          readIntoBuffer(this.buff1);
73          // only fill buff2 if there's more to read. otherwise save heap
74          if (!this.endofchannel) {
75              this.buff2 = ByteBuffer.allocate(capacity);
76              readIntoBuffer(this.buff2);
77          } else {
78              this.buff2 = ByteBuffer.allocate(0);
79          }
80      }
81  
82      /**
83       * If necessary, will move data in the window to make room for additional data from the channel.
84       */
85      private void realignBuffers() throws IOException {
86          logger.debug("realignBuffers() called: buf1 = {}, buf2 = {}", buff1, buff2);
87  
88          final int qtr = this.buff1.capacity() / 2;
89          if (this.endofchannel || (this.buff2.remaining() > qtr)) {
90              logger.debug("after early return from realignBuffers(): buf1 = {}, buf2 = {}", buff1, buff2);
91              return;
92          }
93          // keep track of our position
94          final long offset = (long) this.buff1.position() + this.buff2.position();
95          this.buff1.position(qtr);
96          // push them forward
97          this.buff1.compact();
98  
99          // read from the beginning of the buffer
100         this.buff2.rewind();
101 
102         logger.debug("realignBuffers() called prior to fillDst: buf1 = {}, buf2 = {}", buff1, buff2);
103 
104         filldst(this.buff2, this.buff1);
105         // chuck the bytes read into buff1
106 
107         logger.debug("realignBuffers() called prior prior to buff2 compact: buf1 = {}, buf2 = {}", buff1, buff2);
108 
109         this.buff2.compact();
110 
111         logger.debug("realignBuffers() called prior to readIntoBuffer: buf1 = {}, buf2 = {}", buff1, buff2);
112         readIntoBuffer(this.buff2);
113         // update the offset
114         this.minposition += qtr;
115         // reset our location
116         setOffset(offset - qtr);
117 
118         logger.debug("after realignBuffers(): buf1 = {}, buf2 = {}", buff1, buff2);
119 
120     }
121 
122     /**
123      * Determine if there are bytes available to be read.
124      *
125      * @return true if either buffer has data remaining or we have not reached the end of channel.
126      */
127     private boolean bytesAvailable() {
128         return this.buff1.remaining() > 0 || this.buff2.remaining() > 0 || !this.endofchannel;
129     }
130 
131     /**
132      * Attempt to read data from the open channel into the buffer provided.
133      * <p>
134      * After this call completes, we have either filled the buffer -or- have reached the end of data in the input channel.
135      * The buffer will have its position set to 0, and limit set to the end of the data read, which may be equal to the size
136      * of the buffer.
137      * <p>
138      * Has the side effect of raising the endofchannel flag if we have exhausted the bytes in the input channel. Updates the
139      * estimatedLength with the number of bytes read.
140      *
141      * @param buf the destination buffer.
142      * @return the number of bytes read into the buffer.
143      */
144     private int readIntoBuffer(final ByteBuffer buf) throws IOException {
145         logger.debug("readIntoBuffer() called: {}", buf);
146 
147         final int rem = buf.remaining();
148         int read = 0;
149         while ((read != -1) && (buf.remaining() > 0)) {
150             read = this.in.read(buf);
151         }
152         this.endofchannel = (read == -1);
153         // total amount read in case we hit EOS
154         final int totalRead = rem - buf.remaining();
155         this.estimatedLength += totalRead;
156         buf.flip();
157         return totalRead;
158     }

182     /**
183      * {@inheritDoc}
184      *
185      * @see java.nio.channels.SeekableByteChannel#read(ByteBuffer)
186      */
187     @Override
188     public int read(final ByteBuffer dst) throws IOException {
189         if (!this.isOpen()) {
190             throw new ClosedChannelException();
191         }
192 
193         if (dst == null) {
194             throw new IllegalArgumentException("Destination ByteBuffer cannot be null");
195         }
196 
197         // we have nothing left to read and have consumed both buffers fully.
198         if (this.endofchannel && (this.buff1.remaining() + this.buff2.remaining() == 0)) {
199             return -1;
200         }
201 
202         // no more room in the target buffer, but we might have more to read.
203         // - do we want to possibly throw an exception here?
204         if (dst.remaining() == 0) {
205             return 0;
206         }
207 
208         final int maxWrite = dst.remaining();
209 
210         while (dst.hasRemaining() && bytesAvailable()) {
211             // filling buffers
212             realignBuffers();
213             filldst(this.buff1, dst);
214             filldst(this.buff2, dst);
215         }
216         final int bytesRead = maxWrite - dst.remaining();
217         return (this.endofchannel && bytesRead == 0) ? -1 : bytesRead;
218     }
219 
220     /*
221      * Safely fill a destination buffer avoiding a buffer overflow if necessary. <p> Copies byte from src to dest. The
222      * number of bytes copied is the minimum of the amount of space remaining in the destination buffer
223      */
224     private void filldst(final ByteBuffer src, final ByteBuffer dst) {
225         while (src.hasRemaining() && dst.hasRemaining()) {
226             final int origLimit = src.limit();
227             // avoid buffer overflow
228             final int limit = (src.remaining() > dst.remaining()) ? (src.position() + dst.remaining()) : origLimit;
229             src.limit(limit);
230             dst.put(src);
231             // set it back
232             src.limit(origLimit);
233         }
234     }

246     /**
247      * {@inheritDoc}
248      *
249      * @see java.nio.channels.SeekableByteChannel#position(long)
250      */
251     @Override
252     public SeekableByteChannel position(final long newPosition) throws IOException {
253         // if data hasn't been read in, we'll have to do it below and see if it's available
254         if (this.endofchannel && (newPosition > this.estimatedLength)) {
255             throw new EOFException("Position is beyond EOF");
256         }
257 
258         long tgtPosition = newPosition - this.minposition;
259         // see if we can move there
260         if (tgtPosition < 0) {
261             throw new IllegalStateException("Cannot move to " + newPosition + " in the stream. Minimum position is " + this.minposition);
262         }
263 
264         while (!setOffset(tgtPosition) && !this.endofchannel) {
265             realignBuffers();
266             tgtPosition = newPosition - this.minposition;
267         }
268         if (newPosition > this.estimatedLength) {
269             throw new EOFException("Position is beyond EOF");
270         }
271         return this;
272     }
273 
274     /**
275      * attempts to set position to specified offset in underlying buffers
276      */
277     private boolean setOffset(final long tgtOffset) {
278         logger.debug("setOffset() called tgtOffset = {}, buff1 = {}, buff2 = {}", tgtOffset, buff1, buff2);
279 
280         if (tgtOffset <= this.buff1.limit()) {
281             this.buff1.position((int) tgtOffset);
282             this.buff2.position(0);
283         } else if (tgtOffset <= (this.buff1.limit() + this.buff2.limit())) {
284             this.buff1.position(this.buff1.limit());
285             this.buff2.position((int) (tgtOffset - this.buff1.capacity()));
286         } else {
287             this.buff1.position(this.buff1.limit());
288             this.buff2.position(this.buff2.limit());
289             return false;
290         }
291         return true;
292     }

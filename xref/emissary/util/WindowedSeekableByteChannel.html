14  /**
15   * This class provides a seekable channel for a portion, or window, within the provided ReadableByteChannel. The
16   * underlying window size is configured on instantiation. If you intend to move the positions for stateful processing,
17   * you should provide a buffer size, <code>buffsize</code>, greater than the maximum amount you wish to operate against
18   * at any one time.
19   *
20   * <p>
21   * This implementation should be able to hold a maximum window of ~4GB. This implementation uses on heap buffers so be
22   * wary of using Integer.MAX_VALUE as that can cause and OOME.
23   */
24  public class WindowedSeekableByteChannel implements SeekableByteChannel {
25      private static final Logger logger = LoggerFactory.getLogger(WindowedSeekableByteChannel.class);
26  
27      /**
28       * The input source
29       */
30      private final ReadableByteChannel in;
31  
32      /**
33       * estimated length. We have to estimate because we are buffering into a window and may not be at the end. We read ahead
34       * to keep buffers full, but there can be additional data.
35       */
36      long estimatedLength;
37  
38      /**
39       * The earliest position we can move to. Essentially, position of the underlying Channel that is at position 0 of buff1
40       */
41      long minposition;
42  
43      /* Maximum amount of data allowed in memory */
44      // long maxwindow;
45  
46      /** flag if we've reached the end of the underlying channel */
47      private boolean endofchannel;
48  
49      /**
50       * Internal buffers for windowed content
51       */
52      @Nullable
53      private ByteBuffer buff1;
54      @Nullable
55      private ByteBuffer buff2;
56  
57      /**
58       * Creates a new instance and populates buffers with data.
59       */
60      public WindowedSeekableByteChannel(final ReadableByteChannel in, final int buffsize) throws IOException {
61  
62          logger.debug("WindowSeekableByteChannel created with buffer size = {}", buffsize);
63  
64          if ((in == null) || !in.isOpen()) {
65              throw new IllegalArgumentException("Channel must be open and not null:");
66          }
67  
68          this.in = in;
69          int capacity = buffsize / 2;
70          if ((buffsize % 2) == 1) {
71              capacity++;
72          }
73  
74          this.buff1 = ByteBuffer.allocate(capacity);
75          readIntoBuffer(this.buff1);
76          // only fill buff2 if there's more to read. otherwise save heap
77          if (!this.endofchannel) {
78              this.buff2 = ByteBuffer.allocate(capacity);
79              readIntoBuffer(this.buff2);
80          } else {
81              this.buff2 = ByteBuffer.allocate(0);
82          }
83      }
84  
85      /**
86       * If necessary, will move data in the window to make room for additional data from the channel.
87       */
88      private void realignBuffers() throws IOException {
89          logger.debug("realignBuffers() called: buf1 = {}, buf2 = {}", buff1, buff2);
90  
91          final int qtr = this.buff1.capacity() / 2;
92          if (this.endofchannel || (this.buff2.remaining() > qtr)) {
93              logger.debug("after early return from realignBuffers(): buf1 = {}, buf2 = {}", buff1, buff2);
94              return;
95          }
96          // keep track of our position
97          final long offset = (long) this.buff1.position() + this.buff2.position();
98          this.buff1.position(qtr);
99          // push them forward
100         this.buff1.compact();
101 
102         // read from the beginning of the buffer
103         this.buff2.rewind();
104 
105         logger.debug("realignBuffers() called prior to fillDst: buf1 = {}, buf2 = {}", buff1, buff2);
106 
107         filldst(this.buff2, this.buff1);
108         // chuck the bytes read into buff1
109 
110         logger.debug("realignBuffers() called prior prior to buff2 compact: buf1 = {}, buf2 = {}", buff1, buff2);
111 
112         this.buff2.compact();
113 
114         logger.debug("realignBuffers() called prior to readIntoBuffer: buf1 = {}, buf2 = {}", buff1, buff2);
115         readIntoBuffer(this.buff2);
116         // update the offset
117         this.minposition += qtr;
118         // reset our location
119         setOffset(offset - qtr);
120 
121         logger.debug("after realignBuffers(): buf1 = {}, buf2 = {}", buff1, buff2);
122 
123     }
124 
125     /**
126      * Determine if there are bytes available to be read.
127      *
128      * @return true if either buffer has data remaining or we have not reached the end of channel.
129      */
130     private boolean bytesAvailable() {
131         return this.buff1.remaining() > 0 || this.buff2.remaining() > 0 || !this.endofchannel;
132     }
133 
134     /**
135      * Attempt to read data from the open channel into the buffer provided.
136      * <p>
137      * After this call completes, we have either filled the buffer -or- have reached the end of data in the input channel.
138      * The buffer will have its position set to 0, and limit set to the end of the data read, which may be equal to the size
139      * of the buffer.
140      * <p>
141      * Has the side effect of raising the endofchannel flag if we have exhausted the bytes in the input channel. Updates the
142      * estimatedLength with the number of bytes read.
143      *
144      * @param buf the destination buffer.
145      * @return the number of bytes read into the buffer.
146      */
147     private int readIntoBuffer(final ByteBuffer buf) throws IOException {
148         logger.debug("readIntoBuffer() called: {}", buf);
149 
150         final int rem = buf.remaining();
151         int read = 0;
152         while ((read != -1) && (buf.remaining() > 0)) {
153             read = this.in.read(buf);
154         }
155         this.endofchannel = (read == -1);
156         // total amount read in case we hit EOS
157         final int totalRead = rem - buf.remaining();
158         this.estimatedLength += totalRead;
159         buf.flip();
160         return totalRead;
161     }

185     /**
186      * {@inheritDoc}
187      *
188      * @see java.nio.channels.SeekableByteChannel#read(ByteBuffer)
189      */
190     @Override
191     public int read(final ByteBuffer dst) throws IOException {
192         if (!this.isOpen()) {
193             throw new ClosedChannelException();
194         }
195 
196         if (dst == null) {
197             throw new IllegalArgumentException("Destination ByteBuffer cannot be null");
198         }
199 
200         // we have nothing left to read and have consumed both buffers fully.
201         if (this.endofchannel && (this.buff1.remaining() + this.buff2.remaining() == 0)) {
202             return -1;
203         }
204 
205         // no more room in the target buffer, but we might have more to read.
206         // - do we want to possibly throw an exception here?
207         if (dst.remaining() == 0) {
208             return 0;
209         }
210 
211         final int maxWrite = dst.remaining();
212 
213         while (dst.hasRemaining() && bytesAvailable()) {
214             // filling buffers
215             realignBuffers();
216             filldst(this.buff1, dst);
217             filldst(this.buff2, dst);
218         }
219         final int bytesRead = maxWrite - dst.remaining();
220         return (this.endofchannel && bytesRead == 0) ? -1 : bytesRead;
221     }
222 
223     /*
224      * Safely fill a destination buffer avoiding a buffer overflow if necessary. <p> Copies byte from src to dest. The
225      * number of bytes copied is the minimum of the amount of space remaining in the destination buffer
226      */
227     private static void filldst(final ByteBuffer src, final ByteBuffer dst) {
228         while (src.hasRemaining() && dst.hasRemaining()) {
229             final int origLimit = src.limit();
230             // avoid buffer overflow
231             final int limit = (src.remaining() > dst.remaining()) ? (src.position() + dst.remaining()) : origLimit;
232             src.limit(limit);
233             dst.put(src);
234             // set it back
235             src.limit(origLimit);
236         }
237     }

249     /**
250      * {@inheritDoc}
251      *
252      * @see java.nio.channels.SeekableByteChannel#position(long)
253      */
254     @Override
255     public SeekableByteChannel position(final long newPosition) throws IOException {
256         // if data hasn't been read in, we'll have to do it below and see if it's available
257         if (this.endofchannel && (newPosition > this.estimatedLength)) {
258             throw new EOFException("Position is beyond EOF");
259         }
260 
261         long tgtPosition = newPosition - this.minposition;
262         // see if we can move there
263         if (tgtPosition < 0) {
264             throw new IllegalStateException("Cannot move to " + newPosition + " in the stream. Minimum position is " + this.minposition);
265         }
266 
267         while (!setOffset(tgtPosition) && !this.endofchannel) {
268             realignBuffers();
269             tgtPosition = newPosition - this.minposition;
270         }
271         if (newPosition > this.estimatedLength) {
272             throw new EOFException("Position is beyond EOF");
273         }
274         return this;
275     }
276 
277     /**
278      * attempts to set position to specified offset in underlying buffers
279      */
280     private boolean setOffset(final long tgtOffset) {
281         logger.debug("setOffset() called tgtOffset = {}, buff1 = {}, buff2 = {}", tgtOffset, buff1, buff2);
282 
283         if (tgtOffset <= this.buff1.limit()) {
284             this.buff1.position((int) tgtOffset);
285             this.buff2.position(0);
286         } else if (tgtOffset <= (this.buff1.limit() + this.buff2.limit())) {
287             this.buff1.position(this.buff1.limit());
288             this.buff2.position((int) (tgtOffset - this.buff1.capacity()));
289         } else {
290             this.buff1.position(this.buff1.limit());
291             this.buff2.position(this.buff2.limit());
292             return false;
293         }
294         return true;
295     }

36      @SuppressWarnings("MemberName")
37      protected static SAXBuilder createSAXBuilder(final boolean validate) {
38          SAXBuilder builder;
39          if (validate) {
40              builder = new SAXBuilder(XMLReaders.DTDVALIDATING);
41          } else {
42              builder = new SAXBuilder(XMLReaders.NONVALIDATING);
43              builder.setFeature("http://xml.org/sax/features/validation", false);
44              builder.setFeature("http://apache.org/xml/features/nonvalidating/load-dtd-grammar", false);
45              builder.setFeature("http://apache.org/xml/features/nonvalidating/load-external-dtd", false);
46          }
47  
48          // If you can't completely disable DTDs, then at least do the following:
49          builder.setFeature("http://xml.org/sax/features/external-general-entities", false);
50          builder.setFeature("http://xml.org/sax/features/external-parameter-entities", false);
51          builder.setFeature("http://apache.org/xml/features/xinclude", false);
52          builder.setEntityResolver((publicId, systemId) -> new InputSource(new StringReader("")));
53          builder.setExpandEntities(false);
54          return builder;
55      }

57      protected static Document createDocument(final String xml, @Nullable final XMLFilter filter, final SAXBuilder builder) throws JDOMException {
58          if (filter != null) {
59              builder.setXMLFilter(filter);
60          }
61  
62          try {
63              return builder.build(new StringReader(xml));
64          } catch (IOException iox) {
65              throw new JDOMException(ERR_MSG + iox.getMessage(), iox);
66          }
67      }

77      protected static Document createDocument(final byte[] xml, @Nullable final XMLFilter filter, final SAXBuilder builder,
78              @Nullable final String charset)
79              throws JDOMException {
80          if (filter != null) {
81              builder.setXMLFilter(filter);
82          }
83          final ByteArrayInputStream bais = new ByteArrayInputStream(xml);
84          InputStreamReader isr;
85  
86          if (charset != null) {
87              try {
88                  isr = new InputStreamReader(bais, charset);
89              } catch (UnsupportedEncodingException e) {
90                  isr = new InputStreamReader(bais);
91              }
92          } else {
93              isr = new InputStreamReader(bais);
94          }
95  
96          try {
97              return builder.build(isr);
98          } catch (IOException iox) {
99              throw new JDOMException(ERR_MSG + iox.getMessage(), iox);
100         }
101     }

115     /**
116      * Get a string from a JDOM Document
117      *
118      * @param jdom the jdom document
119      * @return String value in UTF-8
120      */
121     @Nullable
122     public static String toString(final Document jdom) {
123         final XMLOutputter outputter = new XMLOutputter(Format.getPrettyFormat());
124         try (ByteArrayOutputStream os = new ByteArrayOutputStream()) {
125             outputter.output(jdom, os);
126             return os.toString();
127         } catch (IOException iox) {
128             logger.error("ByteArrayOutputStream exception", iox);
129             return null;
130         }
131     }

185     /**
186      * Create a JDOM element, protecting the data with encoding if needed
187      */
188     @SuppressWarnings("PMD.UselessParentheses")
189     public static Element protectedElement(final String name, final String s) {
190         final Element e = new Element(name);
191         int badCount = 0;
192         for (int i = 0; i < s.length(); i++) {
193             final char c = s.charAt(i);
194             if (Character.isISOControl(c) && !Character.isWhitespace(c)) {
195                 badCount++;
196             }
197         }
198         if ((1.0 * badCount) / (1.0 * s.length()) > 0.1) {
199             e.setAttribute("encoding", "base64");
200             final Base64 b64 = new Base64();
201             e.addContent(new String(b64.encode(s.getBytes())));
202         } else if (badCount > 0) {
203             e.setAttribute("encoding", "quoted-printable");
204             final QuotedPrintableCodec qp = new QuotedPrintableCodec();
205             e.addContent(new String(qp.encode(s.getBytes())));
206         } else {
207             e.addContent(s);
208         }
209         return e;
210     }

60      /**
61       * Get an array of UTF-8 characters from the input bytes
62       *
63       * @param byteArray the input bytes
64       * @param charSet <em>derived</em> charSet of the input array
65       * @param start index into input array to start copying
66       * @param end index into input array to stop copying
67       * @return array of UTF8 char
68       */
69      public static char[] getUTFCharArray(final byte[] byteArray, final String charSet, final int start, final int end) {
70          String actualCharSet = charSet;
71          if (actualCharSet != null) {
72              final String jcs = JavaCharSet.get(actualCharSet);
73              if (jcs != null) {
74                  actualCharSet = jcs;
75              }
76          }
77          return jGetUTFCharArray(byteArray, actualCharSet, start, end);
78      }

80      /**
81       * Get an array of UTF-8 characters from the input bytes
82       *
83       * @param byteArray the input bytes
84       * @param charSet <em>JAVA</em> charSet of the input array
85       * @param start byte index into input array to start copying
86       * @param end byte index into input array to stop copying
87       * @return array of UTF8 char
88       */
89      public static char[] jGetUTFCharArray(final byte[] byteArray, @Nullable final String charSet, final int start, final int end) {
90          char[] cbuffer = null;
91  
92          if (byteArray != null) {
93  
94              // Check start value
95              if (start > byteArray.length || start < 0) {
96                  throw new ArrayIndexOutOfBoundsException("start : " + start + ", actual length " + byteArray.length);
97              }
98  
99              // Check end value
100             final int actualEnd;
101             if ((end == -1) || (end > byteArray.length)) {
102                 actualEnd = byteArray.length;
103             } else {
104                 actualEnd = end;
105             }
106 
107             if (actualEnd <= start) {
108                 throw new ArrayIndexOutOfBoundsException("start : " + start + ", end : " + actualEnd + ", actual length " + byteArray.length);
109             }
110 
111             // Convert byteArray to UTF-8
112             if (charSet != null) {
113                 try {
114                     final String converted = new String(byteArray, start, actualEnd - start, charSet);
115                     cbuffer = converted.toCharArray();
116                 } catch (UnsupportedEncodingException uee) {
117                     logger.warn("Unable to convert to " + charSet);
118                     // Convert from Byte Array to Char Array...
119                     cbuffer = byteToCharArray(byteArray, start, actualEnd);
120                 }
121             } else {
122                 // Convert from Byte Array to Char Array...
123                 cbuffer = byteToCharArray(byteArray, start, actualEnd);
124             }
125         }
126 
127         return cbuffer;
128     }

169     /**
170      * Convert bytes to chars using platform default encoding with begin and end points
171      *
172      * @param bArray the input data
173      * @param start byte index into input to start copy
174      * @param end byte index into input to end copy or -1 for end
175      */
176     private static char[] byteToCharArray(final byte[] bArray, final int start, final int end) {
177         final int len = (end == -1 ? bArray.length : (bArray.length < end) ? bArray.length : end) - start;
178         final char[] cArray = new char[len];
179 
180         for (int j = start; j < len; j++) {
181             cArray[j] = (char) (0xFF & bArray[j]);
182         }
183         return cArray;
184     }

186     /**
187      * test for ascii ness
188      *
189      * @param s string to test
190      * @return true if string is ascii
191      */
192     public static boolean isAscii(final String s) {
193         try {
194             final int len = s.length();
195             for (int i = 0; i < len; i++) {
196                 final char c = s.charAt(i);
197                 if (c <= 0 || c > 127) {
198                     return false;
199                 }
200             }
201         } catch (Exception e) {
202             return false;
203         }
204         return true;
205     }

228     /**
229      * Check for valid utf8 data. Borrowed from the unidesc package (GPL) by Bill Poser, converted from C to Java. The check
230      * runs from offs to dlen-1
231      *
232      * @param data the bytes to check for validity
233      * @param offs beginning offset to check
234      * @param dlen ending offset of the range
235      * @return true if valid utf8
236      */
237     @SuppressWarnings("fallthrough")
238     public static boolean isUTF8(final byte[] data, final int offs, final int dlen) {
239         int pos = offs;
240         int a;
241 
242         while (pos < dlen) {
243             try {
244                 final int val = data[pos] & 0xff;
245                 final int len = TrailingBytesForUTF8[val] + 1;
246                 int srcptr = pos + len;
247 
248                 switch (len) {
249                     default:
250                         return false;
251                     // everything else falls through when true
252                     case 4:
253                         a = (data[--srcptr] & 0xff);
254                         if (a < 0x80 || a > 0xbf) {
255                             return false;
256                         }
257                     case 3:
258                         a = (data[--srcptr] & 0xff);
259                         if (a < 0x80 || a > 0xbf) {
260                             return false;
261                         }
262                     case 2:
263                         a = (data[--srcptr] & 0xff);
264                         if (a > 0xbf) {
265                             return false;
266                         }
267 
268                         switch (val) {
269                             // no fall through in this one
270                             case 0xe0:
271                                 if (a < 0xa0) {
272                                     return false;
273                                 }
274                                 break;
275                             case 0xf0:
276                                 if (a < 0x90) {
277                                     return false;
278                                 }
279                                 break;
280                             case 0xf4:
281                                 if (a > 0x8f) {
282                                     return false;
283                                 }
284                                 break;
285                             default:
286                                 if (a < 0x80) {
287                                     return false;
288                                 }
289                         }
290                     case 1:
291                         if (val >= 0x80 && val < 0xc0) {
292                             return false;
293                         }
294                         if (val >= 0xf0) {
295                             return false;
296                         }
297                 }
298                 pos += len;
299             } catch (ArrayIndexOutOfBoundsException x) {
300                 logger.warn("ooops", x);
301                 return false;
302             }
303         }
304         return true;
305     }

307     /**
308      * See if string has multibyte chars (No longer based on org.apache.poi.util.StringUtil) It would be a bad idea to call
309      * this with a very large string
310      *
311      * @param value string to test
312      * @return true if string has at least one multibyte char
313      */
314     public static boolean hasMultibyte(@Nullable final String value) {
315         if (value == null) {
316             return false;
317         }
318         final int cpc = value.codePointCount(0, value.length());
319         final int bc = value.getBytes().length;
320         return cpc != bc;
321     }

60      /**
61       * Get an array of UTF-8 characters from the input bytes
62       *
63       * @param byteArray the input bytes
64       * @param charSet <em>derived</em> charSet of the input array
65       * @param start index into input array to start copying
66       * @param end index into input array to stop copying
67       * @return array of UTF8 char
68       */
69      public static char[] getUtfCharArray(final byte[] byteArray, final String charSet, final int start, final int end) {
70          String actualCharSet = charSet;
71          if (actualCharSet != null) {
72              final String jcs = JavaCharSet.get(actualCharSet);
73              if (jcs != null) {
74                  actualCharSet = jcs;
75              }
76          }
77          return jGetUtfCharArray(byteArray, actualCharSet, start, end);
78      }

80      /**
81       * Get an array of UTF-8 characters from the input bytes
82       *
83       * @param byteArray the input bytes
84       * @param charSet <em>JAVA</em> charSet of the input array
85       * @param start byte index into input array to start copying
86       * @param end byte index into input array to stop copying
87       * @return array of UTF8 char
88       */
89      public static char[] jGetUtfCharArray(final byte[] byteArray, @Nullable final String charSet, final int start, final int end) {
90          char[] cbuffer = null;
91  
92          if (byteArray != null) {
93  
94              // Check start value
95              if (start > byteArray.length || start < 0) {
96                  throw new ArrayIndexOutOfBoundsException("start : " + start + ", actual length " + byteArray.length);
97              }
98  
99              // Check end value
100             final int actualEnd;
101             if ((end == -1) || (end > byteArray.length)) {
102                 actualEnd = byteArray.length;
103             } else {
104                 actualEnd = end;
105             }
106 
107             if (actualEnd <= start) {
108                 throw new ArrayIndexOutOfBoundsException("start : " + start + ", end : " + actualEnd + ", actual length " + byteArray.length);
109             }
110 
111             // Convert byteArray to UTF-8
112             if (charSet != null) {
113                 try {
114                     final String converted = new String(byteArray, start, actualEnd - start, charSet);
115                     cbuffer = converted.toCharArray();
116                 } catch (UnsupportedEncodingException uee) {
117                     logger.warn("Unable to convert to " + charSet);
118                     // Convert from Byte Array to Char Array...
119                     cbuffer = byteToCharArray(byteArray, start, actualEnd);
120                 }
121             } else {
122                 // Convert from Byte Array to Char Array...
123                 cbuffer = byteToCharArray(byteArray, start, actualEnd);
124             }
125         }
126 
127         return cbuffer;
128     }

170     /**
171      * Convert bytes to chars using platform default encoding with begin and end points
172      *
173      * @param bArray the input data
174      * @param start byte index into input to start copy
175      * @param end byte index into input to end copy or -1 for end
176      */
177     private static char[] byteToCharArray(final byte[] bArray, final int start, final int end) {
178         final int len = (end == -1 ? bArray.length : (bArray.length < end) ? bArray.length : end) - start;
179         final char[] cArray = new char[len];
180 
181         for (int j = start; j < len; j++) {
182             cArray[j] = (char) (0xFF & bArray[j]);
183         }
184         return cArray;
185     }

187     /**
188      * test for ascii ness
189      *
190      * @param s string to test
191      * @return true if string is ascii
192      */
193     public static boolean isAscii(final String s) {
194         try {
195             final int len = s.length();
196             for (int i = 0; i < len; i++) {
197                 final char c = s.charAt(i);
198                 if (c <= 0 || c > 127) {
199                     return false;
200                 }
201             }
202         } catch (RuntimeException e) {
203             return false;
204         }
205         return true;
206     }

229     /**
230      * Check for valid utf8 data. Borrowed from the unidesc package (GPL) by Bill Poser, converted from C to Java. The check
231      * runs from offs to dlen-1
232      *
233      * @param data the bytes to check for validity
234      * @param offs beginning offset to check
235      * @param dlen ending offset of the range
236      * @return true if valid utf8
237      */
238     @SuppressWarnings("fallthrough")
239     public static boolean isUtf8(final byte[] data, final int offs, final int dlen) {
240         int pos = offs;
241         int a;
242 
243         while (pos < dlen) {
244             try {
245                 final int val = data[pos] & 0xff;
246                 final int len = trailingBytesForUtF8[val] + 1;
247                 int srcptr = pos + len;
248 
249                 switch (len) {
250                     // everything falls through when true
251                     case 4:
252                         a = (data[--srcptr] & 0xff);
253                         if (a < 0x80 || a > 0xbf) {
254                             return false;
255                         }
256                     case 3:
257                         a = (data[--srcptr] & 0xff);
258                         if (a < 0x80 || a > 0xbf) {
259                             return false;
260                         }
261                     case 2:
262                         a = (data[--srcptr] & 0xff);
263                         if (a > 0xbf) {
264                             return false;
265                         }
266 
267                         switch (val) {
268                             // no fall through in this one
269                             case 0xe0:
270                                 if (a < 0xa0) {
271                                     return false;
272                                 }
273                                 break;
274                             case 0xf0:
275                                 if (a < 0x90) {
276                                     return false;
277                                 }
278                                 break;
279                             case 0xf4:
280                                 if (a > 0x8f) {
281                                     return false;
282                                 }
283                                 break;
284                             default:
285                                 if (a < 0x80) {
286                                     return false;
287                                 }
288                         }
289                     case 1:
290                         if (val >= 0x80 && val < 0xc0) {
291                             return false;
292                         }
293                         if (val >= 0xf0) {
294                             return false;
295                         }
296                         break;
297                     default:
298                         return false;
299                 }
300                 pos += len;
301             } catch (ArrayIndexOutOfBoundsException x) {
302                 logger.warn("ooops", x);
303                 return false;
304             }
305         }
306         return true;
307     }

309     /**
310      * See if string has multibyte chars (No longer based on org.apache.poi.util.StringUtil) It would be a bad idea to call
311      * this with a very large string
312      *
313      * @param value string to test
314      * @return true if string has at least one multibyte char
315      */
316     public static boolean hasMultibyte(@Nullable final String value) {
317         if (value == null) {
318             return false;
319         }
320         final int cpc = value.codePointCount(0, value.length());
321         final int bc = value.getBytes().length;
322         return cpc != bc;
323     }

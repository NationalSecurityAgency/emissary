5   import java.util.Collection;
6   import java.util.List;

10  /**
11   * This class provides routines for approximating the RAM size of Emissary objects - primarily the
12   * {@link IBaseDataObject}.
13   */
14  public class SizeUtil {
15  
16      /**
17       * A flag to indicate whether we are running 32 or 64 bit JVM. Note that this is the JVM bits and not the OS bits
18       */
19      private static final boolean arch64;
20  
21      /**
22       * A reference/pointer size in a non-compressed Ops JVM. Default is 32-bit = 4 bytes
23       */
24      private static final long refSize;
25  
26      /**
27       * Object overhead
28       */
29      private static final long OBJ_OVERHEAD = 8L;
30  
31      static {
32          if (System.getProperty("os.arch").contains("64")) {
33              arch64 = true;
34              refSize = 8L;
35          } else {
36              arch64 = false;
37              refSize = 4L;
38          }
39      }
40  
41      /**
42       * <i>Approximate</i> the amount of RAM consumed by a String.
43       * <p>
44       * See
45       * <a href="https://www.cs.virginia.edu/kim/publicity/pldi09tutorials/memory-efficient-java-tutorial.pdf">tutorial</a>
46       * Slide 26.
47       * <p>
48       * This calculation is affected by the JVM architecture and whether
49       * <a href="https://wikis.oracle.com/display/HotSpotInternals/CompressedOops">CompressedOops</a> are enabled.
50       *
51       * @param str - the String to approximate
52       * @return The approximate size, in bytes, in RAM for str
53       */
54      public static long sizeof(@Nullable String str) {
55          if (str == null) {
56              return 0L;
57          }
58          if (arch64) {
59              return (str.length() * 2L) + 52; // 2 bytes per UTF-16, 52 for JVM overhead & bookkeeping, 64-bit arch
60          } else {
61              return (str.length() * 2L) + 48; // 2 bytes per UTF-16, 48 for JVM overhead & bookkeeping, 32-bit arch
62          }
63      }
64  
65      /**
66       * Approximate the amount of RAM consumed by an individual {@link IBaseDataObject}. The purpose of this method is to
67       * approximate the RAM that will be consumed by a corresponding serialized/deserialized object, hence not all aspects of
68       * the {@link IBaseDataObject} are considered. Additionally, this method does not include all the "outputtable" logic
69       * that may be present in output filter, which could make the actual size smaller than is reported by this method.
70       * 
71       * @param ibdo The IBaseDataObject to approximate the size of
72       * @return The approximate size, in bytes, in RAM for the IBaseDataObject
73       */
74      public static long sizeof(@Nullable IBaseDataObject ibdo) {
75          if (ibdo == null) {
76              return 0L;
77          }
78  
79          long totalSize = 0L;
80  
81          // Primary view (potentially gigantic)
82          totalSize += getPayloadMemorySize(ibdo);
83  
84          // Factor in the extracted records
85          totalSize += getExtractedRecordMemorySize(ibdo);
86  
87          if (ibdo.getParameters() == null) {
88              return totalSize;
89          } // This should never be null, but still check
90  
91          // Finally, estimate the size of the metadata parameters, assuming the underlying Strings are UTF-16
92          for (Map.Entry<String, Collection<Object>> entry : ibdo.getParameters().entrySet()) {
93  
94              // Note: The core multimap in Emissary has CharSequence as keys. This is to allow for more
95              // flexibility in key design. Under the hood, they are still Strings so the following
96              // routines are safe for the time-being, but will need to be changed if the keys change
97  
98              // Get the size of the key
99              String k = entry.getKey();
100             totalSize += sizeof(k);
101             totalSize += refSize;
102 
103             // Similar to the above, the values in Emissary are generic Objects although almost all of them
104             // are Strings. That is why the following check is made in the for loop.
105 
106             // Get the size of the List of values
107             Collection<Object> values = entry.getValue();
108             for (Object v : values) {
109                 if (v instanceof String) {
110                     totalSize += sizeof((String) v);
111                     totalSize += refSize;
112                 }
113                 // TODO: factor in non-String objects
114                 // but this is not that important since there
115                 // are not many of them and will change the overall
116                 // size minimally at this time
117             }
118         }
119 
120         return totalSize;
121     }
122 
123     /**
124      * Estimate the size, in bytes, of the RAM of an entire {@link IBaseDataObject} family tree. This is simply the sum of
125      * the sizes of the individual members of the family tree.
126      *
127      * @param familyTree - {@code List<IBaseDataObject>} representing the family tree for a document object
128      * @return - the approximate size, in bytes, in RAM for the familyTree
129      */
130     public static long sizeof(@Nullable List<IBaseDataObject> familyTree) {
131         if (familyTree == null) {
132             return 0L;
133         }
134 
135         long totalSize = 0L;
136         for (IBaseDataObject ibdo : familyTree) {
137             if (ibdo != null) {
138                 totalSize += sizeof(ibdo);
139                 totalSize += OBJ_OVERHEAD + refSize; // For the pointer to ibdo in familyTree (Object overhead)
140             }
141         }
142         return totalSize;
143     }
144 
145     /**
146      * Approximates the amount of memory consumed by the "extracted records" in a single IBaseDataObject. This is typically
147      * the case when the framework gets eventing datasets. These extracted records are usually treated specially and not run
148      * through the processing pipelines proper, but on output do appear as proper child IBaseDataObjects. In the case of a
149      * large dataset, these extracted records can consume huge amounts of RAM.
150      * 
151      * @param ibdo - The IBaseDataObject to approximate
152      * @return - The approximate memory footprint, in bytes, for extracted records of an IBaseDataObject
153      */
154     public static long getExtractedRecordMemorySize(@Nullable IBaseDataObject ibdo) {
155         if (ibdo == null) {
156             return 0L;
157         }
158 
159         long totalSize = 0L;
160 
161         // Count up the size of any extracted children
162         if (ibdo.hasExtractedRecords()) {
163             List<IBaseDataObject> childObjList = ibdo.getExtractedRecords();
164             if (childObjList != null) {
165                 for (IBaseDataObject child : childObjList) {
166                     if (child != null) {
167                         totalSize += sizeof(child);
168                         totalSize += OBJ_OVERHEAD + refSize; // For the pointer to child in childObjList (Object
169                                                              // overhead)
170                     }
171                 }
172             }
173         }
174 
175         return totalSize;
176     }
177 
178     /**
179      * Approximate the amount of memory consumed by the various "payloads" of an IBaseDataObject. In this case, a payload
180      * refers to the header, footer, data (primary view), and all the alternate views.
181      * 
182      * @param ibdo - The IBaseDataObject to approximate
183      * @return - The approximate memory footprint, in bytes, for the IBaseDataObject
184      */
185     public static long getPayloadMemorySize(@Nullable IBaseDataObject ibdo) {
186         if (ibdo == null) {
187             return 0L;
188         }
189 
190         long totalSize = 0L;
191 
192         // We don't concern ourselves with object references here, because they are likely
193         // tiny compared to the various "payloads" and altViews. A more accurate
194         // version of this method would factor those in, but they are likely to be
195         // negligible
196 
197         // Primary view (potentially gigantic)
198         totalSize += ibdo.dataLength(); // This always returns an int
199 
200         // Header and footer size (probably not big)
201         if (ibdo.footer() != null) {
202             totalSize += ibdo.footer().length;
203         }
204         if (ibdo.header() != null) {
205             totalSize += ibdo.header().length;
206         }
207 
208         // Size up the alternative views
209         for (Map.Entry<String, byte[]> altView : ibdo.getAlternateViews().entrySet()) {
210             if (altView.getValue() != null) {
211                 totalSize += altView.getValue().length;
212             }
213         }
214 
215         return totalSize;
216     }
217 
218     /** This class is not meant to be instantiated. */
219     private SizeUtil() {}
220 }

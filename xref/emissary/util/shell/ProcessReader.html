7   package emissary.util.shell;
8   
9   import org.slf4j.MDC;
10  
11  import java.util.Map;
12  
13  public abstract class ProcessReader extends Thread {
14  
15      private Map<String, String> contextMap;
16  
17      public abstract void finish();
18  
19      /**
20       * Allows parent threads to pass state values retrieved through a call to {@link MDC#getCopyOfContextMap()}
21       * 
22       * @param contextMap Map of captured point-in-time state values
23       */
24      public void setContextMap(Map<String, String> contextMap) {
25          this.contextMap = contextMap;
26      }
27  
28      /**
29       * Applies logger context to the current thread. This method should only be called from within the overridden
30       * {@link Thread#run()} method, since the context is backed by a {@link ThreadLocal ThreadLocal&lt;T&gt;} map.
31       */
32      protected final void applyLogContextMap() {
33          if (contextMap != null) {
34              MDC.setContextMap(contextMap);
35          }
36      }
37  
38      /**
39       * Wrapper method to ensure that the {@link #applyLogContextMap()} method is always invoked before executing the core
40       * functionality of ProcessReader subclasses
41       */
42      @Override
43      public void run() {
44          applyLogContextMap();
45          runImpl();
46      }
47  
48      /**
49       * Abstract method that subclasses should override to implement their core thread-specific functionality
50       */
51      abstract void runImpl();
52  }

9   import java.util.ArrayList;

12  /**
13   * Helper methods to ease handling of Dispose objects
14   * 
15   * Dispose objects are added to BDOs to handle clean-up after a
16   * {@link emissary.core.channels.SeekableByteChannelFactory} is finished with
17   */
18  public final class DisposeHelper {
19      private DisposeHelper() {}
20  
21      /** Metadata key to use to store runnables in on a BDO. */
22      public static final String KEY = "DISPOSE_RUNNABLES";
23      private static final String VALIDATION_MSG_IBDO = "Cannot be null: ibdo";
24      private static final String VALIDATION_MSG_RUNNABLE = "Cannot be null: Runnable";
25  
26      private static final Logger LOGGER = LoggerFactory.getLogger(DisposeHelper.class);
27  
28      /**
29       * Add a Dispose object to a BDO
30       * 
31       * @param ibdo to add the Runnable to
32       * @param newRunnable to handle disposing of the referenced object
33       */
34      public static void add(final IBaseDataObject ibdo, final Runnable newRunnable) {
35          Validate.notNull(ibdo, VALIDATION_MSG_IBDO);
36          Validate.notNull(newRunnable, VALIDATION_MSG_RUNNABLE);
37          if (ibdo.hasParameter(KEY)) {
38              final List<Object> existingRunnables = new ArrayList<>(ibdo.getParameter(KEY));
39              existingRunnables.add(newRunnable);
40              ibdo.putParameter(KEY, existingRunnables);
41          } else {
42              ibdo.putParameter(KEY, newRunnable);
43          }
44      }
45  
46      /**
47       * Add a list of Dispose objects to a BDO (additive to existing Dispose objects)
48       * 
49       * @param ibdo to add the Runnables to
50       * @param newRunnables list of Runnables to add
51       */
52      public static void add(final IBaseDataObject ibdo, final List<Runnable> newRunnables) {
53          Validate.notNull(ibdo, VALIDATION_MSG_IBDO);
54          Validate.notNull(newRunnables, VALIDATION_MSG_RUNNABLE);
55          if (ibdo.hasParameter(KEY)) {
56              final List<Object> existingRunnables = new ArrayList<>(ibdo.getParameter(KEY));
57              existingRunnables.addAll(newRunnables);
58              ibdo.putParameter(KEY, existingRunnables);
59          } else {
60              ibdo.putParameter(KEY, newRunnables);
61          }
62      }
63  
64      /**
65       * Set a single Dispose object to a BDO (overwrites any existing Dispose objects)
66       * 
67       * @param ibdo to add the Runnable to
68       * @param newRunnable to handle disposing of the referenced object
69       */
70      public static void set(final IBaseDataObject ibdo, final Runnable newRunnable) {
71          Validate.notNull(ibdo, VALIDATION_MSG_IBDO);
72          Validate.notNull(newRunnable, VALIDATION_MSG_RUNNABLE);
73          ibdo.setParameter(KEY, newRunnable);
74      }
75  
76      /**
77       * Get the list of Runnables for an object. If an object provided by the key is not a valid Runnable, it will be
78       * ignored.
79       * 
80       * @param ibdo to get Runnables from
81       * @return the list of Runnables
82       */
83      public static List<Runnable> get(final IBaseDataObject ibdo) {
84          Validate.notNull(ibdo, VALIDATION_MSG_IBDO);
85          if (!ibdo.hasParameter(KEY)) {
86              return List.of();
87          }
88  
89          final List<Runnable> validatedAsRunnables = new ArrayList<>();
90          final List<Object> existingRunnables = ibdo.getParameter(KEY);
91  
92          for (final Object possibleRunnable : existingRunnables) {
93              if (possibleRunnable instanceof Runnable) {
94                  validatedAsRunnables.add((Runnable) possibleRunnable);
95              } else if (LOGGER.isWarnEnabled()) {
96                  LOGGER.warn("Not a valid Runnable on object {}", ibdo.shortName());
97              }
98          }
99          return validatedAsRunnables;
100     }
101 
102     /**
103      * Execute Runnables on provided objects. Execution will be for each object in the order provided.
104      * 
105      * @param ibdos to execute runnables from
106      */
107     public static void execute(final List<IBaseDataObject> ibdos) {
108         Validate.notNull(ibdos, "Cannot be null: %s", "ibdos");
109         ibdos.forEach(DisposeHelper::execute);
110     }
111 
112     /**
113      * Execute Runnables for the specified object.
114      * 
115      * @param ibdo to execute Runnables on
116      */
117     public static void execute(final IBaseDataObject ibdo) {
118         Validate.notNull(ibdo, VALIDATION_MSG_IBDO);
119         // Can't be refactored to method::reference as we need to ensure exceptions are not swallowed
120         for (final Runnable runnable : DisposeHelper.get(ibdo)) {
121             try {
122                 runnable.run();
123             } catch (final RuntimeException e) {
124                 LOGGER.warn("Exception while executing Runnable for {}", ibdo.shortName(), e);
125             }
126         }
127     }

1   package emissary.util;

3   import java.util.HashMap;
4   import java.util.Locale;
5   import java.util.Map;
6   
7   /**
8    * Be a HashMap that doesn't care about the case of the keys
9    */
10  public class CaseInsensitiveMap<K, V> extends HashMap<K, V> {
11      // serializable
12      static final long serialVersionUID = -8692638820306464417L;
13  
14      // For remapping string keys to through a case-insensitive layer
15      private final Map<String, String> remap = new HashMap<>();
16  
17      /**
18       * Create one
19       */
20      public CaseInsensitiveMap() {
21          super();
22      }
23  
24      /**
25       * Create one with capacity
26       */
27      public CaseInsensitiveMap(int initialCapacity) {
28          super(initialCapacity);
29      }
30  
31      /**
32       * Create one with capacity and load factor
33       */
34      public CaseInsensitiveMap(int initialCapacity, float loadFactor) {
35          super(initialCapacity, loadFactor);
36      }
37  
38      /**
39       * Create one from another map
40       */
41      public CaseInsensitiveMap(Map<? extends K, ? extends V> map) {
42          super(map.size());
43          putAll(map);
44      }
45  
46      /**
47       * Put an element into the map If it is a string, put it in and also add it to our case remapping map
48       */
49      @Override
50      // @SuppressWarnings("unchecked")
51      public V put(K key, V value) {
52          if (key instanceof String) {
53              String uckey = ((String) key).toLowerCase(Locale.getDefault());
54              if (remap.containsKey(uckey)) {
55                  this.remove(uckey);
56              }
57              remap.put(uckey, (String) key);
58          }
59  
60          return super.put(key, value);
61      }
62  
63  
64      /**
65       * Get an element from the map If the key is a string check our remap first
66       */
67      @Override
68      public V get(Object key) {
69          Object realkey = key;
70  
71          if (key instanceof String) {
72              String strkey = remap.get(((String) key).toLowerCase(Locale.getDefault()));
73              if (strkey != null) {
74                  realkey = strkey;
75              }
76          }
77  
78          return super.get(realkey);
79      }
80  
81      /**
82       * Put all
83       */
84      @Override
85      public void putAll(Map<? extends K, ? extends V> t) {
86          for (Map.Entry<? extends K, ? extends V> entry : t.entrySet()) {
87              this.put(entry.getKey(), entry.getValue());
88          }
89      }
90  
91      /**
92       * Contains key
93       */
94      @Override
95      public boolean containsKey(Object key) {
96          Object realkey = key;
97  
98          if (key instanceof String) {
99              String strkey = remap.get(((String) key).toLowerCase(Locale.getDefault()));
100             if (strkey != null) {
101                 realkey = strkey;
102             }
103         }
104 
105         return super.containsKey(realkey);
106     }
107 
108     /**
109      * Remove a mapping
110      */
111     @Override
112     public V remove(Object key) {
113         Object realkey = key;
114 
115         if (key instanceof String) {
116             String uckey = ((String) key).toLowerCase(Locale.getDefault());
117             String strkey = remap.get(uckey);
118             if (strkey != null) {
119                 realkey = strkey;
120             }
121 
122             remap.remove(uckey);
123         }
124 
125         return super.remove(realkey);
126     }
127 
128 }

14

25      public static boolean isa(final String one, final String two) {
26          String currentTwo = two;
27          try {
28              while (currentTwo != null) {
29                  logger.debug("Check {} and {}", one, currentTwo);
30                  if (one.equals(currentTwo)) {
31                      return true;
32                  }
33                  final Class<?> c = Class.forName(currentTwo).getSuperclass();
34                  if (c == null) {
35                      logger.debug("Out with a null superclass from two {}", currentTwo);
36                      break;
37                  }
38                  currentTwo = c.getName();
39              }
40          } catch (Exception ex) {
41              logger.debug("one={}, two={}", one, currentTwo, ex);
42          }
43          return false;
44      }
45  
46      public static boolean isaImplementation(Class<? extends Object> clazz, Class<? extends Object> iface) {
47          return iface.isAssignableFrom(clazz);
48      }

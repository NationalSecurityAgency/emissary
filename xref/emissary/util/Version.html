28      /**
29       * Read hardwired config info. Cannot be overridden using the normal emissary.config.pkg or emissary.config.dir methods.
30       * Does not create a Configurator due to logging restrictions on this method.
31       */
32      private void readHardWiredConfigInfo() {
33          InputStream rstream = null;
34          try {
35              // This configurator is not overridable by
36              // the normal config.pkg or config.dir mechanisms.
37              // It also cannot call any method in ConfigUtil that
38              // will end up calling a logger.* method, since we
39              // are here producing the header for the log file
40              String rez = Version.class.getName().replace('.', '/') + ConfigUtil.CONFIG_FILE_ENDING;
41              // System.out.println("Reading " + rez);
42              rstream = new ResourceReader().getResourceAsStream(rez);
43              if (rstream != null) {
44                  byte[] data = IOUtils.toByteArray(rstream);
45                  String sdata = new String(data);
46                  String[] lines = sdata.split("[\r\n]+");
47                  if (lines != null) {
48                      // System.out.println(" Got " + lines.length + " lines");
49                      for (int i = 0; i < lines.length; i++) {
50                          // System.out.println("Got line " + lines[i]);
51                          if (lines[i].startsWith("emissary_version")) {
52                              version = getVal(lines[i]);
53                          } else if (lines[i].startsWith("emissary_build")) {
54                              timestamp = getVal(lines[i]);
55                          }
56                      }
57                  }
58              }
59          } catch (IOException iox) {
60              // System.out.println("Bad read: " + iox);
61          } finally {
62              if (rstream != null) {
63                  try {
64                      rstream.close();
65                  } catch (IOException ignore) {
66                  }
67              }
68          }
69      }

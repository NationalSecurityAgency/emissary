3   import emissary.config.ConfigUtil;
4   import emissary.util.io.ResourceReader;
5   
6   import org.apache.commons.io.IOUtils;
7   import org.slf4j.Logger;
8   import org.slf4j.LoggerFactory;
9   
10  import java.io.IOException;
11  import java.io.InputStream;
12  import java.nio.charset.Charset;
13  
14  public final class Version {
15      private static final Logger logger = LoggerFactory.getLogger(Version.class);
16  
17      private String version = "missing version";
18      private String timestamp = "missing timestamp";
19  
20      /**
21       * Build a version object DO NOT call a logger from here
22       */
23      public Version() {
24          readHardWiredConfigInfo();
25      }
26  
27      /**
28       * Read hardwired config info. Cannot be overridden using the normal emissary.config.pkg or emissary.config.dir methods.
29       * Does not create a Configurator due to logging restrictions on this method.
30       */
31      private void readHardWiredConfigInfo() {
32          // This configurator is not overridable by the normal config.pkg or config.dir mechanisms
33          String rez = Version.class.getName().replace('.', '/') + ConfigUtil.CONFIG_FILE_ENDING;
34          try (InputStream rstream = new ResourceReader().getResourceAsStream(rez)) {
35              if (rstream != null) {
36                  for (String line : IOUtils.readLines(rstream, Charset.defaultCharset())) {
37                      if (line.startsWith("emissary_version")) {
38                          version = getVal(line);
39                      } else if (line.startsWith("emissary_build")) {
40                          timestamp = getVal(line);
41                      }
42                  }
43              }
44          } catch (IOException ignore) {
45              // ignore
46          }
47      }
48  
49      /**
50       * Do not call a logger method from here!
51       */
52      private static String getVal(String line) {
53          if (!line.contains(" = ")) {
54              return line;
55          }
56          return line.substring(line.indexOf(" = ") + 3).replaceAll("\"", "");
57      }
58  
59      @Override
60      public String toString() {
61          return version + " " + timestamp;
62      }
63  
64      public String getVersion() {
65          return version;
66      }
67  
68      public String getTimestamp() {
69          return timestamp;
70      }
71  
72      public static void main(String[] args) {
73          logger.info("Emissary version {}", new Version());
74      }
75  }

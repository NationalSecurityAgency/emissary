34      /**
35       * Constructor to load instance using the specified File. If the specified file is invalid, an exception will be thrown
36       * when attempting utilize the <code>execute</code> method.
37       *
38       * @param magicFile the <code>File</code> containing magic number entries
39       */
40      public UnixFile(File magicFile) throws IOException {
41          new UnixFile(magicFile, false);
42      }

44      /**
45       * Constructor to load instance using the specified File. If the specified file is invalid, an exception will be thrown
46       * when attempting utilize the <code>execute</code> method.
47       *
48       * @param magicFile the <code>File</code> containing magic number entries
49       * @param swallowParseException should we swallow Ignorable ParseException or bubble them up
50       */
51      public UnixFile(File magicFile, boolean swallowParseException) throws IOException {
52          if (!magicFile.exists()) {
53              throw new IllegalArgumentException("Magic file not found at: " + magicFile.getAbsolutePath());
54          }
55  
56          this.magicFiles.add(magicFile);
57          util.load(magicFile, swallowParseException);
58      }
59  
60      /**
61       * Load multiple magic files into one identification engine
62       *
63       * @param magicPaths the String names of magic files to load
64       */
65      public UnixFile(List<String> magicPaths) throws IOException {
66          new UnixFile(magicPaths, false);
67      }
68  
69      /**
70       * Load multiple magic files into one identification engine
71       * 
72       * @param magicPaths the String names of magic files to load
73       * @param swallowParseException should we swallow Ignorable ParseException or bubble them up
74       */
75      public UnixFile(List<String> magicPaths, boolean swallowParseException) throws IOException {
76          for (String mPath : magicPaths) {
77              File mFile = new File(mPath);
78              if (!mFile.exists() || !mFile.canRead()) {
79                  throw new IllegalArgumentException("Magic file not found at " + mFile.getAbsolutePath());
80              }
81              this.magicFiles.add(mFile);
82              util.load(mFile, swallowParseException);
83          }
84      }

91      /**
92       * Behaves just like the UNIX file command. First performs a magic number test, then an ascii or binary file test. This
93       * is also the same as calling <code>evaluateByMagicNumber (bytes :
94       * byte[])</code> and then calling <code>evaluateBinaryProperty
95       * (bytes : byte[])</code>
96       */
97      public String execute(@Nullable byte[] bytes) throws IOException {
98          if (bytes == null || bytes.length < 1) {
99              return FILETYPE_EMPTY;
100         }
101 
102         String subjectFileType = evaluateByMagicNumber(bytes);
103 
104         if (subjectFileType != null) {
105             return subjectFileType;
106         } else {
107             return evaluateBinaryProperty(bytes);
108         }
109     }
110 
111     /**
112      * Statically tests a byte array to determine if the file representation can be of type ASCII or is binary. Simply
113      * checks each byte value to be less then greater/equal then 127.
114      */
115     public static String evaluateBinaryProperty(@Nullable byte[] bytes) {
116         if (bytes == null || bytes.length < 1) {
117             return FILETYPE_EMPTY;
118         }
119 
120         for (byte aByte : bytes) {
121             if (aByte < 32) {
122                 return FILETYPE_BINARY;
123             }
124         }
125         return FILETYPE_ASCII;
126     }
127 
128     /**
129      * Evaluates the byte array against the collection of Magic numbers
130      */
131     public String evaluateByMagicNumber(byte[] bytes) throws IOException {
132         return util.describe(bytes);
133     }

135     /**
136      * Test standalone main
137      */
138     @SuppressWarnings("SystemOut")
139     public static void main(String[] args) throws Exception {
140         if (args.length < 2) {
141             log.info("Usage: UnixFile [-v] magicfile file1 [file2 ...]");
142             return;
143         }
144 
145         boolean verbose = false;
146 
147         int argIndex = 0;
148         if (args[argIndex].equalsIgnoreCase("-v")) {
149             verbose = true;
150             argIndex++;
151         }
152 
153         UnixFile unixfile = null;
154 
155         File magicNumberFile = new File(args[argIndex++]);
156         if (!magicNumberFile.exists()) {
157             System.err.println("Could not find magic file: " + magicNumberFile.getAbsolutePath());
158             return;
159         } else {
160             unixfile = new UnixFile(magicNumberFile);
161             if (verbose) {
162                 System.err.println("Using magic file at: " + magicNumberFile.getAbsolutePath());
163             }
164         }
165 
166         while (argIndex < args.length) {
167             File target = new File(args[argIndex]);
168             if (!target.exists()) {
169                 System.out.println(args[argIndex] + ": UNREADABLE");
170             } else if (target.isDirectory()) {
171                 System.out.println(args[argIndex] + ": DIRECTORY");
172             } else {
173                 byte[] data = Executrix.readDataFromFile(args[argIndex]);
174                 System.out.println(args[argIndex] + ": " + unixfile.evaluateByMagicNumber(data));
175             }
176             argIndex++;
177         }
178 
179     }

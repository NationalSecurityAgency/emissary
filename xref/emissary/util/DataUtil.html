16      /**
17       * As {@link IBaseDataObject#pushCurrentForm(String)} but makes sure each form is not already on the stack first.
18       * 
19       * @param d data object to have form pushed into
20       * @param forms that are pushed into current forms if not already present
21       */
22      public static void pushDedupedForms(final IBaseDataObject d, final Collection<String> forms) {
23          for (final String f : forms) {
24              pushDedupedForm(d, f);
25          }
26      }
27  
28      /**
29       * As {@link IBaseDataObject#pushCurrentForm(String)} but makes sure each form is not already on the stack first.
30       * 
31       * @param d data object to have form pushed into
32       * @param form that are pushed into current forms if not already present
33       */
34      public static void pushDedupedForm(final IBaseDataObject d, final String form) {
35          if (d.searchCurrentForm(form) == -1) {
36              d.pushCurrentForm(form);
37          }
38      }

40      /**
41       * Check that a data object is neither null nor empty.
42       *
43       * @param d the data object

46      public static boolean isNotEmpty(@Nullable final IBaseDataObject d) {

86      /**
87       * Escape a string so it is suitable for use in a CSV record
88       */
89      public static String csvescape(final String field) {
90          final String SEP = ",";
91          String s = field;
92          if ((s != null) && (s.contains("\n") || s.contains("\r"))) {
93              s = NL_REPL.matcher(s).replaceAll(" ");
94          }
95  
96          if ((s != null) && (s.contains(SEP) || s.indexOf('"') > -1)) {
97              s = "\"" + s.replace("\"", "\"\"") + "\""; // escape quotes
98          }
99  
100         return s;
101     }

132     /**
133      * Copy a metadata item from one payload to another
134      * 
135      * @param source to copy from
136      * @param target to copy to
137      * @param key metadata key to copy
138      */
139     public static void copyParam(final IBaseDataObject source, final IBaseDataObject target, final String key) {
140         final Object value = source.getParameter(key);
141         if (value != null) {
142             target.setParameter(key, value);
143         }
144     }

7   import java.util.List;
8   import java.util.concurrent.atomic.AtomicInteger;

10  public class CheckCloseChannelFactory implements SeekableByteChannelFactory {
11      private final AtomicInteger instanceNumber = new AtomicInteger(0);
12  
13      public final List<Boolean> isClosedList = Collections.synchronizedList(new ArrayList<>());
14  
15      @Override
16      public SeekableByteChannel create() {
17          isClosedList.add(false);
18  
19          return new AbstractSeekableByteChannel() {
20              final int myInstanceNumber = instanceNumber.getAndIncrement();
21  
22              @Override
23              protected void closeImpl() {
24                  isClosedList.set(myInstanceNumber, true);
25              }
26  
27              @Override
28              protected int readImpl(ByteBuffer byteBuffer) {
29                  return 0;
30              }
31  
32              @Override
33              protected long sizeImpl() {
34                  return 1;
35              }
36          };
37      }
38  }

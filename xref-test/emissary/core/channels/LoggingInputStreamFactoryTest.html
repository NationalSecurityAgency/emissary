26  class LoggingInputStreamFactoryTest extends UnitTest {
27      @Nullable
28      private ListAppender<ILoggingEvent> appender = null;
29      @SuppressWarnings("HidingField")
30      private final Logger logger = (Logger) LoggerFactory.getLogger(LoggingInputStreamFactoryTest.class);
31  
32      @BeforeEach
33      void setup() throws IOException {
34          appender = new ListAppender<>();
35          appender.start();
36          logger.addAppender(appender);
37      }
38  
39      @AfterEach
40      void detach() throws Exception {
41          logger.detachAppender(appender);
42      }

44      @Test
45      void testInputStreamFactory() throws IOException {
46          final byte[] bytes = new byte[100];
47          final TestInputStreamFactory tisf = new TestInputStreamFactory(bytes);
48  
49          assertThrows(NullPointerException.class,
50                  () -> LoggingInputStreamFactory.create(null, "Identifier", logger, false));
51          assertThrows(NullPointerException.class, () -> LoggingInputStreamFactory.create(tisf, null, logger, false));
52          assertThrows(NullPointerException.class,
53                  () -> LoggingInputStreamFactory.create(tisf, "Identifier", null, false));
54  
55          final InputStreamFactory loggingFalseIsf = LoggingInputStreamFactory.create(tisf, "Identifier", logger, false);
56  
57          testLogging(loggingFalseIsf, false);
58  
59          final InputStreamFactory loggingTrueIsf = LoggingInputStreamFactory.create(tisf, "Identifier", logger, true);
60  
61          testLogging(loggingTrueIsf, true);
62      }

64      private void testLogging(final InputStreamFactory isf, final boolean testStackTrace) throws IOException {
65          assertTrue(appender.list.isEmpty());
66  
67          try (InputStream is = isf.create()) {
68              check("Identifier : 0 : created : lST=" + testStackTrace, testStackTrace);
69  
70              assertTrue(is.markSupported());
71  
72              check("Identifier : 0 : markSupported : r=true", testStackTrace);
73  
74              is.mark(100);
75  
76              check("Identifier : 0 : mark : rl=100", testStackTrace);
77  
78              assertEquals(0, is.read());
79  
80              check("Identifier : 0 : read-0 : r=0", testStackTrace);
81  
82              assertEquals(10, is.read(new byte[10]));
83  
84              check("Identifier : 0 : read-1 : r=10", testStackTrace);
85  
86              assertEquals(5, is.read(new byte[10], 5, 5));
87  
88              check("Identifier : 0 : read-3 : off=5 len=5 : r=5", testStackTrace);
89  
90              assertEquals(84, is.available());
91  
92              check("Identifier : 0 : available : r=84", testStackTrace);
93  
94              assertEquals(15, is.skip(15));
95  
96              check("Identifier : 0 : skip : n=15 : r=15", testStackTrace);
97  
98              is.reset();
99  
100             check("Identifier : 0 : reset", testStackTrace);
101         }
102 
103         check("Identifier : 0 : close", testStackTrace);
104     }
105 
106     private void check(final String message, final boolean testStackTrace) {
107         assertEquals(1, appender.list.size());
108 
109         final ILoggingEvent loggingEvent = appender.list.get(0);
110 
111         assertEquals(message, loggingEvent.getFormattedMessage());
112 
113         final IThrowableProxy throwableProxy = loggingEvent.getThrowableProxy();
114 
115         if (testStackTrace) {
116             assertNotNull(throwableProxy);
117             assertEquals("java.lang.Throwable", throwableProxy.getClassName());
118             assertEquals("DEBUG", throwableProxy.getMessage());
119         } else {
120             assertNull(throwableProxy);
121         }
122 
123         appender.list.clear();
124     }

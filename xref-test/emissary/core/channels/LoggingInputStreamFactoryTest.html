25  class LoggingInputStreamFactoryTest extends UnitTest {
26      private ListAppender<ILoggingEvent> appender = null;
27      private final Logger logger = (Logger) LoggerFactory.getLogger(LoggingInputStreamFactoryTest.class);
28  
29      @BeforeEach
30      void setup() throws IOException {
31          appender = new ListAppender<>();
32          appender.start();
33          logger.addAppender(appender);
34      }
35  
36      @AfterEach
37      void detach() throws Exception {
38          logger.detachAppender(appender);
39      }
40  
41      @Test
42      void testInputStreamFactory() throws IOException {
43          final byte[] bytes = new byte[100];
44          final TestInputStreamFactory tisf = new TestInputStreamFactory(bytes);
45  
46          assertThrows(NullPointerException.class,
47                  () -> LoggingInputStreamFactory.create(null, "Identifier", logger, false));
48          assertThrows(NullPointerException.class, () -> LoggingInputStreamFactory.create(tisf, null, logger, false));
49          assertThrows(NullPointerException.class,
50                  () -> LoggingInputStreamFactory.create(tisf, "Identifier", null, false));
51  
52          final InputStreamFactory loggingFalseIsf = LoggingInputStreamFactory.create(tisf, "Identifier", logger, false);
53  
54          testLogging(loggingFalseIsf, false);
55  
56          final InputStreamFactory loggingTrueIsf = LoggingInputStreamFactory.create(tisf, "Identifier", logger, true);
57  
58          testLogging(loggingTrueIsf, true);
59      }
60  
61      private void testLogging(final InputStreamFactory isf, final boolean testStackTrace) throws IOException {
62          assertTrue(appender.list.isEmpty());
63  
64          try (InputStream is = isf.create()) {
65              check("Identifier : 0 : created : lST=" + testStackTrace, testStackTrace);
66  
67              assertTrue(is.markSupported());
68  
69              check("Identifier : 0 : markSupported : r=true", testStackTrace);
70  
71              is.mark(100);
72  
73              check("Identifier : 0 : mark : rl=100", testStackTrace);
74  
75              assertEquals(0, is.read());
76  
77              check("Identifier : 0 : read-0 : r=0", testStackTrace);
78  
79              assertEquals(10, is.read(new byte[10]));
80  
81              check("Identifier : 0 : read-1 : r=10", testStackTrace);
82  
83              assertEquals(5, is.read(new byte[10], 5, 5));
84  
85              check("Identifier : 0 : read-3 : off=5 len=5 : r=5", testStackTrace);
86  
87              assertEquals(84, is.available());
88  
89              check("Identifier : 0 : available : r=84", testStackTrace);
90  
91              assertEquals(15, is.skip(15));
92  
93              check("Identifier : 0 : skip : n=15 : r=15", testStackTrace);
94  
95              is.reset();
96  
97              check("Identifier : 0 : reset", testStackTrace);
98          }
99  
100         check("Identifier : 0 : close", testStackTrace);
101     }
102 
103     private void check(final String message, final boolean testStackTrace) {
104         assertEquals(1, appender.list.size());
105 
106         final ILoggingEvent loggingEvent = appender.list.get(0);
107 
108         assertEquals(message, loggingEvent.getFormattedMessage());
109 
110         final IThrowableProxy throwableProxy = loggingEvent.getThrowableProxy();
111 
112         if (testStackTrace) {
113             assertNotNull(throwableProxy);
114             assertEquals("java.lang.Throwable", throwableProxy.getClassName());
115             assertEquals("DEBUG", throwableProxy.getMessage());
116         } else {
117             assertNull(throwableProxy);
118         }
119 
120         appender.list.clear();
121     }
122 
123     private static class TestInputStreamFactory implements InputStreamFactory {
124         private final byte[] bytes;
125 
126         public TestInputStreamFactory(final byte[] bytes) {
127             this.bytes = bytes;
128         }
129 
130         @Override
131         public InputStream create() {
132             return new BufferedInputStream(new ByteArrayInputStream(bytes));
133         }
134     }

136     public interface ElementDecoder<T, U> {
137         /**
138          * Decodes and XML element value and sets it on the specified IBDO method.
139          * 
140          * @param elements the list of elements to be decoded.
141          * @param ibdo the ibdo to set the values on.
142          * @param ibdoMethod the ibdo method name to use to set the values.
143          * @throws IOException thrown if anything goes wrong.
144          */
145         void decode(List<Element> elements, IBaseDataObject ibdo, BiConsumer<T, U> ibdoMethod) throws IOException;
146     }

460             for (final SeekableByteChannelFactory value : values) {
461                 if (value != null) {
462                     try {
463                         final byte[] bytes = SeekableByteChannelHelper.getByteArrayFromChannel(value,
464                                 BaseDataObject.MAX_BYTE_ARRAY_SIZE);
465                         final Element childElement = preserve(protectedElementBase64(childElementName, bytes));
466 
467                         childElement.setAttribute(LENGTH_ATTRIBUTE_NAME, Integer.toString(bytes.length));
468 
469                         parentElement.addContent(childElement);
470                     } catch (final IOException e) {
471                         LOGGER.error("Could not get bytes from SeekableByteChannel!", e);
472                     }
473                 }
474             }
475         }

538     private static class StringEncoder implements ElementEncoder<String> {
539         @Override
540         public void encode(final List<String> values, final Element parentElement, final String childElementName) {
541             for (int i = values.size() - 1; i >= 0; i--) {
542                 String value = values.get(i);
543 
544                 if (PROCESSING_ERROR.equals(childElementName) && StringUtils.isNotEmpty(value)) {
545                     value = value.substring(0, value.length() - 1);
546                 }
547 
548                 if (value != null) {
549                     parentElement.addContent(preserve(protectedElement(childElementName, value)));

716     /**
717      * Return UTF8 bytes from an XML value, decoding base64 if required
718      * 
719      * @param encoding e.g. 'base64', otherwise it returns the bytes as they are presented
720      * @param elementValue containing the data
721      * @return the data from elementValue, decoded from base64 if required
722      */
723     public static byte[] extractBytes(final String encoding, final String elementValue) {
724         if (BASE64.equalsIgnoreCase(encoding)) {
725             final String newElementValue = elementValue.replace("\n", "");
726             final byte[] bytes = newElementValue.getBytes(StandardCharsets.UTF_8);
727             return BASE64_DECODER.decode(bytes);
728         }
729 
730         return elementValue.getBytes(StandardCharsets.UTF_8);
731     }

757      * Creates a 'protected' element which can be encoded with base64 if it contains non-printable characters
758      * 
759      * See method source for specific definition of 'non-printable'.
760      * 
761      * @param name of the element
762      * @param bytes to wrap, if they contain non-printable characters
763      * @return the created element
764      */
765     public static Element protectedElementBase64(final String name, final byte[] bytes) {
766         final Element element = new Element(name);
767 
768         if (ByteUtil.containsNonIndexableBytes(bytes)) {
769             String base64String = BASE64_NEW_LINE_STRING +
770                     BASE64_ENCODER.encodeToString(bytes) +
771                     BASE64_NEW_LINE_STRING;
772 
773             element.setAttribute(ENCODING_ATTRIBUTE_NAME, BASE64);
774             element.addContent(base64String);
775         } else {
776             element.addContent(new String(bytes, StandardCharsets.UTF_8));
777         }
778 
779         return element;
780     }

782     /**
783      * Creates a 'protected' element which can be hashed with sha256 if it contains non-printable characters
784      * 
785      * See method source for specific definition of 'non-printable'.
786      * 
787      * @param name of the element
788      * @param bytes to wrap, if they contain non-printable characters.
789      * @param alwaysHash overrides the non-printable check and always hashes the bytes.
790      * @return the created element
791      */
792     public static Element protectedElementSha256(final String name, final byte[] bytes, final boolean alwaysHash) {
793         final Element element = new Element(name);
794 
795         if (alwaysHash || ByteUtil.containsNonIndexableBytes(bytes)) {
796             element.setAttribute(IBaseDataObjectXmlCodecs.ENCODING_ATTRIBUTE_NAME, IBaseDataObjectXmlCodecs.SHA256);
797             element.addContent(ByteUtil.sha256Bytes(bytes));
798         } else {
799             element.addContent(new String(bytes, StandardCharsets.UTF_8));
800         }
801 
802         return element;
803     }

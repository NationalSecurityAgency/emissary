7    * For hacking the move test. This agent forces a move even when place is in the namespace.

9   public class BogusHDAgent extends HDMobileAgent {
10  
11      static final long serialVersionUID = -8044923045057998565L;
12  
13      public BogusHDAgent() {
14          super();
15      }
16  
17      /**
18       * Create a new reusable Agent
19       *
20       * @param threadGroup group we operate it
21       * @param threadName symbolic name for this agent thread
22       */
23      public BogusHDAgent(final ThreadGroup threadGroup, final String threadName) {
24          super(threadGroup, threadName);
25      }
26  
27      /**
28       * Get the next key from the directory with error handling Can return null if there is no place to handle the form
29       *
30       * @param place the place we will use to access the directory
31       * @param payload the current payload we care about
32       * @return the fake Directory Entry (local always false)
33       */
34      @Override
35      protected DirectoryEntry getNextKey(final IServiceProviderPlace place, final IBaseDataObject payload) {
36          final DirectoryEntry de = super.getNextKey(place, payload);
37          if (de == null) {
38              return null;
39          }
40  
41          // Compare host and port
42          final DirectoryEntry lastVisited = payload.getLastPlaceVisited();
43          if (lastVisited != null && de.getServiceHostURL().equals(lastVisited.getServiceHostURL())) {
44              // Ok they really *are* local
45              return de;
46          }
47          // They appear local due to being in same JVM but on are on different
48          // host and port, so force them to be non-local
49          return new BogusDirectoryEntry(de);
50      }
51  }

19      /**
20       * Create a new reusable Agent
21       *
22       * @param threadGroup group we operate it
23       * @param threadName symbolic name for this agent thread
24       */
25      public BogusHDAgent(final ThreadGroup threadGroup, final String threadName) {
26          super(threadGroup, threadName);
27      }
28  
29      /**
30       * Get the next key from the directory with error handling Can return null if there is no place to handle the form
31       *
32       * @param place the place we will use to access the directory
33       * @param payload the current payload we care about
34       * @return the fake Directory Entry (local always false)
35       */
36      @Override
37      @Nullable
38      protected DirectoryEntry getNextKey(final IServiceProviderPlace place, final IBaseDataObject payload) {
39          final DirectoryEntry de = super.getNextKey(place, payload);
40          if (de == null) {
41              return null;
42          }
43  
44          // Compare host and port
45          final DirectoryEntry lastVisited = payload.getLastPlaceVisited();
46          if (lastVisited != null && de.getServiceHostUrl().equals(lastVisited.getServiceHostUrl())) {
47              // Ok they really *are* local
48              return de;
49          }
50          // They appear local due to being in same JVM but on are on different
51          // host and port, so force them to be non-local
52          return new BogusDirectoryEntry(de);
53      }

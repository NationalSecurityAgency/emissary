14  public class MetadataDictionaryTest extends UnitTest {
15      private static final String TEST_NAMESPACE = "test_namespace";
16  
17      private MetadataDictionary getDict(boolean bind) {
18          MetadataDictionary m = null;
19          try {
20              Configurator conf = ConfigUtil.getConfigInfo(this.getClass());
21              assertNotNull(conf, "Could not find config for MetadataDictionaryTest");
22              if (bind) {
23                  m = MetadataDictionary.initialize(TEST_NAMESPACE, conf);
24              } else {
25                  m = new MetadataDictionary(TEST_NAMESPACE, conf);
26              }
27          } catch (Exception ex) {
28              fail("Exception configuring dictionary", ex);
29          }
30          assertNotNull(m, "Metadata Dictionary must be created");
31          return m;
32      }
33  
34      private MetadataDictionary getDict() {
35          return getDict(false);
36      }
37  
38      private void clearNamespace() {
39          emissary.core.Namespace.unbind(TEST_NAMESPACE);
40      }
41  
42      @Test
43      void testRename() {
44          MetadataDictionary d = getDict();
45          assertEquals("bar", d.rename("foo"), "Renaming for found value");
46      }
47  
48      @Test
49      void testNonRename() {
50          MetadataDictionary d = getDict();
51          assertEquals("baz", d.rename("baz"), "Original name for not-found value");
52      }
53  
54      @Test
55      void testRegexRename() {
56          MetadataDictionary d = getDict();
57          assertEquals("^bar^", d.regex("abc_bar_def"), "Rename by regex match");
58      }
59  
60      @Test
61      void testRegexNonRename() {
62          MetadataDictionary d = getDict();
63          assertEquals("shazam", d.regex("shazam"), "Original name when no regex match");
64      }
65  
66      @Test
67      void testFullMapRename() {
68          MetadataDictionary d = getDict();
69          assertEquals("NothingHappens", d.map("xyzzy"), "Rename on exact match via map");
70      }
71  
72      @Test
73      void testFullMapRegex() {
74          MetadataDictionary d = getDict();
75          assertEquals("NothingHappensCaseInsensitively", d.map("xyZZy"), "Rename on regex match via map");
76      }
77  
78      @Test
79      void testFullMapNoMatch() {
80          MetadataDictionary d = getDict();
81          assertEquals("Hello", d.map("Hello"), "NO match via map gives original name");
82      }
83  
84  
85      @Test
86      void testNamespace() {
87          try {
88              MetadataDictionary d1 = getDict(true);
89              MetadataDictionary d2 = MetadataDictionary.lookup(TEST_NAMESPACE);
90              assertEquals(d1, d2, "INitialize must bind by default lookup name");
91          } catch (Exception ex) {
92              fail("Exception doing namespace lookup", ex);
93          } finally {
94              clearNamespace();
95          }
96      }
97  
98      @Test
99      void testNullRegex() {
100         MetadataDictionary dict = getDict(true);
101         try {
102             assertNull(dict.regex(null), "regex() should return null if we pass it a null value.");
103         } catch (NullPointerException ex) {
104             fail("An exception was thrown instead of handling a null value gracefully", ex);
105         }
106     }

17      @Test
18      void testSet() {
19          String key1 = "UNKNOWN.FILE_PICK_UP.INPUT.http://localhost:8001/FilePickUpPlace$5050";
20          String key2 = "KNOWN.COOL_STUFF.TRANSFORM.http://localhost:8001/CoolStuffPlace$5050";
21          String key3 = "KNOWN.ONE_THING.ANALYZE.http://localhost:8001/DoOneThingPlace$5050";
22          String key4 = "KNOWN.LAST_PLACE.VERIFY.http://localhost:8001/LastThingPlace$5050";
23  
24          var keySet1 = List.of(key1, key2);
25          var keySet2 = List.of(key3, key4);
26  
27          // create, set and check the history
28          TransformHistory th = new TransformHistory();
29          th.set(keySet1);
30          assertEquals(2, th.getHistory().size());
31          assertTrue(th.get().contains(key1));
32          assertTrue(th.get().contains(key2));
33  
34          // re-set and check that history is cleared
35          th.set(keySet2);
36          assertEquals(2, th.getHistory().size());
37          assertTrue(th.get().contains(key3));
38          assertTrue(th.get().contains(key4));
39  
40          // set using object
41          TransformHistory historyObject = new TransformHistory();
42          historyObject.set(keySet1);
43          th.set(historyObject);
44          assertEquals(2, th.getHistory().size());
45          assertTrue(th.get().contains(key1));
46          assertTrue(th.get().contains(key2));
47  
48          // create using object
49          TransformHistory newHistory = new TransformHistory(historyObject);
50          assertEquals(2, newHistory.getHistory().size());
51          assertTrue(newHistory.get().contains(key1));
52          assertTrue(newHistory.get().contains(key2));
53      }

55      @Test
56      void testAppend() {
57          String key1 = "UNKNOWN.FILE_PICK_UP.INPUT.http://localhost:8001/FilePickUpPlace$5050";
58          String key2 = "KNOWN.COOL_STUFF.COORDINATE.http://localhost:8001/CoolStuffPlace$5050";
59          String key3 = "KNOWN.ONE_THING.ANALYZE.http://localhost:8001/DoOneThingPlace$5050";
60          var keyList = List.of(key1);
61  
62          // create and seed a transform history
63          TransformHistory th = new TransformHistory();
64          th.set(keyList);
65          assertEquals(1, th.getHistory().size());
66          assertTrue(th.get().contains(key1));
67  
68          // now append a value
69          th.append(key2);
70          assertEquals(2, th.getHistory().size());
71          assertTrue(th.get().contains(key2));
72          assertTrue(th.getHistory().get(1).getCoordinated().isEmpty());
73  
74          // now append a coordinated value
75          th.append(key3, true);
76          assertEquals(2, th.getHistory().size());
77          assertFalse(th.get().contains(key3)); // does not include coordinated results
78          assertTrue(th.get(true).contains(key3));
79          assertEquals(key3, th.getHistory().get(1).getCoordinated().get(0));
80      }

82      @Test
83      void testVisitation() {
84          String key1 = "UNKNOWN.FILE_PICK_UP.INPUT.http://localhost:8001/FilePickUpPlace$5050";
85          String key2 = "KNOWN.COOL_STUFF.COORDINATE.http://localhost:8001/CoolStuffPlace$5050";
86          String key3 = "KNOWN.ONE_THING.ANALYZE.http://localhost:8001/DoOneThingPlace$5050";
87          String key4 = "KNOWN.LAST_PLACE.VERIFY.http://localhost:8001/LastThingPlace$5050";
88  
89          TransformHistory th = new TransformHistory();
90  
91          // set and check the history when last place visited was coordinated
92          th.append(key1);
93          th.append(key2);
94          th.append(key3, true);
95          assertEquals(2, th.getHistory().size());
96  
97          assertEquals(key2, th.lastVisit().getKey()); // skips the coordinated place
98          assertEquals(key1, th.penultimateVisit().getKey());
99  
100         // set and check the history with the last place visited was not coordinated
101         th.append(key4);
102         assertEquals(3, th.getHistory().size());
103 
104         assertEquals(key4, th.lastVisit().getKey());
105         assertEquals(key2, th.penultimateVisit().getKey()); // skips the coordinated place
106 
107         // check if a place has been visited
108         assertFalse(th.hasVisited("*.ONE_THING.*.*")); // coordinated place is ignored
109         assertTrue(th.hasVisited("*.LAST_PLACE.*.*"));
110         assertFalse(th.hasVisited("*.NEVER_PLACE.*.*"));
111 
112         // clear and check for null
113         th.clear();
114         assertNull(th.lastVisit());
115         assertNull(th.penultimateVisit());
116     }

121         String key2 = "KNOWN.COOL_STUFF.TRANSFORM.http://localhost:8001/CoolStuffPlace$5050";

123         String key4 = "KNOWN.LAST_PLACE.VERIFY.http://localhost:8001/LastThingPlace$5050";
124         var keyList = List.of(key1, key2, key3);
125 
126         TransformHistory th = new TransformHistory();
127 
128         // empty should return true
129         assertTrue(th.beforeStart());
130 
131         // just one key, no sprout
132         th.append(key1);
133         assertFalse(th.beforeStart());
134 
135         // history with a sprout as the last item
136         th.set(keyList);
137         assertTrue(th.beforeStart());
138 
139         // processing after sprout
140         th.append(key4);
141         assertFalse(th.beforeStart());

157         expected.append("        -> ").append(key2).append(newline);
158         expected.append("        -> ").append(key3).append(newline);
159         expected.append("        -> ").append(key4).append(newline);

161         expected.append("        -> ").append(key6).append(newline);
162 
163         TransformHistory th = new TransformHistory();
164         th.append(key1);
165         th.append(key2);
166         th.append(key3);
167         th.append(key4);
168         th.append(key5, true);
169         th.append(key6);
170

99      private void runTest(@Nullable final String agentClass) {
100         assertNotNull(this.dir1, "Directory1 should have been set up");
101         assertNotNull(this.dir2, "Directory2 should have been set up");
102         assertNotNull(this.toLower, "toLower place should have been set up");
103         assertNotNull(this.toUpper, "toUpper place should have been set up");
104         assertNotNull(this.cache, "cache place should have been set up");
105 
106         if (agentClass != null && !pool.getClassName().equals(agentClass)) {
107             logger.debug("Resetting factory from " + pool.getClassName() + " to " + agentClass);
108             pool.resetFactory(new MobileAgentFactory(agentClass));
109             try {
110                 pool = AgentPool.lookup();
111                 logger.debug("Found pool after reset to " + agentClass);
112                 spool.resetPool();
113             } catch (NamespaceException ex) {
114                 fail("Agent pool is missing after reset to " + agentClass, ex);
115             }
116         } else if (agentClass != null) {
117             logger.debug("Factory already using " + agentClass);
118         }
119 
120         // Create a payload and send it to the spool for UpperPlace
121         IBaseDataObject payload = DataObjectFactory.getInstance(new Object[] {"abcdefghijklmnopqrstuvwxyz".getBytes(), "test_load", "LOWER_CASE"});
122         payload.setFileType(TEXT);
123 
124         spool.send(payload);
125 
126         // Wait up to 10 seconds in 1/4 second increments
127         int waitCounter = 0;
128         while (this.cache.getCacheSize() == 0 && waitCounter < 40) {
129             try {
130                 Thread.sleep(250);
131             } catch (Exception ex) {
132                 if (ex instanceof InterruptedException) {
133                     Thread.currentThread().interrupt();
134                 }
135             }
136             waitCounter++;
137         }
138 
139         // Check agent arrived at cache
140         assertTrue(this.cache.getCacheSize() > 0, "Payload should arrive at cache using " + agentClass);
141 
142         payload = this.cache.pop();
143         assertNotNull(payload, "Payload coming from cache");
144 
145         assertEquals("FINI", payload.currentForm(), "Current form on payload");
146         final List<String> th = payload.transformHistory();
147         assertTrue(th.size() > 5, "Transform history");
148 
149 
150         // Purge the cache for the next test
151         payload = null;
152         do {
153             payload = this.cache.pop();
154         } while (payload != null);
155     }

157     private void runHttpTest() {
158         final Version version = new Version();
159         final EmissaryClient h = new EmissaryClient();
160         final String urlBase = KeyManipulator.getServiceHostUrl(this.dir1.getKey()) + EmissaryClient.context + "/";
161         HttpGet get = new HttpGet(urlBase + "welcome.jsp");
162         EmissaryResponse ws = h.send(get);
163         String msg = ws.getContentString();
164         // TODO consider moving this into the EmissaryResponse ((ws.getStatus() == HttpStatus.SC_OK)))
165         assertEquals(HttpStatus.SC_OK, ws.getStatus(), "welcome.jsp must return good status");
166         assertNotNull(msg, "welcome.jsp must have text");
167         assertTrue(msg.contains(version.toString()), "welcome.jsp contains version");
168 
169         get = new HttpGet(urlBase + "Namespace.action");
170         ws = h.send(get);
171         msg = ws.getContentString();
172         assertEquals(HttpStatus.SC_OK, ws.getStatus(), "Namespace.action must return good status");
173         assertNotNull(msg, "Namespace text must not be mull");
174         assertTrue(msg.contains("MoveSpool"), "Namespace must contain MoveSpool");
175         assertTrue(msg.contains("AgentPool"), "Namespace must contain AgentPool");
176         assertTrue(msg.contains("Directory"), "Namespace must contain directory");
177         assertTrue(msg.contains("CachePlace"), "Namespace must have place name");
178 
179         get = new HttpGet(urlBase + "DumpDirectory.action");
180         ws = h.send(get);
181         msg = ws.getContentString();
182         assertEquals(HttpStatus.SC_OK, ws.getStatus(), "DumpDirectory.action must return good status");
183         assertNotNull(msg, "DumpDirectory must have text");
184         assertTrue(msg.contains("CachePlace"), "DumpDirectory must have place name");
185 
186     }

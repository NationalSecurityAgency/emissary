295     @Test
296     void watch() throws Exception {
297         String agentKey = "MobileAgent-01";
298         String placename = "thePlace";
299         String shortname = "testing.txt";
300 
301         IMobileAgent hdma = mock(IMobileAgent.class);
302         when(hdma.getName()).thenReturn(agentKey);
303         when(hdma.isInUse()).thenReturn(true);
304         when(hdma.agentId()).thenReturn("Agent-1234-" + shortname);
305         when(hdma.getLastPlaceProcessed()).thenReturn("http://host.domain.com:8001/" + placename);
306 
307         Sentinel sm = mock(Sentinel.class);
308         when(sm.getPollingInterval()).thenReturn(5L);
309 
310         AgentTracker tracker;
311         try (MockedStatic<Namespace> namespace = Mockito.mockStatic(Namespace.class);
312                 MockedStatic<Sentinel> sentinel = Mockito.mockStatic(Sentinel.class)) {
313             namespace.when(Namespace::keySet).thenReturn(Set.of(agentKey));
314             namespace.when(() -> Namespace.lookup(agentKey)).thenReturn(hdma);
315             sentinel.when(() -> Sentinel.lookup()).thenReturn(sm);
316             protocol.run();
317             tracker = protocol.trackers.get(agentKey);
318         }
319 
320         assertNotNull(tracker);
321         assertEquals(0, tracker.getTimer());
322         assertEquals(placename, tracker.getPlaceName());
323         assertEquals(shortname, tracker.getShortName());

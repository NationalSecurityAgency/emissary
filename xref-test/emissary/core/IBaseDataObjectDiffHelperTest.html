51      private void verifyDiff(final List<String> expectedDifferencesForward, final List<String> expectedDifferencesReverse) {
52          verifyDiff(expectedDifferencesForward, expectedDifferencesReverse, EMPTY_OPTIONS);
53      }

55      private void verifyDiff(final List<String> expectedDifferencesForward, final List<String> expectedDifferencesReverse,
56              final DiffCheckConfiguration options) {
57          IBaseDataObjectDiffHelper.diff(ibdo1, ibdo1, differences, options);
58          assertEquals(0, differences.size());
59          IBaseDataObjectDiffHelper.diff(ibdo1, ibdo2, differences, options);
60          replaceVariableText(differences);
61          assertIterableEquals(expectedDifferencesForward, differences);
62          differences.clear();
63          IBaseDataObjectDiffHelper.diff(ibdo2, ibdo1, differences, options);
64          replaceVariableText(differences);
65          assertIterableEquals(expectedDifferencesReverse, differences);
66          differences.clear();
67      }
68  
69      private static void replaceVariableText(final List<String> differences) {
70          for (int i = 0; i < differences.size(); i++) {
71              final String difference = differences.remove(i);
72              final String differenceNohash = difference.replaceAll("@[0-9a-fA-F]{5,8}", "@xxxxxxxx");
73              final String differenceNohashNouuid =
74                      differenceNohash.replaceAll("[0-9a-fA-F]{8}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{4}-[0-9a-fA-F]{12}",
75                              "xxxxxxxx-xxxx-xxxx-xxxx-xxxxxxxxxxxx");
76  
77              differences.add(i, differenceNohashNouuid);
78          }
79      }
80  
81      private void verifyDiffList(final List<String> expectedDifferences, @Nullable final List<IBaseDataObject> list1,
82              @Nullable final List<IBaseDataObject> list2) {
83          IBaseDataObjectDiffHelper.diff(list1, list1, "test", differences, EMPTY_OPTIONS);
84          assertEquals(0, differences.size());
85          IBaseDataObjectDiffHelper.diff(list1, list2, "test", differences, EMPTY_OPTIONS);
86          assertIterableEquals(expectedDifferences, differences);
87          differences.clear();
88      }

101         checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(ibdoList1, ibdoList2, "id", null, EMPTY_OPTIONS));
102         checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(ibdo1, ibdo2, differences, null));
103         checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(new Object(), new Object(), null, differences));
104         checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(new Object(), new Object(), "id", null));
105 
106         // Integers
107         checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(0, 0, null, differences));
108         checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(0, 0, "id", null));
109 
110         // Booleans
111         checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(false, false, null, differences));
112         checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(false, false, "id", null));
113 
114         // Maps
115         checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(null, new HashMap<>(), "id", differences));
116         checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(new HashMap<>(), null, "id", differences));
117         checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(new HashMap<>(), new HashMap<>(), null, differences));
118         checkThrowsNull(() -> IBaseDataObjectDiffHelper.diff(new HashMap<>(), new HashMap<>(), "id", null));

175     @Test
176     void testDiffParameters() {
177         final DiffCheckConfiguration checkKeyValueDiffParameter = DiffCheckConfiguration.configure().enableKeyValueParameterDiff().build();
178         final DiffCheckConfiguration checkDetailedDiffParameter = DiffCheckConfiguration.configure().enableDetailedParameterDiff().build();
179 
180         ibdo1.putParameter("STRING", "string");
181         ibdo1.putParameter("LIST", Arrays.asList("first", "second", "third"));
182 
183         verifyDiff(List.of("meta are not equal-Differing Keys: [LIST, STRING] : []"),
184                 List.of("meta are not equal-Differing Keys: [] : [LIST, STRING]"));
185         verifyDiff(List.of("meta are not equal-Differing Keys/Values: {LIST=[first, second, third], STRING=[string]} : {}"),
186                 List.of("meta are not equal-Differing Keys/Values: {} : {LIST=[first, second, third], STRING=[string]}"),
187                 checkKeyValueDiffParameter);
188         verifyDiff(List.of("meta are not equal: {LIST=[first, second, third], STRING=[string]} : {}"),
189                 List.of("meta are not equal: {} : {LIST=[first, second, third], STRING=[string]}"), checkDetailedDiffParameter);
190 
191         ibdo1.clearParameters();
192         ibdo2.clearParameters();
193         ibdo1.putParameter("STRING", "string");
194         ibdo1.putParameter("Integer", Integer.valueOf(1));
195         ibdo2.putParameter("STRING", "string");
196 
197         verifyDiff(List.of("meta are not equal-Differing Keys: [Integer] : []"), List.of("meta are not equal-Differing Keys: [] : [Integer]"));
198         verifyDiff(List.of("meta are not equal-Differing Keys/Values: {Integer=[1]} : {}"),
199                 List.of("meta are not equal-Differing Keys/Values: {} : {Integer=[1]}"), checkKeyValueDiffParameter);
200         verifyDiff(List.of("meta are not equal: {Integer=[1], STRING=[string]} : {STRING=[string]}"),
201                 List.of("meta are not equal: {STRING=[string]} : {Integer=[1], STRING=[string]}"), checkDetailedDiffParameter);
202 
203         ibdo1.clearParameters();
204         ibdo2.clearParameters();
205         ibdo1.putParameter("STRING", "string");
206         ibdo2.putParameter("STRING", "string");
207         ibdo2.putParameter("Integer", Integer.valueOf(1));
208 
209         verifyDiff(List.of("meta are not equal-Differing Keys: [] : [Integer]"), List.of("meta are not equal-Differing Keys: [Integer] : []"));
210         verifyDiff(List.of("meta are not equal-Differing Keys/Values: {} : {Integer=[1]}"),
211                 List.of("meta are not equal-Differing Keys/Values: {Integer=[1]} : {}"), checkKeyValueDiffParameter);
212         verifyDiff(List.of("meta are not equal: {STRING=[string]} : {Integer=[1], STRING=[string]}"),
213                 List.of("meta are not equal: {Integer=[1], STRING=[string]} : {STRING=[string]}"), checkDetailedDiffParameter);
214 
215         ibdo1.clearParameters();
216         ibdo2.clearParameters();
217         ibdo1.putParameter("STRING", "string");
218         ibdo2.putParameter("STRING", "string");
219 
220         verifyDiff(Collections.emptyList(), Collections.emptyList());
221         verifyDiff(Collections.emptyList(), Collections.emptyList(), checkKeyValueDiffParameter);
222         verifyDiff(Collections.emptyList(), Collections.emptyList(), checkDetailedDiffParameter);
223     }

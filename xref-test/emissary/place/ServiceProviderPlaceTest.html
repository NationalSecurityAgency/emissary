212     private void runFileConfiguredTest(boolean usePackage, int ctorType) throws IOException, EmissaryException {
213         Path cfg = null;
214         OutputStream fos = null;
215         try {
216             // Write out the config data to the temp config dir
217             if (usePackage) {
218                 cfg = Paths.get(CFGDIR, thisPackage.getName() + ".MyFileConfigedTestPlace" + ConfigUtil.CONFIG_FILE_ENDING);
219             } else {
220                 cfg = Paths.get(CFGDIR, "MyFileConfigedTestPlace" + ConfigUtil.CONFIG_FILE_ENDING);
221             }
222             fos = Files.newOutputStream(cfg);
223             fos.write(configData);
224 
225             MyFileConfigedTestPlace mtp = null;
226 
227             if (ctorType == 1) {
228                 // No config specified, auto-discovered
229                 mtp = new MyFileConfigedTestPlace("http://example.com:8001/MyFileConfigedTestPlace");
230             } else if (ctorType == 2) {
231                 // String name of config file
232                 mtp = new MyFileConfigedTestPlace(cfg.getFileName().toString(), null, "http://example.com:8001/MyTestPlace");
233             } else if (ctorType == 3) {
234                 // Stream input of config
235                 InputStream config = new ByteArrayInputStream(configData);
236                 mtp = new MyFileConfigedTestPlace(config, null, "http://example.com:8001/MyTestPlace");
237             }
238 
239             assertNotNull(mtp, "Test run with bad ctypeType arg?");
240 
241             assertTrue(mtp.getFinishedSuperConstructor(), "MyFileConfigedTestPlace did not initialize");
242 
243             mtp.shutDown();
244         } catch (IOException iox) {
245             fail("Could not initialize MyFileConfigedTestPlace with one arg constructor", iox);
246         } finally {
247             // Clean up the tmp config settings
248             restoreConfig();
249 
250             assert cfg != null;
251             Files.deleteIfExists(cfg);
252 
253             if (fos != null) {
254                 try {
255                     fos.close();
256                 } catch (IOException ignore) {
257                 }
258             }
259             Files.deleteIfExists(cfg);
260         }
261     }

460             InputStream config = new ByteArrayInputStream(configKeysData);
461             PlaceTest tp = new PlaceTest(config);
462             assertEquals("TPlaceName", tp.getPlaceName(), "Configured place name");
463             assertEquals("TPROXY", tp.getPrimaryProxy(), "Primary proxy");
464             assertEquals("TPROXY.TNAME.ID.http://localhost:8001/TPlaceName", tp.getKey(), "Key generation uses first key");

472 
473             // Check keys
474             Set<String> keys = tp.getKeys();
475             assertNotNull(keys, "Keys as a set");
476             assertEquals(4, keys.size(), "Size of Keys");
477 
478             // Check namespace
479             for (String k : keys) {
480                 String sl = KeyManipulator.getServiceLocation(k);

482                     Object place = Namespace.lookup(sl);

488             }

492             assertNotNull(proxies, "Proxies as a set");
493             assertEquals(4, proxies.size(), "Size of proxies set");
494             assertTrue(proxies.contains("TPROXY"), "Proxies does not contain original in set");
495             assertTrue(proxies.contains("TP2"), "Proxies does not contain original in set");
496             assertTrue(proxies.contains("TP3"), "Proxies does not contain original in set");

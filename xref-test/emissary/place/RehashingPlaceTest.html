22      private static final byte[] configKeyData = ("TGT_HOST = \"myhost.example.com\"\n" + "TGT_PORT = \"9999\"\n"
23              + "SERVICE_KEY = \"TPROXY.TRANSFORM.TNAME.http://@{TGT_HOST}:@{TGT_PORT}/TPlaceName$5050\"\n" + "SERVICE_DESCRIPTION = \"test place\"\n")
24              .getBytes();
25  
26      @Test
27      void testRehash() throws Exception {
28          byte[] DATA = "abcdefghijklmnopqrstuvwxyz".getBytes();
29          byte[] DOUBLE_DATA = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ".getBytes();
30  
31  
32          InputStream config = new ByteArrayInputStream(configKeyData);
33          RehashPlaceTest tp = new RehashPlaceTest(config);
34          IBaseDataObject payload = new BaseDataObject(DATA, "file1", "FORM1");
35          KffDataObjectHandler kff = tp.getKffHandler();
36          assertNotNull(kff, "Kff handler must be iinitialized in a place " + "that implements the RehashingPlace interface");
37  
38          // Simulate previous activity that would set the length and
39          // hash values to the current data
40          kff.hash(payload);
41          payload.setParameter(SessionParser.ORIG_DOC_SIZE_KEY, payload.dataLength());
42  
43          // Save off the previous hash values
44          String oldhash = KffDataObjectHandler.getHashValue(payload);
45  
46          // Tell the place how to change the data
47          tp.setUseData(DOUBLE_DATA);
48  
49          // Process the payload
50          tp.agentProcessCall(payload);
51  
52          // Check the new ORIG SIZE parameter
53          assertEquals("" + payload.dataLength(), payload.getStringParameter(SessionParser.ORIG_DOC_SIZE_KEY), "Original Size must be reset on rehash");
54  
55          // Check the new hash values
56          String newhash = KffDataObjectHandler.getHashValue(payload);
57  
58          assertNotEquals(oldhash, newhash, "Hash value must change in a rehashing place when the data changes");
59  
60      }
61  
62      private static final class RehashPlaceTest extends ServiceProviderPlace implements RehashingPlace {
63  
64          @Nullable

66

68              super(config);
69          }

72          public void process(IBaseDataObject d) {
73              assertNotNull(d);
74              d.setData(useData);
75          }
76  
77          public KffDataObjectHandler getKffHandler() {
78              return kff;
79          }
80  
81          public void setUseData(byte[] data) {
82              useData = data;
83          }

24  class VersionPlaceTest extends UnitTest {
25      @Nullable
26      private IBaseDataObject payload;
27      @Nullable
28      private VersionPlace place;
29      private Version version;
30      @Nullable
31      private String versionDate;
32      private Path gitRepositoryFile;
33  
34      @Override
35      @BeforeEach
36      public void setUp() throws Exception {
37          payload = DataObjectFactory.getInstance();
38          version = new Version();
39          versionDate = version.getTimestamp().replaceAll("\\D", "");
40          gitRepositoryFile = Paths.get(new ResourceReader().getResource("emissary/util/test.git.properties").toURI());
41      }
42  
43      @Override
44      @AfterEach
45      public void tearDown() throws Exception {
46          super.tearDown();
47          place.shutDown();
48          place = null;
49          payload = null;
50          versionDate = null;
51      }
52  
53      @Test
54      void testAddVersionToPayload() throws ResourceException, IOException {
55          // create the place, using the normal class cfg
56          place = new MyVersionPlace();
57  
58          place.process(payload);
59          assertEquals(payload.getStringParameter("EMISSARY_VERSION"), version.getVersion() + "-" + versionDate,
60                  "added version should contain the date.");
61      }
62  
63      @Test
64      void testAddVersionWithoutDate() throws ResourceException, IOException {
65          // create the place with the test cfg, having INCLUDE_DATE = "false"
66          InputStream is = new ResourceReader().getConfigDataAsStream(this.getClass());
67          place = new MyVersionPlace(is);
68  
69          place.process(payload);
70          assertFalse(payload.getStringParameter("EMISSARY_VERSION").contains(versionDate), "the date should not be added to the version");
71          assertEquals(payload.getStringParameter("EMISSARY_VERSION"), version.getVersion(), "the version should be added");
72      }
73  
74      @Test
75      void testAddVersionHash() throws ResourceException, IOException {
76          // create the place, using the normal class cfg
77          place = new MyVersionPlace();
78  
79          place.process(payload);
80          assertEquals(payload.getStringParameter("EMISSARY_VERSION_HASH").substring(0, 7),
81                  GitRepositoryState.getRepositoryState(gitRepositoryFile).getCommitIdAbbrev(),
82                  "EMISSARY_VERSION_HASH should contain (at least) the abbreviated hash");
83      }
84  
85      class MyVersionPlace extends VersionPlace {
86          MyVersionPlace() throws IOException {
87              super(new ResourceReader().getConfigDataAsStream(VersionPlace.class));
88          }
89  
90          MyVersionPlace(InputStream is) throws IOException {
91              super(is);
92          }
93  
94          @Override
95          GitRepositoryState initGitRepositoryState() {
96              return GitRepositoryState.getRepositoryState(gitRepositoryFile);
97          }
98      }
99  }

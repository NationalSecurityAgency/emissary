57      @Test
58      void testWrite_3args() throws Exception {
59          final byte[] onekbuff = this.onekstring.getBytes();
60          this.channel.write(onekbuff, 0, onekbuff.length);
61          this.channel.commit();
62          assertEquals(this.channel.position(), onekbuff.length, "Position is 1k " + this.channel.position());
63          StringBuilder fiveKStr = new StringBuilder();
64          for (int i = 0; i < 5; i++) {
65              fiveKStr.append(this.onekstring);
66          }
67          final byte[] fiveK = fiveKStr.toString().getBytes();
68          this.channel.write(fiveK, 0, fiveK.length);
69          this.channel.commit();
70          assertEquals(this.channel.position(), onekbuff.length + fiveK.length, "Position is 6k " + this.channel.position());
71          final Path channelPath = this.channel.path;
72          this.channel.close();
73          try (SeekableByteChannel sbc = Files.newByteChannel(channelPath)) {
74              final ByteBuffer buff = ByteBuffer.allocate(1024);
75              for (int i = 0; i < 6; i++) {
76                  buff.clear();
77                  sbc.read(buff);
78                  buff.flip();
79                  final String read = new String(buff.array());
80                  assertEquals(read, this.onekstring, "Read one k back\n" + read + "\n" + this.onekstring);
81              }
82          }
83      }

85      @Test
86      void testWrite_ByteBuffer() throws Exception {
87          final ByteBuffer buff = ByteBuffer.allocateDirect(6 * 1024);
88          for (int i = 0; i < 6; i++) {
89              buff.put(this.onekstring.getBytes());
90          }
91          buff.flip();
92          this.channel.write(buff);
93          assertEquals(this.channel.position(), buff.capacity(), "6k written " + this.channel.position() + " " + buff.capacity());
94      }

45      /**
46       * Test of freeChannel method, of class JournaledChannelPool.
47       */
48      @Test
49      void testGetAndFree() throws IOException, InterruptedException {
50          final SeekableByteChannel out = this.instance.getFree();
51          final int created = this.instance.getCreatedCount();
52          assertTrue(created > 0, "Should have recorded creation of at least 1 object");
53          final int free = this.instance.getFreeSize();
54          // out is a wrapper, closing it will delegate to freeChannel
55          out.close();
56          assertEquals((free + 1), this.instance.getFreeSize(), "Free should be one more than " + free);
57      }

59      /**
60       * Test of getJournalEntries method, of class JournaledChannelPool.
61       */
62      @Test
63      void testGetJournalEntries() throws IOException, InterruptedException {
64          final String onetext = "one line of text";
65          final String twotext = "two lines of text\nthe second line";
66          try (KeyedOutput k1 = this.instance.getFree(); KeyedOutput k2 = this.instance.getFree()) {
67              writeText(k1, onetext);
68              writeText(k2, twotext);
69          }
70          final ArrayList<JournalEntry> result = new ArrayList<>();
71          for (final Path journalPath : JournalReader.getJournalPaths(this.directory)) {
72              try (JournalReader jr = new JournalReader(journalPath)) {
73                  Journal j = jr.getJournal();
74                  result.addAll(j.getEntries());
75              }
76          }
77          assertEquals(4, result.size(), "Expected 4 Journal Entries " + result.size());
78          int jrnltot = 0;
79          for (final JournalEntry e : result) {
80              jrnltot += (int) e.getOffset();
81          }
82          final int totlen = onetext.length() + twotext.length();
83          assertEquals(totlen, jrnltot, "Total length should be " + totlen);
84      }
85  
86      private static void writeText(final KeyedOutput ko, final String text) throws IOException {
87          try (BufferedWriter bw = new BufferedWriter(Channels.newWriter(ko, StandardCharsets.UTF_8))) {
88              bw.write(text);
89              bw.flush();
90              ko.commit();
91          }
92      }

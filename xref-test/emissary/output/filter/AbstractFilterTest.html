35      AbstractFilter getDenylistFilter(final List<String> denylist) {
36          Configurator config = new ServiceConfigGuide();
37          config.addEntry("OUTPUT_SPEC_FOO", "SPEC");
38          config.addEntry("OUTPUT_TYPE", "TYPE");
39          denylist.forEach(entry -> config.addEntry("DENYLIST", entry));
40          AbstractFilter f = getAbstractFilterInstance();
41          f.initialize(new ServiceConfigGuide(), "FOO", config);
42          return f;
43      }

69          @Test
70          void testDefaultDenylistFiletypeFormat() {
71              String pattern = f.getDenylistFiletypeFormat();
72              assertEquals("^[a-zA-Z0-9_\\-]+$", pattern);
73  
74              assertFalse(f.matchesDenylistFiletypeFormatPattern(""),
75                      String.format("Unexpected match [empty string] for Pattern %s", pattern));
76  
77              String formatString = "Fi1e%sTyp3";
78  
79              String noMatchChars = " !@#$%^&*()+=\\/.,";
80              for (int i = 0; i < noMatchChars.length(); i++) {
81                  char noMatchChar = noMatchChars.charAt(i);
82                  String noMatch = String.format(formatString, noMatchChar);
83                  assertFalse(f.matchesDenylistFiletypeFormatPattern(noMatch),
84                          String.format("Unexpected match %s for Pattern %s", noMatch, pattern));
85              }
86  
87              String nullInsert = String.format(formatString, "");
88              assertTrue(f.matchesDenylistFiletypeFormatPattern(nullInsert),
89                      String.format("Expected match %s for Pattern %s", nullInsert, pattern));
90  
91              String matchChars = "-_";
92              for (int i = 0; i < matchChars.length(); i++) {
93                  char matchChar = matchChars.charAt(i);
94                  String match = String.format(formatString, matchChar);
95                  assertTrue(f.matchesDenylistFiletypeFormatPattern(match),
96                          String.format("Expected match %s for Pattern %s", match, pattern));
97              }
98          }

138     @Test
139     void testIncorrectConfigs() {
140         List<String> invalidEntries = Arrays.asList(
141                 "type*.view", "*type.view", "ty*pe.view", "*.view", ".view",
142                 "type*.*view", "*type.*view", "ty*pe.*view", "*.*view", ".*view", "*view",
143                 "type*.vi*ew", "*type.vi*ew", "ty*pe.vi*ew", "*.vi*ew", ".vi*ew", "vi*ew",
144                 "type*.view*", "*type.view*", "ty*pe.view*", "*.view*", ".view*",
145                 "type.", "*", "type.*", ".", "type.view.", ".type.view", ".type.view.",
146                 "type..view", "type.view.view.view", "type.view*.view*", "type.view**", "view**");
147         for (String entry : invalidEntries) {
148             final Configurator config = new ServiceConfigGuide();
149             config.addEntry("DENYLIST", entry);
150             AbstractFilter f = getAbstractFilterInstance();
151             EmissaryRuntimeException e = assertThrows(
152                     EmissaryRuntimeException.class,
153                     () -> f.initialize(new ServiceConfigGuide(), "FOO", config));
154             assertTrue(e.getMessage().contains(String.format("Invalid filter configuration: `DENYLIST = \"%s\"`", entry)));
155         }

160         AbstractFilter f = getDenylistFilter(Arrays.asList(
161                 "JSON_ML", "JSON_LANG_*",

173             assertTrue(f.denyListContains(filetype1, denied), denied + " should be in denyList");
174         }

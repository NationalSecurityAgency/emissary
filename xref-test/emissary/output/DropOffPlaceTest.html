73          payload.setFileType("FTYPE");
74          payload.setFilename("/this/is/a/testfile");
75          final List<IBaseDataObject> payloadList = new ArrayList<>();
76          payloadList.add(payload);
77          final List<IBaseDataObject> val = this.place.agentProcessHeavyDuty(payloadList);
78          assertEquals(1, payloadList.size(), "All payloads still on list");
79          assertEquals(0, val.size(), "Nothing returned from drop off");
80          assertEquals(0, payloadList.get(0).currentFormSize(), "All current forms removed");

84      @Test
85      void testOutputMessageContainsPayloadSize() throws Exception {
86          final LoggerContext lc = (LoggerContext) LoggerFactory.getILoggerFactory();
87          final Logger rootLogger = lc.getLogger(Logger.ROOT_LOGGER_NAME);
88  
89          ListAppender<ILoggingEvent> appender = new ListAppender<>();
90          byte[] content = "This is the data".getBytes();
91          String expected = "payload size: " + content.length + " bytes";
92  
93          try {
94  
95              appender.start();
96              rootLogger.addAppender(appender);
97  
98              final IBaseDataObject payload = DataObjectFactory.getInstance();
99              payload.setData(content);
100 
101             payload.setCurrentForm("FOO");
102             payload.setFileType("FTYPE");
103             payload.setFilename("/this/is/a/testfile");
104             final List<IBaseDataObject> payloadList = new ArrayList<>();
105             payloadList.add(payload);
106 
107             assertDoesNotThrow(() -> {
108                 assertNotNull(this.place);
109                 return this.place.agentProcessHeavyDuty(payloadList);
110             });
111             assertTrue(appender.list.stream().anyMatch(i -> i.getFormattedMessage().endsWith(expected)));
112         } finally {
113             rootLogger.detachAppender(appender);
114             appender.stop();
115         }
116     }

154     public static void cleanupDirectoryRecursively(Path path) throws IOException {
155         Files.walkFileTree(path, new SimpleFileVisitor<>() {
156             @Override
157             public FileVisitResult visitFile(Path file, BasicFileAttributes attrs) throws IOException {
158                 Files.delete(file);
159                 return FileVisitResult.CONTINUE;
160             }
161 
162             @Override
163             public FileVisitResult postVisitDirectory(Path dir, IOException exc) throws IOException {
164                 Files.delete(dir);
165                 return FileVisitResult.CONTINUE;
166             }
167         });

406         child.addExtractedRecords(childRecords);

411 
412         final List<IBaseDataObject> gchildRecords = new ArrayList<>();
413         for (int i = 1; i < 5; i++) {
414             gchildRecords.add(DataObjectFactory.getInstance(new byte[0], "item-att-2-att-1-att-" + i, "GCHILD_RECORD_FORM"));

417 
418         family.add(parent);
419         family.add(child);
420         family.add(child2);

422

424 
425         // Child 1 subtree has a FOO param

428                 "Propagation of configured parent type must use closest available entry on child");

435         assertEquals("FOO", gchildRecords.get(0)
436                 .getStringParameter("PARENT_FOO"), "Propagation of configured parent type must use closest available entry on grandchild records");

441         // populate the tld with all the event date options (note: year of 2016)
442         IBaseDataObject tld = new BaseDataObject();
443         tld.setParameter(EVENT_DATE, "2016-02-13 23:13:03");
444         tld.setParameter(FILE_DATE, "2016-01-05 17:45:53");
445 
446         // populate the child with all the event date options (note: year of 2015)
447         IBaseDataObject d = new BaseDataObject();
448         d.setParameter(EVENT_DATE, "2015-02-13 23:13:03");
449         d.setParameter(FILE_DATE, "2015-01-05 17:45:53");
450 
451         Date start = new Date(); // use this for relative comparison
452 
453         // hit on child EventDate
454         assertEquals("2015-02-13 23:13:03", TimeUtil.getDateAsISO8601(this.util.getEventDate(d, tld).toInstant()));
455 
456         // removing child EventDate should hit on child FILE_DATE
457         d.deleteParameter(EVENT_DATE);
458         assertEquals("2015-01-05 17:45:53", TimeUtil.getDateAsISO8601(this.util.getEventDate(d, tld).toInstant()));
459 
460         // removing child FILE_DATE should hit on tld EventDate
461         d.deleteParameter(FILE_DATE);
462         assertEquals("2016-02-13 23:13:03", TimeUtil.getDateAsISO8601(this.util.getEventDate(d, tld).toInstant()));
463 
464         // removing tld EventDate should hit on tld FILE_DATE
465         tld.deleteParameter(EVENT_DATE);
466         assertEquals("2016-01-05 17:45:53", TimeUtil.getDateAsISO8601(this.util.getEventDate(d, tld).toInstant()));
467 
468         // removing tld FILE_DATE should default to now as configured by default
469         tld.deleteParameter(FILE_DATE);
470         assertNotNull(this.util.getEventDate(d, tld));
471         assertNotEquals(-1, this.util.getEventDate(d, tld).compareTo(start));
472 
473         // changing the configuration to not default to now should return null
474         Configurator cfg = new ServiceConfigGuide();
475         final List<String> dates = new ArrayList<>();
476         dates.add(EVENT_DATE);
477         dates.add(FILE_DATE);
478         cfg.addEntries("DATE_PARAMETER", dates);
479         cfg.addEntry("DEFAULT_EVENT_DATE_TO_NOW", "false");
480         this.util = new DropOffUtil(cfg);
481 
482         assertNull(this.util.getEventDate(d, tld));
483     }

436         final List<IBaseDataObject> family = new ArrayList<>();
437 
438         final IBaseDataObject parent = DataObjectFactory.getInstance(new byte[0], "item", "PARENT_FORM");
439         parent.setFileType("PARENT_FTYPE");

443         child.putParameter("FOO", "CHILD_FOO");
444 
445         final List<IBaseDataObject> childRecords = new ArrayList<>();
446         for (int i = 1; i < 5; i++) {
447             childRecords.add(DataObjectFactory.getInstance(new byte[0], "item-att-1-att-" + i, "RECORD_FORM"));
448         }
449         child.addExtractedRecords(childRecords);
450 
451         final IBaseDataObject child2 = DataObjectFactory.getInstance(new byte[0], "item-att-2", "CHILD2_FORM");
452         // child2 does not have a FOO param
453         final IBaseDataObject grandchild = DataObjectFactory.getInstance(new byte[0], "item-att-2-att-1", "GRANDCHILD_FORM");
454 
455         final List<IBaseDataObject> gchildRecords = new ArrayList<>();
456         for (int i = 1; i < 5; i++) {
457             gchildRecords.add(DataObjectFactory.getInstance(new byte[0], "item-att-2-att-1-att-" + i, "GCHILD_RECORD_FORM"));
458         }
459         grandchild.addExtractedRecords(gchildRecords);
460 
461         family.add(parent);
462         family.add(child);
463         family.add(child2);
464         family.add(grandchild);
465 
466         this.util.processMetadata(family);
467 
468         // Child 1 subtree has a FOO param
469         assertEquals("FOO",
470                 child.getStringParameter("PARENT_FOO"),
471                 "Propagation of configured parent type must use closest available entry on child");
472         assertEquals("FOO",

477                 "Propagation of configured parent type must use closest available entry on child");

483                 .getParameterAsString("PARENT_FOO"), "Propagation of configured parent type must use closest available entry on child records");

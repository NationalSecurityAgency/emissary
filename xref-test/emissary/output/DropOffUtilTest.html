406 
407         final List<IBaseDataObject> gchildRecords = new ArrayList<>();
408         for (int i = 1; i < 5; i++) {
409             gchildRecords.add(DataObjectFactory.getInstance(new byte[0], "item-att-2-att-1-att-" + i, "GCHILD_RECORD_FORM"));
410         }
411         grandchild.addExtractedRecords(gchildRecords);
412 
413         family.add(parent);
414         family.add(child);
415         family.add(child2);
416         family.add(grandchild);
417

419 
420         // Child 1 subtree has a FOO param

423                 "Propagation of configured parent type must use closest available entry on child");

430         assertEquals("FOO", gchildRecords.get(0)
431                 .getStringParameter("PARENT_FOO"), "Propagation of configured parent type must use closest available entry on grandchild records");

435     void testGetEventDate() {
436         // populate the tld with all the event date options (note: year of 2016)
437         IBaseDataObject tld = new BaseDataObject();
438         tld.setParameter(EVENT_DATE, "2016-02-13 23:13:03");
439         tld.setParameter(FILE_DATE, "2016-01-05 17:45:53");
440 
441         // populate the child with all the event date options (note: year of 2015)
442         IBaseDataObject d = new BaseDataObject();
443         d.setParameter(EVENT_DATE, "2015-02-13 23:13:03");
444         d.setParameter(FILE_DATE, "2015-01-05 17:45:53");
445 
446         Date start = new Date(); // use this for relative comparison
447 
448         // hit on child EventDate
449         assertEquals("2015-02-13 23:13:03", TimeUtil.getDateAsISO8601(this.util.getEventDate(d, tld).toInstant()));
450 
451         // removing child EventDate should hit on child FILE_DATE
452         d.deleteParameter(EVENT_DATE);
453         assertEquals("2015-01-05 17:45:53", TimeUtil.getDateAsISO8601(this.util.getEventDate(d, tld).toInstant()));
454 
455         // removing child FILE_DATE should hit on tld EventDate
456         d.deleteParameter(FILE_DATE);
457         assertEquals("2016-02-13 23:13:03", TimeUtil.getDateAsISO8601(this.util.getEventDate(d, tld).toInstant()));
458 
459         // removing tld EventDate should hit on tld FILE_DATE
460         tld.deleteParameter(EVENT_DATE);
461         assertEquals("2016-01-05 17:45:53", TimeUtil.getDateAsISO8601(this.util.getEventDate(d, tld).toInstant()));
462 
463         // removing tld FILE_DATE should default to now as configured by default
464         tld.deleteParameter(FILE_DATE);
465         assertNotNull(this.util.getEventDate(d, tld));
466         assertNotEquals(-1, this.util.getEventDate(d, tld).compareTo(start));
467 
468         // changing the configuration to not default to now should return null
469         Configurator cfg = new ServiceConfigGuide();
470         final List<String> dates = new ArrayList<>();
471         dates.add(EVENT_DATE);
472         dates.add(FILE_DATE);
473         cfg.addEntries("DATE_PARAMETER", dates);
474         cfg.addEntry("DEFAULT_EVENT_DATE_TO_NOW", "false");
475         this.util = new DropOffUtil(cfg);
476 
477         assertNull(this.util.getEventDate(d, tld));
478     }

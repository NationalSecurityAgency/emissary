194         byte[] no = new byte[] {'a', 'A', 'z', 'Z', '0', '9', '~'};
195         byte[] yes = new byte[] {' ', '\t', '\n', '\r', (byte) 12, (byte) (0xff & 254)};
196         for (int pos = 0; pos < yes.length; pos++) {
197             assertTrue(ByteUtil.isControlOrBlankSpace(yes, pos), "Should be control at pos " + pos);
198         }
199         for (int pos = 0; pos < no.length; pos++) {
200             assertFalse(ByteUtil.isControlOrBlankSpace(no, pos), "Should not be control at pos " + pos);
201         }

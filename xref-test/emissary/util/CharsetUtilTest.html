25      @Test
26      void testUTF8() {
27          assertTrue(CharsetUtil.isUTF8("This is a test."), "Ascii is utf8");
28          assertTrue(CharsetUtil.isUTF8("!@#$%^&*(F)=+-_[]{}\\|'\";:,.></?`~"), "Punctuation is utf8");
29          assertTrue(CharsetUtil.isUTF8("0123456789 9876543210"), "Numbers are utf8");
30  
31          assertTrue(CharsetUtil.isUTF8("This is a bytes array test. 123 #$%".getBytes()), "Ascii bytes are utf8");
32  
33          for (int i = 0; i < S.length; i++) {
34              assertTrue(CharsetUtil.isUTF8(S[i]), "Foreign strings from java, entry " + i + " of " + S.length + " == " + S[i] + "/" + S[i].length());
35              assertTrue(CharsetUtil.isUTF8(S[i].getBytes()),
36                      "Foreign bytes from java, entry " + i + " of " + S.length + " == " + S[i] + "/" + S[i].getBytes().length);
37  
38          }
39      }
40  
41      @Test
42      void testMultibyte() {
43          for (int i = 0; i < S.length; i++) {
44              assertTrue(CharsetUtil.hasMultibyte(S[i]),
45                      "Foreign strings have multibyte, entry " + i + " of " + S.length + " == " + S[i] + "/" + S[i].length());
46          }
47  
48          assertFalse(CharsetUtil.hasMultibyte(null), "Null does not have multibyte");
49  
50          assertFalse(CharsetUtil.hasMultibyte(""), "Empty does not have multibyte");
51  
52          assertFalse(CharsetUtil.hasMultibyte("1234 abcde !@#$%"), "Ascii does not have multibyte");
53      }
54  
55      @Test
56      void testNotUTF8() {
57          byte[] b = new byte[] {(byte) 192, (byte) 192, (byte) 224, (byte) 224, (byte) 192, (byte) 0, (byte) 192, (byte) 224};
58          assertFalse(CharsetUtil.isUTF8(b), "Bad utf-8 stream is not utf-8");
59      }
60  
61      @Test
62      void testIsAscii() {
63          assertTrue(CharsetUtil.isAscii("abcdefg 1234567"), "Ascii is easy");
64          assertFalse(CharsetUtil.isAscii("Шарифа"), "This is not ascii");
65      }

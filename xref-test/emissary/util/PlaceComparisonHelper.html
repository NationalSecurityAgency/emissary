18  /**
19   * Helper class to use during development of a major refactoring or replacement of a Place.
20   */
21  public class PlaceComparisonHelper {
22  
23      private PlaceComparisonHelper() {}
24  
25      // Config key to use in Place configuration to define another place to compare
26      public static final String CFG_PLACE_TO_COMPARE = "PLACE_TO_COMPARE";
27  
28      /**
29       * Given a config which contains a 'PLACE_TO_COMPARE' directive, instantiate the referenced place with its usual
30       * configuration
31       * 
32       * @param configG from an existing place which is scanned for PLACE_TO_COMPARE, which should reference a place that the
33       *        developer wants to compare against
34       * @return the instantiated place if configured
35       * @throws ReflectiveOperationException if there is a problem.
36       */
37      @Nullable
38      public static ServiceProviderPlace getPlaceToCompare(final Configurator configG) throws ReflectiveOperationException {
39          if (configG == null) {
40              return null;
41          }
42  
43          // Find the config line, which could be null if not defined
44          final String compareToPlaceCfg = StringUtils.trimToNull(configG.findStringEntry(CFG_PLACE_TO_COMPARE));
45          if (StringUtils.isNotBlank(compareToPlaceCfg)) {
46              final Class<? extends ServiceProviderPlace> clazz = Class
47                      .forName(compareToPlaceCfg).asSubclass(ServiceProviderPlace.class);
48              return clazz.getDeclaredConstructor(String.class)
49                      .newInstance(compareToPlaceCfg + ".cfg");
50          }
51  
52          return null;
53      }
54  
55      /**
56       * Used to compare a 'new' place with another, usually during development aimed at replacing the 'old' place.
57       * 
58       * @param newResults an empty list to add results into from running the newPlace
59       * @param ibdoForNewPlace the actual BDO being passed in for comparison
60       * @param newPlace the new place we're comparing 'from'
61       * @param newMethodName to use when comparing (e.g. processHeavyDuty)
62       * @param oldPlace to compare against
63       * @param oldMethodName to use when comparing (e.g. processHeavyDuty)
64       * @param options {@link DiffCheckConfiguration} to configure diffing options
65       * @return a readable string of differences to log or use elsewhere.
66       * @throws ReflectiveOperationException if there is a problem.
67       */
68      @SuppressWarnings("unchecked")
69      public static String compareToPlace(final List<IBaseDataObject> newResults, final IBaseDataObject ibdoForNewPlace,
70              final ServiceProviderPlace newPlace, final String newMethodName, final ServiceProviderPlace oldPlace, final String oldMethodName,
71              final DiffCheckConfiguration options) throws ReflectiveOperationException {
72  
73          Validate.notNull(newResults, "Required: newResults not null");
74          Validate.notNull(ibdoForNewPlace, "Required: ibdoForNewPlace not null");
75          Validate.notNull(newPlace, "Required: newPlace not null");
76          Validate.notNull(newMethodName, "newMethodName: newResults not null");
77          Validate.notNull(oldPlace, "Required: oldPlace not null");
78          Validate.notNull(oldMethodName, "Required: oldMethodName not null");
79          Validate.notNull(options, "Required: options not null");
80  
81          // Generate an identifier from the simple class names, e.g. Comparison[ColorPlace==ColourPlace]
82          final String oldPlaceName = oldPlace.getClass().getSimpleName();
83          final String newPlaceName = newPlace.getClass().getSimpleName();
84          final String identifier = String.format("Comparison[%s==%s]", oldPlaceName, newPlaceName);
85  
86          // Get the method to run (e.g. processHeavyDuty)
87          final Method oldProcess = oldPlace.getClass().getDeclaredMethod(oldMethodName, IBaseDataObject.class);
88          final Method newProcess = newPlace.getClass().getDeclaredMethod(newMethodName, IBaseDataObject.class);
89  
90          // Clone the data before running old or new methods
91          final IBaseDataObject ibdoForOldPlace = IBaseDataObjectHelper.clone(ibdoForNewPlace, true);
92  
93          // Actually run the places to get results
94          final List<IBaseDataObject> oldResults = (List<IBaseDataObject>) oldProcess.invoke(oldPlace, ibdoForOldPlace);
95          newResults.addAll((List<IBaseDataObject>) newProcess.invoke(newPlace, ibdoForNewPlace));
96  
97          // Now generate the 'diff' for the results
98          return checkDifferences(ibdoForOldPlace, ibdoForNewPlace, oldResults, newResults, identifier, options);
99      }
100 
101     /**
102      * Given two BDOs and results from two processing place runs, compare them and log any differences.
103      * 
104      * @param ibdoForOldPlace likely a cloned object of the 'new' place object
105      * @param ibdoForNewPlace the 'main' BDO that was originally passed in from upstream
106      * @param oldResults from the 'old' run with the 'ibdoForOldPlace' object
107      * @param newResults from the 'new' run with the 'ibdoForNewPlace' object
108      * @param identifier to highlight any differences in logs
109      * @param options {@link DiffCheckConfiguration} to configure diffing options
110      * @return the string of differences, or null if there aren't any
111      */
112     @Nullable
113     public static String checkDifferences(final IBaseDataObject ibdoForOldPlace, final IBaseDataObject ibdoForNewPlace,
114             final List<IBaseDataObject> oldResults, final List<IBaseDataObject> newResults, final String identifier,
115             final DiffCheckConfiguration options) {
116         Validate.notNull(ibdoForOldPlace, "Required: ibdoForOldPlace not null");
117         Validate.notNull(ibdoForNewPlace, "Required: ibdoForNewPlace not null");
118         Validate.notNull(oldResults, "Required: oldResults not null");
119         Validate.notNull(newResults, "Required: newResults not null");
120         Validate.notNull(identifier, "Required: identifier not null");
121         Validate.notNull(options, "Required: options not null");
122 
123         final List<String> parentDifferences = new ArrayList<>();
124         final List<String> childDifferences = new ArrayList<>();
125 
126         IBaseDataObjectDiffHelper.diff(ibdoForOldPlace, ibdoForNewPlace, parentDifferences, options);
127         IBaseDataObjectDiffHelper.diff(oldResults, newResults, identifier, childDifferences, options);
128 
129         if (!parentDifferences.isEmpty() || !childDifferences.isEmpty()) {
130             final StringBuilder sb = new StringBuilder();
131             for (int i = 0; i < parentDifferences.size(); i++) {
132                 if (i != 0) {
133                     sb.append(StringUtils.LF);
134                 }
135                 sb.append(identifier).append(": PDiff: ");
136                 sb.append(parentDifferences.get(i));
137             }
138             if (!parentDifferences.isEmpty() && !childDifferences.isEmpty()) {
139                 sb.append(StringUtils.LF);
140             }
141             for (int i = 0; i < childDifferences.size(); i++) {
142                 if (i != 0) {
143                     sb.append(StringUtils.LF);
144                 }
145                 sb.append(identifier).append(": CDiff: ");
146                 sb.append(childDifferences.get(i));
147             }
148 
149             return sb.toString();
150         }
151 
152         return null;
153     }

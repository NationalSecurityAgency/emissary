1   package emissary.util;

3   import emissary.test.core.junit5.UnitTest;
4   
5   import org.junit.jupiter.api.Test;
6   
7   import static org.junit.jupiter.api.Assertions.assertEquals;
8   import static org.junit.jupiter.api.Assertions.fail;
9   
10  /**
11   * Unit tests for {@link ClassLookupCache}.
12   */
13  class ClassLookupCacheTest extends UnitTest {
14  
15      // We just need some class name to look up and its corresponding
16      // class object.
17      private static final Class<?> TEST_CLASS_OBJECT = ClassLookupCacheTest.class;
18      private static final String TEST_CLASS_NAME = TEST_CLASS_OBJECT.getName();
19  
20      /**
21       * Call {@link ClassLookupCache#lookup(String)} and ensure that it returns the given class.
22       *
23       * @param className The class name to look up.
24       * @param expectedClazz The expected class object.
25       */
26      private static void assertLookup(final String className, final Class<?> expectedClazz) {
27          try {
28              final Class<?> clazz = ClassLookupCache.lookup(className);
29              assertEquals(expectedClazz, clazz, "lookup of " + className + " should return known class object");
30          } catch (ClassNotFoundException e) {
31              fail("could not find the class " + className, e);
32          }
33      }
34  
35      @Test
36      void testThatLookupsWork() {
37          // Just make sure that lookup(name) is working at all.
38          assertLookup(TEST_CLASS_NAME, TEST_CLASS_OBJECT);
39      }
40  
41      @Test
42      void testThatRecentLookupsAreBeingCached() {
43          // Do a regular lookup.
44          assertLookup(TEST_CLASS_NAME, TEST_CLASS_OBJECT);
45  
46          // Now make sure the lookup got cached. Since we're on the
47          // same thread and we know the same class was just looked up
48          // successfully and very little time has elapsed, it's a
49          // reasonable expectation that the class is still cached.
50          assertEquals(TEST_CLASS_OBJECT,
51                  ClassLookupCache.get(TEST_CLASS_NAME),
52                  "the recent lookup of " + TEST_CLASS_NAME + " should cache the class object");
53      }
54  }

149         assertTrue(Executrix.writeDataToFile("aaa".getBytes(), 0, 3, TMPDIR + "/foo.dat", false), "File written");
150         byte[] data = Executrix.readFile(TMPDIR + "/foo.dat");
151         assertNotNull(data, "Data must be read");
152         assertEquals(3, data.length, "Data must all be read");
153 
154         // append
155         assertTrue(Executrix.writeDataToFile("aaa".getBytes(), 0, 3, TMPDIR + "/foo.dat", true), "File written");
156         data = Executrix.readFile(TMPDIR + "/foo.dat");
157         assertNotNull(data, "Data must be read");
158         assertEquals(6, data.length, "Data must all be read");
159 
160         data = Executrix.readFile(TMPDIR + "/foo.dat", 3);
161         assertNotNull(data, "Data must be read");
162         assertEquals(3, data.length, "Data must be read up to limit");
163 
164         data = Executrix.readFile(TMPDIR + "/foo.dat", 3000);
165         assertNotNull(data, "Data must be read");
166         assertEquals(6, data.length, "Data must be read up to actual size");
167 
168         assertFalse(Executrix.writeDataToFile(null, TMPDIR + "/foo.dat"), "Should not write null data");
169         assertFalse(Executrix.writeDataToFile(null, TMPDIR + "/foo.dat", false), "Should not write null data");
170         assertFalse(Executrix.writeDataToFile(null, 0, 3, TMPDIR + "/foo.dat", false), "Should not write null data");
171 
172         assertTrue(Executrix.writeDataToFile("aaa".getBytes(), TMPDIR + "/foo.dat", false), "Overwrite longer file should truncate previous data");
173         data = Executrix.readFile(TMPDIR + "/foo.dat");
174         assertNotNull(data, "Data must be read");
175         assertEquals(3, data.length, "Data must be read up to actual size");
176 
177         data = Executrix.readDataFromFile(TMPDIR + "/foo.dat");
178         assertNotNull(data, "Data must be read");
179         assertEquals(3, data.length, "Data must be read up to actual size");
180 
181         data = Executrix.readDataFromFile(TMPDIR + "/filedoesnotexist.dat");
182         assertNull(data, "Read non existent does not throw");
183 
184         assertFalse(Executrix.writeDataToFile("aaa".getBytes(), null), "Write to null path");
185         assertFalse(Executrix.writeDataToFile("aaa".getBytes(), null, false), "Write to null path");
186         assertFalse(Executrix.writeDataToFile("aaa".getBytes(), 0, 3, null, false), "Write to null path");
187 
188         RandomAccessFile raf = new RandomAccessFile(TMPDIR + "/foo.dat", "rw");
189         data = Executrix.readDataFromFile(raf);
190         assertNotNull(data, "Data must be read from raf");
191         assertEquals(3, data.length, "Data must all be read from raf");
192         raf.close();
193 
194         raf = new RandomAccessFile(TMPDIR + "/foo.dat", "rw");
195         data = Executrix.readDataFromFile(raf, 2, 1);
196         assertNotNull(data, "Data must be read from raf");
197         assertEquals(1, data.length, "Data must all be read from raf");
198 
199         raf.seek(0);
200         data = Executrix.readDataFromFile(raf, 100, 200);
201         assertNull(data, "Data requested out of bounds");
202 
203         raf.seek(0);
204         data = Executrix.readDataFromFile(raf, 0, 100);
205         assertNotNull(data, "Available data read from raf");
206         assertEquals(3, data.length, "All available data read from raf");
207 
208         raf.seek(0);
209         final FileChannel channel = raf.getChannel();
210         data = Executrix.readDataFromChannel(channel);
211         assertNotNull(data, "Available data read from channel");
212         assertEquals(3, data.length, "All available data read from channel");
213 
214         data = Executrix.readDataFromChannel(channel, 0, 3);
215         assertNotNull(data, "Available data read from channel");
216         assertEquals(3, data.length, "All available data read from channel");
217 
218         data = Executrix.readDataFromChannel(channel, 0, 1);
219         assertNotNull(data, "Available data read from channel");
220         assertEquals(1, data.length, "All limit data read from channel");
221 
222         data = Executrix.readDataFromChannel(channel, 0, 100);
223         assertNotNull(data, "Available data read from channel");
224         assertEquals(3, data.length, "All limit data read from channel");
225 
226         data = Executrix.readDataFromChannel(channel, 50, 100);
227         assertNull(data, "Out of bounds data on channel returned non-null");
228 
229         channel.close();
230         raf.close();
231 
232         final File f = new File(TMPDIR + "/foo.dat");
233         Files.deleteIfExists(f.toPath());
234     }

589     private static void readAndNuke(final String name) throws IOException {

39       * Test the date string against the expected output
40       *
41       * @param date the string representation of a date
42       * @param expected the expected parsed and formatted date
43       * @param msg the error message to display if the test fails
44       */
45      private static void test(@Nullable String date, long expected, String msg) {
46          ZonedDateTime unknownParse = FlexibleDateTimeParser.parse(date);
47          assertEquals(expected, unknownParse == null ? 0L : unknownParse.toEpochSecond(), "Error on: " + msg);
48      }

50      /**
51       * Test the date string against the expected output
52       *
53       * @param date the string representation of a date
54       * @param expected the expected parsed and formatted date
55       * @param formatter run a date format manually and compare the output to the unknown parser
56       */
57      private static void test(String date, long expected, DateTimeFormatter formatter) {
58          long unknownParse = FlexibleDateTimeParser.parse(date).toEpochSecond();
59          assertEquals(expected, unknownParse, "Flexible parse failed for " + formatter);
60  
61          long knownParse = FlexibleDateTimeParser.parse(date, formatter).toEpochSecond();
62          assertEquals(expected, knownParse, "Manual parse failed for " + formatter);
63  
64          // this is as close as I can get to testing the expected date format
65          assertEquals(unknownParse, knownParse, "Parsed date/times are not the same");
66      }

83       * 
84       * @param dateString the string representation of a date

86       */
87      private static void testAllOffsets(String dateString, long expected) {
88          String[] offsets = {"EST-0500 -0500", // zZZ
89                  "EST EST-0500", // zzZ
90                  "EST-0500 EST", // zZz
91                  "EST-0500 -05", // zZX
92                  "-0500 EST-0500", // ZzZ
93                  "-0500 -05", // ZX
94                  "-05", // X
95                  "-05 EST", // Xz
96                  "-05 -0500", // XZ
97                  "-05 (EST-0500)"}; // XzZ
98  
99          for (String offset : offsets) {
100             String dateAndOffset = dateString + " " + offset;
101             testExtensiveParsing(dateAndOffset, expected, "Did not parse this string correctly: " + dateAndOffset);
102         }

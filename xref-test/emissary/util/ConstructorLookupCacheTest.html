5   import jakarta.annotation.Nullable;
6   import org.junit.jupiter.api.Test;

8   import java.lang.reflect.Constructor;
9   import java.util.List;

11  import static org.junit.jupiter.api.Assertions.assertEquals;

13  /**
14   * Unit tests for {@link ConstructorLookupCache}.
15   */
16  class ConstructorLookupCacheTest extends UnitTest {
17  
18      // We need a known class and constructors to look up.
19      private static final class ExampleClass {
20  
21          @SuppressWarnings("unused")
22          public ExampleClass() {}
23  
24          @SuppressWarnings("unused")
25          public ExampleClass(Integer arg1) {}
26  
27          @SuppressWarnings("unused")
28          public ExampleClass(String arg1, String arg2, List<String> arg3) {}
29  
30          @SuppressWarnings("unused")
31          public ExampleClass(int arg1, double arg2, boolean arg3, String arg4) {}
32  
33          /**
34           * @return A constructor for this class that accepts the given arguments, or {@code null} if none.
35           */
36          @Nullable
37          public static Constructor<ExampleClass> getConstructor(Class<?>... args) {
38              try {
39                  return ExampleClass.class.getConstructor(args);
40              } catch (NoSuchMethodException e) {
41                  return null;
42              }
43          }
44      }
45  
46      private static final Class<ExampleClass> EXAMPLE_CLASS = ExampleClass.class;
47  
48      private static final Class<?>[] ARGS_NONE = new Class<?>[] {};
49      private static final Constructor<ExampleClass> EXAMPLE_CONSTRUCTOR_NONE = ExampleClass.getConstructor(ARGS_NONE);
50  
51      private static final Class<?>[] ARGS_INT = new Class<?>[] {Integer.class};
52      private static final Constructor<ExampleClass> EXAMPLE_CONSTRUCTOR_INT = ExampleClass.getConstructor(ARGS_INT);
53  
54      private static final Class<?>[] ARGS_STRINGS = new Class<?>[] {String.class, String.class, List.class};
55      private static final Constructor<ExampleClass> EXAMPLE_CONSTRUCTOR_STRINGS = ExampleClass.getConstructor(ARGS_STRINGS);
56  
57      private static final Class<?>[] ARGS_PRIM = new Class<?>[] {Integer.TYPE, Double.TYPE, Boolean.TYPE, String.class};
58      private static final Constructor<ExampleClass> EXAMPLE_CONSTRUCTOR_PRIM = ExampleClass.getConstructor(ARGS_PRIM);
59  
60      /**
61       * Call {@link ConstructorLookupCache#lookup} and ensure that it returns the given constructor.
62       *
63       * @param clazz The class of interest.
64       * @param args The constructor arguments you want to use.
65       * @param expectedConstructor The expected constructor object.
66       */
67      private static void assertLookup(final Class<?> clazz, final Class<?>[] args, final Constructor<?> expectedConstructor) {
68          final Constructor<?> constructor = ConstructorLookupCache.lookup(clazz, args);
69          assertEquals(expectedConstructor, constructor, "lookup of " + clazz.getName() + " constructor should return a known constructor object");
70      }
71  
72      @Test
73      void testThatNoArgLookupsWork() {
74          assertLookup(EXAMPLE_CLASS, ARGS_NONE, EXAMPLE_CONSTRUCTOR_NONE);
75      }
76  
77      @Test
78      void testThatSingleArgLookupsWork() {
79          assertLookup(EXAMPLE_CLASS, ARGS_INT, EXAMPLE_CONSTRUCTOR_INT);
80      }
81  
82      @Test
83      void testThatMultiArgLookupsWork() {
84          assertLookup(EXAMPLE_CLASS, ARGS_STRINGS, EXAMPLE_CONSTRUCTOR_STRINGS);
85      }
86  
87      @Test
88      void testThatPrimitiveArgLookupsWork() {
89          assertLookup(EXAMPLE_CLASS, ARGS_PRIM, EXAMPLE_CONSTRUCTOR_PRIM);
90      }
91  
92      /**
93       * Do a constructor lookup and then ensure that it is cached. If we check the cache immediately after a successful
94       * lookup, within the same thread, it is reasonable that we should find it cached.
95       *
96       * @param clazz The class of interest.
97       * @param args The constructor arguments you want to use.
98       * @param expectedConstructor The expected constructor object.
99       */
100     private static void assertCaching(final Class<?> clazz, final Class<?>[] args, final Constructor<?> expectedConstructor) {
101         assertLookup(clazz, args, expectedConstructor);
102         assertEquals(expectedConstructor,
103                 ConstructorLookupCache.get(clazz, args),
104                 "the recent lookup of the " + clazz.getName() + " constructor should be cached");
105     }
106 
107     @Test
108     void testThatRecentLookupsAreBeingCached() {
109         assertCaching(EXAMPLE_CLASS, ARGS_NONE, EXAMPLE_CONSTRUCTOR_NONE);
110         assertCaching(EXAMPLE_CLASS, ARGS_INT, EXAMPLE_CONSTRUCTOR_INT);
111         assertCaching(EXAMPLE_CLASS, ARGS_STRINGS, EXAMPLE_CONSTRUCTOR_STRINGS);
112         assertCaching(EXAMPLE_CLASS, ARGS_PRIM, EXAMPLE_CONSTRUCTOR_PRIM);
113     }
114 }

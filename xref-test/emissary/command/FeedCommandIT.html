79          // setup
80          // Add the required parameters
81          arguments.addAll(Arrays.asList(PROJECT_BASE_ARGS.get(0), baseDir.toString(), INPUT_ARGS.get(0), inputDir.toString()));

83          // test
84          command = FeedCommand.parse(FeedCommand.class, arguments);

87          assertEquals(1, command.getPriorityDirectories().size());
88          assertEquals(inputDir.toString() + "/", command.getPriorityDirectories().get(0).getDirectoryName());
89          assertEquals(baseDir, command.getProjectBase());

91          // verify defaults

96          assertEquals("", command.getEatPrefix());
97          assertEquals("localhost", command.getHost());
98          assertEquals(baseDir + "/config/logback.xml", command.getLogbackConfig());
99          assertTrue(command.getOutputRoot().endsWith("/DoneParsedData"));

112     static class CartesianArgumentsProvider implements ArgumentsProvider {
113         List<List<String>> cartesian = Lists.cartesianProduct(
114                 PROJECT_BASE_ARGS,
115                 INPUT_ARGS,
116                 STRING_ARGS,
117                 BOOLEAN_ARGS,
118                 INT_ARGS);
119 
120         @Override
121         public Stream<? extends Arguments> provideArguments(ExtensionContext context) {
122             return cartesian.stream().map(list -> Arguments.of((Object[]) list.toArray(new String[0])));
123         }
124     }

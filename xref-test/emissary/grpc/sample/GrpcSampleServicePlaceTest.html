360     class SynchronicityTests {
361         private static final String SYNCHRONICITY_1_ID = "SYNCHRONICITY_ENDPOINT_1";
362         private static final String SYNCHRONICITY_2_ID = "SYNCHRONICITY_ENDPOINT_2";
363 
364         private Server startSynchronicityServer(CountDownLatch startedLatch, CountDownLatch releaseLatch) throws IOException {
365             return ServerBuilder.forPort(0)
366                     .addService(new GrpcSampleServiceImpl.CountdownLatchedIdentityServiceImpl(startedLatch, releaseLatch))
367                     .build()
368                     .start();
369         }
370 
371         @Test
372         void testBlockedResponsesAreProcessedSequentially() throws IOException, InterruptedException {
373             Server synchronicityOneServer = null;
374             Server synchronicityTwoServer = null;
375 
376             try {
377                 CountDownLatch startedLatchOne = new CountDownLatch(1);
378                 CountDownLatch releaseLatchOne = new CountDownLatch(1);
379                 synchronicityOneServer = startSynchronicityServer(startedLatchOne, releaseLatchOne);
380 
381                 CountDownLatch startedLatchTwo = new CountDownLatch(1);
382                 CountDownLatch releaseLatchTwo = new CountDownLatch(1);
383                 synchronicityTwoServer = startSynchronicityServer(startedLatchTwo, releaseLatchTwo);
384 
385                 samplePlace = placeFactory.buildPlace(
386                         new ConfigEntry(GrpcSampleServicePlace.GRPC_HOST + ENDPOINT_1_ID, null),
387                         new ConfigEntry(GrpcSampleServicePlace.GRPC_HOST + ENDPOINT_2_ID, null),
388                         new ConfigEntry(GrpcSampleServicePlace.GRPC_HOST + SYNCHRONICITY_1_ID, ENDPOINT_HOST),
389                         new ConfigEntry(GrpcSampleServicePlace.GRPC_HOST + SYNCHRONICITY_2_ID, ENDPOINT_HOST),
390                         new ConfigEntry(GrpcSampleServicePlace.GRPC_PORT + SYNCHRONICITY_1_ID, getPort(synchronicityOneServer)),
391                         new ConfigEntry(GrpcSampleServicePlace.GRPC_PORT + SYNCHRONICITY_2_ID, getPort(synchronicityTwoServer)));
392 
393                 IBaseDataObject o = new BaseDataObject(INPUT_DATA, FILENAME);
394 
395                 AtomicReference<Map<String, byte[]>> viewRef = new AtomicReference<>();
396                 AtomicReference<Throwable> errorRef = new AtomicReference<>();
397 
398                 Thread clientThread = new Thread(() -> {
399                     try {
400                         samplePlace.processAllTargetsSequentially(o);
401                         viewRef.set(o.getAlternateViews());
402                     } catch (Throwable t) {
403                         errorRef.set(t);
404                     }
405                 });
406 
407                 clientThread.start();
408 
409                 assertTrue(startedLatchOne.await(1, TimeUnit.SECONDS), "First server should have received request");
410                 assertEquals(1L, startedLatchTwo.getCount(), "Second server should be waiting to receive request");
411                 assertNull(viewRef.get());
412                 assertNull(errorRef.get());
413 
414                 assertTrue(clientThread.isAlive());
415                 releaseLatchOne.countDown();
416 
417                 assertTrue(startedLatchTwo.await(1, TimeUnit.SECONDS));
418                 assertNull(viewRef.get());
419                 assertNull(errorRef.get());
420 
421                 assertTrue(clientThread.isAlive());
422                 releaseLatchTwo.countDown();
423 
424                 clientThread.join(1000); // Wait for thread to die
425                 assertFalse(clientThread.isAlive());
426 
427                 assertEquals(2, viewRef.get().size());
428                 assertArrayEquals(INPUT_DATA, viewRef.get().get(SYNCHRONICITY_1_ID));
429                 assertArrayEquals(INPUT_DATA, viewRef.get().get(SYNCHRONICITY_2_ID));
430                 assertNull(errorRef.get());
431             } finally {
432                 if (synchronicityOneServer != null) {
433                     synchronicityOneServer.shutdownNow();
434                 }
435                 if (synchronicityTwoServer != null) {
436                     synchronicityTwoServer.shutdownNow();
437                 }
438             }
439         }

442         void testFutureResponsesAreProcessedInParallel() throws IOException, InterruptedException {
443             Server synchronicityOneServer = null;
444             Server synchronicityTwoServer = null;
445 
446             try {
447                 CountDownLatch startedLatch = new CountDownLatch(2);
448                 CountDownLatch releaseLatch = new CountDownLatch(1);
449 
450                 synchronicityOneServer = startSynchronicityServer(startedLatch, releaseLatch);
451                 synchronicityTwoServer = startSynchronicityServer(startedLatch, releaseLatch);

455                         new ConfigEntry(GrpcSampleServicePlace.GRPC_HOST + ENDPOINT_2_ID, null),

458                         new ConfigEntry(GrpcSampleServicePlace.GRPC_PORT + SYNCHRONICITY_1_ID, getPort(synchronicityOneServer)),
459                         new ConfigEntry(GrpcSampleServicePlace.GRPC_PORT + SYNCHRONICITY_2_ID, getPort(synchronicityTwoServer)));
460 
461                 IBaseDataObject o = new BaseDataObject(INPUT_DATA, FILENAME);
462 
463                 AtomicReference<Map<String, byte[]>> viewRef = new AtomicReference<>();
464                 AtomicReference<Throwable> errorRef = new AtomicReference<>();
465 
466                 Thread clientThread = new Thread(() -> {

471                         errorRef.set(t);

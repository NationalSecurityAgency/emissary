66              new ConfigEntry(GrpcSampleServicePlace.GRPC_HOST + ENDPOINT_2_ID, ENDPOINT_HOST),

73      static Stream<Integer> grpcCodes() {
74          return Arrays.stream(Status.Code.values()).map(Status.Code::value);
75      }
76  
77      static Stream<Integer> recoverableGrpcCodes() {
78          return IntStream.of(Status.Code.RESOURCE_EXHAUSTED.value(), Status.Code.UNAVAILABLE.value()).boxed();
79      }
80  
81      static Stream<Integer> nonRecoverableGrpcCodes() {
82          Set<Integer> exclude = recoverableGrpcCodes().collect(Collectors.toSet());
83          return grpcCodes().filter(i -> !exclude.contains(i));
84      }
85  
86      static String createSampleForm(String formSuffix) {
87          return String.join("-", GrpcSampleServicePlace.SERVICE_PROXY, formSuffix);
88      }
89  
90      static byte[] getSampleAltView(IBaseDataObject o) {
91          return o.getAlternateView(GrpcSampleServicePlace.ALTERNATE_VIEW_NAME);
92      }

107     class ManagedChannelHandlingTests extends UnitTest {
108         private static final String TARGET_ID = ENDPOINT_1_ID;
109         private static final String VALIDATION_PARAM = ConnectionFactory.GRPC_POOL_TEST_BEFORE_BORROW;
110         private static final String PASSIVATION_PARAM = GrpcSampleServicePlace.GRPC_POOL_KILL_AFTER_RETURN;
111         private AtomicBoolean connectionValidated;
112 
113         @BeforeEach
114         void initialize() {
115             connectionValidated = new AtomicBoolean(false);
116         }
117 
118         @Test
119         void testConnectionIsNotValidated() {
120             samplePlace = placeFactory.buildPlace(new ConfigEntry(VALIDATION_PARAM, Boolean.FALSE.toString()));
121             samplePlace.setValidationCheck(connectionValidated);
122             assertFalse(connectionValidated.get());
123             ManagedChannel channel = samplePlace.acquireChannel(TARGET_ID);
124             assertFalse(connectionValidated.get());
125             samplePlace.returnChannel(channel, TARGET_ID);
126         }
127 
128         @Test
129         void testConnectionIsValidated() {
130             samplePlace = placeFactory.buildPlace(new ConfigEntry(VALIDATION_PARAM, Boolean.TRUE.toString()));
131             samplePlace.setValidationCheck(connectionValidated);
132             assertFalse(connectionValidated.get());
133             ManagedChannel channel = samplePlace.acquireChannel(TARGET_ID);
134             assertTrue(connectionValidated.get());
135             samplePlace.returnChannel(channel, TARGET_ID);
136         }
137 
138         @Test
139         void testConnectionIsNotPassivated() {
140             samplePlace = placeFactory.buildPlace(new ConfigEntry(PASSIVATION_PARAM, Boolean.FALSE.toString()));
141             ManagedChannel channel = samplePlace.acquireChannel(TARGET_ID);
142             assertFalse(channel.isShutdown());
143             samplePlace.returnChannel(channel, TARGET_ID);
144             assertFalse(channel.isShutdown());
145         }
146 
147         @Test
148         void testConnectionIsPassivated() {
149             samplePlace = placeFactory.buildPlace(new ConfigEntry(PASSIVATION_PARAM, Boolean.TRUE.toString()));
150             ManagedChannel channel = samplePlace.acquireChannel(TARGET_ID);
151             assertFalse(channel.isShutdown());
152             samplePlace.returnChannel(channel, TARGET_ID);
153             assertTrue(channel.isShutdown());
154         }

162         @BeforeEach

164             samplePlace = placeFactory.buildPlace(
165                     new ConfigEntry(RetryHandler.GRPC_RETRY_MAX_ATTEMPTS, Integer.toString(RETRY_ATTEMPTS)));
166             dataObject = new BaseDataObject(INPUT_DATA, FILENAME, createSampleForm(TARGET_ID));
167         }
168 
169         @Test
170         void testGrpcSuccess() {
171             samplePlace.process(dataObject);
172             assertArrayEquals(ENDPOINT_1_PROCESSED_DATA, getSampleAltView(dataObject));
173         }
174 
175         @ParameterizedTest
176         @MethodSource("emissary.grpc.sample.GrpcSampleServicePlaceTest#recoverableGrpcCodes")
177         void testGrpcRecoverableCodes(int code) {
178             Status status = Status.fromCodeValue(code);
179             Runnable invocation = () -> samplePlace.throwExceptionsDuringProcess(dataObject, new StatusRuntimeException(status));
180             ServiceNotAvailableException e = assertThrows(ServiceNotAvailableException.class, invocation::run);
181             assertTrue(e.getMessage().endsWith(status.getCode().name()));
182             assertTrue(dataObject.getAlternateViewNames().isEmpty());
183         }
184 
185         @ParameterizedTest
186         @MethodSource("emissary.grpc.sample.GrpcSampleServicePlaceTest#nonRecoverableGrpcCodes")
187         void testGrpcNonRecoverableCodes(int code) {
188             Status status = Status.fromCodeValue(code);
189             Runnable invocation = () -> samplePlace.throwExceptionsDuringProcess(dataObject, new StatusRuntimeException(status));

216         @Test

222         @ParameterizedTest
223         @MethodSource("emissary.grpc.sample.GrpcSampleServicePlaceTest#recoverableGrpcCodes")
224         void testGrpcSuccessAfterRecoverableCodes(int code) {
225             Status status = Status.fromCodeValue(code);
226             AtomicInteger attemptNumber = new AtomicInteger(0);
227 
228             Runnable invocation = () -> samplePlace.throwExceptionsDuringProcessWithSuccessfulRetry(
229                     dataObject, new StatusRuntimeException(status), RETRY_ATTEMPTS, attemptNumber);
230             invocation.run();
231 
232             assertArrayEquals(ENDPOINT_1_PROCESSED_DATA, getSampleAltView(dataObject));
233             assertEquals(RETRY_ATTEMPTS, attemptNumber.get());
234         }

238         void testGrpcFailureAfterMaxRecoverableCodes(int code) {
239             Status status = Status.fromCodeValue(code);
240             AtomicInteger attemptNumber = new AtomicInteger(0);
241             int retryAttempts = RETRY_ATTEMPTS + 1;

285         @Test
286         void testMismatchedHostPortConfigs() {
287             IllegalArgumentException e = placeFactory.getBuildPlaceException(IllegalArgumentException.class,
288                     new ConfigEntry(GrpcRoutingPlace.GRPC_HOST + ENDPOINT_3_ID, ENDPOINT_HOST));
289 
290             assertEquals("gRPC hostname target-IDs do not match gRPC port number target-IDs", e.getMessage());
291         }
292 
293         @Test
294         void testNoHostPortConfigs() {
295             NullPointerException e = placeFactory.getBuildPlaceException(NullPointerException.class,
296                     new ConfigEntry(GrpcRoutingPlace.GRPC_HOST + ENDPOINT_1_ID, null),
297                     new ConfigEntry(GrpcRoutingPlace.GRPC_HOST + ENDPOINT_2_ID, null),
298                     new ConfigEntry(GrpcRoutingPlace.GRPC_PORT + ENDPOINT_1_ID, null),
299                     new ConfigEntry(GrpcRoutingPlace.GRPC_PORT + ENDPOINT_2_ID, null));
300 
301             assertEquals("Missing required arguments: GRPC_HOST_${Target-ID} and GRPC_PORT_${Target-ID}", e.getMessage());
302         }
303 
304         @Test
305         void testEndpointOneServiceRouting() {
306             samplePlace = placeFactory.buildPlace();
307             dataObject = new BaseDataObject(INPUT_DATA, FILENAME, createSampleForm(ENDPOINT_1_ID));
308             samplePlace.process(dataObject);
309             byte[] altView = getSampleAltView(dataObject);
310             assertArrayEquals(ENDPOINT_1_PROCESSED_DATA, altView);
311             assertFalse(Arrays.equals(ENDPOINT_2_PROCESSED_DATA, altView));
312         }

316             samplePlace = placeFactory.buildPlace();

318             samplePlace.process(dataObject);
319             byte[] altView = getSampleAltView(dataObject);

321             assertFalse(Arrays.equals(ENDPOINT_1_PROCESSED_DATA, altView));
322         }

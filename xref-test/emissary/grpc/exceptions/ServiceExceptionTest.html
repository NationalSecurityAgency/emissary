12  class ServiceExceptionTest extends UnitTest {
13      @Test
14      void testHandleGrpcStatusRuntimeExceptionWithNullCode() {
15          StatusRuntimeException exception = new StatusRuntimeException(Status.UNKNOWN.withDescription("test"));
16          exception = new StatusRuntimeException(Status.fromCodeValue(Status.UNKNOWN.getCode().value()), exception.getTrailers());
17          StatusRuntimeException finalException = exception;
18          ServiceException se =
19                  assertThrows(ServiceException.class, () -> ServiceException.handleGrpcStatusRuntimeException(finalException));
20          assertEquals(ServiceException.GRPC_ERROR_PREFIX + "This is an unhandled code type. Please add it to the list of gRPC exceptions: UNKNOWN",
21                  se.getMessage());
22      }

21      @Test
22      void testThreadPoolStuff() throws Exception {
23          ServerCommand cmd = ServerCommand.parse(ServerCommand.class, "-h", "host1", "-p", "3001");
24          EmissaryServer server = new EmissaryServer(cmd);
25          Server jettyServer = server.configureServer();
26          QueuedThreadPool pool = (QueuedThreadPool) jettyServer.getThreadPool();
27          assertEquals(10, pool.getMinThreads());
28          assertEquals(250, pool.getMaxThreads());
29          assertEquals(50, pool.getLowThreadsThreshold());
30          assertEquals(new Long(TimeUnit.MINUTES.toMillis(15)).intValue(), pool.getIdleTimeout());
31          assertEquals(9, pool.getThreadsPriority());
32      }
33  
34      @Test
35      void testSSLWorks() throws Exception {
36          ServerCommand cmd = ServerCommand.parse(ServerCommand.class, "-p", "3443", "--ssl");
37          EmissaryServer server = new EmissaryServer(cmd);
38          try {
39              server.startServer();
40              EmissaryClient client = new EmissaryClient();
41              String hostPort = cmd.getHost() + ":" + cmd.getPort(); // will be key in response
42              String endpoint = cmd.getScheme() + "://" + hostPort + "/api/version";
43              MapResponseEntity versionMap = client.send(new HttpGet(endpoint)).getContent(MapResponseEntity.class);
44              Map<String, String> response = versionMap.getResponse();
45              assertEquals(new Version().getVersion(), response.get(hostPort));
46          } finally {
47              // throws some stopping warning, good place to start when fixing the stop
48              server.stop();
49          }
50      }

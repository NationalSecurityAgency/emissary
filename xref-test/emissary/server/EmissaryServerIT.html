25      @Test
26      void testThreadPoolStuff() throws Exception {
27          ServerCommand cmd = ServerCommand.parse(ServerCommand.class, "-h", "host1", "-p", "3001");
28          EmissaryServer server = EmissaryServer.init(cmd);
29          Server jettyServer = server.configureServer();
30          QueuedThreadPool pool = (QueuedThreadPool) jettyServer.getThreadPool();
31          assertEquals(10, pool.getMinThreads());
32          assertEquals(250, pool.getMaxThreads());
33          assertEquals(50, pool.getLowThreadsThreshold());
34          assertEquals(Long.valueOf(TimeUnit.MINUTES.toMillis(15)).intValue(), pool.getIdleTimeout());
35          assertEquals(9, pool.getThreadsPriority());
36      }

38      @Test
39      void testSSLWorks() throws Exception {
40          ServerCommand cmd = ServerCommand.parse(ServerCommand.class, "-p", "3443", "--ssl", "--disableSniHostCheck");
41          EmissaryServer server = EmissaryServer.init(cmd);
42          try {
43              server.startServer();
44              EmissaryClient client = new EmissaryClient();
45              String hostPort = cmd.getHost() + ":" + cmd.getPort(); // will be key in response
46              String endpoint = cmd.getScheme() + "://" + hostPort + "/api/version";
47              MapResponseEntity versionMap = client.send(new HttpGet(endpoint)).getContent(MapResponseEntity.class);
48              Map<String, String> response = versionMap.getResponse();
49              assertEquals(new Version().getVersion(), response.get(hostPort));
50          } finally {
51              // throws some stopping warning, good place to start when fixing the stop
52              server.stop();
53          }
54      }

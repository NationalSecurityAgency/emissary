16

18      public static void tearDownClass() {

21  
22      // Handle setting up all the Jersey Test framework components
23      @Override
24      protected Application configure() {
25          UnitTest.setupSystemProperties();
26          // Tells Jersey to use first available port, fixes address already in use exception
27          forceSet(TestProperties.CONTAINER_PORT, "0");
28          final ResourceConfig application = new ResourceConfig();
29          application.register(MultiPartFeature.class);
30          application.property(MustacheMvcFeature.TEMPLATE_BASE_PATH, "/templates");
31          application.register(MustacheMvcFeature.class).packages("emissary.server.mvc");
32          application.register(MustacheMvcFeature.class).packages("emissary.server.api");
33  
34          return application;

33      public static final int TEST_PORT = 123456;
34      public static final String TEST_NODE = "localhost";
35      public static final String TEST_NODE_PORT = TEST_NODE + ":" + TEST_PORT;
36      public static final String DIRNAME = "http://" + TEST_NODE_PORT + "/DirectoryPlace";
37      public static final String SELF = "*.*.*.http://localhost:9999/DirectoryPlace";
38      public static final String PEER1 = "*.*.*.http://remoteHost:8888/DirectoryPlace";
39      public static final String PEER2 = "*.*.*.http://remoteHost2:8888/DirectoryPlace";
40      public static final Set<String> PEERS = new HashSet<>(Arrays.asList(SELF, PEER1, PEER2));
41  
42      @BeforeEach
43      public void setup() throws Exception {
44          String projectBase = System.getenv(ConfigUtil.PROJECT_BASE_ENV);
45          ServerCommand cmd = ServerCommand.parse(ServerCommand.class, "-b ", projectBase, "-m", "cluster", "-p", "123456");
46          cmd.setupServer();
47          EmissaryServer server = EmissaryServer.init(cmd, new TestEmissaryNode(EmissaryNode.Mode.CLUSTER));
48          Namespace.bind("EmissaryServer", server);
49  
50          DirectoryPlace directoryPlace = new DirectoryPlace(DIRNAME, server.getNode());
51          directoryPlace.addPeerDirectories(PEERS, false);
52          Namespace.bind(DIRNAME, directoryPlace);
53      }
54  
55      @AfterEach
56      public void cleanup() {
57          Namespace.unbind(DIRNAME);
58          Namespace.unbind("EmissaryServer");
59      }
60  
61      @Test
62      void peers() {
63          // test
64          Response response = target("peers").request().get();
65  
66          // verify
67          assertEquals(200, response.getStatus());
68          PeersResponseEntity entity = response.readEntity(PeersResponseEntity.class);
69          assertEquals(0, entity.getErrors().size());
70          assertTrue(CollectionUtils.isEmpty(entity.getCluster()));
71          assertEquals(TEST_NODE_PORT, entity.getLocal().getHost());
72          assertTrue(entity.getLocal().getPeers().containsAll(PEERS));
73      }
74

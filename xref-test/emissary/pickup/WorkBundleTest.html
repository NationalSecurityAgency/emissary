27      private static boolean compareWorkerUnits(WorkUnit w1, WorkUnit w2) {
28          boolean matchFileNames;
29          if (w1.getFileName() == null && w2.getFileName() == null) {
30              matchFileNames = true;
31          } else {
32              assertNotNull(w1.getFileName());
33              matchFileNames = w1.getFileName().equals(w2.getFileName());
34          }
35  
36          boolean matchTxid;
37          if (w1.getTransactionId() == null && w2.getTransactionId() == null) {
38              matchTxid = true;
39          } else {
40              assertNotNull(w1.getTransactionId());
41              matchTxid = w1.getTransactionId().equals(w2.getTransactionId());
42          }
43  
44          return matchFileNames && matchTxid && (w1.failedToParse() == w2.failedToParse()) && (w1.failedToProcess() == w2.failedToProcess());
45      }

58              ic++;
59          }
60          assertEquals(3, ic, "Files from iterator");
61  
62          WorkBundle w2 = new WorkBundle(w1);
63          assertNotEquals(w1.getBundleId(), w2.getBundleId(), "Copy ctor new id");
64          assertEquals(w1.getWorkUnitList(), w2.getWorkUnitList(), "Failed to properly copy WorkUnit list");
65  
66          String xml = w2.toXml();
67          assertNotNull(xml, "XML failed to generate");
68  
69          WorkBundle w3 = WorkBundle.buildWorkBundle(xml);
70          assertNotNull(w3, "Failed to buildworkbundle from xml");

72              if (!compareWorkerUnits(w2.getWorkUnitList().get(i), w3.getWorkUnitList().get(i))) {
73                  fail("BuildWorkBundle did not generate equivalent workBundle");

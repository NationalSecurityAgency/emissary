57              ps.shutDown();

99      class TestErrorPickUpSpace extends PickUpSpace {
100         public TestQueueServer queServer;
101         public PickupQueue queue;
102         public int expectedErrors;
103         public int numErrors = 0;
104         public int expectedBundles;
105         public int numBundles = 0;
106         public int bundleCompletedCount = 0;
107         public int bundleFailedCount = 0;
108 
109         public TestErrorPickUpSpace(int errors, int bundles) throws IOException {
110             super(tepusConfigStream, null, "http://localhost:8005/TestErrorPickUpSpace");
111             this.expectedErrors = errors;
112             this.expectedBundles = bundles;
113             queue = new PickupQueue(13);
114             queServer = new TestQueueServer(this, queue, 50, "QueServerTest");
115             queServer.start();
116         }
117 
118         @Override
119         public boolean enque(WorkBundle path) {
120             assertTrue(queue.enque(path), "Bundle enqueued");
121             return true;
122         }
123 
124         @Override
125         public int getQueSize() {
126             return queue.getQueSize();
127         }
128 
129         @Override
130         public boolean take() {
131             if (numErrors++ < expectedErrors || numBundles >= expectedBundles) {
132                 return false;
133             }
134             numBundles++;
135             WorkBundle w = new WorkBundle("/output/root", "/eat/prefix");
136             w.addFileName("file1.txt");
137             assertTrue(enque(w), "Bundle enqueueued");
138             return true;
139         }
140 
141         @Override
142         public void bundleCompleted(String bundleId, boolean itWorked) {
143             if (itWorked) {
144                 bundleCompletedCount++;
145             } else {
146                 bundleFailedCount++;
147             }
148         }
149 
150         @Override
151         public void shutDown() {
152             queServer.shutdown();
153             super.shutDown();
154         }
155     }

351     @Test
352     void testFailurePutsPendingBundleBackToOutboundQueue() {
353         String C1 = "INITIAL.INPUT.A.http://otherhost:7001/FilePickUpClient";
354         String C2 = "INITIAL.INPUT.A.http://otherhost:8001/FilePickUpClient";
355         String C3 = "INITIAL.INPUT.A.http://otherhost:9001/FilePickUpClient";
356 
357         // Set up some fake clients
358         mws.addPickUp(C1);
359         mws.addPickUp(C2);
360         mws.addPickUp(C3);
361         assertEquals(3, mws.getPickUpPlaceCount(), "Pickups added must count");
362 
363         // Set up some fake work
364         WorkBundle wb = new WorkBundle("/fake/root", "/fake/eat");
365         wb.addFileName("faker.txt");
366 
367         // Give the fake work to the space
368         mws.addOutboundBundle(wb);
369         assertEquals(1, mws.getBundlesProcessed(), "Adding bundle processes it");
370         assertEquals(1, mws.getOutboundQueueSize(), "Budnle is outbound");
371         assertEquals(0, mws.getRetriedCount(), "No retries yet");
372         assertEquals(0, mws.getPendingQueueSize(), "No pending items yet");
373 
374         // Take the fake work
375         WorkBundle taken = mws.take(C2);
376         assertEquals(wb.getBundleId(), taken.getBundleId(), "wb taken must be the one put in");
377 
378         // Check the counts after taking the work
379         assertEquals(1, mws.getBundlesProcessed(), "Taking bundle doesn't change  processes count");
380         assertEquals(0, mws.getOutboundQueueSize(), "No bundles outbound");
381         assertEquals(0, mws.getRetriedCount(), "No retries yet");
382         assertEquals(1, mws.getPendingQueueSize(), "Taken item is pending");
383 
384         // Fail the fake client that took the work
385         mws.removePickUp(C2);
386         assertEquals(2, mws.getPickUpPlaceCount(), "Pickups removed must count");
387 
388         // Make sure the bundle went back to outbound queue
389         assertEquals(1, mws.getOutboundQueueSize(), "Failed bundle back to  outbound");
390         assertEquals(1, mws.getRetriedCount(), "Retry counter bumped on failure");
391         assertEquals(0, mws.getPendingQueueSize(), "Failed item no longer pending");
392     }

60      @Test
61      void testDataObjectCreatedCallbackWithMissingPath() {
62          client.dataObjectCreated(payload, new File("/eat/prefix/foo/bar"));
63          assertEquals("bar", payload.getStringParameter(INPUT_FILENAME), "Input filename must be recorded");
64          assertEquals("/foo", payload.getStringParameter("TARGETBIN"), "TargetBin must be recorded without prefix");
65          assertEquals(bundle.getPriority(), payload.getPriority(), "Priority must be transferred from bundle to payload");
66      }

68      @Test
69      void testDataObjectCreatedCallbackWithSimpleCaseId() {
70          bundle.setCaseId("PETERPAN");
71          client.dataObjectCreated(payload, new File("/eat/prefix/foo/bar"));
72          assertEquals(bundle.getCaseId(), payload.getStringParameter("DATABASE_CASE_ID"), "Simple case id must be transferred from bundle to payload");
73      }
74  
75      @Test
76      void testDataObjectCreatedCallbackWithComplexCaseId() {
77          String originalShortName = payload.shortName();
78          bundle.setCaseId("PROJECT:PETERPAN");
79          client.dataObjectCreated(payload, new File("/eat/prefix/foo/bar"));
80          assertEquals("PETERPAN", payload.getStringParameter("PROJECT"), "Complex case id must be transferred from bundle to payload");
81          assertEquals(originalShortName, payload.shortName(), "Complex case id must not mess with payload filename when not simple");
82      }
83  
84      @Test
85      void testDataObjectCreatedCallbackWithComplexCaseIdInSimpleMode() {
86          bundle.setCaseId("PROJECT:PETERPAN");
87          payload.putParameter("SIMPLE_MODE", "true");
88          client.dataObjectCreated(payload, new File("/eat/prefix/foo/bar"));
89          assertEquals("PETERPAN", payload.getStringParameter("PROJECT"), "Complex case id must be transferred from bundle to payload");
90          assertTrue(payload.shortName().startsWith("PETERPAN-"),
91                  "Complex case in simple mode must add fn hash to payload not " + payload.shortName());
92          assertEquals("/foo/bar", payload.getStringParameter(ORIGINAL_FILENAME), "Original-Filename should be set in simple mode");
93      }

95      @Test

97          bundle.setCaseId("PETERPAN");
98          client.nullifyCaseIdInHook = true;

100         assertEquals(TimeUtil.getCurrentDateOrdinal(),

105     @Test
106     void testCreateFileName() throws Exception {
107         // Perform the default filename creation strategy
108         String filePath = "/foo/bar";
109         MessageDigest digest = MessageDigest.getInstance("SHA-256");
110         digest.reset();
111         byte[] resultHash = digest.digest(filePath.getBytes());
112         String resultString = Hexl.toUnformattedHexString(resultHash);
113 
114         // Compare with the output of the createFilename() method
115         assertEquals("PETERPAN-" + resultString, client.createFilename(filePath, "PETERPAN"), "File names do not match");
116 
117         // Compare with what was set in the payload for simple mode
118         bundle.setCaseId("PROJECT:PETERPAN");
119         payload.putParameter("SIMPLE_MODE", "true");
120         client.dataObjectCreated(payload, new File(filePath));
121         assertEquals("PETERPAN-" + resultString, payload.getFilename(), "Payload filename is not set to correct value");
122     }

141     @Test
142     void testAll() {
143 
144         assertTrue(Namespace.exists("http://localhost:8005/WorkSpace"), "WorkSpace should exist in namespace as " + space.getKey());
145 
146         pause(100);
147 
148         int byteSize = 0;
149         int fileCount = 0;
150         int bundleCount = 0;
151         int clientCount = 1;
152         int expectedOutbound = 0;
153         int expectedPending = 0;
154         int expectedRetries = 0;
155 
156         checkFileCounts(space, fileCount, byteSize, bundleCount, clientCount, expectedOutbound, expectedPending, expectedRetries);
157 
158         Thread tspacethr = new Thread(space, "WorkSpaceTestFPC");
159         tspacethr.setDaemon(true);
160         tspacethr.start();
161 
162         logger.debug("WorkSpace is started!");
163 
164         pause(1500);
165 
166         for (File f : workingFiles) {
167             // explicit cast to address warning for implicit narrowing conversion
168             byteSize += (int) f.length();
169         }
170         bundleCount++;
171         fileCount += 2;
172 
173         // BadPlace should have crashed and been
174         // deregistered when it received a bundle
175         expectedOutbound++;
176         clientCount--;
177 
178         // Account for retries
179         expectedRetries++;
180 
181         checkFileCounts(space, fileCount, byteSize, bundleCount, clientCount, expectedOutbound, expectedPending, expectedRetries);
182         pause(1000);
183 
184         // Add another broken FilePickUpCLient, but one that receives
185         // ok and fails to process
186         logger.debug("STARTING BROKEN DURING PROCESSING CLIENT");
187         badplace =
188                 (BreakableFilePickUpClient) addPlace("http://localhost:9005/FilePickUpClient", BreakableFilePickUpClient.class.getName(),
189                         peer2.getKey());
190         badplace.setBrokenDuringProcessing(true); // crash after receiving a bundle
191         space.addPickUp(badplace.getKey());
192         clientCount++;
193 
194         pause(1000); // it dies
195         space.removePickUp(badplace.getKey());
196         expectedRetries++;
197         clientCount--;
198         checkFileCounts(space, fileCount, byteSize, bundleCount, clientCount, expectedOutbound, expectedPending, expectedRetries);
199 
200         pause(1000);
201 
202 
203         // Start a GOOD FilePickUpClient
204         logger.debug("STARTING GOOD CLIENT");
205         goodplace =
206                 (BreakableFilePickUpClient) addPlace("http://localhost:9005/FilePickUpClient", BreakableFilePickUpClient.class.getName(),
207                         peer2.getKey());
208         space.addPickUp(goodplace.getKey());
209 
210         clientCount++;
211         expectedOutbound--;
212         pause(1000);
213 
214         checkFileCounts(space, fileCount, byteSize, bundleCount, clientCount, expectedOutbound, expectedPending, expectedRetries);
215 
216         pause(500);
217         checkFileLocations();
218         logger.debug("Failed Space Place all tests completed!");
219     }

221     private void checkFileLocations() {
222         // Detailed debugging help on the structure of what is left in the file system
223         if (logger.isDebugEnabled()) {
224             StringBuilder sb = new StringBuilder();
225             new Executrix().execute(new String[] {"find", TMPDIR + "/filepicktest", TMPDIR + "/data", "-print"}, sb);
226             logger.debug("Files:\n" + sb);
227         }
228 
229         // Assert things about where the files are located
230         int counter = 0;
231         for (File f : workingFiles) {
232             assertFalse(f.exists(), "File[" + counter + "] should not exist in input area any more - " + f);
233             counter++;
234         }
235 
236         counter = 0;
237         for (String fn : workingFilePaths) {
238             File fp = new File(goodplace.getInProcessArea() + "/" + fn);
239             File fd = new File(space.getOutputRoot() + "/" + fn);
240             assertFalse(fp.exists(), "File[" + counter + "] should not exist in in-process area any more - " + fn + " - " + fp.getPath());
241             assertTrue(fd.exists(), "File[" + counter + "] should exist in the output area - " + fn + " - " + fd.getPath());
242             counter++;
243         }
244     }

246     private static void checkFileCounts(WorkSpace space, int files, int bytes, int bundles, int places, int outbound, int pending, int retried) {
247         assertEquals(files, space.getFilesProcessed(), "files processed");
248         assertEquals(bytes, space.getBytesProcessed(), "bytes processed");
249         assertEquals(bundles, space.getBundlesProcessed(), "bundles processed");
250         assertEquals(places, space.getPickUpPlaceCount(), "pickup place count");
251         assertEquals(outbound, space.getOutboundQueueSize(), "Outbound queue count");
252         assertEquals(pending, space.getPendingQueueSize(), "Pending queue count");
253         assertEquals(retried, space.getRetriedCount(), "Retried bundle count");
254     }

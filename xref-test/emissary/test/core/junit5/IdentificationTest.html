28  public abstract class IdentificationTest extends UnitTest {
29  
30      protected static final Logger logger = LoggerFactory.getLogger(IdentificationTest.class);
31  
32      @Nullable
33      @SuppressWarnings("NonFinalStaticField")
34      protected static IServiceProviderPlace place = null;
35  
36      @BeforeEach
37      public void setUpPlace() throws Exception {
38          place = createPlace();
39      }
40  
41      @AfterEach
42      public void tearDownPlace() {
43          if (place != null) {
44              place.shutDown();
45              place = null;
46          }
47      }
48  
49      /**
50       * Derived classes must implement this
51       */
52      public abstract IServiceProviderPlace createPlace() throws IOException;
53  
54      public static Stream<? extends Arguments> data() {
55          return getMyTestParameterFiles(IdentificationTest.class);
56      }
57  
58      @ParameterizedTest
59      @MethodSource("data")
60      public void testIdentificationPlace(String resource) {
61          logger.debug("Running {} test on resource {}", place.getClass().getName(), resource);
62  
63          try (InputStream doc = new ResourceReader().getResourceAsStream(resource)) {
64              byte[] data = IOUtils.toByteArray(doc);
65              String expectedAnswer = resolveFormForResource(resource);
66              IBaseDataObject payload = DataObjectFactory.getInstance(data, resource, Form.UNKNOWN);
67              processPreHook(payload, resource);
68              place.agentProcessHeavyDuty(payload);
69              processPostHook(payload, resource);
70              checkAnswersPreHook(payload, resource, expectedAnswer);
71              checkAnswers(payload, resource, expectedAnswer);
72              checkAnswersPostHook(payload, resource, expectedAnswer);
73          } catch (Exception ex) {
74              logger.error("Error running test {}", resource, ex);
75              fail("Cannot run test " + resource, ex);
76          }
77      }
78  
79      /**
80       * Resolves the expected form for the test resource.
81       *
82       * @param resource complete file path of the test resource
83       * @return form expected for the resource
84       */
85      protected String resolveFormForResource(String resource) {
86          return resource.replaceAll("^.*/([^/@]+)(@\\d+)?\\.dat$", "$1");
87      }

94          // Nothing to do here
95      }

98          // Nothing to do here
99      }
100 
101     protected void checkAnswersPostHook(IBaseDataObject payload, String resource, String expected) {
102         // Nothing to do here
103     }
104 
105     protected void checkAnswers(IBaseDataObject payload, String resource, String expected) {
106         List<String> expectedForms = Splitter.on('#').splitToList(expected);
107         for (int i = 0; i < expectedForms.size(); i++) {
108             String result = payload.currentFormAt(i).replaceAll("^LANG-", "");
109             if (expectedForms.get(i).indexOf("(") > 0) {
110                 assertEquals(expectedForms.get(i), result, "Current form is wrong in " + resource);
111             } else {
112                 assertEquals(expectedForms.get(i), result.replaceAll("\\(.*\\)", ""), "Current form is wrong in " + resource);
113             }
114         }
115     }

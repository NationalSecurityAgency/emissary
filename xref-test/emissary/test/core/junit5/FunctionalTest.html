49      /**
50       * Start a directory on the specified port of localhost for testing.
51       */
52      protected IDirectoryPlace startDirectory(int port) {
53          String url = "http://localhost:" + port + "/DirectoryPlace";
54          System.setProperty(EmissaryNode.NODE_PORT_PROPERTY, "" + port);
55          // Instantiate the place
56          IServiceProviderPlace thePlace;
57          // TODO maybe uncomment
58          // MDC.put(MDCConstants.SERVICE_LOCATION, "DIR-" + port);
59          try {
60              // No parent
61              thePlace = PlaceStarter.createPlace(url, null, "emissary.directory.DirectoryPlace", null);
62              logger.debug("Started directory on port {} as {}", port, thePlace);
63          } finally {
64              // TODO maybe uncomment
65              // MDC.remove(MDCConstants.SERVICE_LOCATION);
66          }
67          return (IDirectoryPlace) thePlace;
68      }

70      /**
71       * Start a localhost jetty on the specified port. Create a emissary/test123 user in a /tmp/jetty-users.properties
72       */
73      protected void startJetty(int port) throws Exception {
74          String PROJECT_BASE = System.getenv(ConfigUtil.PROJECT_BASE_ENV);
75          // Set up a password file
76          File realmFile = new File(PROJECT_BASE + "/config", "jetty-users.properties");
77          try (FileOutputStream ros = new FileOutputStream(realmFile)) {
78              ros.write("emissary: test123, emissary".getBytes());
79          }
80  
81          String nodeName = "localhost";
82          String[] args = new String[] {"-b", PROJECT_BASE, "-p", Integer.toString(port), "-h", nodeName, "--mobileAgents", Integer.toString(3)};
83          try {
84              ServerCommand cmd = ServerCommand.parse(ServerCommand.class, args);
85  
86              jserver = EmissaryServer.init(cmd);
87              jetty = jserver.startServer();
88          } catch (Exception ignored) {
89              // Ignore
90          }
91  
92          // Wait for jetty to come up
93          try {
94              Thread.sleep(500);
95          } catch (InterruptedException ignored) {
96              // Ignore
97          }
98          referenceServices(port);
99      }

101     /**
102      * Collect local references to the emissary services
103      */
104     protected void referenceServices(int port) throws Exception {
105         // The configuration should cause peer1 to be created.
106         // Check node.cfg, node-localhost.cfg or node-localhost-8005.cfg if not
107         directory = (IDirectoryPlace) Namespace.lookup("http://localhost:" + port + "/DirectoryPlace");
108 
109         try {
110             pool = AgentPool.lookup();
111         } catch (NamespaceException ex) {
112             logger.warn("Agent pool is missing");
113         }
114         try {
115             spool = MoveSpool.lookup();
116         } catch (NamespaceException ex) {
117             logger.warn("Move spool is missing");
118         }
119     }
120 
121     protected IServiceProviderPlace addPlace(String key, String clsName) {
122         if (directory != null) {
123             return addPlace(key, clsName, directory.getKey());

133     protected IServiceProviderPlace addPlace(String key, String clsName, InputStream configStream) {
134         if (directory != null) {
135             return PlaceStarter.createPlace(key, configStream, clsName, directory.getKey());
136         } else {
137             return PlaceStarter.createPlace(key, configStream, clsName, null);

141     protected void demolishServer() {
142         // Stop the places
143         for (String key : Namespace.keySet()) {
144             try {
145                 Object obj = Namespace.lookup(key);
146                 if (obj instanceof IServiceProviderPlace) {
147                     logger.debug("Stopping {}", obj);
148                     ((IServiceProviderPlace) obj).shutDown();
149                 }
150             } catch (NamespaceException ignored) {
151                 // empty catch block
152             }
153         }
154 
155         if (directory != null && !directory.isShutdownInitiated()) {
156             directory.shutDown();
157             directory = null;
158         }
159 
160         if (jserver != null && jserver.isServerRunning()) {
161             jserver.stop();
162             jetty = null;
163             assertFalse(EmissaryServer.getInstance().isServerRunning(), "Server did not stop and unbind");
164             jserver = null;
165         }
166 
167         if (pool != null) {
168             pool.close();
169             pool = null;
170         }
171 
172         // Clean out namespace
173         for (String key : Namespace.keySet()) {
174             Namespace.unbind(key);
175         }
176 
177         File realmFile = new File(TMPDIR, "jetty-users.properties");
178         try {
179             Files.deleteIfExists(realmFile.toPath());
180         } catch (IOException e) {
181             logger.debug("Unable to delete temporary realmFile {}", realmFile, e);
182         }
183     }

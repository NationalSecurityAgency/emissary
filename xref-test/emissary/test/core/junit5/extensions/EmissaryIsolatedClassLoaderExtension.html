16  public class EmissaryIsolatedClassLoaderExtension implements InvocationInterceptor {
17  
18      @Override
19      public void interceptTestMethod(Invocation<Void> invocation, ReflectiveInvocationContext<Method> invocationContext,
20              ExtensionContext extensionContext) throws Throwable {
21          if (extensionContext.getRequiredTestClass().getClassLoader().getClass().getName().equals(TestClassLoader.class.getName())) {
22              invocation.proceed();
23              return;
24          }
25          invocation.skip();
26          runTestWithIsolatedClassPath(invocationContext, extensionContext);
27      }
28  
29      private static void runTestWithIsolatedClassPath(ReflectiveInvocationContext<Method> invocationContext,
30              ExtensionContext extensionContext) throws Throwable {
31          // get the test class and method
32          Class<?> testClass = extensionContext.getRequiredTestClass();
33          Method testMethod = invocationContext.getExecutable();
34  
35          // get the original and isolated class loaders
36          ClassLoader originalClassLoader = Thread.currentThread().getContextClassLoader();
37          @SuppressWarnings("BanClassLoader")
38          URLClassLoader isolatedClassLoader = new TestClassLoader();
39  
40          // set the isolated class loader
41          Thread.currentThread().setContextClassLoader(isolatedClassLoader);
42          try {
43              // run test using isolated class loader
44              Class<?> isolatedTestClass = isolatedClassLoader.loadClass(testClass.getName());
45              Object testInstance = ReflectionUtils.newInstance(isolatedTestClass);
46              Method method = ReflectionUtils.findMethod(isolatedTestClass, testMethod.getName())
47                      .orElseThrow(() -> new UnsupportedOperationException("Method not found " + testMethod.getName()));
48              ReflectionUtils.invokeMethod(method, testInstance);
49          } finally {
50              // revert to the original class loader
51              Thread.currentThread().setContextClassLoader(originalClassLoader);
52          }
53      }
54  
55      @SuppressWarnings("BanClassLoader")
56      public static class TestClassLoader extends URLClassLoader {
57          public TestClassLoader() {
58              super(getUrlsFromSystemClassLoader());
59          }
60  
61          private static URL[] getUrlsFromSystemClassLoader() {
62              ClassLoader systemClassLoader = getSystemClassLoader();
63              if (systemClassLoader instanceof URLClassLoader) {
64                  return ((URLClassLoader) systemClassLoader).getURLs();
65              }
66              String[] paths = System.getProperty("java.class.path").split("[" + File.pathSeparator + "]");
67              return Arrays.stream(paths).map(path -> {
68                  try {
69                      return Paths.get(path).toUri().toURL();
70                  } catch (MalformedURLException e) {
71                      throw new IllegalArgumentException("Invalid class path item: " + path, e);
72                  }
73              }).toArray(URL[]::new);
74          }
75  
76          @Override
77          public Class<?> loadClass(String name) throws ClassNotFoundException {
78              if (name.startsWith("emissary")) {
79                  return super.findClass(name);
80              }
81              return super.loadClass(name);
82          }
83      }

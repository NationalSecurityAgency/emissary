120     public static synchronized void setupSystemProperties() {
121         // mostly just to get the system properties set
122         // synchronized since multiple threads are testing at the same time
123         String projectBase = System.getenv("PROJECT_BASE");
124         if (projectBase == null) {
125             fail("PROJECT_BASE is not set");
126         }
127         // setup the environment stuff
128         try {
129             ServerCommand command = ServerCommand.parse(ServerCommand.class, "-m", "cluster");
130             command.setupCommand();
131             EmissaryServer.init(command);
132         } catch (EmissaryException e) {
133             fail("Unable to setup Emissary environment", e);
134         }
135     }

137     public void assertMaxNonSystemThreadCount(int max) {
138         ThreadDump td = new ThreadDump();
139         List<ThreadInfo> ti = td.getThreadInfo(true);
140         if (ti.size() > max) {
141             StringBuilder sb = new StringBuilder();
142             for (ThreadInfo t : ti) {
143                 sb.append(t.getThreadName()).append(" ");
144             }
145             assertTrue(max <= ti.size(), "Not expecting " + ti.size() + " threads from " + this.getClass().getName() + ": " + sb);
146         }
147     }

204     /**
205      * Pause for the specified number of millis without throwing any exceptions when it doesn't work
206      *
207      * @param millis how long to pause
208      */
209     @SuppressWarnings("ThreadPriorityCheck")
210     protected void pause(long millis) {
211         Thread.yield();
212         try {
213             Thread.sleep(millis);
214         } catch (Exception ex) {
215             // empty exception block
216             if (ex instanceof InterruptedException) {
217                 Thread.currentThread().interrupt();
218             }
219             logger.debug("Thread interrupted", ex);
220         }
221     }

259     /**
260      * Get an JDOM XML document corresponding to a test resource
261      *
262      * @param resource the resource under test
263      * @return the answer document
264      */
265     @Nullable
266     protected Document getAnswerDocumentFor(String resource) {
267         int datPos = resource.lastIndexOf(ResourceReader.DATA_SUFFIX);
268         if (datPos == -1) {
269             logger.debug("Resource is not a DATA file {}", resource);
270             return null;
271         }
272 
273         String aname = "";
274         if (answerFiles.isEmpty()) {
275             aname = resource.substring(0, datPos) + ResourceReader.XML_SUFFIX;
276         } else {
277             // if answer files are in different directory than data files, this will be used to find matching answer to data pair
278             String testFileName = FilenameUtils.getBaseName(resource);
279             for (String answer : answerFiles) {
280                 if (FilenameUtils.getBaseName(answer).equals(testFileName)) {
281                     aname = answer;
282                     break;
283                 }
284             }
285         }
286         return getAnswerDocumentValidated(aname);
287     }
288 
289     @Nullable
290     protected Document getAnswerDocumentValidated(String aname) {
291         SchemaFactory schemaFactory = SchemaFactory.newInstance(XMLConstants.W3C_XML_SCHEMA_NS_URI);
292         try (InputStream is = new ResourceReader().getResourceAsStream(aname)) {
293             Schema schema = schemaFactory.newSchema(new ResourceReader().getResource(getAnswerXsd()));
294             XMLReaderJDOMFactory factory = new XMLReaderSchemaFactory(schema);
295             SAXBuilder sb2 = new SAXBuilder(factory);
296             return sb2.build(is);
297         } catch (JDOMException je) {
298             if (Boolean.getBoolean("validateAnswerFiles")) {
299                 fail("Answer document " + aname + " failed to validate against schema " + getAnswerXsd(), je);
300             }
301             logger.error("Answer document {} failed to validate against schema {}", aname, getAnswerXsd(), je);
302             return getAnswerDocumentNonValidated(aname);
303         } catch (Exception ex) {
304             logger.debug("No answer document provided for {}", aname, ex);
305         }
306         return null;
307     }
308

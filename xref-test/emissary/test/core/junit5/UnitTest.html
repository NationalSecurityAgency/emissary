105     public static synchronized void setupSystemProperties() {
106         // mostly just to get the system properties set
107         // synchronized since multiple threads are testing at the same time
108         String projectBase = System.getenv("PROJECT_BASE");
109         if (projectBase == null) {
110             fail("PROJECT_BASE is not set");
111         }
112         // setup the environment stuff
113         try {
114             ServerCommand command = ServerCommand.parse(ServerCommand.class, "-m", "cluster");
115             command.setupCommand();
116             EmissaryServer.init(command);
117         } catch (EmissaryException e) {
118             fail("Unable to setup Emissary environment", e);
119         }
120     }

122     public void assertMaxNonSystemThreadCount(int max) {
123         ThreadDump td = new ThreadDump();
124         List<ThreadInfo> ti = td.getThreadInfo(true);
125         if (ti.size() > max) {
126             StringBuilder sb = new StringBuilder();
127             for (ThreadInfo t : ti) {
128                 sb.append(t.getThreadName()).append(" ");
129             }
130             assertTrue(max <= ti.size(), "Not expecting " + ti.size() + " threads from " + this.getClass().getName() + ": " + sb);
131         }
132     }

179     /**
180      * Pause for the specified number of millis without throwing any exceptions when it doesn't work
181      *
182      * @param millis how long to pause
183      */
184     @SuppressWarnings("ThreadPriorityCheck")
185     protected void pause(long millis) {
186         Thread.yield();
187         try {
188             Thread.sleep(millis);
189         } catch (Exception ex) {
190             // empty exception block
191             if (ex instanceof InterruptedException) {
192                 Thread.currentThread().interrupt();
193             }
194             logger.debug("Thread interrupted", ex);
195         }
196     }

231     /**
232      * Get an JDOM XML document corresponding to a test resource
233      */
234     @Nullable
235     protected Document getAnswerDocumentFor(String resource) {
236         int datPos = resource.lastIndexOf(ResourceReader.DATA_SUFFIX);
237         if (datPos == -1) {
238             logger.debug("Resource is not a DATA file {}", resource);
239             return null;
240         }
241 
242         String aname = "";
243         if (answerFiles.isEmpty()) {
244             aname = resource.substring(0, datPos) + ResourceReader.XML_SUFFIX;
245         } else {
246             // if answer files are in different directory than data files, this will be used to find matching answer to data pair
247             String testFileName = FilenameUtils.getBaseName(resource);
248             for (String answer : answerFiles) {
249                 if (FilenameUtils.getBaseName(answer).equals(testFileName)) {
250                     aname = answer;
251                     break;
252                 }
253             }
254         }
255 
256         SAXBuilder builder = new SAXBuilder(XMLReaders.NONVALIDATING);
257         Document answerDoc;
258         try (InputStream is = new ResourceReader().getResourceAsStream(aname)) {
259             answerDoc = builder.build(is);
260         } catch (Exception ex) {
261             logger.debug("No answer document provided for {}", aname, ex);
262             return null;
263         }
264         return answerDoc;
265     }

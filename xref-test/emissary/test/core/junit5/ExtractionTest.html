80      /**
81       * Allow overriding the initial form in extensions to this test.
82       * 
83       * By default, get the initial form from the filename in the form {@code INITIAL_FORM@2.dat} where {@code INITIAL_FORM}
84       * will be the initial form.
85       *
86       * @param resource to get the form from
87       * @return the initial form
88       */
89      @ForOverride
90      protected String getInitialForm(final String resource) {
91          return resource.replaceAll("^.*/([^/@]+)(@\\d+)?\\.dat$", "$1");
92      }

94      @ParameterizedTest
95      @MethodSource("data")
96      public void testExtractionPlace(String resource) {
97          logger.debug("Running {} test on resource {}", place.getClass().getName(), resource);
98  
99          // Need a pair consisting of a .dat file and a .xml file (answers)
100         Document controlDoc = getAnswerDocumentFor(resource);
101         if (controlDoc == null) {
102             fail("No answers provided for test " + resource);
103         }
104 
105         try (InputStream doc = new ResourceReader().getResourceAsStream(resource)) {
106             byte[] data = IOUtils.toByteArray(doc);
107             String initialForm = getInitialForm(resource);
108             IBaseDataObject payload = DataObjectFactory.getInstance(data, resource, initialForm);
109             setupPayload(payload, controlDoc);
110             processPreHook(payload, controlDoc);
111             List<IBaseDataObject> attachments = processHeavyDutyHook(place, payload);
112             processPostHook(payload, attachments);
113             checkAnswersPreHook(controlDoc, payload, attachments, resource);
114             checkAnswers(controlDoc, payload, attachments, resource);
115             checkAnswersPostHook(controlDoc, payload, attachments, resource);
116         } catch (Exception ex) {
117             logger.error("Error running test {}", resource, ex);
118             fail("Cannot run test " + resource, ex);
119         }
120     }

150     protected void checkAnswers(Document answers, IBaseDataObject payload, List<IBaseDataObject> attachments, String tname)
151             throws DataConversionException {
152         Element root = answers.getRootElement();
153         Element parent = root.getChild("answers");
154         if (parent == null) {
155             parent = root;
156         }
157 
158         // Check the payload
159         checkAnswers(parent, payload, attachments, tname);
160 
161         // Check each attachment
162         for (int attNum = 1; attNum <= attachments.size(); attNum++) {
163             String atname = tname + Family.SEP + attNum;
164             Element el = parent.getChild("att" + attNum);
165             if (el != null) {
166                 checkAnswersPreHook(el, payload, attachments.get(attNum - 1), atname);
167                 checkAnswers(el, attachments.get(attNum - 1), null, atname);
168                 checkAnswersPostHook(el, payload, attachments.get(attNum - 1), atname);
169             }
170         }

173     protected void checkAnswers(Element el, IBaseDataObject payload, List<IBaseDataObject> attachments, String tname) throws DataConversionException {
174 
175         int numAtt = JDOMUtil.getChildIntValue(el, "numAttachments");
176         if (numAtt > -1) {
177             assertEquals(numAtt, attachments != null ? attachments.size() : 0, "Number of attachments in " + tname);
178         }
179 
180         for (Element currentForm : el.getChildren("currentForm")) {
181             String cf = currentForm.getTextTrim();
182             if (cf != null) {
183                 Attribute index = currentForm.getAttribute("index");
184                 if (index != null) {
185                     assertEquals(payload.currentFormAt(index.getIntValue()), cf,
186                             String.format("Current form '%s' not found at position [%d] in %s, %s", cf, index.getIntValue(), tname,
187                                     payload.getAllCurrentForms()));
188                 } else {
189                     assertTrue(payload.searchCurrentForm(cf) > -1,
190                             String.format("Current form %s not found in %s, %s", cf, tname, payload.getAllCurrentForms()));
191                 }
192             }
193         }
194 
195         String cf = el.getChildTextTrim("currentForm");
196         if (cf != null) {
197             assertTrue(payload.searchCurrentForm(cf) > -1,
198                     String.format("Current form '%s' not found in %s, %s", cf, tname, payload.getAllCurrentForms()));
199         }
200 
201         String ft = el.getChildTextTrim("fileType");
202         if (ft != null) {
203             assertEquals(ft, payload.getFileType(), String.format("Expected File Type '%s' in %s", ft, tname));
204         }
205 
206         int cfsize = JDOMUtil.getChildIntValue(el, "currentFormSize");
207         if (cfsize > -1) {
208             assertEquals(cfsize, payload.currentFormSize(), "Current form size in " + tname);
209         }
210 
211         String classification = el.getChildTextTrim("classification");
212         if (classification != null) {
213             assertEquals(classification, payload.getClassification(),
214                     String.format("Classification in '%s' is '%s', not expected '%s'", tname, payload.getClassification(), classification));
215         }
216 
217         int dataLength = JDOMUtil.getChildIntValue(el, "dataLength");
218         if (dataLength > -1) {
219             assertEquals(dataLength, payload.dataLength(), "Data length in " + tname);
220         }
221 
222         String shortName = el.getChildTextTrim("shortName");
223         if (shortName != null && shortName.length() > 0) {
224             assertEquals(shortName, payload.shortName(), "Shortname does not match expected in " + tname);
225         }
226 
227         String fontEncoding = el.getChildTextTrim("fontEncoding");
228         if (StringUtils.isNotBlank(fontEncoding)) {
229             assertEquals(fontEncoding, payload.getFontEncoding(), "Font encoding does not match expected in " + tname);
230         }
231 
232         String broke = el.getChildTextTrim("broken");
233         if (broke != null && broke.length() > 0) {
234             assertEquals(broke, payload.isBroken() ? "true" : "false", "Broken status in " + tname);
235         }
236 
237         // Check specified metadata
238         for (Element meta : el.getChildren("meta")) {
239             String key = meta.getChildTextTrim("name");
240             checkForMissingNameElement("meta", key, tname);
241             checkStringValue(meta, payload.getStringParameter(key), tname);
242         }
243 
244         // Check specified nometa
245         for (Element meta : el.getChildren("nometa")) {
246             String key = meta.getChildTextTrim("name");
247             checkForMissingNameElement("nometa", key, tname);
248             assertFalse(payload.hasParameter(key),
249                     String.format("Metadata element '%s' in '%s' should not exist, but has value of '%s'", key, tname,
250                             payload.getStringParameter(key)));
251         }
252 
253         // Check the primary view. Even though there is only one
254         // primary view there can be multiple elements to test it
255         // with differing matchMode operators
256         for (Element dataEl : el.getChildren("data")) {
257             byte[] payloadData = payload.data();
258             checkStringValue(dataEl, new String(payloadData), tname);
259         }
260 
261         // Check each alternate view
262         for (Element view : el.getChildren("view")) {
263             String viewName = view.getChildTextTrim("name");
264             String lengthStr = view.getChildTextTrim("length");
265             byte[] viewData = payload.getAlternateView(viewName);
266             assertNotNull(viewData, String.format("Alternate View '%s' is missing in %s", viewName, tname));
267             if (lengthStr != null) {
268                 assertEquals(Integer.parseInt(lengthStr), viewData.length,
269                         String.format("Length of Alternate View '%s' is wrong in %s", viewName, tname));
270             }
271             checkStringValue(view, new String(viewData), tname);
272         }
273 
274         // Check for noview items
275         for (Element view : el.getChildren("noview")) {
276             String viewName = view.getChildTextTrim("name");
277             byte[] viewData = payload.getAlternateView(viewName);
278             assertNull(viewData, String.format("Alternate View '%s' is present, but should not be, in %s", viewName, tname));
279         }
280 
281 
282         // Check each extract
283         String extractCountStr = el.getChildTextTrim("extractCount");
284 
285         if (payload.hasExtractedRecords()) {
286             List<IBaseDataObject> extractedChildren = payload.getExtractedRecords();
287             int foundCount = extractedChildren.size();
288 
289             if (extractCountStr != null) {
290                 assertEquals(Integer.parseInt(extractCountStr), foundCount,
291                         String.format("Number of extracted children in '%s' is %s, not expected %s", tname, foundCount, extractCountStr));
292             }
293 
294             int attNum = 1;
295             for (IBaseDataObject extractedChild : extractedChildren) {
296                 Element extel = el.getChild("extract" + attNum);
297                 if (extel != null) {
298                     checkAnswers(extel, extractedChild, NO_ATTACHMENTS, String.format("%s::extract%d", tname, attNum));
299                 }
300                 attNum++;
301             }
302         } else {
303             if (extractCountStr != null) {
304                 assertEquals(0, Integer.parseInt(extractCountStr),
305                         String.format("No extracted children in '%s' when expecting %s", tname, extractCountStr));
306             }
307         }
308     }

316     protected void checkStringValue(Element meta, String data, String tname) {
317         String key = meta.getChildTextTrim("name");
318         String value = meta.getChildText("value");
319         String matchMode = "equals";
320         Attribute mm = meta.getAttribute("matchMode");
321 
322         if (value == null) {
323             return; // checking the value is optional
324         }
325 
326         if (mm != null) {
327             matchMode = mm.getValue();
328         }
329 
330         if (matchMode.equals("equals")) {
331             assertEquals(value, data,
332                     String.format("%s element '%s' problem in %s value '%s' does not equal '%s'", meta.getName(), key, tname, data, value));
333         } else if (matchMode.equals("index") || matchMode.equals("contains")) {
334             assertTrue(data.contains(value),
335                     String.format("%s element '%s' problem in %s value '%s' does not index '%s'", meta.getName(), key, tname, data, value));
336         } else if (matchMode.equals("!index") || matchMode.equals("!contains")) {
337             assertFalse(data.contains(value),
338                     String.format("%s element '%s' problem in %s value '%s' should not be indexed in '%s'", meta.getName(), key, tname, value, data));
339         } else if (matchMode.equals("match")) {
340             assertTrue(data.matches(value),
341                     String.format("%s element '%s' problem in %s value '%s' does not match '%s'", meta.getName(), key, tname, data, value));
342         } else if (matchMode.equals("base64")) {
343             // decode value as a base64 encoded byte[] array and use the string
344             // representation of the byte array for comparison to the incoming value
345             value = new String(DatatypeConverter.parseBase64Binary(value));
346             assertEquals(value, data,
347                     String.format("%s element '%s' problem in %s value '%s' does not match '%s'", meta.getName(), key, tname, data, value));
348         } else if ("collection".equalsIgnoreCase(matchMode)) {
349             Attribute separatorAttribute = meta.getAttribute("collectionSeparator");
350             String separator = null != separatorAttribute ? separatorAttribute.getValue() : ","; // comma is default
351             // separator
352             Collection<String> expectedValues = Arrays.asList(value.split(separator));
353             Collection<String> actualValues = Arrays.asList(data.split(separator));
354             assertTrue(CollectionUtils.isEqualCollection(expectedValues, actualValues),
355                     String.format(
356                             "%s element '%s' problem in %s did not have equal collection, value '%s' does not equal '%s' split by separator '%s'",
357                             meta.getName(), key, tname, data, value, separator));
358 
359         } else {
360             fail(String.format("Problematic matchMode specified for test '%s' on %s in element %s", matchMode, key, meta.getName()));
361         }
362     }

364     protected void setupPayload(IBaseDataObject payload, Document doc) {
365         kff.hash(payload);
366         Element root = doc.getRootElement();
367         Element setup = root.getChild("setup");
368         boolean didSetFiletype = false;
369         if (setup != null) {
370             List<Element> cfChildren = setup.getChildren("initialForm");
371             if (!cfChildren.isEmpty()) {
372                 payload.popCurrentForm(); // remove default
373             }
374             for (Element cf : cfChildren) {
375                 payload.enqueueCurrentForm(cf.getTextTrim());
376             }
377 
378             final String classification = setup.getChildTextTrim("classification");
379             if (StringUtils.isNotBlank(classification)) {
380                 payload.setClassification(classification);
381             }
382 
383             final String fontEncoding = setup.getChildTextTrim("fontEncoding");
384             if (StringUtils.isNotBlank(fontEncoding)) {
385                 payload.setFontEncoding(fontEncoding);
386             }
387 
388             for (Element meta : setup.getChildren("meta")) {
389                 String key = meta.getChildTextTrim("name");
390                 String value = meta.getChildTextTrim("value");
391                 payload.appendParameter(key, value);
392             }
393 
394             for (Element altView : setup.getChildren("altView")) {
395                 String name = altView.getChildTextTrim("name");
396                 byte[] value = altView.getChildText("value").getBytes(StandardCharsets.UTF_8);
397                 payload.addAlternateView(name, value);
398             }
399 
400             final String fileType = setup.getChildTextTrim("fileType");
401             if (StringUtils.isNotBlank(fileType)) {
402                 payload.setFileType(fileType);
403                 didSetFiletype = true;
404             }
405 
406             final String inputAlternateView = setup.getChildTextTrim("inputAlternateView");
407             if (StringUtils.isNotBlank(inputAlternateView)) {
408                 final byte[] data = payload.data();
409                 payload.addAlternateView(inputAlternateView, data);
410                 payload.setData(INCORRECT_VIEW_MESSAGE);
411             }
412 
413             final String badAlternateView = setup.getChildTextTrim("badAlternateView");
414             if (StringUtils.isNotBlank(badAlternateView)) {
415                 payload.addAlternateView(badAlternateView, INCORRECT_VIEW_MESSAGE);
416             }
417         }
418         if (!didSetFiletype) {
419             payload.setFileType(payload.currentForm());
420         }

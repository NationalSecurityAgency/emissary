102     private static Element getAttributeFromDataChild(List<Element> dataList, String matchMode) {
103         Element data = null;
104         // Having different matchModes in the same data necessitates having to go through each child and
105         // filter for the correct one
106         try {
107             data = dataList.stream().filter(item -> item.getAttribute("matchMode").getValue().equals(matchMode)).findFirst().get();
108         } catch (NoSuchElementException e) {
109             fail("Attribute " + matchMode + " does not exist in data");
110         }
111         return data;
112     }

130     @Test
131     void testProcessingErrorTag() throws IOException, JDOMException {
132         IBaseDataObject d = DataObjectFactory.getInstance();
133         SAXBuilder builder = new SAXBuilder(XMLReaders.NONVALIDATING);
134         InputStream inputStream = TestExtractionTest.class.getResourceAsStream(PROC_ERROR_RESOURCE);
135         assertNotNull(inputStream, "Could not locate: " + PROC_ERROR_RESOURCE);
136         Document answerDoc = builder.build(inputStream);
137         inputStream.close();
138 
139         ExtractionTest test = spy(ExtractionTest.class);
140         // answer file is expecting 2 processing errors
141         Element answers = answerDoc.getRootElement().getChild("answers");
142         AssertionError error;
143 
144         // test with no processing error present in payload
145         error = assertThrows(AssertionError.class, () -> test.checkAnswers(answers, d, null, PROC_ERROR_RESOURCE),
146                 "Test should fail, no processing errors added to payload.");
147         assertTrue(error.getMessage().endsWith("expected: not <null>"), "Verify correct error message (assertNotNull)");
148 
149         // add first processing error to payload, verify fails at assertEquals
150         d.addProcessingError("test processing error 1");
151         error = assertThrows(AssertionError.class, () -> test.checkAnswers(answers, d, null, PROC_ERROR_RESOURCE),
152                 "Test should fail, only one processing error added to payload.");
153         assertTrue(error.getMessage().endsWith("expected: <test processing error 1;test processing error 2;> but was: <test processing error 1;>"),
154                 "Verify correct error message (assertEquals)");
155 
156         // add second procError, should pass through checkAnswers now
157         d.addProcessingError("test processing error 2");
158         test.checkAnswers(answers, d, null, PROC_ERROR_RESOURCE);
159     }

161     @Test
162     void testOsSpecificTest() throws IOException, JDOMException {
163         IBaseDataObject d = DataObjectFactory.getInstance();
164         SAXBuilder builder = new SAXBuilder(XMLReaders.NONVALIDATING);
165         InputStream inputStream = TestExtractionTest.class.getResourceAsStream(OS_SPECIFIC_RESOURCE);
166         assertNotNull(inputStream, "Could not locate: " + OS_SPECIFIC_RESOURCE);
167         Document answerDoc = builder.build(inputStream);
168         inputStream.close();
169 
170         ExtractionTest test = spy(ExtractionTest.class);
171         Element answers = answerDoc.getRootElement().getChild("answers");
172 
173         // done to test when no os specified
174         d.appendParameter("Test", "Test");
175 
176         // append bdo data for os specific dataLength & data
177         if (OSReleaseUtil.isUbuntu()) {
178             d.setData("ubuntu-dataLength-test".getBytes());
179             d.appendParameter("ubuntu-test", "ubuntu-test");
180         } else if (OSReleaseUtil.isCentOs()) {
181             d.setData("centos-dataLength-test (make different than ubuntu)".getBytes());
182             d.appendParameter("centos-test", "centos-test");
183         } else if (OSReleaseUtil.isRhel()) {
184             d.setData("rhel-dataLength-test".getBytes());
185             d.appendParameter("rhel-test", "rhel-test");
186         } else if (OSReleaseUtil.isMac()) {
187             d.setData("mac-dataLength-test".getBytes());
188             d.appendParameter("mac-test", "mac-test");
189         }
190 
191         // verify if invalid os is specified, AssertionError is thrown
192         // with all things appended above, this should only fail at checking last element
193         AssertionError error = assertThrows(AssertionError.class, () -> test.checkAnswers(answers, d, null, OS_SPECIFIC_RESOURCE),
194                 "Test should fail, last meta has a fake os.");
195         assertTrue(error.getMessage().endsWith("Provided OS=FAKE-OS"));
196     }

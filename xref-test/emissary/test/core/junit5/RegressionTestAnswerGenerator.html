30      protected static final Logger logger = LoggerFactory.getLogger(RegressionTestAnswerGenerator.class);

32      @Override
33      public void generateAnswerFiles(final String resource, final IServiceProviderPlace place, final IBaseDataObject initialIbdo,
34              final ElementEncoders encoders, final AtomicReference<Class<?>> answerFileClassRef, final String logbackLoggerName) {
35          try {
36              // Clone the BDO to create an 'after' copy
37              final IBaseDataObject finalIbdo = IBaseDataObjectHelper.clone(initialIbdo);
38              // Actually process the BDO and keep the children
39              final List<IBaseDataObject> finalResults;
40              final List<LogbackTester.SimplifiedLogEvent> finalLogEvents;
41              if (logbackLoggerName == null) {
42                  finalResults = place.agentProcessHeavyDuty(finalIbdo);
43                  finalLogEvents = new ArrayList<>();
44              } else {
45                  try (LogbackTester logbackTester = new LogbackTester(logbackLoggerName)) {
46                      finalResults = place.agentProcessHeavyDuty(finalIbdo);
47                      finalLogEvents = logbackTester.getSimplifiedLogEvents();
48                  }
49              }
50  
51              // Allow overriding things before serializing to XML
52              tweakInitialIbdoBeforeSerialization(resource, initialIbdo);
53              tweakFinalIbdoBeforeSerialization(resource, finalIbdo);
54              tweakFinalResultsBeforeSerialization(resource, finalResults);
55              tweakFinalLogEventsBeforeSerialization(resource, finalLogEvents);
56  
57              // Generate the full XML (setup & answers from before & after)
58              writeAnswerXml(resource, initialIbdo, finalIbdo, finalResults, finalLogEvents, encoders,
59                      answerFileClassRef);
60          } catch (final Exception e) {
61              logger.error("Error running test {}", resource, e);
62              fail("Unable to generate answer file", e);
63          }
64      }
65  
66      @Nullable
67      @Override
68      public Document getAnswerDocumentFor(final String resource, final AtomicReference<Class<?>> answerFileClassRef) {
69          try {
70              /* XML builder to read XML answer file in */
71              final Path path = getXmlPath(resource, answerFileClassRef);
72              return path == null ? null : new SAXBuilder(XMLReaders.NONVALIDATING).build(path.toFile());
73          } catch (final JDOMException | IOException e) {
74              // Fail if invalid XML document
75              fail(String.format("No valid answer document provided for %s", resource), e);
76              return null;
77          }
78      }
79  
80      @ForOverride
81      @Override
82      protected void tweakInitialIbdoBeforeSerialization(final String resource, final IBaseDataObject initialIbdo) {
83          initialIbdo.clearData();
84      }
85  
86      @Override
87      protected void tweakFinalIbdoBeforeSerialization(final String resource, final IBaseDataObject finalIbdo) {
88          try {
89              final Path datFileUrl = Paths.get(new ResourceReader().getResource(resource).toURI());
90              final InitialFinalFormFormat datFile = new InitialFinalFormFormat(datFileUrl);
91              if (!finalIbdo.currentForm().equals(datFile.getFinalForm())) {
92                  final String format = "Final form from place [%s] didn't match final form in filename [%s]";
93                  fail(String.format(format, finalIbdo.currentForm(), datFile.getFinalForm()));
94              }
95          } catch (final URISyntaxException e) {
96              fail("Couldn't get path for resource: " + resource, e);
97          }
98  
99          fixDisposeRunnables(finalIbdo);
100     }

64          inputStream.close();
65          String matchMode = "!index";
66  
67          WhyDoYouMakeMeDoThisExtractionTest test = new WhyDoYouMakeMeDoThisExtractionTest("nonsense");

69          List<Element> dataList = answerDoc.getRootElement().getChild("answers").getChildren("data");
70          Element bangIndexData = getAttributeFromDataChild(dataList, matchMode);
71  
72          test.checkStringValue(bangIndexData, "time:20221229", "testCheckStringValue!IndexTrue");
73          assertThrows(AssertionError.class, () -> test.checkStringValue(bangIndexData, "timestamp:20221229", "testCheckStringValue!IndexFalse"));
74  
75          matchMode = "!contains";
76  
77          Element bangContainsData = getAttributeFromDataChild(dataList, matchMode);
78  
79          test.checkStringValue(bangContainsData, "time:20221229", "testCheckStringValue!ContainsTrue");
80          assertThrows(AssertionError.class, () -> test.checkStringValue(bangContainsData, "timestamp:20221229", "testCheckStringValue!ContainsFalse"));

82      }

84      private Element getAttributeFromDataChild(List<Element> dataList, String matchMode) {
85          Element data = null;
86          // Having different matchModes in the same data necessitates having to go through each child and filter for the correct
87          // one
88          try {
89              data = dataList.stream().filter(item -> item.getAttribute("matchMode").getValue().equals(matchMode)).findFirst().get();
90          } catch (NoSuchElementException e) {
91              fail("Attribute " + matchMode + " does not exist in data");
92          }
93          return data;
94      }

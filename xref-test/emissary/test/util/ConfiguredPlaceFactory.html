14  /**
15   * Create instances of a {@link emissary.place.ServiceProviderPlace} with custom configurations for testing. Removes
16   * need for a separate .cfg file and simplifies process of using multiple configurations in the same test class.
17   * Override priority is as follows:
18   * <ol>
19   * <li>Highest priority configs are passed as params into the {@link #buildPlace(ConfigEntry...)} method</li>
20   * <li>Next priority configs are defaults passed into the {@link #ConfiguredPlaceFactory} constructor</li>
21   * <li>Lowest priority configs are found in the main {@code T} class .cfg file, if it exists</li>
22   * </ol>
23   * <i>Note</i>: setting the value of a {@link ConfigEntry} to {@code null} will remove existing instances of the
24   * configuration.
25   *
26   * @param <T> The Emissary place to create variations of
27   */
28  public class ConfiguredPlaceFactory<T extends IServiceProviderPlace> {
29      private final String placeName;
30      private final Constructor<T> placeConstructor;
31      private final Configurator defaultConfigurator;
32  
33      public ConfiguredPlaceFactory(Class<T> place, ConfigEntry... defaultConfigs) {
34          this(place, null, defaultConfigs);
35      }
36  
37      public ConfiguredPlaceFactory(Class<T> place, Configurator baseConfigurator, ConfigEntry... defaultConfigs) {
38          placeName = place.getName();
39          placeConstructor = getPlaceConstructor(place);
40          defaultConfigurator = baseConfigurator != null ? baseConfigurator : loadConfigFile(place);
41          addAndReplaceConfigEntries(defaultConfigurator, defaultConfigs);
42      }
43  
44      private Constructor<T> getPlaceConstructor(Class<T> place) {
45          try {
46              return place.getDeclaredConstructor(Configurator.class);
47          } catch (ReflectiveOperationException e) {
48              throw new IllegalStateException("Failed to create ConfiguredPlaceFactory instance for " + place.getName(), e);
49          }
50      }
51  
52      /**
53       * Gets the configurations for a given {@link emissary.place.ServiceProviderPlace} that are used in production from its
54       * .cfg file. Config files are expected to match the names of the place. E.g. {@code foo.BarPlace.java} expects
55       * {@code foo.BarPlace.cfg}. If no such file exists, an empty {@link Configurator} object will be returned.
56       *
57       * @param place The place to get configurations for
58       * @return Configurations based on the place's name if it exists, otherwise an empty {@link Configurator}
59       */
60      private Configurator loadConfigFile(Class<T> place) {
61          try {
62              return ConfigUtil.getConfigInfo(place);
63          } catch (IOException e) {
64              return new ServiceConfigGuide();
65          }
66      }
67  
68      /**
69       * Create a new instance of a {@link emissary.place.ServiceProviderPlace} for testing with optional configurations.
70       * Configs override any matching instances found in the actual .cfg file and/or default test configs.
71       * <p>
72       * <i>Note</i>: setting the value of a {@link ConfigEntry} to {@code null} will remove existing instances of the
73       * configuration.
74       *
75       * @param optionalConfigs list of new or overriding place configurations
76       * @return new instance of place
77       */
78      public T buildPlace(ConfigEntry... optionalConfigs) {
79          try {
80              Configurator classConfigs = newInstanceConfigurator(optionalConfigs);
81              return placeConstructor.newInstance(classConfigs);
82          } catch (Exception e) {
83              throw new IllegalStateException("Failed to create instance of " + placeName, e);
84          }
85      }
86  
87      /**
88       * Returns the Exception that would be thrown if the place is built with the given configurations. Actually throws an
89       * Exception if the build doesn't throw anything.
90       *
91       * @param optionalConfigs list of new or overriding place configurations
92       * @return an Exception object
93       */
94      public Exception getBuildPlaceException(ConfigEntry... optionalConfigs) {
95          return getBuildPlaceException(Exception.class, optionalConfigs);
96      }
97  
98      /**
99       * Returns the Exception that would be thrown if the place is built with the given configurations. Actually throws an
100      * Exception if the provided type doesn't match, or if the build doesn't throw anything.
101      *
102      * @param exceptionType Exception base class
103      * @param optionalConfigs list of new or overriding place configurations
104      * @return an Exception object
105      * @param <E> The Exception type that is expected to be thrown
106      */
107     public <E extends Exception> E getBuildPlaceException(Class<E> exceptionType, ConfigEntry... optionalConfigs) {
108         try {
109             Configurator classConfigs = newInstanceConfigurator(optionalConfigs);
110             placeConstructor.newInstance(classConfigs);
111         } catch (Exception e) {
112             return exceptionType.cast(e.getCause());
113         }
114         throw new IllegalStateException(
115                 String.format("Succeeded building %s but expected to throw %s", placeName, exceptionType.getName()));
116     }
117 
118     private Configurator newInstanceConfigurator(ConfigEntry... optionalConfigs) {
119         if (optionalConfigs.length == 0) {
120             return defaultConfigurator;
121         }
122         Configurator configurator = copyConfigurator(defaultConfigurator);
123         addAndReplaceConfigEntries(configurator, optionalConfigs);
124         return configurator;
125     }
126 
127     private static void addAndReplaceConfigEntries(Configurator configurator, ConfigEntry... configEntries) {
128         Set<String> encounteredKeys = new HashSet<>();
129         for (ConfigEntry entry : configEntries) {
130             String key = entry.getKey();
131             // Keep track of keys we've removed, so we don't accidentally remove them more than once
132             if (!encounteredKeys.contains(key)) {
133                 removeAllConfigEntriesForKey(configurator, key);
134                 encounteredKeys.add(key);
135             }
136             if (entry.getValue() != null) {
137                 configurator.addEntry(key, entry.getValue());
138             }
139         }
140     }
141 
142     private static void removeAllConfigEntriesForKey(Configurator configurator, String key) {
143         for (String value : configurator.findEntries(key)) {
144             configurator.removeEntry(key, value);
145         }
146     }
147 
148     private static Configurator copyConfigurator(Configurator source) {
149         Configurator target = new ServiceConfigGuide();
150         source.getEntries().forEach(entry -> target.addEntry(entry.getKey(), entry.getValue()));
151         return target;
152     }

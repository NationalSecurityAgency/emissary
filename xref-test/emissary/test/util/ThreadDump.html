3   import java.lang.management.ManagementFactory;
4   import java.lang.management.ThreadInfo;
5   import java.lang.management.ThreadMXBean;
6   import java.util.ArrayList;
7   import java.util.Arrays;
8   import java.util.List;

10  public class ThreadDump {
11      public static final List<String> SYSTEM_THREADS = List.of("Reference Handler", "Finalizer", "Signal Dispatcher");
12  
13      ThreadMXBean tmbean = ManagementFactory.getThreadMXBean();
14  
15      public List<ThreadInfo> getThreadInfo() {
16          final long[] tids = this.tmbean.getAllThreadIds();
17          return Arrays.asList(this.tmbean.getThreadInfo(tids, Integer.MAX_VALUE));
18      }
19  
20      public List<ThreadInfo> getThreadInfo(final boolean excludeSystem) {
21          final List<ThreadInfo> t = getThreadInfo();
22          final List<ThreadInfo> l = new ArrayList<>();
23          if (excludeSystem) {
24              for (final ThreadInfo ti : t) {
25                  if (ti == null) {
26                      continue;
27                  }
28                  final String name = ti.getThreadName();
29                  boolean isSystem = false;
30                  for (final String sysname : SYSTEM_THREADS) {
31                      if (sysname.equals(name)) {
32                          isSystem = true;
33                          break;
34                      }
35                  }
36                  if (!isSystem) {
37                      l.add(ti);
38                  }
39              }
40              return l;
41          }
42          return t;
43      }

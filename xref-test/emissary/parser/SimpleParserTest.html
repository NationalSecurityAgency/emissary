52      @Test
53      void testDecompWithHeaderAndFooter() throws ParserException {
54  
55          SimpleParser p = new SimpleParser(DATA) {
56  
57              @Override
58              public DecomposedSession getNextSession() throws ParserException {
59                  if (isFullyParsed()) {
60                      throw new ParserEOFException("Sessions completed");
61                  }
62  
63                  List<PositionRecord> emptyList = Collections.emptyList();
64                  InputSession i = new InputSession(O, H, F, D);
65  
66                  i.addHeaderRecs(null);
67                  i.addFooterRecs(null);
68                  i.addDataRecs(null);
69  
70                  i.addHeaderRecs(emptyList);
71                  i.addFooterRecs(emptyList);
72                  i.addDataRecs(emptyList);
73  
74                  Map<String, Object> map = new HashMap<>();
75                  map.put("KEY1", "VAL1");
76                  map.put("KEY2", new PositionRecord(2, 5));
77                  map.put("KEY3", new Object());
78                  i.addMetaDataRec("FOO", "BAR");
79                  i.addMetaDataRec("BAZ", (String) null);
80                  i.addMetaData(map);
81  
82                  // i.validateAll(); is protected so...
83                  i.setOverall(O);
84                  i.setValid(true);
85                  setFullyParsed(true);
86  
87                  String s = i.toString();
88                  assertNotNull(s, "InputSession to string");
89  
90                  return decomposeSession(i);
91              }
92          };
93  
94          DecomposedSession d = p.getNextSession();
95          assertNotNull(d, "Decomposition");
96          assertTrue(d.isValid(), "Session decomposed");
97          assertTrue(d.hasHeader(), "Header decomposition");
98          assertTrue(d.hasFooter(), "Footer decomposition");
99          assertTrue(d.hasData(), "Data decomposition");
100         assertTrue(d.hasMetaData(), "Metadata decompsition");
101         byte[] header = d.getHeader();
102         assertNotNull(header, "Header decomposition");
103         assertEquals(10, header.length, "Header decomp size");
104         byte[] footer = d.getFooter();
105         assertNotNull(footer, "Footer decomposition");
106         assertEquals(10, footer.length, "Footer decomp size");
107         byte[] data = d.getData();
108         assertNotNull(data, "Data decomposition");
109         Map<String, Collection<Object>> meta = d.getMetaData();
110         assertNotNull(meta, "MetaData map extraction");
111         assertTrue(meta.size() > 0, "MetaData map population " + meta.keySet());
112         assertEquals("BAR", d.getStringMetadataItem("FOO"), "MetaData map value");
113         assertTrue(meta.containsKey(SessionParser.ORIG_DOC_SIZE_KEY), "MetaData should always have doc size");
114         assertFalse(meta.containsKey("BAZ"), "Null metadata values not carried");
115         assertTrue(meta.containsKey("KEY1"), "Map string entry");
116         assertTrue(meta.containsKey("KEY2"), "Map posrec entry");
117         assertFalse(meta.containsKey("KEY3"), "Map object entry");
118         assertEquals(80, data.length, "Data decomp size");
119         assertFalse(d.hasClassification(), "SimleParser should not set classification");
120 
121 
122         int oldMetaSize = meta.size();
123         Map<String, Object> newMeta = new HashMap<>();
124         newMeta.put("NEWKEY1", "NEWVALUE1");
125         newMeta.put("NEWKEY2", "NEWVALUE2");
126         d.addMetaData(newMeta);
127         meta = d.getMetaData();
128         assertEquals(newMeta.size() + oldMetaSize, meta.size(), "Consolidated meta");
129     }

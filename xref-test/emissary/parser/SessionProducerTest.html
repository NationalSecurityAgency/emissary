25      @Test
26      void testZoneAssignments() {
27          SessionParser parser = Mockito.mock(SessionParser.class);
28          SessionProducer sp = new SessionProducer(parser, "UNKNOWN");
29          DecomposedSession d = new DecomposedSession();
30          d.setHeader("The Header".getBytes());
31          d.setFooter("The Footer".getBytes());
32          d.setData("The Data".getBytes());
33          d.setClassification("Purple Nurple");
34          d.addMetaData("foo", "bar");
35          IBaseDataObject payload = sp.createAndLoadDataObject(d, "name");
36          assertEquals("The Header", new String(payload.header()), "Incorrect header");
37          assertEquals("The Footer", new String(payload.footer()), "Incorrect footer");
38          assertEquals("The Data", new String(payload.data()), "Incorrect data");
39          assertEquals("Purple Nurple", payload.getClassification(), "Incorrect classification");
40          assertEquals("bar", payload.getStringParameter("foo"), "Incorrect metadata");
41          assertEquals("name", payload.shortName(), "Incorrect name");
42      }

44      @Test
45      void testAlternateViewAssignment() {
46          SessionParser parser = Mockito.mock(SessionParser.class);
47          SessionProducer sp = new SessionProducer(parser, "UNKNOWN");
48          DecomposedSession d = new DecomposedSession();
49          d.addMetaData("FOO", "BAR");
50          d.addMetaData("ALT_VIEW_FOOVIEW", "This is the view data".getBytes());
51          d.addMetaData("ALT_VIEW_BARVIEW", "This is string data for a view");
52          IBaseDataObject payload = sp.createAndLoadDataObject(d, "name");
53          assertNotNull(payload.getAlternateView("FOOVIEW"), "Alt view must be created from byte array");
54          assertNotNull(payload.getAlternateView("BARVIEW"), "Alt veiw must be created from string");
55          assertEquals("This is the view data", new String(payload.getAlternateView("FOOVIEW")), "Byte array view data incorrect");
56          assertEquals("This is string data for a view", new String(payload.getAlternateView("BARVIEW")), "String view data incorrect");
57          assertNull(payload.getParameter("ALT_VIEW_FOOVIEW"), "View data must be removed from normal metadata");
58          assertNull(payload.getParameter("ALT_VIEW_BARVIEW"), "View data must be removed from normal metadata");
59      }

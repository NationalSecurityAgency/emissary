36      @Test
37      void testSortPlaces() throws IOException {
38          List<String> somePlaces = new ArrayList<>();
39          EmissaryNode node = new EmissaryNode();
40          String location = "http://" + node.getNodeName() + ":" + node.getNodePort();
41          somePlaces.add(location + "/" + CoordinationPlace.class.getSimpleName());
42          somePlaces.add(location + "/" + DelayPlace.class.getSimpleName());
43          somePlaces.add(location + "/" + DevNullPlace.class.getSimpleName());
44          somePlaces.add(location + "/" + FilePickUpPlace.class.getSimpleName());
45          somePlaces.add(location + "/" + FilePickUpClient.class.getSimpleName());
46  
47  
48          Startup startup = new Startup(node.getNodeConfigurator(), node);
49          startup.sortPlaces(somePlaces);
50          // TODO: figure out why the key isn't http://localhost:8001. Seems
51          // this way when running too, so get("") works. This test is just testing the
52          // PickUp stuff is pulled out separately
53          Set<String> pickups = startup.pickupLists.get("");
54          Set<String> places = startup.placeLists.get("");
55          assertIterableEquals(Arrays.asList(location + "/" + "FilePickUpPlace", location + "/" + "FilePickUpClient"), pickups);
56          assertIterableEquals(Arrays.asList(location + "/" + "CoordinationPlace", location + "/" + "DelayPlace", location + "/" + "DevNullPlace"),
57                  places);
58      }

60      @Test
61      void testInvisPlaceStart() throws IOException {
62          // setup node, startup, and DirectoryPlace
63          EmissaryNode node = new EmissaryNode();
64          String location = "http://" + node.getNodeName() + ":" + node.getNodePort();
65          dirStartUp();
66  
67          // test if place is already started before startup
68          Startup.activeDirPlaces.add(location + "/PlaceAlreadyStartedTest");
69          Startup.placeAlreadyStarted.add(location + "/PlaceAlreadyStartedTest");
70          assertTrue(Startup.verifyNoInvisiblePlacesStarted());
71          Startup.activeDirPlaces.clear();
72          Startup.placeAlreadyStarted.clear();
73  
74          // test if place is started up normally and is active dir
75          Startup.places.put(location, DevNullPlace.class.getSimpleName());
76          Startup.activeDirPlaces.add(DevNullPlace.class.getSimpleName());
77          assertTrue(Startup.verifyNoInvisiblePlacesStarted());
78          Startup.activeDirPlaces.clear();
79          Startup.places.remove(location, DevNullPlace.class.getSimpleName());
80  
81          // test unannounced place with active dir
82          Startup.activeDirPlaces.add(location + "/PlaceStartUnannouncedTest");
83          assertFalse(Startup.verifyNoInvisiblePlacesStarted());
84          Startup.activeDirPlaces.clear();
85  
86          // teardown
87          dirTeardown();
88      }

90      @Test
91      void verifyNoInvisiblePlacesStartedHandlesNullLocalPlace() {
92          try (MockedStatic<DirectoryPlace> dirPlace = Mockito.mockStatic(DirectoryPlace.class)) {
93  
94              DirectoryEntry entry = mock(DirectoryEntry.class);
95              when(entry.getLocalPlace()).thenReturn(null);
96  
97              List<DirectoryEntry> dirEntries = new ArrayList<>();
98              dirEntries.add(0, entry);
99  
100             IDirectoryPlace directoryPlace = mock(IDirectoryPlace.class);
101             when(directoryPlace.getEntries()).thenReturn(dirEntries);
102 
103             dirPlace.when(DirectoryPlace::lookup).thenReturn(directoryPlace);
104 
105             assertTrue(Startup.verifyNoInvisiblePlacesStarted());
106         }
107     }

109     @Nullable
110     private DirectoryPlace server = null;
111     @Nullable
112     private DirectoryPlace client = null;
113 
114     public void dirStartUp() throws IOException {
115         // server directory
116         InputStream serverConfigStream = new ResourceReader().getConfigDataAsStream(this);
117         String serverLocation = "http://localhost:8001/TestServerDirectoryPlace";
118         this.server = spy(new DirectoryPlace(serverConfigStream, serverLocation, new EmissaryNode()));
119         serverConfigStream.close();
120         Namespace.bind(serverLocation, this.server);
121 
122         // client directory
123         InputStream clientConfigStream = new ResourceReader().getConfigDataAsStream(this);
124         String clientLocation = "http://localhost:8001/DirectoryPlace";
125         this.client = spy(new DirectoryPlace(clientConfigStream, this.server.getDirectoryEntry().getKey(), clientLocation, new EmissaryNode()));
126         clientConfigStream.close();
127         Namespace.bind(clientLocation, this.client);
128     }
129 
130     public void dirTeardown() {
131         this.server.shutDown();
132         this.server = null;
133         this.client.shutDown();
134         this.client = null;
135         for (String s : Namespace.keySet()) {
136             Namespace.unbind(s);
137         }
138     }

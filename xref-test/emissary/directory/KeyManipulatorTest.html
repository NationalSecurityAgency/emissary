30      void testCost() {
31          final String s = "UNKNOWN.FOOPLACE.ID.http://hosta.domain.com:8001/thePlace";
32          final String t = "UNKNOWN.FOOPLACE.ID.http://hostb.domain.com:8001/thePlace$5050";
33  
34          assertEquals(5050, KeyManipulator.getExpense(t, -99), "Expense");
35          assertEquals(-99, KeyManipulator.getExpense(s, -99), "Default expense");

37          assertEquals(5150, KeyManipulator.getExpense(KeyManipulator.addExpense(t, 5150)), "Expense addition");
38          assertEquals(5050, KeyManipulator.getExpense(KeyManipulator.addExpense(t, 5050)), "Expense no change during addition");
39          assertEquals(100, KeyManipulator.getExpense(KeyManipulator.addExpense(s, 100)), "Expense addition to default");
40  
41  
42          assertEquals(5050 + IDirectoryPlace.REMOTE_EXPENSE_OVERHEAD,
43                  KeyManipulator.getExpense(KeyManipulator.addRemoteCostIfMoved(s, t)),
44                  "Remote expense addition");
45          assertEquals(5050, KeyManipulator.getExpense(KeyManipulator.addRemoteCostIfMoved(t, t)), "Remote expense no addition");

74      @Test
75      void testMatch() {
76          final String t = "UNKNOWN.FOOPLACE.ID.http://hostb.domain.com:8001/thePlace$5050";
77          assertTrue(KeyManipulator.gmatch(t, "UNKNOWN.*"), "Gmatch on data type");
78          assertTrue(KeyManipulator.gmatch(t, "*.FOOPLACE.*"), "Gmatch on serivce type");
79          assertTrue(KeyManipulator.gmatch(t, "*.ID.*"), "Gmatch on data type");
80  
81          assertTrue(KeyManipulator.gmatch(t, "*.*.*.*"), "Gmatch on four wilds");
82          assertTrue(KeyManipulator.gmatch(t, "UNKNOWN.*.*.*"), "Gmatch on three wilds");
83          assertTrue(KeyManipulator.gmatch(t, "UNKNOWN.FOOPLACE.*.*"), "Gmatch on two wilds");
84          assertTrue(KeyManipulator.gmatch(t, "UNKNOWN.FOOPLACE.ID.*"), "Gmatch on one wild");
85  
86          assertFalse(KeyManipulator.gmatch(t, "*.*.*.*x"), "Gmatch on four wilds");
87          assertFalse(KeyManipulator.gmatch(t, "A.*.*.*"), "Gmatch on wilds wrong first");
88          assertFalse(KeyManipulator.gmatch(t, "*.B.*.*"), "Gmatch on wilds wrong second");
89          assertFalse(KeyManipulator.gmatch(t, "*.*.DORK.*"), "Gmatch on wilds wrong third");
90          assertFalse(KeyManipulator.gmatch(t, "*.*.*.tcp://hostb.domain.com:8001/thePlace$5050"), "Gmatch on wilds wrong fourth");
91  
92          assertTrue(KeyManipulator.gmatch(t, "UNKNOWN.?OOPLACE.ID.http://hostb.domain.com:8001/thePlace$5050"), "Gmatch on single pos wildcard");
93          assertTrue(KeyManipulator.gmatch(t, "UNKNOWN.???PLACE.ID.http://hostb.domain.com:8001/thePlace$5050"), "Gmatch on triple pos wildcard");
94          assertTrue(KeyManipulator.gmatch(t, "UNKNOWN.?OOPLACE.?D.http://hostb.domain.com:?001/thePlace$5050"), "Gmatch on disjoint ? wildcard");
95          assertFalse(KeyManipulator.gmatch(t, "UNKNOWN.?FOOPLACE.ID.http://hostb.domain.com:8001/thePlace$5050"),
96                  "No Gmatch on wrong single pos wildcard");
97          assertFalse(KeyManipulator.gmatch(t, "BLUBBER.*"), "No Gmatch on wrong data type");
98          assertTrue(KeyManipulator.gmatch(t, "UNKNOWN.*.ID.*"), "Gmatch on data id");
99  
100     }

150     @Test
151     void testProxyKeyGen() {
152         final String s = "UNKNOWN.FOOPLACE.ID.http://hosta.domain.com:8001/thePlace$9090";
153         final String proxyHost = "*.*.*.http://hostb.otherdomain.com:9001/OtherPlace";
154         assertEquals("UNKNOWN.FOOPLACE.ID.http://hostb.otherdomain.com:9001/OtherPlace$9090",
155                 KeyManipulator.makeProxyKey(s, proxyHost, 9090),
156                 "Proxy key construction");
157         assertEquals("UNKNOWN.FOOPLACE.ID.http://hostb.otherdomain.com:9001/OtherPlace$9090",
158                 KeyManipulator.makeProxyKey(s, proxyHost, 8080),
159                 "Proxy key construction");
160 
161         final String t = "UNKNOWN.FOOPLACE.ID.http://hosta.domain.com:8001/thePlace";
162         assertEquals("UNKNOWN.FOOPLACE.ID.http://hostb.otherdomain.com:9001/OtherPlace$9090",
163                 KeyManipulator.makeProxyKey(t, proxyHost, 9090),
164                 "Proxy key construction");
165         assertEquals("UNKNOWN.FOOPLACE.ID.http://hostb.otherdomain.com:9001/OtherPlace$8080",
166                 KeyManipulator.makeProxyKey(t, proxyHost, 8080),
167                 "Proxy key construction");
168         assertEquals("UNKNOWN.FOOPLACE.ID.http://hostb.otherdomain.com:9001/OtherPlace",
169                 KeyManipulator.makeProxyKey(t, proxyHost, -1),
170                 "Proxy key dflt exp construction");
171 
172         final String u = "UNKNOWN.FOOPLACE.ID.http://hostb.otherdomain.com:9001/OtherPlace";
173         assertEquals(u, KeyManipulator.makeProxyKey(u, proxyHost, -1), "Proxy key is local already");
174         assertEquals(u + "$9090", KeyManipulator.makeProxyKey(u, proxyHost, 9090), "Proxy key is local already but needs cost");
175     }

161     @Test
162     void testCopy() {
163         this.dl.add(this.d);
164         this.dl.add(this.d2);
165         final DirectoryEntryList dl2 = new DirectoryEntryList();
166         dl2.addAll(this.dl);
167         assertEquals(this.dl.size(), dl2.size(), "Copy size");
168 
169         final DirectoryEntryList dl3 = new DirectoryEntryList(this.dl, true);
170         assertEquals(this.dl.size(), dl3.size(), "Copy ctor size");
171 
172         // Assure that the copy is deep
173         final DirectoryEntry e1 = dl3.get(0);
174         e1.setCost(cost * 4);
175         assertEquals(cost, this.dl.get(0).getCost(), "Cost of original entry unchanged");
176 
177         // Test possible NPE path
178         final DirectoryEntryList dl4 = new DirectoryEntryList(null);
179         assertEquals(0, dl4.size(), "Copy ctor size on null");
180     }

101     @Test
102     void testXmlToObject() throws JDOMException {
103         final int exp = DirectoryEntry.calculateExpense(cost, quality);
104         final String xml =
105                 "<entry><key>" + key + "</key><cost>" + cost + "</cost><quality>" + quality
106                         + "</quality><description>This is the description</description><expense>" + exp + "</expense><place>" + key
107                         + "</place></entry>";
108         final Document jdom = JDOMUtil.createDocument(xml, false);
109         final DirectoryEntry dxml = DirectoryEntry.fromXML(jdom.getRootElement());
110         assertNotNull(dxml, "DirectoryEntry created from xml");
111         assertEquals(cost, dxml.getCost(), "Cost from xml");
112         assertEquals(quality, dxml.getQuality(), "Quality from xml");
113         assertEquals(exp, dxml.getExpense(), "Expense from xml");
114         assertEquals("This is the description", dxml.getDescription(), "Description from xml");
115         assertEquals(key, dxml.getKey(), "Key from xml");
116         assertEquals(key + KeyManipulator.DOLLAR + exp, dxml.getFullKey(), "Full key from xml");
117     }

130         this.d.proxyFor(proxyKey);
131         assertEquals(cost, this.d.getCost(), "Proxy cost");
132         assertEquals(quality, this.d.getQuality(), "Proxy qual");
133         assertEquals(KeyManipulator.getServiceType(key), this.d.getServiceType(), "Proxy service type");

135         assertEquals(KeyManipulator.getDataType(key), KeyManipulator.getDataType(this.d.getKey()), "Proxy data type");
136         assertEquals(DirectoryEntry.calculateExpense(cost, quality), this.d.getExpense(), "Proxy expense");
137 
138         // Proxy key cost does not change it
139         proxyKey = "*.*.*.http://host2.domain.com:9001/OtherPlace$12345";

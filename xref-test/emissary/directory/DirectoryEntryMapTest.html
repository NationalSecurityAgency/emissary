62      @Test
63      void testRemoveEntry() {
64          // remove first on list
65          DirectoryEntry removed = this.dm.removeEntry(this.d.getKey());
66          assertNotNull(removed, "Removed entry");
67          assertEquals(this.d.getKey(), removed.getKey(), "Removed entry");
68          assertEquals(this.d, removed, "No copy on remove");

70          // Add it back and remove second on list
71          this.dm.addEntry(this.d);
72          removed = this.dm.removeEntry(this.d2.getKey());
73          assertNotNull(removed, "Removed entry");
74          assertEquals(this.d2.getKey(), removed.getKey(), "Remove entry");
75  
76          // no such entry on list
77          DirectoryEntry notfound = this.dm.removeEntry("FOO.BAR.BAZ.http://host:8001/FooPlace");
78          assertNull(notfound, "Removed entry not found");

80          // no such list
81          notfound = this.dm.removeEntry("BAR.FOO.BAZ.http://host:8001/FooPlace");

147     @Test
148     void testCollectAllMatching() {
149         // Create a bunch of entries spread out onto several data identifiers
150         final String[] phases = new String[] {"ID", "TRANSFORM", "ANALYZE", "IO"};
151         final int phasesLength = phases.length;
152         final String[] forms = new String[] {"UNKNOWN", "PETERPAN", "WENDY", "CROC", "MR_SMEE"};
153         final int formsLength = forms.length;
154         for (int i = 1; i < 10000; i++) {
155             String sb = forms[i % formsLength] +
156                     ".FOOPLACE." +
157                     phases[i % phasesLength] +
158                     ".http://host-" +
159                     i +
160                     ".domain.com:7001/ThePlace";
161             this.dm.addEntry(new DirectoryEntry(sb, "blah", i % 100, 50));
162             // dm.addEntry(new DirectoryEntry(String.format("%s.FOOPLACE.%s.http://host-%d.domain.com:7001/ThePlace",
163             // forms[i % formsLength], phases[i
164             // % phasesLength], i), "blah", i % 100, 50));
165         }
166         final int[] expectedCount = new int[] {2501, 2002, 501, 2002, 501, 10002, 10002, 1};
167         final String[] searchPatterns =
168                 new String[] {"*.*.ID.*", "UNKNOWN.*.*.*", "UNKNOWN.*.ID.*", "?NKNOWN*.*.*.*", "UNKNOWN.*.TRANSFORM.*", "*.FOOPLACE.*.*", "*.*.*.*",
169                         "*.FOOPLACE.*.http://host-1001.domain.com:7001/ThePlace"};
170         for (int i = 0; i < searchPatterns.length; i++) {
171             final long start = System.currentTimeMillis();
172             final List<DirectoryEntry> matches = this.dm.collectAllMatching(searchPatterns[i]);
173             final long end = System.currentTimeMillis();
174             assertEquals(expectedCount[i], matches.size(), "Match must produce results using pattern " + searchPatterns[i]);
175             assertTrue((end - start) < 1000, "Search using pattern " + searchPatterns[i] + " must be under 1sec");
176         }
177     }
